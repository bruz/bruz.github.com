291   leftAlign: function() {
292     var val = 0, controlSize = this.get('controlSize') ;

294     // what. the. heck?
295     // no, I don't want to shift the menu to the left. Yet.
296     if(controlSize === SC.SMALL_CONTROL_SIZE) val = -14 ;

302   /**
303     override this method to implement your own sorting of the menu. By
304     default, menu items are sorted using the value shown or the sortKey
305 
306     @param{SC.Array} objects the unsorted array of objects to display.
307     @returns sorted array of objects
308   */
309   sortObjects: function(objects) {
310     if(!this.get('disableSort')){
311       var nameKey = this.get('itemSortKey') || this.get('itemTitleKey') ;
312       objects = objects.sort(function(a,b) {
313         if (nameKey) {
314           a = a.get ? a.get(nameKey) : a[nameKey] ;
315           b = b.get ? b.get(nameKey) : b[nameKey] ;
316         }
317         return (a<b) ? -1 : ((a>b) ? 1 : 0) ;
318       }) ;
319     }
320     return objects ;
321   },

323   /**
324     render method
325 
326     @private
327   */
328   render: function(context,firstTime) {
329     sc_super();
330     var layoutWidth, items, len, nameKey, iconKey, valueKey, separatorKey, showCheckbox,
331       currentSelectedVal, shouldLocalize, isSeparator, itemList, isChecked,
332       idx, name, icon, value, item;
333 
334     items = this.get('items') ;
335     items = this.sortObjects(items) ;
336     len = items.length ;
337 
338     //Get the namekey, iconKey and valueKey set by the user
339     nameKey = this.get('itemTitleKey') ;
340     iconKey = this.get('itemIconKey') ;
341     valueKey = this.get('itemValueKey') ;
342     separatorKey = this.get('itemSeparatorKey');
343     showCheckbox = this.get('showCheckbox') ;
344 
345     //get the current selected value
346     currentSelectedVal = this.get('value') ;
347 
348     // get the localization flag.
349     shouldLocalize = this.get('localize') ;
350 
351     //itemList array to set the menu items
352     itemList = [] ;
353 
354     //to set the 'checkbox' property of menu items
355     isChecked = YES ;
356 
357     //index for finding the first item in the list
358     idx = 0 ;
359 
360     items.forEach(function(object) {
361     if (object) {
362 
363       //Get the name value. If value key is not specified convert obj
364       //to string
365       name = nameKey ? (object.get ?
366         object.get(nameKey) : object[nameKey]) : object.toString() ;
367 
368       // localize name if specified.
369       name = shouldLocalize? name.loc() : name ;
370 
371       //Get the icon value
372       icon = iconKey ? (object.get ? 
373         object.get(iconKey) : object[iconKey]) : null ;
374       if (SC.none(object[iconKey])) icon = null ;
375 
376       // get the value using the valueKey or the object
377         value = (valueKey) ? (object.get ?
378         object.get(valueKey) : object[valueKey]) : object ;
379 
380       if (!SC.none(currentSelectedVal) && !SC.none(value)){
381         if( currentSelectedVal === value ) {
382           this.set('title', name) ;
383           this.set('icon', icon) ;
384         }
385       }
386 
387       //Check if the item is currentSelectedItem or not
388       if(value === this.get('value')) {
389 
390         //set the _itemIdx - To change the prefMatrix accordingly.
391         this.set('_itemIdx', idx) ;
392         isChecked = !showCheckbox ? NO : YES ;
393       }
394       else {
395         isChecked = NO ;
396       }
397       
398       // get the separator
399       isSeparator = separatorKey ? (object.get ? object.get(separatorKey) : object[separatorKey]) : NO;
400 
401       //Set the first item from the list as default selected item
402       if (idx === 0) {
403         this._defaultVal = value ;
404         this._defaultTitle = name ;
405         this._defaultIcon = icon ;
406       }
407 
408       var item = SC.Object.create({
409         separator: isSeparator,
410         title: name,
411         icon: icon,
412         value: value,
413         isEnabled: YES,
414         checkbox: isChecked,
415         action: this.displaySelectedItem
416       }) ;
417 
418       //Set the items in the itemList array
419       itemList.push(item);
420 
421     }
422 
423     idx += 1 ;
424 
425     this.set('_itemList', itemList) ;
426     }, this ) ;
427 
428     if(firstTime) {
429       this.invokeLast(function() {
430         var value = this.get('value') ;
431         if(SC.none(value)) {
432           this.set('value', this._defaultVal) ;
433           this.set('title', this._defaultTitle) ;
434           this.set('icon', this._defaultIcon) ;
435         }
436       });
437     }
438 
439     //Set the preference matrix for the menu pane
440     this.changeSelectButtonPreferMatrix(this._itemIdx) ;
441 
442   },

444   /**
445     Button action handler
446 
447     @private
448     @param {DOMMouseEvent} evt mouseup event that triggered the action
449   */
450   _action: function( evt )
451   {
452     var buttonLabel, menuWidth, scrollWidth, lastMenuWidth, offsetWidth,
453       items, elementOffsetWidth, largestMenuWidth, item, element, idx,
454       currSel, itemList, menuControlSize, menuHeightPadding, customView,
455       customMenuView, menu, itemsLength;
456       
457     buttonLabel = this.$('.sc-button-label')[0] ;
458     // Get the length of the text on the button in pixels
459     menuWidth = this.get('layer').offsetWidth ;
460     scrollWidth = buttonLabel.scrollWidth ;
461     lastMenuWidth = this.get('lastMenuWidth') ;
462     if(scrollWidth) {
463        // Get the original width of the label in the button
464        offsetWidth = buttonLabel.offsetWidth ;
465        if(scrollWidth && offsetWidth) {
466           menuWidth = menuWidth + scrollWidth - offsetWidth ;
467        }
468     }
469     if (!lastMenuWidth || (menuWidth > lastMenuWidth)) {
470       lastMenuWidth = menuWidth ;
471     }
472 
473     items = this.get('_itemList') ;
474 
475     var customViewClassName = this.get('customViewClassName') ;
476     var customViewMenuOffsetWidth = this.get('customViewMenuOffsetWidth') ;
477     var className = 'sc-view sc-pane sc-panel sc-palette sc-picker sc-menu select-button sc-scroll-view sc-menu-scroll-view sc-container-view menuContainer sc-button-view sc-menu-item sc-regular-size' ;
478     className = customViewClassName ? (className + ' ' + customViewClassName) : className ;
479     
480     SC.prepareStringMeasurement("", className);
481     for (idx = 0, itemsLength = items.length; idx < itemsLength; ++idx) {
482       //getting the width of largest menu item
483       item = items.objectAt(idx) ;
484       elementOffsetWidth = SC.measureString(item.title).width;
485 
486       if (!largestMenuWidth || (elementOffsetWidth > largestMenuWidth)) {
487         largestMenuWidth = elementOffsetWidth ;
488       }
489     }
490     SC.teardownStringMeasurement();
491 
492     lastMenuWidth = (largestMenuWidth + this.menuItemPadding > lastMenuWidth) ?
493                       largestMenuWidth + this.menuItemPadding : lastMenuWidth ;
494 
495     // Get the window size width and compare with the lastMenuWidth.
496     // If it is greater than windows width then reduce the maxwidth by 25px
497     // so that the ellipsis property is enabled by default
498     var maxWidth = SC.RootResponder.responder.get('currentWindowSize').width;
499     if(lastMenuWidth > maxWidth) {
500       lastMenuWidth = (maxWidth - 25) ;
501     }
502 
503     this.set('lastMenuWidth',lastMenuWidth) ;
504     currSel = this.get('_currentSelItem') ;
505     itemList = this.get('_itemList') ;
506     menuControlSize = this.get('controlSize') ;
507     menuHeightPadding = this.get('menuPaneHeightPadding') ;
508 
509     // get the user defined custom view
510     customView = this.get('exampleView') ;
511     customMenuView = customView ? customView : SC.MenuItemView ;
512 
513     menu  = SC.MenuPane.create({
514 
515       /**
516         Class name - select-button-item
517       */
518       classNames: ['select-button'],
519 
520       /**
521         The menu items are set from the itemList property of SelectButton
522 
523         @property
524       */
525       items: itemList,
526 
527       /**
528         Example view which will be used to create the Menu Items
529 
530         @default SC.MenuItemView
531         @type SC.View
532       */
533       exampleView: customMenuView,
534 
535       /**
536         This property enables all the items and makes them selectable.
537 
538         @property
539       */
540       isEnabled: YES,
541 
542       menuHeightPadding: menuHeightPadding,
543 
544       preferType: SC.PICKER_MENU,
545       itemHeightKey: 'height',
546       layout: { width: lastMenuWidth },
547       controlSize: menuControlSize,
548       itemWidth: lastMenuWidth,
549       contentView: SC.View.extend({
550       })
551     }) ;
552 
553     // no menu to toggle... bail...
554     if (!menu) return NO ;
555     menu.popup(this, this.preferMatrix) ;
556     menu.set('currentSelectedMenuItem', currSel) ;
557     return YES ;
558   },

560   /**
561      Action method for the select button menu items
562 
563   */
564   displaySelectedItem: function() {
565     var menuView, currSel, itemViews, title, val, itemIdx = 0, button, object,
566       len, found = null, objTmp;
567     
568     //Get MenuPane, currentSelectedMenuItem & menuItemView
569     // Get the main parent view to show the menus
570       
571     menuView = this.parentMenu() ;
572     currSel = menuView.get('currentSelectedMenuItem') ;
573     itemViews = menuView.menuItemViews ;
574     
575     //  Fetch the index of the current selected item
576     if (currSel && itemViews) {
577       itemIdx = itemViews.indexOf(currSel) ;
578     }
579 
580     // Get the select button View
581     button = menuView.get('anchor') ;
582 
583     // set the value and title
584     object = menuView.get('items') ;
585     len = object.length ;
586 
587     while (!found && (--len >= 0)) {
588       objTmp = object[len];
589       title = !SC.none(objTmp.title) ? objTmp.title: object.toString() ;
590       val =  !SC.none(objTmp.value) ? objTmp.value: title ;
591 
592       if (title === this.get('value') && (itemIdx === len)) {
593         found = object ;
594         button.set('value', val) ;
595         button.set('title', title) ;
596       }
597     }
598 
599     // set the icon, currentSelectedItem and itemIdx
600     button.set('icon', this.get('icon')).set('_currentSelItem', currSel).
601       set('_itemIdx', itemIdx) ;
602   },

604   /**
605      Set the "top" attribute in the prefer matrix property which will
606      position menu such that the selected item in the menu will be
607      place aligned to the item on the button when menu is opened.
608   */
609   changeSelectButtonPreferMatrix: function() {
610     var preferMatrixAttributeTop = 0 ,
611       itemIdx = this.get('_itemIdx') ,
612       leftAlign = this.get('leftAlign'), defPreferMatrix, tempPreferMatrix ;
613 
614     if(this.get('positionMenuBelow')) {
615       defPreferMatrix = [leftAlign, 4, 3] ;
616       this.set('preferMatrix', defPreferMatrix) ;
617     }
618     else {
619       if(itemIdx) {
620         preferMatrixAttributeTop = itemIdx * this.CUSTOM_MENU_ITEM_HEIGHT ;
621       }
622       tempPreferMatrix = [leftAlign, -preferMatrixAttributeTop, 2] ;
623       this.set('preferMatrix', tempPreferMatrix) ;
624     }
625   },

646   keyDown: function(event) {

648       return YES;
649     }
650     else {
651       arguments.callee.base.apply(this,arguments);
652     }

655   /**
656     @private
657 
658     Pressing the Up or Down arrow key should display the menu pane
659   */
660   interpretKeyEvents: function(event) {
661     if (event) {
662       if ((event.keyCode === 38 || event.keyCode === 40)) {
663         this._action() ;
664       }
665       else if (event.keyCode === 27) {
666         this.resignFirstResponder() ;
667       }
668     }
669     return arguments.callee.base.apply(this,arguments);
670   }
