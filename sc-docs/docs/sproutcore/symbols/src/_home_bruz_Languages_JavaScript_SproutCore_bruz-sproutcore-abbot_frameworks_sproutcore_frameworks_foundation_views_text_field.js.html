141   /**
142     The current selection of the text field, returned as an SC.TextSelection
143     object.
144 
145     Note that if the selection changes a new object will be returned -- it is
146     not the case that a previously-returned SC.TextSelection object will
147     simply have its properties mutated.
148 
149     @property {SC.TextSelection}
150   */
151   selection: function(key, value) {
152     var element = this.$input()[0],
153         range, start, end;
154 
155     // Are we being asked to set the value, or return the current value?
156     if (value === undefined) {
157       // The client is retrieving the value.
158       if (element) {
159         start = null;
160         end = null;
161 
162         if (!element.value) {
163           start = end = 0 ;
164         }
165         else {
166           // In IE8, input elements don't have hasOwnProperty() defined.
167           if ('selectionStart' in element) {
168             start = element.selectionStart ;
169           }
170           if ('selectionEnd' in element) {
171             end = element.selectionEnd ;
172           }
173 
174           // Support Internet Explorer.
175           if (start === null  ||  end === null ) {
176             var selection = document.selection ;
177             if (selection) {
178               var type = selection.type ;
179               if (type  &&  (type === 'None'  ||  type === 'Text')) {
180                 range = selection.createRange() ;
181 
182                 if (!this.get('isTextArea')) {
183                   // Input tag support.  Figure out the starting position by
184                   // moving the range's start position as far left as possible
185                   // and seeing how many characters it actually moved over.
186                   var length = range.text.length ;
187                   start = Math.abs(range.moveStart('character', 0 - (element.value.length + 1))) ;
188                   end = start + length ;
189                 }
190                 else {
191                   // Textarea support.  Unfortunately, this case is a bit more
192                   // complicated than the input tag case.  We need to create a
193                   // "dummy" range to help in the calculations.
194                   var dummyRange = range.duplicate() ;
195                   dummyRange.moveToElementText(element) ;
196                   dummyRange.setEndPoint('EndToStart', range) ;
197                   start = dummyRange.text.length ;
198                   end = start + range.text.length ;
199                 }
200               }
201             }
202           }
203         }
204         return SC.TextSelection.create({ start:start, end:end }) ;
205       }
206       else {
207         return null;
208       }
209     }
210     else {
211       // The client is setting the value.  Make sure the new value is a text
212       // selection object.
213       if (!value  ||  !value.kindOf  ||  !value.kindOf(SC.TextSelection)) {
214         throw "When setting the selection, you must specify an SC.TextSelection instance.";
215       }
216 
217       if (element) {
218         if (element.setSelectionRange) {
219           element.setSelectionRange(value.get('start'), value.get('end')) ;
220         }
221         else {
222           // Support Internet Explorer.
223           range = element.createTextRange() ;
224           start = value.get('start') ;
225           range.move('character', start) ;
226           range.moveEnd('character', value.get('end') - start) ;
227           range.select() ;
228         }
229       }
230       return value;
231     }
232

234     // There are certain ways users can add/remove text that we can't identify
235     // via our key/mouse down/up handlers (such as the user choosing Paste
236     // from a menu).  So that's why we need to update our 'selection' property
237     // whenever the field's value changes.
238   }.property('fieldValue').cacheable(),

255   accessoryViewObserver: function() {
256     var classNames,
257         viewProperties = ['leftAccessoryView', 'rightAccessoryView'],
258         len = viewProperties.length , i, viewProperty, previousView, 
259         accessoryView;
260         
261     for (i=0; i<len; i++) {
262       viewProperty = viewProperties[i] ;
263 
264       // Is there an accessory view specified?
265       previousView = this['_'+viewProperty] ;
266       accessoryView = this.get(viewProperty) ;
267 
268       // If the view is the same, there's nothing to do.  Otherwise, remove
269       // the old one (if any) and add the new one.
270       if (! (previousView
271              &&  accessoryView
272              &&  (previousView === accessoryView) ) ) {
273 
274         // If there was a previous previous accessory view, remove it now.
275         if (previousView) {
276           // Remove the "sc-text-field-accessory-view" class name that we had
277           // added earlier.
278           classNames = previousView.get('classNames') ;
279           classNames = classNames.without('sc-text-field-accessory-view') ;
280           previousView.set('classNames', classNames) ;
281           this.removeChild(previousView) ;
282           previousView = null ;
283           this['_'+viewProperty] = null ;
284         }
285 
286         // If there's a new accessory view to add, do so now.
287         if (accessoryView) {
288           // If the user passed in a class rather than an instance, create an
289           // instance now.
290           if (accessoryView.isClass) {
291             accessoryView = accessoryView.create({
292               layoutView: this
293             }) ;
294           }
295 
296           // Add in the "sc-text-field-accessory-view" class name so that the
297           // z-index gets set correctly.
298           classNames = accessoryView.get('classNames') ;
299           var className = 'sc-text-field-accessory-view' ;
300           if (classNames.indexOf(className) < 0) {
301             classNames = SC.clone(classNames);
302             classNames.push(className) ;
303             accessoryView.set('classNames', classNames);
304           }
305 
306           // Actually add the view to our hierarchy and cache a reference.
307           this.appendChild(accessoryView) ;
308           this['_'+viewProperty] = accessoryView ;
309         }
310       }
311     }

314   layoutChildViewsIfNeeded: function(isVisible) {
315     // For the right accessory view, adjust the positioning such that the view
316     // is right-justified, unless 'right' is specified.
317     if (!isVisible) isVisible = this.get('isVisibleInWindow') ;
318     if (isVisible && this.get('childViewsNeedLayout')) {
319       var rightAccessoryView = this.get('rightAccessoryView') ;
320       if (rightAccessoryView  &&  rightAccessoryView.get) {
321         var layout = rightAccessoryView.get('layout') ;
322         if (layout) {
323           // Clear out any 'left' value.
324           layout.left = null;
325 
326           // Unless the user specified a 'right' value, specify a default to
327           // right-justify the view.
328           if (!layout.right) layout.right = 0 ;
329 
330           rightAccessoryView.adjust({ layout: layout }) ;
331         }
332       }
333     }

337   render: function(context, firstTime) {
338     sc_super() ;
339     var v, accessoryViewWidths, leftAdjustment, rightAdjustment;
340 
341     // always have at least an empty string
342     v = this.get('fieldValue');
343     if (SC.none(v)) v = '';
344     v = String(v);
345 
346     // update layer classes always
347     context.setClass('not-empty', v.length > 0);
348 
349     // If we have accessory views, we'll want to update the padding on the
350     // hint to compensate for the width of the accessory view.  (It'd be nice
351     // if we could add in the original padding, too, but there's no efficient
352     // way to do that without first rendering the element somewhere on/off-
353     // screen, and we don't want to take the performance hit.)
354     accessoryViewWidths = this._getAccessoryViewWidths() ;
355     leftAdjustment  = accessoryViewWidths['left'] ;
356     rightAdjustment = accessoryViewWidths['right'] ;
357 
358     if (leftAdjustment)  leftAdjustment  += 'px' ;
359     if (rightAdjustment) rightAdjustment += 'px' ;
360 
361    this._renderField(context, firstTime, v, leftAdjustment, rightAdjustment) ;
362     if(SC.browser.mozilla) this.invokeLast(this._applyFirefoxCursorFix);

376   _renderField: function(context, firstTime, value, leftAdjustment, rightAdjustment) {
377     // TODO:  The cleanest thing might be to create a sub- rendering context
378     //        here, but currently SC.RenderContext will render sibling
379     //        contexts as parent/child.
380 
381     var hint = this.get('hint'), disabled, name, adjustmentStyle, type, 
382         hintElements, element, paddingElementStyle, fieldClassNames,
383         spellCheckEnabled=this.get('spellCheckEnabled'), spellCheckString,
384         maxLength = this.get('maxLength');
385         
386     context.setClass('text-area', this.get('isTextArea'));
387     
388     spellCheckString = spellCheckEnabled ? ' spellcheck="true"' : ' spellcheck="false"';
389     if (firstTime || this._forceRenderFirstTime) {
390       this._forceRenderFirstTime = NO;
391       disabled = this.get('isEnabled') ? '' : 'disabled="disabled"' ;
392       name = this.get('layerId');
393       
394       context.push('<span class="border"></span>');
395 
396       // Render the padding element, with any necessary positioning
397       // adjustments to accommodate accessory views.
398       adjustmentStyle = '' ;
399       if (leftAdjustment  ||  rightAdjustment) {
400         adjustmentStyle = 'style="' ;
401         if (leftAdjustment)  adjustmentStyle += 'left: '  + leftAdjustment  + '; ' ;
402         if (rightAdjustment) adjustmentStyle += 'right: ' + rightAdjustment + ';' ;
403         adjustmentStyle += '"' ;
404       }
405       context.push('<span class="padding" '+adjustmentStyle+'>');
406                   
407       value = this.get('escapeHTML')?SC.RenderContext.escapeHTML(value):value;
408       if(!this.get('_supportsPlaceHolder') && (!value || (value && value.length===0))) {
409         value = this.get('hint');
410         context.setClass('sc-hint', YES);
411       } 
412       
413       //for gecko pre 1.9 vertical aligment is completely broken so we need
414       //different styling.
415       fieldClassNames = (SC.browser.mozilla &&
416                           (parseFloat(SC.browser.mozilla)<1.9 || 
417                           SC.browser.mozilla.match(/1\.9\.0|1\.9\.1/))) ?
418                           "field oldGecko": "field";
419       
420       // Render the input/textarea field itself, and close off the padding.
421       if (this.get('isTextArea')) {
422         context.push('<textarea class="',fieldClassNames,'" name="', name, 
423                       '" ', disabled, ' placeholder="',hint, '"',
424                       spellCheckString,' maxlength="', maxLength, '">', 
425                       value, '</textarea></span>') ;
426       }
427       else {
428         type = this.get('isPassword') ? 'password' : 'text' ;
429         context.push('<input class="',fieldClassNames,'" type="', type,
430                       '" name="', name, '" ', disabled, ' value="', value,
431                       '" placeholder="',hint,'"', spellCheckString, 
432                       ' maxlength="', maxLength, '" /></span>') ;
433       }
434 
435     }
436     else {
437       if(!this.get('_supportsPlaceHolder')){
438         var val = this.get('value');
439         if((!val || (val && val.length===0))){
440           if(this.get('hintON') && !this.get('isFirstResponder')){
441             //console.log('hint on render');
442             context.setClass('sc-hint', YES);
443             this.$input().val(hint);
444           }else{
445             // console.log('removing hint on render');
446             context.setClass('sc-hint', NO);
447             this.$input().val('');
448           }
449         }
450       }
451       
452       // Enable/disable the actual input/textarea as appropriate.
453       element = this.$input()[0];
454       if (element) {
455         if (!this.get('isEnabled')) {
456           element.disabled = 'true' ;
457         }
458         else {
459           element.disabled = null ;
460         }
461 
462         // Adjust the padding element to accommodate any accessory views.
463         paddingElementStyle = element.parentNode.style;
464         if (leftAdjustment) {
465           if (paddingElementStyle.left !== leftAdjustment) {
466             paddingElementStyle.left = leftAdjustment ;
467           }
468         }
469         else {
470           paddingElementStyle.left = null ;
471         }
472 
473         if (rightAdjustment) {
474           if (paddingElementStyle.right !== rightAdjustment) {
475             paddingElementStyle.right = rightAdjustment ;
476           }
477         }
478         else {
479           paddingElementStyle.right = null ;
480         }
481       }
482     }

485   _getAccessoryViewWidths: function() {
486     var widths = {},
487         accessoryViewPositions = ['left', 'right'],
488         numberOfAccessoryViewPositions = accessoryViewPositions.length, i,
489         position, accessoryView, frames, width, layout, offset, frame;
490     for (i = 0;  i < numberOfAccessoryViewPositions;  i++) {
491       position = accessoryViewPositions[i];
492       accessoryView = this.get(position + 'AccessoryView');
493       if (accessoryView) {
494         // need acessoryView as an instance, not class...
495         if (accessoryView.isClass) {
496           accessoryView = accessoryView.create({
497             layoutView: this
498           });
499         }
500         // sanity check
501         if (accessoryView.get) {
502           frame = accessoryView.get('frame');
503           if (frame) {
504             width = frame.width;
505             if (width) {
506               // Also account for the accessory view's inset.
507               layout = accessoryView.get('layout');
508               if (layout) {
509                 offset = layout[position];
510                 width += offset;
511               }
512               widths[position] = width;
513             }
514           }
515         }
516       }
517     }
518     return widths;

525   didCreateLayer: function() {
526     sc_super(); 
527     // For some strange reason if we add focus/blur events to textarea
528     // inmediately they won't work. However if I add them at the end of the
529     // runLoop it works fine.
530     if(!this.get('_supportsPlaceHolder') && this.get('hintON')){
531       var currentValue = this.$input().val();
532       if(!currentValue || (currentValue && currentValue.length===0)){
533         // console.log('hint on didcreatelayer');
534         this.$input().val(this.get('hint'));
535       }
536     }
537     if(this.get('isTextArea')) {
538       this.invokeLast(this._addTextAreaEvents);
539     }
540     else {
541       this._addTextAreaEvents();
542       
543       // In Firefox, for input fields only (that is, not textarea elements),
544       // if the cursor is at the end of the field, the "down" key will not
545       // result in a "keypress" event for the document (only for the input
546       // element), although it will be bubbled up in other contexts.  Since
547       // SproutCore's event dispatching requires the document to see the
548       // event, we'll manually forward the event along.
549       if (SC.browser.mozilla) {
550         var input = this.$input();
551         SC.Event.add(input, 'keypress', this, this._firefox_dispatch_keypress);
552       }
553     }

557   /** 
558     Adds all the textarea events. This functions is called by didCreateLayer
559     at different moments depending if it is a textarea or not. Appending 
560     events to text areas is not reliable unless the element is already added 
561     to the DOM.

565     var input = this.$input();

570     // our key/mouse down/up handlers (such as the user choosing Select All
571     // from a menu).
572     SC.Event.add(input, 'select', this, this._textField_selectionDidChange);
573         
574     if(SC.browser.mozilla){
575       // cache references to layer items to improve firefox hack perf
576       this._cacheInputElement = this.$input();
577       this._cachePaddingElement = this.$('.padding');
578     }

657   /**
658     Move magic number out so it can be over-written later in inline editor
659   */
660   _topOffsetForFirefoxCursorFix: 3,
661 
662 
663   /**
664     Mozilla had this bug until firefox 3.5 or gecko 1.8 They rewrote input text
665     and textareas and now they work better. But we have to keep this for older
666     versions.
667   */
668   _applyFirefoxCursorFix: function() {
669     // Be extremely careful changing this code.  !!!!!!!! 
670     // Contact me if you need to change or improve the code. After several 
671     // iterations the new way to apply the fix seems to be the most 
672     // consistent.
673     // This fixes: selection visibility, cursor visibility, and the ability 
674     // to fix the cursor at any position. As of FF 3.5.3 mozilla hasn't fixed this 
675     // bug, even though related bugs that I've found on their database appear
676     // as fixed.  
677     
678     // UPDATE: Things seem to be working on FF3.6 therefore we are disabling the
679     // hack for the latest versions of FF.
680     // 
681     // Juan Pinzon
682     
683     if (parseFloat(SC.browser.mozilla)<1.9 && !this.get('useStaticLayout')) {
684       var top, left, width, height, p, layer, element, textfield;
685       
686       // I'm caching in didCreateLayer this elements to improve perf
687       element = this._cacheInputElement;
688       textfield = this._cachePaddingElement;
689       if(textfield && textfield[0]){
690         layer = textfield[0];
691         p = SC.$(layer).offset() ;
692       
693         // this is to take into account an styling issue.
694         // this is counterproductive in FF >= 3.6
695         if(SC.browser.compareVersion(1,9,2) < 0 && 
696            element[0].tagName.toLowerCase()==="input") {
697           top = p.top+this._topOffsetForFirefoxCursorFix; 
698         }
699         else top = p.top;
700         left = p.left;
701         width = layer.offsetWidth;
702         height = layer.offsetHeight ;
703       
704         var style = 'position: fixed; top: %@px; left: %@px; width: %@px; height: %@px;'.fmt(top, left, width, height) ;
705         // if the style is the same don't re-apply
706         if(!this._prevStyle || this._prevStyle!=style) element.attr('style', style) ;
707         this._prevStyle = style;
708       }
709     }
710     return this ;
711   },

718     document (only for the input element), although it will be bubbled up in

791   /** @private
792     Simply allow keyDown & keyUp to pass through to the default web browser
793     implementation.
794   */
795   keyDown: function(evt) {
796     // Handle return and escape.  this way they can be passed on to the
797     // responder chain.
798     // If the event is triggered by a return while entering IME input,
799     // don't got through this path.
800     var which = evt.which;
801     if ((which === 13 && !evt.isIMEInput) && !this.get('isTextArea')) return NO ;
802     if (which === 27) return NO ;
803 
804     // handle tab key
805     if (which === 9 && this.get('defaultTabbingEnabled')) {
806       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
807       if (view) view.becomeFirstResponder();
808       else evt.allowDefault();
809       return YES ; // handled
810     }
811  
812     // validate keyDown...
813     if (this.performValidateKeyDown(evt)) {
814       this._isKeyDown = YES ;
815       evt.allowDefault();
816     } else {
817       evt.stop();
818     }
819     
820     if (this.get('applyImmediately')) {
821       // We need this invokeLater as we need to get the value of the field
822       // once the event has been processed. I tried with invokeLast , but
823       // I guess the field doesn't repaint until js execution finishes and 
824       // therefore the field value doesn't update if we don't give it a break.
825       this.invokeLater(this.fieldValueDidChange,1); // notify change
826     }
827     return YES;
828   },
829 
830   keyUp: function(evt) {
831     if(SC.browser.mozilla && evt.keyCode===13) this.fieldValueDidChange(); 
832     // The caret/selection could have moved.  In some browsers, though, the
833     // element's values won't be updated until after this event is finished
834     // processing.
835     this.notifyPropertyChange('selection');
836     this._isKeyDown = NO;
837     evt.allowDefault();
838     return YES;
839   },

846       return YES;
