 11 /** @namespace
 12 
 13   This bit of meta-programming magic can install a benchmark handler on any
 14   object method.  When a benchmark is installed, the time required to execute
 15   the method will be printed to the console log everytime the method is 
 16   called.
 17 
 18   This class can be used to implement benchmarking.  To use this object, just
 19   call start() with a key name and end() with a keyname.  The benchmark will 
 20   be logged.  If you set verbose = true, then benchmark will log everytime it 
 21   saves a bench.  Otherwise, it just keeps stats.  You can get the stats by
 22   calling report().
 23 
 24   Benchmark does not require anything other than the date.js class.  It also
 25   does not rely on SC.Object so that you can benchmark code in that area as
 26   well.
 27   
 28   The benchmark has three types of reports.
 29   
 30   report(): Returns an abbreviated list with just the durations of the bench. 
 31             Also, it averages multiple runs. Everything is reported on the top
 32             level only.
 33             
 34   timelineReport(): Returns an list of benchmarks and sub-benchmarks. If the
 35                     the globalStartTime is set, then it will show relative
 36                     time from that time.
 37   
 38   timelineChart(): Displays a chart of all the benchmarks (not sub-benchmarks)
 39                    relative to the first time capture or to the globalStartTime.
 40                    Hide this by calling hideChart()
 41 */
 42 SC.Benchmark = {
 43 
 44   /**
 45     If true, then benchmarks will be logged to the console as they are 
 46     recorded.
 47   
 48     @property {Boolean}
 49   */
 50   verbose: NO,
 51   
 52   /**
 53     If false, benchmarking will be disabled.  You might want to disable this
 54     during production to maximize performance.
 55   
 56     @property {Boolean}
 57   */
 58   enabled: YES,
 59   
 60   /** 
 61      This hash stores collected stats.  It contains key value pairs.  The value
 62      will be a hash with the following properties:
 63    
 64     * * *runs*: the number of times this stat has run
 65     * * *amt*: the total time consumed by this (divide by runs to get avg)
 66     * * *name*: an optional longer name you assigned to the stat key.  Set this  using name().
 67     * * *_starts*: this array is used internally.
 68     * * *_times*: this array is used internally.
 69     
 70     @property {Object}
 71   */
 72   stats: {},
 73 
 74   /**
 75     If set, one can tell when the benchmark is started relatively to the global start time.
 76   
 77     @property {Number}
 78   */
 79   globalStartTime: null,
 80 
 81    /**
 82     Call this method at the start of whatever you want to collect.
 83     If a parentKey is passed, then you will attach the stat to the parent, 
 84     otherwise it will be on the top level. If topLevelOnly is passed, then 
 85     recursive calls to the start will be ignored and only the top level call 
 86     will be benchmarked.
 87     
 88     @param {String} key 
 89       A unique key that identifies this benchmark.  All calls to start/end 
 90       with the same key will be groups together.
 91     
 92     @param {String} parentKey
 93       A unique key that identifies the parent benchmark.  All calls to 
 94       start/end with the same key will be groups together.
 95     
 96     @param {Boolean} topLevelOnly
 97       If true then recursive calls to this method with the same key will be 
 98       ignored.  
 99     
100     @param {Number} time
101       Only pass if you want to explicitly set the start time.  Otherwise the 
102       start time is now.
103       
104     @returns {String} the passed key
105   */
106   start: function(key, parentKey, time, topLevelOnly) {
107     if (!this.enabled) return ;
108 
109     var start = (time || Date.now()), stat;
110 
111     if (parentKey) stat = this._subStatFor(key, parentKey) ;
112     else stat = this._statFor(key) ;
113     
114     if (topLevelOnly && stat._starts.length > 0) stat._starts.push('ignore');
115     else stat._starts.push(start) ;
116 
117     stat._times.push({start: start, _subStats: {}});
118     return key;
119   },
120 
121   /**
122     Call this method at the end of whatever you want to collect.  This will
123     save the collected benchmark.
124     
125     @param {String} key
126       The benchmark key you used when you called start()
127     
128     @param {String} parentKey
129       The benchmark parent key you used when you called start()
130     
131     @param {Number} time
132       Only pass if you want to explicitly set the end time.  Otherwise start 
133       time is now.
134   */
135   end: function(key, parentKey, time) {
136     var stat;
137     if (!this.enabled) return ;
138     if(parentKey)
139     {
140       stat = this._subStatFor(key, parentKey) ;
141     }
142     else
143     {
144       stat = this._statFor(key) ;
145     }
146     var start = stat._starts.pop() ;
147     if (!start) {
148       console.log('SC.Benchmark "%@" ended without a matching start.  No information was saved.'.fmt(key));
149       return ;
150     }
151 
152     // top level only.
153     if (start == 'ignore') return ; 
154     
155     var end = (time || Date.now()) ;
156     var dur = end - start;
157 
158     stat._times[stat._times.length-1].end = end;
159     stat._times[stat._times.length-1].dur = dur;
160 
161     stat.amt += dur ;
162     stat.runs++ ;
163     
164     if (this.verbose) this.log(key) ;
165   },
166   
167   /* 
168     Set the inital global start time.
169   */
170   setGlobalStartTime: function(time)
171   {
172     this.globalStartTime = time;
173   },
174 
175   /**
176     This is a simple way to benchmark a function.  The function will be 
177     run with the name you provide the number of times you indicate.  Only the
178     function is a required param.
179   */  
180   bench: function(func, key, reps) {
181     if (!key) key = "bench%@".fmt(this._benchCount++) ;
182     if (!reps) reps = 1 ;
183     var ret ;
184     
185     while(--reps >= 0) {
186       var timeKey = SC.Benchmark.start(key) ;
187       ret = func();
188       SC.Benchmark.end(timeKey) ; 
189     }
190     
191     return ret ;
192   },
193   
194   /**  
195     This bit of metaprogramming magic install a wrapper around a method and
196     benchmark it whenever it is run.
197   */  
198   install: function(object,method, topLevelOnly) {
199     
200     // vae the original method.
201     object['b__' + method] = object[method] ;
202     var __func = object['b__' + method];
203     
204     // replace with this helper.
205     object[method] = function() {
206       var key = '%@(%@)'.fmt(method, $A(arguments).join(', ')) ;
207       SC.Benchmark.start(key, topLevelOnly) ;
208       var ret = __func.apply(this, arguments) ;
209       SC.Benchmark.end(key) ;
210       return ret ;
211     } ;
212   },
213   
214   /**
215     Restore the original method, deactivating the benchmark.
216   
217     @param {object} object the object to change
218     @param {string} method the method name as a string.
219   
220   */  
221   restore: function(object,method) {
222     object[method] = object['b__' + method] ;
223   },
224   
225   /**
226     This method will return a string containing a report of the stats
227     collected so far.  If you pass a key, only the stats for that key will
228     be returned.  Otherwise, all keys will be used.
229   */
230   report: function(key) {
231     if (key) return this._genReport(key) ;
232     var ret = [] ;
233     for(var k in this.stats) {
234       if (!this.stats.hasOwnProperty(k)) continue ;
235       ret.push(this._genReport(k)) ;
236     }
237     return ret.join("\n") ;
238   },

240   /**
241     Generate a human readable benchmark report. Pass in appName if you desire.
242 
243     @param {string} application name.
244   */
245   timelineReport: function(appName) 
246   {
247     appName = (appName) ? 'SproutCore Application' : appName;
248     var ret = [appName, 'User-Agent: %@'.fmt(navigator.userAgent), 'Report Generated: %@ (%@)'.fmt(new Date().toString(), Date.now()), ''] ;
249 
250     var chart = this._compileChartData(true);
251     for(var i=0; i<chart.length; i++)
252     {
253       if(chart[i][4])
254       {
255         ret.push(this._timelineGenSubReport(chart[i]));
256       }
257       else
258       {
259         ret.push(this._timelineGenReport(chart[i]));
260       }
261     }
262     return ret.join("\n") ;
263   },

265   /**
266     Generate a human readable benchmark chart. Pass in appName if you desire.
267 
268   */
269   timelineChart: function(appName) {
270     var i=0;
271     // Hide the chart if there is an existing one.
272     this.hideChart();
273     
274     // Compile the data.
275     var chart = this._compileChartData(false);
276     var chartLen = chart.length;
277     
278     // Return if there is nothing to draw.
279     if(chartLen === 0) return;
280     
281     // Get the global start of the graph.
282     var gStart = this.globalStartTime ? this.globalStartTime : chart[0][1];
283     var maxDur = chart[chartLen-1][2]-gStart;
284     var maxHeight = 50+chartLen*30;
285     var incr = Math.ceil(maxDur/200)+1;
286     var maxWidth = incr*50;
287     
288     // Create the basic graph element.
289     var graph = document.createElement('div');
290     graph.className = 'sc-benchmark-graph';
291     document.body.appendChild(graph);
292 
293     // Set the title.
294     var title = document.createElement('div');
295     title.innerHTML = ((appName) ? appName : 'SproutCore Application') + (' - Total Captured Time: ' + maxDur +' ms - Points Captured: ' + chartLen) + ' [<a href="javascript:SC.Benchmark.hideChart();">Hide Chart</a>]';
296     title.className = 'sc-benchmark-title'; 
297     graph.appendChild(title);
298 
299 
300     var topBox = document.createElement('div');
301     topBox.className = 'sc-benchmark-top'; 
302     topBox.style.width = maxWidth + 'px';
303     graph.appendChild(topBox);
304 
305     // Draw the tick marks.
306     for(i=0;i<incr; i++)
307     {
308       var tick = document.createElement('div');
309       tick.className = 'sc-benchmark-tick';
310       tick.style.left = (i*50)+'px';
311       tick.style.height = maxHeight+'px';
312       var tickLabel = document.createElement('div');
313       tickLabel.className = 'sc-benchmark-tick-label';
314       tickLabel.style.left = (i*50)+'px';
315       tickLabel.innerHTML = i*200+" ms";
316       graph.appendChild(tick);
317       graph.appendChild(tickLabel);
318     }
319     
320     // For each item in the chart, print it out on the screen.
321     for(i=0;i<chartLen; i++)
322     {
323     	var row = document.createElement('div');
324     	row.style.top = (75+(i*30))+'px';
325     	row.style.width = maxWidth+'px';
326     	row.className = (i%2===0) ? 'sc-benchmark-row even' : 'sc-benchmark-row';
327     	graph.appendChild(row);
328 
329       var div = document.createElement('div');
330       var start = chart[i][1];
331       var end = chart[i][2];
332       var duration = chart[i][3];
333       
334       div.innerHTML = ' ' + (chart[i][0] + " <span class='sc-benchmark-emphasis'>" + duration + 'ms</span>');
335       
336       div.className = 'sc-benchmark-bar';
337       div.style.cssText = 'left:'+ (((start-gStart)/4))+'px; width: '+((duration/4))+
338                           'px; top: '+(53+(i*30))+'px;';
339       div.title = "start: " + (start-gStart) + " ms, end: " + (end-gStart) + ' ms, duration: ' + duration + ' ms';
340       graph.appendChild(div);
341     }
342 
343     // Save the graph.
344     this._graph = graph;
345   },

348     Hide chart.
349     
350   */

353     if(this._graph) {
354       try{ 
355         document.body.removeChild(this._graph);
356       }catch(e){}
357     }

362     This method is just like report() except that it will log the results to

366     // log each line to make this easier to read on an iPad
367     var lines = this.report(key).split('\n'),
368         len   = lines.length, idx;
369     for(idx=0;idx<len;idx++) console.log(lines[idx]);

390   // Generates, sorts, and returns the array of all the data that has been captured.
391   _compileChartData: function(showSub)
392   {
393     var chart = [], dispKey;
394     for(var key in this.stats) 
395     {
396       var stat = this.stats[key];
397       for(var i=0; i<stat._times.length; i++)
398       {
399         var st = stat._times[i];
400         dispKey = (stat._times.length > 1) ? (i+1)+' - '+key : key;
401         chart.push([dispKey, st.start, st.end, st.dur, false]);
402         if(showSub)
403         {
404           var subStats = st._subStats;
405           for(var k in subStats) 
406           {
407            
408             var subStat = subStats[k];
409             for(var j=0; j<subStat._times.length; j++)
410             {
411               var s = subStat._times[j];
412               dispKey = (subStat._times.length > 1) ? (j+1)+' - '+k : k;
413               chart.push([dispKey, s.start, s.end, s.dur, true]);
414          
415             }
416           }
417         }
418       }
419     }
420     
421     chart.sort(function(a,b)
422     {
423       if(a[1] < b[1]) 
424       {
425         return -1;
426       }
427       else if(a[1] == b[1])
428       {
429         if(a[3] && !b[3]) return -1;
430         if(!a[3] && b[3]) return 1;
431         return 0;
432       }
433       return 1;
434     });
435 
436     return chart;
437   },

439   // Generate the traditional report show multiple runs averaged.
440   _genReport: function(key) {
441     var stat = this._statFor(key) ;
442     var avg = (stat.runs > 0) ? (Math.floor(stat.amt * 1000 / stat.runs) / 1000) : 0 ;
443      
444     return 'BENCH %@ msec: %@ (%@x)'.fmt(avg, (stat.name || key), stat.runs) ;  
445   },
446 
447   // Generate the report in the form of at time line. This returns the parent.
448   _timelineGenReport: function(val) 
449   {
450     if(this.globalStartTime)
451     {
452       return 'BENCH start: %@ msec, duration: %@ msec,  %@'.fmt((val[1]-this.globalStartTime), val[3], val[0]) ;  
453     } 
454     else
455     {
456       return 'BENCH duration: %@ msec, %@'.fmt( val[3],  val[0]) ;  
457     }
458   },
459   
460   // Generate the report in the form of at time line. This returns the children.
461   _timelineGenSubReport: function(val) 
462   {
463     if(this.globalStartTime)
464     {
465       return '   CHECKPOINT BENCH start: %@ msec, duration: %@ msec,  %@'.fmt((val[1]-this.globalStartTime), val[3], val[0]) ;  
466     } 
467     else
468     {
469       return '   CHECKPOINT BENCH duration: %@ msec, %@'.fmt( val[3], val[0]) ;  
470     }
471   },

473   // returns a stats hash for the named key and parent key.  If the hash does not exist yet,
474   // creates it.
475   _subStatFor: function(key, parentKey) {
476     var parentTimeLen = this.stats[parentKey]._times.length;
477     if(parentTimeLen === 0) return;
478     var parentSubStats = this.stats[parentKey]._times[this.stats[parentKey]._times.length-1]._subStats;
479     var ret = parentSubStats[key] ;
480     if (!ret) {
481       parentSubStats[key] = {
482         runs: 0, amt: 0, name: key, _starts: [], _times: []      
483       };
484       ret = parentSubStats[key];
485     }
486     return ret ;
487   },
488 
489   // returns a stats hash for the named key.  If the hash does not exist yet,
490   // creates it.
491   _statFor: function(key) {
492     var ret = this.stats[key] ;
493     if (!ret) {
494       ret = this.stats[key] = {
495         runs: 0, amt: 0, name: key, _starts: [], _times: []      
496       };
497       ret = this.stats[key];
498     }
499     return ret ;
500   },
