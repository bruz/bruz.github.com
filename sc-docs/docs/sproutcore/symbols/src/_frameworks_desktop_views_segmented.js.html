  8 /**
  9   @class
 10 
 11   SegmentedView is a special type of button that can display multiple
 12   segments.  Each segment has a value assigned to it.  When the user clicks
 13   on the segment, the value of that segment will become the new value of 
 14   the control.
 15   
 16   You can also optionally configure a target/action that will fire whenever
 17   the user clicks on an item.  This will give your code an opportunity to take
 18   some action depending on the new value.  (of course, you can always bind to
 19   the value as well, which is generally the preferred approach.)
 20   
 21   h1. Defining Your Segments
 22   
 23   You define your segments by providing a items array, much like you provide
 24   to a RadioView.  Your items array can be as simple as an array of strings 
 25   or as complex as full model objects.  Based on how you configure your
 26   itemKey properties, the segmented view will read the properties it needs 
 27   from the array and construct the button.
 28   
 29   You can define the following properties on objects you pass in:
 30   
 31   | *itemTitleKey* | the title of the button |
 32   | *itemValueKey* | the value of the button |
 33   | *itemWidthKey* | the preferred width. if omitted, it autodetects |
 34   | *itemIconKey*  | an icon |
 35   | *itemActionKey* | an optional action to fire when pressed |
 36   | *itemTargetKey* | an optional target for the action |
 37 
 38   @extends SC.View
 39   @since SproutCore 1.0
 40 */

168   /**
169     The array of itemKeys that will be searched to build the displayItems
170     array.  This is used internally by the class.  You will not generally
171     need to access or edit this array.
172     
173     @property {Array}
174   */
175   itemKeys: 'itemTitleKey itemValueKey itemIsEnabledKey itemIconKey itemWidthKey itemToolTipKey'.w(),
176   
177   /**
178     This computed property is generated from the items array based on the 
179     itemKey properties that you set.  The return value is an array of arrays
180     that contain private information used by the SegmentedView to render. 
181     
182     You will not generally need to access or edit this property.
183     
184     @property {Array}
185   */
186   displayItems: function() {
187     var items = this.get('items'), loc = this.get('localize'),
188       keys=null, itemType, cur, ret = [], max = items.get('length'), idx, 
189       item, fetchKeys = SC._segmented_fetchKeys, fetchItem = SC._segmented_fetchItem;
190     
191     // loop through items and collect data
192     for(idx=0;idx<max;idx++) {
193       item = items.objectAt(idx) ;
194       if (SC.none(item)) continue; //skip is null or undefined
195       
196       // if the item is a string, build the array using defaults...
197       itemType = SC.typeOf(item);
198       if (itemType === SC.T_STRING) {
199         cur = [item.humanize().titleize(), item, YES, null, null,  null, idx] ;
200         
201       // if the item is not an array, try to use the itemKeys.
202       } else if (itemType !== SC.T_ARRAY) {
203         // get the itemKeys the first time
204         if (keys===null) {
205           keys = this.itemKeys.map(fetchKeys,this);
206         }
207         
208         // now loop through the keys and try to get the values on the item
209         cur = keys.map(fetchItem, item);
210         cur[cur.length] = idx; // save current index
211         
212         // special case 1...if title key is null, try to make into string
213         if (!keys[0] && item.toString) cur[0] = item.toString(); 
214         
215         // special case 2...if value key is null, use item itself
216         if (!keys[1]) cur[1] = item;
217         
218         // special case 3...if isEnabled is null, default to yes.
219         if (!keys[2]) cur[2] = YES ; 
220       }
221       
222       // finally, be sure to loc the title if needed
223       if (loc && cur[0]) cur[0] = cur[0].loc();
224 
225       // finally, be sure to loc the toolTip if needed
226       if (loc && cur[5] && SC.typeOf(cur[5]) === SC.T_STRING) cur[5] = cur[5].loc();
227       
228       // add to return array
229       ret[ret.length] = cur;
230     }
231     
232     // all done, return!
233     return ret ;
234   }.property('items', 'itemTitleKey', 'itemValueKey', 'itemIsEnabledKey', 'localize', 'itemIconKey', 'itemWidthKey', 'itemToolTipKey'),
235

270   render: function(context, firstTime) { 
271     
272     // collect some data 
273     var items = this.get('displayItems');
274     
275     var theme = this.get('theme');
276     if (theme) context.addClass(theme);
277     if (firstTime || this.get('renderLikeFirstTime')) {
278       this._seg_displayItems = items; // save for future
279       this.renderDisplayItems(context, items) ;
280       context.addStyle('text-align', this.get('align'));
281       this.set('renderLikeFirstTime',NO);
282     }else{
283     // update selection and active state
284       var activeIndex = this.get('activeIndex'),
285           value = this.get('value'),
286           isArray = SC.isArray(value);
287       if (isArray && value.get('length')===1) {
288         value = value.objectAt(0); isArray = NO ;
289       }
290       var names = {}, // reuse  
291           loc = items.length, cq = this.$('.sc-segment'), item;
292       while(--loc>=0) {
293         item = items[loc];
294         names.sel = isArray ? (value.indexOf(item[1])>=0) : (item[1]===value);
295         names.active = (activeIndex === loc);
296         names.disabled = !item[2];
297         SC.$(cq[loc]).setClass(names);
298       }
299       names = items = value = items = null; // cleanup
300     }
301   },

303   /**
304     Actually generates the segment HTML for the display items.  This method 
305     is called the first time a view is constructed and any time the display
306     items change thereafter.  This will construct the HTML but will not set
307     any "transient" states such as the global isEnabled property or selection.
308   */
309   renderDisplayItems: function(context, items) {
310     var value       = this.get('value'),
311         isArray     = SC.isArray(value),
312         activeIndex = this.get('activeIndex'),
313         len         = items.length,
314         title, icon, url, className, ic, item, toolTip, width, i, stylesHash,
315         classArray;
316 
317     for(i=0; i< len; i++){
318       ic = context.begin('a').attr('role', 'button');
319       item=items[i];
320       title = item[0]; 
321       icon = item[3];
322       toolTip = item[5];
323       
324       stylesHash = {};
325       classArray = [];
326 
327       if (this.get('layoutDirection') == SC.LAYOUT_HORIZONTAL) {
328         stylesHash['display'] = 'inline-block' ;
329       }
330 
331       classArray.push('sc-segment');
332       
333       if(!item[2]){
334         classArray.push('disabled');
335       }
336       if(i===0){
337         classArray.push('sc-first-segment');
338       }
339       if(i===(len-1)){
340         classArray.push('sc-last-segment');
341       }
342       if(i!==0 && i!==(len-1)){
343         classArray.push('sc-middle-segment');
344       }      
345       if( isArray ? (value.indexOf(item[1])>=0) : (item[1]===value)){
346         classArray.push('sel');
347       }
348       if(activeIndex === i) {
349         classArray.push('active') ;
350       }
351       if(item[4]){
352         width=item[4];
353         stylesHash['width'] = width+'px';
354       }
355       ic.addClass(classArray);
356       ic.addStyle(stylesHash);
357       if(toolTip) {
358         ic.attr('title', toolTip) ;
359       }
360 
361       if (icon) {
362         url = (icon.indexOf('/')>=0) ? icon : SC.BLANK_IMAGE_URL;
363         className = (url === icon) ? '' : icon ;
364         icon = '<img src="'+url+'" alt="" class="icon '+className+'" />';
365       } else {
366         icon = '';
367       }
368       ic.push('<span class="sc-button-inner"><label class="sc-button-label">',
369               icon+title, '</label></span>');
370       ic.end();
371     }

378   /** 
379     Determines the index into the displayItems array where the passed mouse
380     event occurred.
381   */
382   displayItemIndexForEvent: function(evt) {
383     return this.displayItemIndexForPosition(evt.pageX, evt.pageY);
384   },
385   
386   /**
387     Determines an item index based on a position. The position does not have to be within the view's
388     bounding rectangle. If no item is at that position, this will return -1.
389     
390     NOTE: Eventually, this sort of function should be implemented in a renderer.
391   */
392   displayItemIndexForPosition: function(pageX, pageY) {
393     // find the segments
394     var segments = this.$('.sc-segment'), len = segments.length, idx, segment, r;
395     
396     // loop through them (yes, this comment is mostly because it looks nice in TextMate)
397     for (idx = 0; idx < len; idx++) {
398       // get the segment
399       segment = segments[idx];
400       
401       // get its rectangle
402       r = segment.getBoundingClientRect();
403       
404       // based on orientation, check the position left-to-right or up-to-down.
405       if (this.get('layoutDirection') == SC.LAYOUT_VERTICAL) {
406         // if it fits, return it right away
407         if (pageY > r.top && pageY < r.bottom) return idx;
408       }
409       else {
410         // if it fits, return it right away.
411         if (pageX > r.left && pageX < r.right) return idx;
412       }
413     }
414     
415     // if we didn't find anything, return the old standard -1 for "not found."
416     return -1;
417   },
418   
419   keyDown: function(evt) {
420     // handle tab key
421     var i, item, items, len, value, isArray;
422     if (evt.which === 9) {
423       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
424       if(view) view.becomeFirstResponder();
425       else evt.allowDefault();
426       return YES ; // handled
427     }    
428     if (!this.get('allowsMultipleSelection') && !this.get('allowsEmptySelection')){
429       items = this.get('displayItems');
430       len = items.length;
431       value = this.get('value');
432       isArray = SC.isArray(value);
433       if (evt.which === 39 || evt.which === 40) {  
434         for(i=0; i< len-1; i++){
435           item=items[i];
436           if( isArray ? (value.indexOf(item[1])>=0) : (item[1]===value)){
437             this.triggerItemAtIndex(i+1);
438           }
439         }
440         return YES ; // handled
441       }
442       else if (evt.which === 37 || evt.which === 38) {
443         for(i=1; i< len; i++){
444           item=items[i];
445           if( isArray ? (value.indexOf(item[1])>=0) : (item[1]===value)){
446             this.triggerItemAtIndex(i-1);
447           }
448         }
449         return YES ; // handled
450       }
451     }
452     return YES; 
453   },

455   mouseDown: function(evt) {
456     if (!this.get('isEnabled')) return YES; // nothing to do
457     var idx = this.displayItemIndexForEvent(evt);
458     
459     // if mouse was pressed on a button, then start detecting pressed events
460     if (idx>=0) {
461       this._isMouseDown = YES ;
462       this.set('activeIndex', idx);
463     }
464     
465     return YES ;
466   },
467   
468   mouseUp: function(evt) {
469     var idx = this.displayItemIndexForEvent(evt);
470     // if mouse was pressed on a button then detect where we where when we
471     // release and use that one.
472     if (this._isMouseDown && (idx>=0)) this.triggerItemAtIndex(idx);
473     
474     // cleanup
475     this._isMouseDown = NO ;
476     this.set('activeIndex', -1);
477     return YES ;
478   },
479   
480   mouseMoved: function(evt) {
481     if (this._isMouseDown) {
482       var idx = this.displayItemIndexForEvent(evt);
483       this.set('activeIndex', idx);
484     }
485     return YES;
486   },
487   
488   mouseExited: function(evt) {
489     // if mouse was pressed down initially, start detection again
490     if (this._isMouseDown) {
491       var idx = this.displayItemIndexForEvent(evt);
492       this.set('activeIndex', idx);
493     }
494     return YES;
495   },
496   
497   mouseEntered: function(evt) {
498     // if mouse was down, hide active index
499     if (this._isMouseDown) {
500       var idx = this.displayItemIndexForEvent(evt);
501       this.set('activeIndex', -1);
502     }
503     return YES ;
504   },

508   touchStart: function(touch) {
509     if (!this.get('isEnabled')) return YES; // nothing to do
510     var idx = this.displayItemIndexForEvent(touch);
511     
512     // if mouse was pressed on a button, then start detecting pressed events
513     if (idx>=0) {
514       this._isTouching = YES ;
515       this.set('activeIndex', idx);
516     }
517     
518     return YES ;
519   },
520   
521   touchEnd: function(touch) {
522     var idx = this.displayItemIndexForEvent(touch);
523     // if mouse was pressed on a button then detect where we where when we
524     // release and use that one.
525     if (this._isTouching && (idx>=0)) this.triggerItemAtIndex(idx);
526     
527     // cleanup
528     this._isTouching = NO ;
529     this.set('activeIndex', -1);
530     return YES ;
531   },
532   
533   touchesDragged: function(evt, touches) {
534     var isTouching = this.touchIsInBoundary(evt);
535 
536     if (isTouching) {
537       if (!this._isTouching) {
538         this._touchDidEnter(evt);
539       }
540       var idx = this.displayItemIndexForEvent(evt);
541       this.set('activeIndex', idx);
542     } else {
543       if (this._isTouching) this._touchDidExit(evt);
544     }
545     
546     this._isTouching = isTouching;
547     
548     return YES;
549   },
550   
551   _touchDidExit: function(evt) {
552     var idx = this.displayItemIndexForEvent(evt);
553     this.set('activeIndex', -1);
554 
555     return YES;
556   },
557   
558   _touchDidEnter: function(evt) {
559     // if mouse was down, hide active index
560     var idx = this.displayItemIndexForEvent(evt);
561     this.set('activeIndex', idx);
562 
563     return YES ;
564   },

566   /** 
567     Simulates the user clicking on the segment at the specified index. This
568     will update the value if possible and fire the action.
569   */
570   triggerItemAtIndex: function(idx) {
571     var items = this.get('displayItems'),
572         item  = items.objectAt(idx),
573         sel, value, val, empty, mult;
574         
575     if (!item[2]) return this; // nothing to do!
576 
577     empty = this.get('allowsEmptySelection');
578     mult = this.get('allowsMultipleSelection');
579     
580     
581     // get new value... bail if not enabled. Also save original for later.
582     sel = item[1];
583     value = val = this.get('value') ;
584     if (!SC.isArray(value)) value = [value]; // force to array
585     
586     // if we do not allow multiple selection, either replace the current
587     // selection or deselect it
588     if (!mult) {
589       // if we allow empty selection and the current value is the same as
590       // the selected value, then deselect it.
591       if (empty && (value.get('length')===1) && (value.objectAt(0)===sel)){
592         value = [];
593       
594       // otherwise, simply replace the value.
595       } else value = [sel] ;
596       
597     // if we do allow multiple selection, then add or remove item to the
598     // array.
599     } else {
600       if (value.indexOf(sel) >= 0) {
601         if (value.get('length')>1 || (value.objectAt(0)!==sel) || empty) {
602           value = value.without(sel);
603         }
604       } else value = value.concat([sel]) ;
605     }
606     
607     // normalize back to non-array form
608     switch(value.get('length')) {
609       case 0:
610         value = null;
611         break;
612       case 1:
613         value = value.objectAt(0);
614         break;
615       default:
616         break;
617     }
618     
619     // also, trigger target if needed.
620     var actionKey = this.get('itemActionKey'),
621         targetKey = this.get('itemTargetKey'),
622         action, target = null,
623         resp = this.getPath('pane.rootResponder');
624 
625     if (actionKey && (item = this.get('items').objectAt(item[6]))) {
626       // get the source item from the item array.  use the index stored...
627       action = item.get ? item.get(actionKey) : item[actionKey];
628       if (targetKey) {
629         target = item.get ? item.get(targetKey) : item[targetKey];
630       }
631       if (resp) resp.sendAction(action, target, this, this.get('pane'));
632     }
633 
634     // Only set value if there is no action and a value is defined.
635     if(!action && val !== undefined) {
636       this.set('value', value);
637     }
638     
639     // if an action/target is defined on self use that also
640     action =this.get('action');
641     if (action && resp) {
642       resp.sendAction(action, this.get('target'), this, this.get('pane'));
643     }
644   },
