 13 /**
 14   @namespace
 15 
 16   This mixin defines the common interface implemented by enumerable objects 
 17   in SproutCore.  Most of these methods follow the standard Array iteration
 18   API defined up to JavaScript 1.8 (excluding language-specific features that
 19   cannot be emulated in older versions of JavaScript).
 20   
 21   This mixin is applied automatically to the Array class on page load, so you
 22   can use any of these methods on simple arrays.  If Array already implements
 23   one of these methods, the mixin will not override them.
 24   
 25   h3. Writing Your Own Enumerable
 26 
 27   To make your own custom class enumerable, you need two items:
 28   
 29   1. You must have a length property.  This property should change whenever
 30      the number of items in your enumerable object changes.  If you using this
 31      with an SC.Object subclass, you should be sure to change the length 
 32      property using set().
 33      
 34   2. If you must implement nextObject().  See documentation.
 35     
 36   Once you have these two methods implement, apply the SC.Enumerable mixin
 37   to your class and you will be able to enumerate the contents of your object
 38   like any other collection.
 39   
 40   h3. Using SproutCore Enumeration with Other Libraries
 41   
 42   Many other libraries provide some kind of iterator or enumeration like
 43   facility.  This is often where the most common API conflicts occur. 
 44   SproutCore's API is designed to be as friendly as possible with other 
 45   libraries by implementing only methods that mostly correspond to the
 46   JavaScript 1.8 API.  
 47   
 48   @since SproutCore 1.0
 49 */

 59   /**
 60     Implement this method to make your class enumerable.
 61     
 62     This method will be call repeatedly during enumeration.  The index value
 63     will always begin with 0 and increment monotonically.  You don't have to
 64     rely on the index value to determine what object to return, but you should
 65     always check the value and start from the beginning when you see the
 66     requested index is 0.
 67     
 68     The previousObject is the object that was returned from the last call
 69     to nextObject for the current iteration.  This is a useful way to 
 70     manage iteration if you are tracing a linked list, for example.
 71     
 72     Finally the context paramter will always contain a hash you can use as 
 73     a "scratchpad" to maintain any other state you need in order to iterate
 74     properly.  The context object is reused and is not reset between 
 75     iterations so make sure you setup the context with a fresh state whenever
 76     the index parameter is 0.
 77     
 78     Generally iterators will continue to call nextObject until the index
 79     reaches the your current length-1.  If you run out of data before this 
 80     time for some reason, you should simply return undefined.
 81     
 82     The default impementation of this method simply looks up the index.
 83     This works great on any Array-like objects.
 84     
 85     @param index {Number} the current index of the iteration
 86     @param previousObject {Object} the value returned by the last call to nextObject.
 87     @param context {Object} a context object you can use to maintain state.
 88     @returns {Object} the next object in the iteration or undefined   
 89   */ 
 90   nextObject: function(index, previousObject, context) {
 91     return this.objectAt ? this.objectAt(index) : this[index] ;
 92   },

 97     object if the enumerable contains only one item.

100     always return the same value each time it is called.  If your enumerable
101     contains only one object, this method should always return that object.

143     parameters are optional):

313   /** 
314     Returns an array sorted by the value of the passed key parameters.
315     null objects will be sorted first.  You can pass either an array of keys
316     or multiple parameters which will act as key names
317     
318     @param {String} key one or more key names
319     @returns {Array}
320   */
321   sortProperty: function(key) {
322     var keys = (typeof key === SC.T_STRING) ? arguments : key,
323         len  = keys.length,
324         src;
325      
326     // get the src array to sort   
327     if (this instanceof Array) src = this;
328     else {
329       src = [];
330       this.forEach(function(i) { src.push(i); });
331     }
332     
333     if (!src) return [];
334     return src.sort(function(a,b) {
335       var idx, key, aValue, bValue, ret = 0;
336       
337       for(idx=0;ret===0 && idx<len;idx++) {
338         key = keys[idx];
339         aValue = a ? (a.get ? a.get(key) : a[key]) : null;
340         bValue = b ? (b.get ? b.get(key) : b[key]) : null;
341         ret = SC.compare(aValue, bValue);
342       }
343       return ret ;
344     });

590   /**
591     This will combine the values of the enumerator into a single value. It 
592     is a useful way to collect a summary value from an enumeration.  This
593     corresponds to the reduce() method defined in JavaScript 1.8.
594     
595     The callback method you provide should have the following signature (all
596     parameters are optional):
597     
598     {{{
599       function(previousValue, item, index, enumerable) ;      
600     }}}
601     
602     - *previousValue* is the value returned by the last call to the iterator.
603     - *item* is the current item in the iteration.
604     - *index* is the current index in the iteration
605     - *enumerable* is the enumerable object itself.
606 
607     Return the new cumulative value.
608 
609     In addition to the callback you can also pass an initialValue.  An error
610     will be raised if you do not pass an initial value and the enumerator is
611     empty.
612 
613     Note that unlike the other methods, this method does not allow you to 
614     pass a target object to set as this for the callback.  It's part of the
615     spec. Sorry.
616     
617     @params callback {Function} the callback to execute
618     @params initialValue {Object} initial value for the reduce
619     @params reducerProperty {String} internal use only.  May not be available.
620     @returns {Array} A filtered array.
621   */
622   reduce: function(callback, initialValue, reducerProperty) {
623     if (typeof callback !== "function") throw new TypeError() ;
624     var len = this.get ? this.get('length') : this.length ;
625 
626     // no value to return if no initial value & empty
627     if (len===0 && initialValue === undefined) throw new TypeError();
628     
629     var ret  = initialValue;
630     var last = null ;
631     var context = SC.Enumerator._popContext();
632     for(var idx=0;idx<len;idx++) {
633       var next = this.nextObject(idx, last, context) ;
634       
635       // while ret is still undefined, just set the first value we get as ret.
636       // this is not the ideal behavior actually but it matches the FireFox
637       // implementation... :(
638       if (next !== null) {
639         if (ret === undefined) {
640           ret = next ;
641         } else {
642           ret = callback.call(null, ret, next, idx, this, reducerProperty);
643         }
644       }
645       last = next ;
646     }
647     last = null ;
648     context = SC.Enumerator._pushContext(context);
649     
650     // uh oh...we never found a value!
651     if (ret === undefined) throw new TypeError() ;
652     return ret ;
653   },

655   /**
656     Invokes the named method on every object in the receiver that
657     implements it.  This method corresponds to the implementation in
658     Prototype 1.6.
659     
660     @param methodName {String} the name of the method
661     @param args {Object...} optional arguments to pass as well.
662     @returns {Array} return values from calling invoke.
663   */
664   invoke: function(methodName) {
665     var len = this.get ? this.get('length') : this.length ;
666     if (len <= 0) return [] ; // nothing to invoke....
667     
668     var idx;
669     
670     // collect the arguments
671     var args = [] ;
672     var alen = arguments.length ;
673     if (alen > 1) {
674       for(idx=1;idx<alen;idx++) args.push(arguments[idx]) ;
675     }
676     
677     // call invoke
678     var ret = [] ;
679     var last = null ;
680     var context = SC.Enumerator._popContext();
681     for(idx=0;idx<len;idx++) {
682       var next = this.nextObject(idx, last, context) ;
683       var method = next ? next[methodName] : null ;
684       if (method) ret[idx] = method.apply(next, args) ;
685       last = next ;
686     }
687     last = null ;
688     context = SC.Enumerator._pushContext(context);
689     return ret ;
690   },
