 13 /** @class
 14 
 15   A SelectionSet contains a set of objects that represent the current 
 16   selection.  You can select objects by either adding them to the set directly
 17   or indirectly by selecting a range of indexes on a source object.
 18   
 19   @extends SC.Object
 20   @extends SC.Enumerable
 21   @extends SC.Freezable
 22   @extends SC.Copyable
 23   @since SproutCore 1.0
 24 */

 35   /**
 36     Total number of indexes in the selection set
 37     
 38     @property {Number}

 40   length: function() {
 41     var ret     = 0,
 42         sets    = this._sets,
 43         objects = this._objects;
 44     if (objects) ret += objects.get('length');
 45     if (sets) sets.forEach(function(s) { ret += s.get('length'); });
 46     return ret ;

 53   /**
 54     A set of all the source objects used in the selection set.  This 
 55     property changes automatically as you add or remove index sets.
 56     
 57     @property {SC.Array}
 58   */
 59   sources: function() {
 60     var ret  = [],
 61         sets = this._sets,
 62         len  = sets ? sets.length : 0,
 63         idx, set, source;
 64         
 65     for(idx=0;idx<len;idx++) {
 66       set = sets[idx];
 67       if (set && set.get('length')>0 && set.source) ret.push(set.source);
 68     }
 69     return ret ;
 70   }.property().cacheable(),

 72   /**
 73     Returns the index set for the passed source object or null if no items are
 74     seleted in the source.
 75     
 76     @param {SC.Array} source the source object
 77     @returns {SC.IndexSet} index set or null
 78   */
 79   indexSetForSource: function(source) {
 80     if (!source || !source.isSCArray) return null; // nothing to do
 81     
 82     var cache   = this._indexSetCache,
 83         objects = this._objects,
 84         ret, idx;
 85 
 86     // try to find in cache
 87     if (!cache) cache = this._indexSetCache = {};
 88     ret = cache[SC.guidFor(source)];
 89     if (ret && ret._sourceRevision && (ret._sourceRevision !== source.propertyRevision)) {
 90       ret = null;
 91     }
 92 
 93     // not in cache.  generate from index sets and any saved objects
 94     if (!ret) {
 95       ret = this._indexSetForSource(source, NO);
 96       if (ret && ret.get('length')===0) ret = null;
 97     
 98       if (objects) {
 99         if (ret) ret = ret.copy();
100         objects.forEach(function(o) {
101           if ((idx = source.indexOf(o)) >= 0) {
102             if (!ret) ret = SC.IndexSet.create();
103             ret.add(idx);
104           }
105         }, this);
106       }
107       
108       if (ret) {
109         ret = cache[SC.guidFor(source)] = ret.frozenCopy();
110         ret._sourceRevision = source.propertyRevision;
111       }
112     }
113     
114     return ret;
115   },

117   /** 
118     @private
119   
120     Internal method gets the index set for the source, ignoring objects
121     that have been added directly.
122   */
123   _indexSetForSource: function(source, canCreate) {
124     if (canCreate === undefined) canCreate = YES;
125 
126     var guid  = SC.guidFor(source),
127         index = this[guid],
128         sets  = this._sets,
129         len   = sets ? sets.length : 0,
130         ret   = null;
131                 
132     if (index >= len) index = null;
133     if (SC.none(index)) {
134       if (canCreate && !this.isFrozen) {
135         this.propertyWillChange('sources');
136         if (!sets) sets = this._sets = [];
137         ret = sets[len] = SC.IndexSet.create();
138         ret.source = source ;
139         this[guid] = len;
140         this.propertyDidChange('sources');
141       }
142       
143     } else ret = sets ? sets[index] : null;
144     return ret ;
145   },

147   /**
148     Add the passed index, range of indexSet belonging to the passed source
149     object to the selection set.
150     
151     The first parameter you pass must be the source array you are selecting
152     from.  The following parameters may be one of a start/length pair, a 
153     single index, a range object or an IndexSet.  If some or all of the range
154     you are selecting is already in the set, it will not be selected again.
155     
156     You can also pass an SC.SelectionSet to this method and all the selected
157     sets will be added from their instead.
158     
159     @param {SC.Array} source source object or object to add.
160     @param {Number} start index, start of range, range or IndexSet
161     @param {Number} length length if passing start/length pair.
162     @returns {SC.SelectionSet} receiver
163   */
164   add: function(source, start, length) {
165     
166     if (this.isFrozen) throw SC.FROZEN_ERROR ;
167 
168     var sets, len, idx, set, oldlen, newlen, setlen, objects;
169     
170     // normalize
171     if (start === undefined && length === undefined) {
172       if (!source) throw "Must pass params to SC.SelectionSet.add()";
173       if (source.isIndexSet) return this.add(source.source, source);
174       if (source.isSelectionSet) {
175         sets = source._sets;
176         objects = source._objects;
177         len  = sets ? sets.length : 0;
178 
179         this.beginPropertyChanges();
180         for(idx=0;idx<len;idx++) {
181           set = sets[idx];
182           if (set && set.get('length')>0) this.add(set.source, set);
183         }
184         if (objects) this.addObjects(objects);
185         this.endPropertyChanges();
186         return this ;
187         
188       }
189     }
190 
191     set    = this._indexSetForSource(source, YES);
192     oldlen = this.get('length');
193     setlen = set.get('length');
194     newlen = oldlen - setlen;
195         
196     set.add(start, length);
197 
198     this._indexSetCache = null;
199 
200     newlen += set.get('length');
201     if (newlen !== oldlen) {
202       this.propertyDidChange('length');
203       this.enumerableContentDidChange();
204       if (setlen === 0) this.notifyPropertyChange('sources');
205     }
206 
207     return this ;
208   },
209 
210   /**
211     Removes the passed index, range of indexSet belonging to the passed source
212     object from the selection set.
213     
214     The first parameter you pass must be the source array you are selecting
215     from.  The following parameters may be one of a start/length pair, a 
216     single index, a range object or an IndexSet.  If some or all of the range
217     you are selecting is already in the set, it will not be selected again.
218     
219     @param {SC.Array} source source object. must not be null
220     @param {Number} start index, start of range, range or IndexSet
221     @param {Number} length length if passing start/length pair.
222     @returns {SC.SelectionSet} receiver
223   */
224   remove: function(source, start, length) {
225     
226     if (this.isFrozen) throw SC.FROZEN_ERROR ;
227     
228     var sets, len, idx, set, oldlen, newlen, setlen, objects;
229     
230     // normalize
231     if (start === undefined && length === undefined) {
232       if (!source) throw "Must pass params to SC.SelectionSet.remove()";
233       if (source.isIndexSet) return this.remove(source.source, source);
234       if (source.isSelectionSet) {
235         sets = source._sets;
236         objects = source._objects;
237         len  = sets ? sets.length : 0;
238             
239         this.beginPropertyChanges();
240         for(idx=0;idx<len;idx++) {
241           set = sets[idx];
242           if (set && set.get('length')>0) this.remove(set.source, set);
243         }
244         if (objects) this.removeObjects(objects);
245         this.endPropertyChanges();
246         return this ;
247       }
248     }
249     
250     // save starter info
251     set    = this._indexSetForSource(source, YES);
252     oldlen = this.get('length');
253     newlen = oldlen - set.get('length');
254 
255     // if we have objects selected, determine if they are in the index 
256     // set and remove them as well.
257     if (set && (objects = this._objects)) {
258       
259       // convert start/length to index set so the iterator below will work...
260       if (length !== undefined) {
261         start = SC.IndexSet.create(start, length);
262         length = undefined;
263       }
264       
265       objects.forEach(function(object) {
266         idx = source.indexOf(object);
267         if (start.contains(idx)) {
268           objects.remove(object);
269           newlen--;
270         }
271       }, this);
272     }
273     
274     // remove indexes from source index set
275     set.remove(start, length);
276     setlen = set.get('length');
277     newlen += setlen;
278 
279     // update caches; change enumerable...
280     this._indexSetCache = null;
281     if (newlen !== oldlen) {
282       this.propertyDidChange('length');
283       this.enumerableContentDidChange();
284       if (setlen === 0) this.notifyPropertyChange('sources');
285     }
286 
287     return this ;
288   },

291   /**
292     Returns YES if the selection contains the named index, range of indexes.
293     
294     @param {Object} source source object for range
295     @param {Number} start index, start of range, range object, or indexSet
296     @param {Number} length optional range length
297     @returns {Boolean}
298   */
299   contains: function(source, start, length) {
300     if (start === undefined && length === undefined) {
301       return this.containsObject(source);
302     }
303     
304     var set = this.indexSetForSource(source);
305     if (!set) return NO ;
306     return set.contains(start, length);
307   },

309   /**
310     Returns YES if the index set contains any of the passed indexes.  You
311     can pass a single index, a range or an index set.
312     
313     @param {Object} source source object for range
314     @param {Number} start index, range, or IndexSet
315     @param {Number} length optional range length
316     @returns {Boolean}
317   */
318   intersects: function(source, start, length) {
319     var set = this.indexSetForSource(source, NO);
320     if (!set) return NO ;
321     return set.intersects(start, length);

350     Adds objects in the passed enumerable to the selection set.  Unlike adding

392   /**
393     Removes the objects from the selection set.  Note that if the selection
394     set also selects a range of indexes that includes this object, it may 
395     still be in the selection set.
396     
397     @param {Object} object 
398     @returns {SC.SelectionSet} receiver
399   */
400   removeObjects: function(objects) {
401     var cur = this._objects,
402         oldlen, newlen, sets;
403         
404     if (!cur) return this;
405 
406     oldlen = cur.get('length');
407 
408     cur.removeEach(objects);
409     newlen = cur.get('length');
410     
411     // also remove from index sets, if present
412     if (sets = this._sets) {
413       sets.forEach(function(set) {
414         oldlen += set.get('length');
415         set.removeObjects(objects);
416         newlen += set.get('length');
417       }, this);
418     }
419     
420     this._indexSetCache = null;
421     if (newlen !== oldlen) {
422       this.propertyDidChange('length');
423       this.enumerableContentDidChange();
424     }
425     return this;

428   /**
429     Returns YES if the selection contains the passed object.  This will search
430     selected ranges in all source objects.
431     
432     @param {Object} object the object to search for
433     @returns {Boolean}
434   */
435   containsObject: function(object) {
436     // fast path
437     var objects = this._objects ;
438     if (objects && objects.contains(object)) return YES ;
439     
440     var sets = this._sets,
441         len  = sets ? sets.length : 0,
442         idx, set;
443     for(idx=0;idx<len;idx++) {
444       set = sets[idx];
445       if (set && set.indexOf(object)>=0) return YES;
446     }
447     
448     return NO ;
449   },

456   /**
457     Constrains the selection set to only objects found in the passed source
458     object.  This will remove any indexes selected in other sources, any 
459     indexes beyond the length of the content, and any objects not found in the
460     set.
461     
462     @param {Object} source the source to limit
463     @returns {SC.SelectionSet} receiver
464   */
465   constrain: function(source) {
466     var set, len, max, objects;
467     
468     this.beginPropertyChanges();
469     
470     // remove sources other than this one
471     this.get('sources').forEach(function(cur) {
472       if (cur === source) return; //skip
473       var set = this._indexSetForSource(source, NO);
474       if (set) this.remove(source, set);
475     },this); 
476     
477     // remove indexes beyond end of source length
478     set = this._indexSetForSource(source, NO);
479     if (set && ((max=set.get('max'))>(len=source.get('length')))) {
480       this.remove(source, len, max-len);
481     }
482     
483     // remove objects not in source
484     if (objects = this._objects) {
485       objects.forEach(function(cur) {
486         if (source.indexOf(cur)<0) this.removeObject(cur);
487       },this);
488     }
489     
490     this.endPropertyChanges();
491     return this ;
492   },

494   /**
495     Returns YES if the passed index set or selection set contains the exact 
496     same source objects and indexes as  the receiver.  If you pass any object 
497     other than an IndexSet or SelectionSet, returns NO.
498     
499     @param {Object} obj another object.
500     @returns {Boolean}
501   */
502   isEqual: function(obj) {
503     var left, right, idx, len, sources, source;
504     
505     // fast paths
506     if (!obj || !obj.isSelectionSet) return NO ;
507     if (obj === this) return YES;
508     if ((this._sets === obj._sets) && (this._objects === obj._objects)) return YES;
509     if (this.get('length') !== obj.get('length')) return NO;
510     
511     // check objects
512     left = this._objects;
513     right = obj._objects;
514     if (left || right) {
515       if ((left ? left.get('length'):0) !== (right ? right.get('length'):0)) {
516         return NO;
517       }
518       if (left && !left.isEqual(right)) return NO ;
519     }
520 
521     // now go through the sets
522     sources = this.get('sources');
523     len     = sources.get('length');
524     for(idx=0;idx<len;idx++) {
525       source = sources.objectAt(idx);
526       left = this._indexSetForSource(source, NO);
527       right = this._indexSetForSource(source, NO);
528       if (!!right !== !!left) return NO ;
529       if (left && !left.isEqual(right)) return NO ;
530     }
531     
532     return YES ;
533   },

535   /**
536     Clears the set.  Removes all IndexSets from the object
537     
538     @returns {SC.SelectionSet}
539   */
540   clear: function() {
541     if (this.isFrozen) throw SC.FROZEN_ERROR;
542     if (this._sets) this._sets.length = 0 ; // truncate
543     if (this._objects) this._objects = null;
544     
545     this._indexSetCache = null;
546     this.propertyDidChange('length');
547     this.enumerableContentDidChange();
548     this.notifyPropertyChange('sources');
549     
550     return this ;
551   },

553   /**
554    Clones the set into a new set.  
555    
556    @returns {SC.SelectionSet}
557   */
558   copy: function() {
559     var ret  = this.constructor.create(),
560         sets = this._sets,
561         len  = sets ? sets.length : 0 ,
562         idx, set;
563     
564     if (sets && len>0) {
565       sets = ret._sets = sets.slice();
566       for(idx=0;idx<len;idx++) {
567         if (!(set = sets[idx])) continue ;
568         set = sets[idx] = set.copy();
569         ret[SC.guidFor(set.source)] = idx;
570       }
571     }
572     
573     if (this._objects) ret._objects = this._objects.copy();
574     return ret ;
575   },

577   /**
578     @private 
579     
580     Freezing a SelectionSet also freezes its internal sets.
581   */
582   freeze: function() {
583     if (this.isFrozen) return this ;
584     var sets = this._sets,
585         loc  = sets ? sets.length : 0,
586         set ;
587         
588     while(--loc >= 0) {
589       if (set = sets[loc]) set.freeze();
590     }
591     
592     if (this._objects) this._objects.freeze();
593     return sc_super();
594   },

600   /** @private */
601   toString: function() {
602     var sets = this._sets || [];
603     sets = sets.map(function(set) { 
604       return set.toString().replace("SC.IndexSet", SC.guidFor(set.source)); 
605     }, this);
606     if (this._objects) sets.push(this._objects.toString());
607     return "SC.SelectionSet:%@<%@>".fmt(SC.guidFor(this), sets.join(','));  
608   },

610   /** @private */
611   firstObject: function() {
612     var sets    = this._sets,
613         objects = this._objects;
614         
615     // if we have sets, get the first one
616     if (sets && sets.get('length')>0) {
617       var set  = sets ? sets[0] : null,
618           src  = set ? set.source : null,
619           idx  = set ? set.firstObject() : -1;
620       if (src && idx>=0) return src.objectAt(idx);
621     }
622     
623     // otherwise if we have objects, get the first one
624     return objects ? objects.firstObject() : undefined;

628   /** @private
629     Implement primitive enumerable support.  Returns each object in the 
630     selection.
631   */
632   nextObject: function(count, lastObject, context) { 
633     var objects, ret;
634     
635     // TODO: Make this more efficient.  Right now it collects all objects
636     // first.  
637     
638     if (count === 0) {
639       objects = context.objects = [];
640       this.forEach(function(o) { objects.push(o); }, this);
641       context.max = objects.length;
642     }
643 
644     objects = context.objects ;
645     ret = objects[count];
646     
647     if (count+1 >= context.max) {
648       context.objects = context.max = null;
649     }
650     
651     return ret ;
652   },

654   /** 
655     Iterates over the selection, invoking your callback with each __object__.
656     This will actually find the object referenced by each index in the 
657     selection, not just the index.
658 
659     The callback must have the following signature:
660     
661     {{{
662       function callback(object, index, source, indexSet) { ... }
663     }}}
664     
665     If you pass a target, it will be used when the callback is called.
666     
667     @param {Function} callback function to invoke.  
668     @param {Object} target optional content. otherwise uses window
669     @returns {SC.SelectionSet} receiver
670   */
671   forEach: function(callback, target) {
672     var sets = this._sets,
673         objects = this._objects,
674         len = sets ? sets.length : 0,
675         set, idx;
676         
677     for(idx=0;idx<len;idx++) {
678       set = sets[idx];
679       if (set) set.forEachObject(callback, target);
680     }
681     
682     if (objects) objects.forEach(callback, target);
683     return this ;
684   }
