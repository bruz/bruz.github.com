 11 /** @class
 12 
 13   A RecordAttribute describes a single attribute on a record.  It is used to
 14   generate computed properties on records that can automatically convert data
 15   types and verify data.
 16   
 17   When defining an attribute on an SC.Record, you can configure it this way: 
 18   
 19   {{{
 20     title: SC.Record.attr(String, { 
 21       defaultValue: 'Untitled',
 22       isRequired: YES|NO
 23     })
 24   }}}
 25   
 26   In addition to having predefined transform types, there is also a way to 
 27   set a computed relationship on an attribute. A typical example of this would
 28   be if you have record with a parentGuid attribute, but are not able to 
 29   determine which record type to map to before looking at the guid (or any
 30   other attributes). To set up such a computed property, you can attach a 
 31   function in the attribute definition of the SC.Record subclass:
 32   
 33   {{{
 34     relatedToComputed: SC.Record.toOne(function() {
 35       return (this.readAttribute('relatedToComputed').indexOf("foo")==0) ? MyApp.Foo : MyApp.Bar;
 36     })
 37   }}}
 38   
 39   Notice that we are not using .get() to avoid another transform which would 
 40   trigger an infinite loop.
 41   
 42   You usually will not work with RecordAttribute objects directly, though you
 43   may extend the class in any way that you like to create a custom attribute.
 44 
 45   A number of default RecordAttribute types are defined on the SC.Record.
 46   
 47   @extends SC.Object
 48   @since SproutCore 1.0
 49 */
 50 SC.RecordAttribute = SC.Object.extend(
 51   /** @scope SC.RecordAttribute.prototype */ {
 52 
 53   /**
 54     The default value.  If attribute is null or undefined, this default value
 55     will be substituted instead.  Note that defaultValues are not converted
 56     so the value should be in the output type expected by the attribute.
 57     
 58     If you use a defaultValue function, the arguments given to it is the
 59     record instance and the key.
 60     
 61     @property {Object|function}
 62   */
 63   defaultValue: null,
 64   
 65   /**
 66     The attribute type.  Must be either an object class or a property path
 67     naming a class.  The built in handler allows all native types to pass 
 68     through, converts records to ids and dates to UTF strings.
 69     
 70     If you use the attr() helper method to create a RecordAttribute instance,
 71     it will set this property to the first parameter you pass.
 72     
 73     @property {Object|String}
 74   */
 75   type: String,
 76   
 77   /**
 78     The underlying attribute key name this attribute should manage.  If this
 79     property is left empty, then the key will be whatever property name this
 80     attribute assigned to on the record.  If you need to provide some kind
 81     of alternate mapping, this provides you a way to override it.
 82     
 83     @property {String}
 84   */
 85   key: null,
 86   
 87   /**
 88     If YES, then the attribute is required and will fail validation unless
 89     the property is set to a non-null or undefined value.
 90     
 91     @property {Boolean}
 92   */
 93   isRequired: NO,
 94   
 95   /**
 96     If NO then attempts to edit the attribute will be ignored.
 97     
 98     @property {Boolean}
 99   */
100   isEditable: YES,  
101   
102   /**
103     If set when using the Date format, expect the ISO8601 date format.  
104     This is the default.
105     
106     @property {Boolean}
107   */
108   useIsoDate: YES,
109   
110   /**
111     Can only be used for toOne or toMany relationship attributes. If YES,
112     this flag will ensure that any related objects will also be marked
113     dirty when this record dirtied. 
114     
115     Useful when you might have multiple related objects that you want to 
116     consider in an 'aggregated' state. For instance, by changing a child
117     object (image) you might also want to automatically mark the parent 
118     (album) dirty as well.
119     
120     @property {Boolean}
121   */
122   aggregate: NO,
123   
124   // ..........................................................
125   // HELPER PROPERTIES
126   // 
127   
128   /**
129     Returns the type, resolved to a class.  If the type property is a regular
130     class, returns the type unchanged.  Otherwise attempts to lookup the 
131     type as a property path.
132     
133     @property {Object}
134   */
135   typeClass: function() {
136     var ret = this.get('type');
137     if (SC.typeOf(ret) === SC.T_STRING) ret = SC.objectForPropertyPath(ret);
138     return ret ;
139   }.property('type').cacheable(),
140   
141   /**
142     Finds the transform handler. 
143     
144     @property {Function}
145   */
146   transform: function() {
147     var klass      = this.get('typeClass') || String,
148         transforms = SC.RecordAttribute.transforms,
149         ret ;
150         
151     // walk up class hierarchy looking for a transform handler
152     while(klass && !(ret = transforms[SC.guidFor(klass)])) {
153       // check if super has create property to detect SC.Object's
154       if(klass.superclass.hasOwnProperty('create')) klass = klass.superclass ;
155       // otherwise return the function transform handler
156       else klass = SC.T_FUNCTION ;
157     }
158     
159     return ret ;
160   }.property('typeClass').cacheable(),
161   
162   // ..........................................................
163   // LOW-LEVEL METHODS
164   // 
165   
166   /** 
167     Converts the passed value into the core attribute value.  This will apply 
168     any format transforms.  You can install standard transforms by adding to
169     the SC.RecordAttribute.transforms hash.  See 
170     SC.RecordAttribute.registerTransform() for more.
171     
172     @param {SC.Record} record the record instance
173     @param {String} key the key used to access this attribute on the record
174     @param {Object} value the property value
175     @returns {Object} attribute value
176   */
177   toType: function(record, key, value) {
178     var transform = this.get('transform'),
179         type      = this.get('typeClass');
180     
181     if (transform && transform.to) {
182       value = transform.to(value, this, type, record, key) ;
183     }
184     return value ;
185   },
186 
187   /** 
188     Converts the passed value from the core attribute value.  This will apply 
189     any format transforms.  You can install standard transforms by adding to
190     the SC.RecordAttribute.transforms hash.  See 
191     SC.RecordAttribute.registerTransform() for more.
192 
193     @param {SC.Record} record the record instance
194     @param {String} key the key used to access this attribute on the record
195     @param {Object} value the property value
196     @returns {Object} attribute value
197   */
198   fromType: function(record, key, value) {
199     var transform = this.get('transform'),
200         type      = this.get('typeClass');
201     
202     if (transform && transform.from) {
203       value = transform.from(value, this, type, record, key);
204     }
205     return value;
206   },
207 
208   /**
209     The core handler.  Called from the property.
210     
211     @param {SC.Record} record the record instance
212     @param {String} key the key used to access this attribute on the record
213     @param {Object} value the property value if called as a setter
214     @returns {Object} property value
215   */
216   call: function(record, key, value) {
217     var attrKey = this.get('key') || key, nvalue;
218     
219     if ((value !== undefined) && this.get('isEditable')) {
220       // careful: don't overwrite value here.  we want the return value to 
221       // cache.
222       nvalue = this.fromType(record, key, value) ; // convert to attribute.
223       record.writeAttribute(attrKey, nvalue); 
224     } 
225 
226     nvalue = value = record.readAttribute(attrKey);
227     if (SC.none(value) && (value = this.get('defaultValue'))) {
228        if (typeof value === SC.T_FUNCTION) {
229         value = this.defaultValue(record, key, this);
230         // write default value so it doesn't have to be executed again
231         if ((nvalue !== value)  &&  record.get('store').readDataHash(record.get('storeKey'))) {
232           record.writeAttribute(attrKey, value, true);
233         }
234       }
235     } else value = this.toType(record, key, value);
236     
237     return value ;
238   },
239 
240   // ..........................................................
241   // INTERNAL SUPPORT
242   // 
243   
244   /** @private - Make this look like a property so that get() will call it. */
245   isProperty: YES,
246   
247   /** @private - Make this look cacheable */
248   isCacheable: YES,
249   
250   /** @private - needed for KVO property() support */
251   dependentKeys: [],
252   
253   /** @private */
254   init: function() {
255     sc_super();
256     // setup some internal properties needed for KVO - faking 'cacheable'
257     this.cacheKey = "__cache__" + SC.guidFor(this) ;
258     this.lastSetValueKey = "__lastValue__" + SC.guidFor(this) ;
259   }
260   
261 }) ;

267 /**
268   The default method used to create a record attribute instance.  Unlike 
269   create(), takes an attributeType as the first parameter which will be set 
270   on the attribute itself.  You can pass a string naming a class or a class
271   itself.
272   
273   @param {Object|String} attributeType the assumed attribute type
274   @param {Hash} opts optional additional config options
275   @returns {SC.RecordAttribute} new instance
276 */
277 SC.RecordAttribute.attr = function(attributeType, opts) {
278   if (!opts) opts = {} ;
279   if (!opts.type) opts.type = attributeType || String ;
280   return this.create(opts);
281 };

283 /** @private
284   Hash of registered transforms by class guid. 
285 */
286 SC.RecordAttribute.transforms = {};

288 /**
289   Call to register a transform handler for a specific type of object.  The
290   object you pass can be of any type as long as it responds to the following
291   methods:
292 
293   | *to(value, attr, klass, record, key)* | converts the passed value (which will be of the class expected by the attribute) into the underlying attribute value |
294   | *from(value, attr, klass, record, key)* | converts the underyling attribute value into a value of the class |
295   
296   @param {Object} klass the type of object you convert
297   @param {Object} transform the transform object
298   @returns {SC.RecordAttribute} receiver
299 */
300 SC.RecordAttribute.registerTransform = function(klass, transform) {
301   SC.RecordAttribute.transforms[SC.guidFor(klass)] = transform;
302 };

363 /** @private - generic converter for SC.Record-type records */
364 SC.RecordAttribute.registerTransform(SC.Record, {
365 
366   /** @private - convert a record id to a record instance */
367   to: function(id, attr, recordType, parentRecord) {
368     var store = parentRecord.get('store');
369     if (SC.none(id) || (id==="")) return null;
370     else return store.find(recordType, id);
371   },

373   /** @private - convert a record instance to a record id */
374   from: function(record) { return record ? record.get('id') : null; }

378 SC.RecordAttribute.registerTransform(SC.T_FUNCTION, {
379

381   to: function(id, attr, recordType, parentRecord) {

384     return store.find(recordType, id);

388   from: function(record) { return record.get('id'); }

391 /** @private - generic converter for Date records */
392 SC.RecordAttribute.registerTransform(Date, {
393 
394   /** @private - convert a string to a Date */
395   to: function(str, attr) {
396     if (str === null)
397       return null;
398 
399     var ret ;
400     str = str.toString() || '';
401     
402     if (attr.get('useIsoDate')) {
403       var regexp = "([0-9]{4})(-([0-9]{2})(-([0-9]{2})" +
404              "(T([0-9]{2}):([0-9]{2})(:([0-9]{2})(\\.([0-9]+))?)?" +
405              "(Z|(([-+])([0-9]{2}):([0-9]{2})))?)?)?)?",
406           d      = str.match(new RegExp(regexp)),
407           offset = 0,
408           date   = new Date(d[1], 0, 1),
409           time ;
410 
411       if (d[3]) { date.setMonth(d[3] - 1); }
412       if (d[5]) { date.setDate(d[5]); }
413       if (d[7]) { date.setHours(d[7]); }
414       if (d[8]) { date.setMinutes(d[8]); }
415       if (d[10]) { date.setSeconds(d[10]); }
416       if (d[12]) { date.setMilliseconds(Number("0." + d[12]) * 1000); }
417       if (d[14]) {
418          offset = (Number(d[16]) * 60) + Number(d[17]);
419          offset *= ((d[15] == '-') ? 1 : -1);
420       }
421 
422       offset -= date.getTimezoneOffset();
423       time = (Number(date) + (offset * 60 * 1000));
424       
425       ret = new Date();
426       ret.setTime(Number(time));
427     } else ret = new Date(Date.parse(str));
428     return ret ;
429   },
430   
431   _dates: {},
432 
433   _zeropad: function(num) { 
434     return ((num<0) ? '-' : '') + ((num<10) ? '0' : '') + Math.abs(num); 
435   },
436   
437   /** @private - convert a date to a string */
438   from: function(date) { 
439     var ret = this._dates[date.getTime()];
440     if (ret) return ret ; 
441     
442     // figure timezone
443     var zp = this._zeropad,
444         tz = 0-date.getTimezoneOffset()/60;
445         
446     tz = (tz === 0) ? 'Z' : '%@:00'.fmt(zp(tz));
447     
448     this._dates[date.getTime()] = ret = "%@-%@-%@T%@:%@:%@%@".fmt(
449       zp(date.getFullYear()),
450       zp(date.getMonth()+1),
451       zp(date.getDate()),
452       zp(date.getHours()),
453       zp(date.getMinutes()),
454       zp(date.getSeconds()),
455       tz) ;
456     
457     return ret ;
458   }

461 if (SC.DateTime && !SC.RecordAttribute.transforms[SC.guidFor(SC.DateTime)]) {
462   /**
463     Registers a transform to allow SC.DateTime to be used as a record attribute,
464     ie SC.Record.attr(SC.DateTime);
465   
466     Because SC.RecordAttribute is in the datastore framework and SC.DateTime in
467     the foundation framework, and we don't know which framework is being loaded
468     first, this chunck of code is duplicated in both frameworks.
469   
470     IF YOU EDIT THIS CODE MAKE SURE YOU COPY YOUR CHANGES to record_attribute.js. 
471   */
472 
473   SC.RecordAttribute.registerTransform(SC.DateTime, {
474   
475     /** @private
476       Convert a String to a DateTime
477     */
478     to: function(str, attr) {
479       if (SC.none(str) || SC.instanceOf(str, SC.DateTime)) return str;
480       var format = attr.get('format');
481       return SC.DateTime.parse(str, format ? format : SC.DateTime.recordFormat);
482     },
483   
484     /** @private
485       Convert a DateTime to a String
486     */
487     from: function(dt, attr) {
488       if (SC.none(dt)) return dt;
489       var format = attr.get('format');
490       return dt.toFormattedString(format ? format : SC.DateTime.recordFormat);
491     }
492   });
