 14 /**
 15   @class
 16   
 17   A split view is used to show views that the user can resize or collapse.
 18   To use a split view you need to set a topLeftView, a bottomRightView and,
 19   optionally, a splitDividerView.  You can also set various other properties
 20   to control the minimum and maximum thickness allowed for the flexible views.
 21   
 22   h2. Example
 23   
 24   {{{
 25     SC.SplitView.design({
 26       
 27       // the left view...
 28       topLeftView: SC.View.design({
 29         // view contents
 30       }),
 31       
 32       // the right view
 33       bottomRightView: SC.View.design({
 34         // view contents
 35       })
 36       
 37     })
 38   }}}
 39   
 40   When the user clicks and drags on a split divider view, it will
 41   automatically resize the views immediately before and after the split
 42   divider view. You can constrain the resizing allowed by the split view
 43   either by setting a minThickness and maxThickness property on the views
 44   themselves or by implementing the method splitViewConstrainThickness on
 45   a delegate object.
 46   
 47   In addition to resizing views, users can also collapse views by double
 48   clicking on a split divider view.  When a view is collapsed, it's isVisible
 49   property is set to NO and its space it removed from the view.  Double
 50   clicking on a divider again will restore a collapsed view.  A user can also
 51   start to drag the divider to show the collapsed view.
 52   
 53   You can programmatically control collapsing behavior using various 
 54   properties on either the split view or its child views, and/or by 
 55   implementing the method splitViewCanCollapse on a delegate object.
 56   
 57   Finally, SplitViews can layout their child views either horizontally or
 58   vertically.  To choose the direction of layout set the layoutDirection
 59   property on the view (or the :direction option with the view helper).
 60   This property should be set when the view is created. Changing it
 61   dynamically will have an unknown effect.
 62   
 63   @property {Boolean} layoutDirection Either SC.HORIZONTAL or SC.VERTICAL.
 64   Defaults to SC.HORIZONTAL. Use the :direction option with the split_view
 65   viewhelper.
 66   
 67   @property {Boolean} canCollapseViews Set to NO when you don't want any of
 68   the child views to collapse. Defaults to YES. 
 69   
 70   In addition, the top/left and bottom/right child views can have these
 71   properties:
 72   
 73   @extends SC.View
 74   @since SproutCore 1.0
 75   
 76   @author Charles Jolley
 77   @author Lawrence Pit
 78   @author Erich Ocean
 79 */

165   /**

184         view = this[viewAry[idx]] = this.createChildView(view, {

196   /**
197     Layout the views.
198     
199     This method needs to be called anytime you change the view thicknesses
200     to make sure they are arranged properly.  This will set up the views so
201     that they can resize appropriately.
202   */
203   updateChildLayout: function() {
204     var topLeftView = this.get('topLeftView') ,
205         bottomRightView = this.get('bottomRightView') ,
206         dividerView = this.get('dividerView') ,
207         direction = this.get('layoutDirection') ,
208         topLeftThickness = this._desiredTopLeftThickness ;
209     var dividerThickness = this.get('dividerThickness') ;
210         dividerThickness = (!SC.none(dividerThickness)) ? dividerThickness : 7 ;
211     var splitViewThickness = (direction === SC.LAYOUT_HORIZONTAL) ? this.get('frame').width : this.get('frame').height ,
212         bottomRightThickness = splitViewThickness - dividerThickness - topLeftThickness ,
213         autoresizeBehavior = this.get('autoresizeBehavior') ,
214         layout , isCollapsed ;
215     
216     
217     // top/left view
218     isCollapsed = topLeftView.get('isCollapsed') || NO ;
219     topLeftView.setIfChanged('isVisible', !isCollapsed) ;
220     layout = SC.clone(topLeftView.get('layout'));
221     if (direction === SC.LAYOUT_HORIZONTAL) {
222       layout.top = 0 ;
223       layout.left = 0 ;
224       layout.bottom = 0 ;
225       switch (autoresizeBehavior) {
226         case SC.RESIZE_BOTH:
227           throw "SC.RESIZE_BOTH is currently unsupported.";
228         case SC.RESIZE_TOP_LEFT:
229           layout.right = bottomRightThickness + dividerThickness ;
230           delete layout.width ;
231           break ;
232         case SC.RESIZE_BOTTOM_RIGHT:
233           delete layout.right ;
234           delete layout.height ;
235           layout.width = topLeftThickness ;
236           break ;
237       }
238     } else {
239       layout.top = 0;
240       layout.left = 0 ;
241       layout.right = 0 ;
242       switch (autoresizeBehavior) {
243         case SC.RESIZE_BOTH:
244           throw "SC.RESIZE_BOTH is currently unsupported.";
245         case SC.RESIZE_TOP_LEFT:
246           layout.bottom = bottomRightThickness + dividerThickness ;
247           delete layout.height ;
248           break ;
249         case SC.RESIZE_BOTTOM_RIGHT:
250           delete layout.bottom ;
251           delete layout.width ;
252           layout.height = topLeftThickness ;
253           break ;
254       }
255     }
256     topLeftView.set('layout', layout);
257     
258     // split divider view
259     if (dividerView) {
260       layout = SC.clone(dividerView.get('layout'));
261       if (direction === SC.LAYOUT_HORIZONTAL) {
262         layout.width = dividerThickness;
263         delete layout.height ;
264         layout.top = 0 ;
265         layout.bottom = 0 ;
266         switch (autoresizeBehavior) {
267           case SC.RESIZE_BOTH:
268             throw "SC.RESIZE_BOTH is currently unsupported.";
269             // delete layout.left ;
270             // delete layout.right ;
271             // layout.centerX = topLeftThickness + (dividerThickness / 2) ;
272             // delete layout.centerY ;
273             //break ;
274           case SC.RESIZE_TOP_LEFT:
275             delete layout.left ;
276             layout.right = bottomRightThickness ;
277             delete layout.centerX ;
278             delete layout.centerY ;
279             break ;
280           case SC.RESIZE_BOTTOM_RIGHT:
281             layout.left = topLeftThickness ;
282             delete layout.right ;
283             delete layout.centerX ;
284             delete layout.centerY ;
285             break ;
286         }
287       } else {
288         delete layout.width ;
289         layout.height = dividerThickness ;
290         layout.left = 0 ;
291         layout.right = 0 ;
292         switch (autoresizeBehavior) {
293           case SC.RESIZE_BOTH:
294             throw "SC.RESIZE_BOTH is currently unsupported.";
295             // delete layout.top ;
296             // delete layout.bottom ;
297             // delete layout.centerX ;
298             // layout.centerY = topLeftThickness + (dividerThickness / 2) ;
299             //break ;
300           case SC.RESIZE_TOP_LEFT:
301             delete layout.top ;
302             layout.bottom = bottomRightThickness ;
303             delete layout.centerX ;
304             delete layout.centerY ;
305             break ;
306           case SC.RESIZE_BOTTOM_RIGHT:
307             layout.top = topLeftThickness ;
308             delete layout.bottom ;
309             delete layout.centerX ;
310             delete layout.centerY ;
311             break ;
312         }
313       }
314       dividerView.set('layout', layout);
315     }
316     
317     // bottom/right view
318     isCollapsed = bottomRightView.get('isCollapsed') || NO ;
319     bottomRightView.setIfChanged('isVisible', !isCollapsed) ;
320     layout = SC.clone(bottomRightView.get('layout'));
321     if (direction === SC.LAYOUT_HORIZONTAL) {
322       layout.top = 0 ;
323       layout.bottom = 0 ;
324       layout.right = 0 ;
325       switch (autoresizeBehavior) {
326         case SC.RESIZE_BOTH:
327           throw "SC.RESIZE_BOTH is currently unsupported.";
328           //break ;
329         case SC.RESIZE_BOTTOM_RIGHT:
330           layout.left = topLeftThickness + dividerThickness ;
331           delete layout.width ;
332           break ;
333         case SC.RESIZE_TOP_LEFT:
334           delete layout.left ;
335           layout.width = bottomRightThickness ;
336           break ;
337       }
338     } else {
339       layout.left = 0 ;
340       layout.right = 0 ;
341       layout.bottom = 0 ;
342       switch (autoresizeBehavior) {
343         case SC.RESIZE_BOTH:
344           throw "SC.RESIZE_BOTH is currently unsupported.";
345           //break ;
346         case SC.RESIZE_BOTTOM_RIGHT:
347           layout.top = topLeftThickness + dividerThickness ;
348           delete layout.height ;
349           break ;
350         case SC.RESIZE_TOP_LEFT:
351           delete layout.top ;
352           layout.height = bottomRightThickness ;
353           break ;
354       }
355     }
356     bottomRightView.set('layout', layout);
357     
358     this.notifyPropertyChange('topLeftThickness')
359         .notifyPropertyChange('bottomRightThickness');
360   },

362   /** @private */
363   renderLayout: function(context, firstTime) {
364     if (firstTime || this._recalculateDivider) {
365       if (!this.get('thumbViewCursor')) {
366         this.set('thumbViewCursor', SC.Cursor.create()) ;
367       }
368       
369       var layoutDirection = this.get('layoutDirection') ,
370           fr = this.get('frame'),
371           splitViewThickness, elemRendered = this.$(),
372           desiredThickness = this.get('defaultThickness') ,
373           autoResizeBehavior = this.get('autoresizeBehavior') ;
374       var dividerThickness = this.get('dividerThickness') ;
375       dividerThickness = (!SC.none(dividerThickness)) ? dividerThickness : 7 ;
376       // Turn a flag on to recalculate the spliting if the desired thickness
377       // is a percentage
378       if(this._recalculateDivider===undefined && desiredThickness<1) {
379         this._recalculateDivider=YES;
380       }
381       else if(this._recalculateDivider) this._recalculateDivider=NO;
382       
383       
384       if(elemRendered[0]) {
385         splitViewThickness = (layoutDirection === SC.LAYOUT_HORIZONTAL) ? 
386               elemRendered[0].offsetWidth : elemRendered[0].offsetHeight ;
387       }else{
388         splitViewThickness = (layoutDirection === SC.LAYOUT_HORIZONTAL) ? 
389               fr.width : fr.height ;
390       }
391       // if default thickness is < 1, convert from percentage to absolute
392       if (SC.none(desiredThickness) || 
393         (desiredThickness > 0 && desiredThickness < 1)) {
394         desiredThickness =  Math.floor((splitViewThickness - 
395                             (dividerThickness))* (desiredThickness || 0.5)) ;
396       }
397       if (autoResizeBehavior === SC.RESIZE_BOTTOM_RIGHT) {
398         this._desiredTopLeftThickness = desiredThickness ;
399       } else { // (autoResizeBehavior === SC.RESIZE_TOP_LEFT)
400         this._desiredTopLeftThickness =  splitViewThickness 
401                                       - dividerThickness - desiredThickness ;
402       }
403       
404       // make sure we don't exceed our min and max values, and that collapse 
405       // settings are respected
406       // cached values are required by _updateTopLeftThickness() below...
407       this._topLeftView = this.get('topLeftView') ;
408       this._bottomRightView = this.get('bottomRightView') ;
409       this._topLeftViewThickness = this.thicknessForView(this.get('topLeftView'));
410       this._bottomRightThickness = this.thicknessForView(this.get('bottomRightView'));
411       this._dividerThickness = this.get('dividerThickness') ;
412       this._layoutDirection = this.get('layoutDirection') ;
413       
414       // this handles min-max settings and collapse parameters
415       this._updateTopLeftThickness(0) ;
416       
417       // update the cursor used by thumb views
418       this._setCursorStyle() ;
419       
420       // actually set layout for our child views
421       this.updateChildLayout() ;
422     }
423     sc_super() ;
424   },

437   /**
438     Update the split view's layout based on mouse movement.
439     
440     Call this method in the mouseDown: method of your thumb view. The split view
441     will begin tracking the mouse and will update its own layout to reflect the movement 
442     of the mouse. As a result, the position of your thumb view will also be updated.
443     
444     @returns {Boolean}
445   */
446   mouseDownInThumbView: function(evt, thumbView) {
447     var responder = this.getPath('pane.rootResponder') ;
448     if (!responder) return NO ; // nothing to do
449       
450     // we're not the source view of the mouseDown:, so we need to capture events manually to receive them
451     responder.dragDidStart(this) ;
452     
453     // cache for later
454     this._mouseDownX = evt.pageX ;
455     this._mouseDownY = evt.pageY ;
456     this._thumbView = thumbView ;
457     this._topLeftView = this.get('topLeftView') ;
458     this._bottomRightView = this.get('bottomRightView') ;
459     this._topLeftViewThickness = this.thicknessForView(this.get('topLeftView'));
460     this._bottomRightThickness = this.thicknessForView(this.get('bottomRightView'));
461     this._dividerThickness = this.get('dividerThickness') ;
462     this._layoutDirection = this.get('layoutDirection') ;
463     
464     this.beginLiveResize() ;
465     this._inLiveResize = YES ;
466     
467     return YES ;
468   },
469   
470   mouseDragged: function(evt) {
471     var offset = (this._layoutDirection === SC.LAYOUT_HORIZONTAL) ? evt.pageX 
472                 - this._mouseDownX : evt.pageY - this._mouseDownY ;
473     this._updateTopLeftThickness(offset) ;
474     return YES;
475   },
476   
477   mouseUp: function(evt) {
478     if (this._inLiveResize === YES) {
479     	this._thumbView = null ; // avoid memory leaks
480     	this._inLiveResize = NO ;
481     	this.endLiveResize() ;
482     	return YES ;
483 		}
484 		
485 		return NO ;
486   },

489     return this.mouseDragged(evt);

492   touchEnd: function(evt){
493     return this.mouseUp(evt);

497   doubleClickInThumbView: function(evt, thumbView) {
498     var view = this._topLeftView,
499         isCollapsed = view.get('isCollapsed') || NO ;
500     if (!isCollapsed && !this.canCollapseView(view)) {
501       view = this._bottomRightView ;
502       isCollapsed = view.get('isCollapsed') || NO ;
503       if (!isCollapsed && !this.canCollapseView(view)) return NO;
504     }
505     
506     if (!isCollapsed) {
507       // remember thickness in it's uncollapsed state
508       this._uncollapsedThickness = this.thicknessForView(view)  ;
509       // and collapse
510       // this.setThicknessForView(view, 0) ;
511       if (view === this._topLeftView) {
512         this._updateTopLeftThickness(this.topLeftThickness()*-1) ;
513       } else {
514         this._updateBottomRightThickness(this.bottomRightThickness()*-1) ;
515       }
516       
517       // if however the splitview decided not to collapse, clear:
518       if (!view.get("isCollapsed")) {
519         this._uncollapsedThickness = null;
520       }
521     } else {
522       // uncollapse to the last thickness in it's uncollapsed state
523       if (view === this._topLeftView) {
524         this._updateTopLeftThickness(this._uncollapsedThickness) ;
525       } else {
526         this._updateBottomRightThickness(this._uncollapsedThickness) ;
527       }
528       view._uncollapsedThickness = null ;
529     }
530     this._setCursorStyle() ;
531     return true ;
532   },

534   /** @private */
535   _updateTopLeftThickness: function(offset) {
536     var topLeftView = this._topLeftView ,
537         bottomRightView = this._bottomRightView,
538         // the current thickness, not the original thickness
539         topLeftViewThickness = this.thicknessForView(topLeftView), 
540         bottomRightViewThickness = this.thicknessForView(bottomRightView),
541         minAvailable = this._dividerThickness ,
542         maxAvailable = 0,
543         proposedThickness = this._topLeftViewThickness + offset,
544         direction = this._layoutDirection,
545         bottomRightCanCollapse = this.canCollapseView(bottomRightView),
546         thickness = proposedThickness,
547         // constrain to thickness set on top/left
548         max = this.get('topLeftMaxThickness'),
549         min = this.get('topLeftMinThickness'),
550         bottomRightThickness, tlCollapseAtThickness, brCollapseAtThickness;
551     
552     if (!topLeftView.get("isCollapsed")) {
553       maxAvailable += topLeftViewThickness ;
554     }
555     if (!bottomRightView.get("isCollapsed")) {
556       maxAvailable += bottomRightViewThickness ;
557     }
558     
559     if (!SC.none(max)) thickness = Math.min(max, thickness) ;
560     if (!SC.none(min)) thickness = Math.max(min, thickness) ;
561     
562     // constrain to thickness set on bottom/right
563     max = this.get('bottomRightMaxThickness') ;
564     min = this.get('bottomRightMinThickness') ;
565     bottomRightThickness = maxAvailable - thickness ;
566     if (!SC.none(max)) {
567       bottomRightThickness = Math.min(max, bottomRightThickness) ;
568     }
569     if (!SC.none(min)) {
570       bottomRightThickness = Math.max(min, bottomRightThickness) ;
571     }
572     thickness = maxAvailable - bottomRightThickness ;
573     
574     // constrain to thickness determined by delegate.
575     thickness = this.invokeDelegateMethod(this.delegate, 
576       'splitViewConstrainThickness', this, topLeftView, thickness) ;
577     
578     // cannot be more than what's available
579     thickness = Math.min(thickness, maxAvailable) ;
580     
581     // cannot be less than zero
582     thickness = Math.max(0, thickness) ;
583     
584     tlCollapseAtThickness = topLeftView.get('collapseAtThickness') ;
585     if (!tlCollapseAtThickness) tlCollapseAtThickness = 0 ;
586     brCollapseAtThickness = bottomRightView.get('collapseAtThickness') ;
587     brCollapseAtThickness = SC.none(brCollapseAtThickness) ?
588                       maxAvailable : (maxAvailable - brCollapseAtThickness);
589     
590     if ((proposedThickness <= tlCollapseAtThickness) && 
591           this.canCollapseView(topLeftView)) {
592       // want to collapse top/left, check if this doesn't violate the max thickness of bottom/right
593       max = bottomRightView.get('maxThickness');
594       if (!max || (minAvailable + maxAvailable) <= max) {
595         // collapse top/left view, even if it has a minThickness
596         thickness = 0 ;
597       }
598     } else if (proposedThickness >= brCollapseAtThickness && 
599               this.canCollapseView(bottomRightView)) {
600       // want to collapse bottom/right, check if this doesn't violate the max thickness of top/left
601       max = topLeftView.get('maxThickness');
602       if (!max || (minAvailable + maxAvailable) <= max) {
603         // collapse bottom/right view, even if it has a minThickness
604         thickness = maxAvailable;
605       }
606     }
607     
608     // now apply constrained value
609     if (thickness != this.thicknessForView(topLeftView)) {
610       this._desiredTopLeftThickness = thickness ;
611       
612       // un-collapse if needed.
613       topLeftView.set('isCollapsed', thickness === 0) ;
614       bottomRightView.set('isCollapsed', thickness >= maxAvailable) ;
615       
616       this.updateChildLayout(); // updates child layouts
617       this.displayDidChange(); // updates cursor
618     }
619   },

622   _updateBottomRightThickness: function(offset) {
623     var topLeftView = this._topLeftView ,
624         bottomRightView = this._bottomRightView,
625         topLeftViewThickness = this.thicknessForView(topLeftView), // the current thickness, not the original thickness
626         bottomRightViewThickness = this.thicknessForView(bottomRightView),
627         minAvailable = this._dividerThickness ,
628         maxAvailable = 0,
629         proposedThickness = this._topLeftViewThickness + offset,
630         direction = this._layoutDirection,
631         bottomRightCanCollapse = this.canCollapseView(bottomRightView),
632         thickness = proposedThickness,
633         // constrain to thickness set on top/left
634         max = this.get('topLeftMaxThickness'),
635         min = this.get('topLeftMinThickness'),
636         bottomRightThickness, tlCollapseAtThickness, brCollapseAtThickness;
637     
638     if (!topLeftView.get("isCollapsed")) maxAvailable += topLeftViewThickness ;
639     if (!bottomRightView.get("isCollapsed")) maxAvailable += bottomRightViewThickness ;
640     
641     if (!SC.none(max)) thickness = Math.min(max, thickness) ;
642     if (!SC.none(min)) thickness = Math.max(min, thickness) ;
643     
644     // constrain to thickness set on bottom/right
645     max = this.get('bottomRightMaxThickness') ;
646     min = this.get('bottomRightMinThickness') ;
647     bottomRightThickness = maxAvailable - thickness ;
648     if (!SC.none(max)) bottomRightThickness = Math.min(max, bottomRightThickness) ;
649     if (!SC.none(min)) bottomRightThickness = Math.max(min, bottomRightThickness) ;
650     thickness = maxAvailable - bottomRightThickness ;
651     
652     // constrain to thickness determined by delegate.
653     thickness = this.invokeDelegateMethod(this.delegate, 'splitViewConstrainThickness', this, topLeftView, thickness) ;
654     
655     // cannot be more than what's available
656     thickness = Math.min(thickness, maxAvailable) ;
657     
658     // cannot be less than zero
659     thickness = Math.max(0, thickness) ;
660     
661     tlCollapseAtThickness = topLeftView.get('collapseAtThickness') ;
662     if (!tlCollapseAtThickness) tlCollapseAtThickness = 0 ;
663     brCollapseAtThickness = bottomRightView.get('collapseAtThickness') ;
664     brCollapseAtThickness = SC.none(brCollapseAtThickness) ? maxAvailable : (maxAvailable - brCollapseAtThickness);
665     
666     if ((proposedThickness <= tlCollapseAtThickness) && this.canCollapseView(topLeftView)) {
667       // want to collapse top/left, check if this doesn't violate the max thickness of bottom/right
668       max = bottomRightView.get('maxThickness');
669       if (!max || (minAvailable + maxAvailable) <= max) {
670         // collapse top/left view, even if it has a minThickness
671         thickness = 0 ;
672       }
673     } else if (proposedThickness >= brCollapseAtThickness && this.canCollapseView(bottomRightView)) {
674       // want to collapse bottom/right, check if this doesn't violate the max thickness of top/left
675       max = topLeftView.get('maxThickness');
676       if (!max || (minAvailable + maxAvailable) <= max) {
677         // collapse bottom/right view, even if it has a minThickness
678         thickness = maxAvailable;
679       }
680     }
681     
682     // now apply constrained value
683     if (thickness != this.thicknessForView(topLeftView)) {
684       this._desiredTopLeftThickness = thickness ;
685       
686       // un-collapse if needed.
687       topLeftView.set('isCollapsed', thickness === 0) ;
688       bottomRightView.set('isCollapsed', thickness >= maxAvailable) ;
689       
690       this.updateChildLayout(); // updates child layouts
691       this.displayDidChange(); // updates cursor
692     }
693   },

695   /** 
696     This observes 'layoutDirection' to update the cursor style immediately
697     after the value of the layoutDirection of Split view is changed
698 
699     @private 
700   */
701   _setCursorStyle: function() {
702     var topLeftView = this._topLeftView,
703         bottomRightView = this._bottomRightView,
704         thumbViewCursor = this.get('thumbViewCursor'),
705         // updates the cursor of the thumb view that called 
706         // mouseDownInThumbView() to reflect the status of the drag
707         tlThickness = this.thicknessForView(topLeftView),
708         brThickness = this.thicknessForView(bottomRightView);
709     this._layoutDirection = this.get('layoutDirection') ;
710     if (topLeftView.get('isCollapsed') || 
711       tlThickness === this.get("topLeftMinThickness") || 
712       brThickness == this.get("bottomRightMaxThickness")) {
713       thumbViewCursor.set('cursorStyle', 
714         this._layoutDirection === SC.LAYOUT_HORIZONTAL ? "e-resize" : "s-resize") ;
715     } else if (bottomRightView.get('isCollapsed') || 
716       tlThickness === this.get("topLeftMaxThickness") || 
717       brThickness == this.get("bottomRightMinThickness")) {
718       thumbViewCursor.set('cursorStyle', 
719         this._layoutDirection === SC.LAYOUT_HORIZONTAL ? "w-resize" : "n-resize") ;
720     } else {
721       if(SC.browser.msie) {
722         thumbViewCursor.set('cursorStyle', 
723           this._layoutDirection === SC.LAYOUT_HORIZONTAL ? "e-resize" : "n-resize") ;
724       }
725       else {
726         thumbViewCursor.set('cursorStyle', 
727           this._layoutDirection === SC.LAYOUT_HORIZONTAL ? "ew-resize" : "ns-resize") ;
728       }
729     }
730   }.observes('layoutDirection'),
