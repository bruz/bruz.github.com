  9 /** @class
 10 
 11   SC.PopupButtonView displays a pop-up menu when clicked, from which the user
 12   can select an item.
 13 
 14   To use, create the SC.PopupButtonView as you would a standard SC.ButtonView,
 15   then set the menu property to an instance of SC.MenuPane. For example:
 16 
 17 {{{
 18 SC.PopupButtonView.design({
 19   layout: { width: 200, height: 18 },
 20   menuBinding: 'MyApp.menuController.menuPane'
 21 });
 22 }}}
 23 
 24   You would then have your MyApp.menuController return an instance of the menu
 25   to display.
 26 
 27   @extends SC.ButtonView
 28   @author Santosh Shanbhogue
 29   @author Tom Dale
 30   @copyright 2008-2010, Sprout Systems, Inc. and contributors.
 31   @version 1.0
 32 */
 33 SC.PopupButtonView = SC.ButtonView.extend(
 34 /** @scope SC.PopupButtonView.prototype */ {
 35   classNames: ['sc-popup-button'],
 36 
 37   // ..........................................................
 38   // PROPERTIES
 39   //
 40 
 41   /**
 42     The prefer matrix to use when displaying the menu.
 43 
 44     @property
 45   */
 46   preferMatrix: null,
 47 
 48   /**
 49     The SC.MenuPane that should be displayed when the button is clicked.
 50 
 51     @type {SC.MenuPane}
 52     @default null
 53   */
 54   menu: null,
 55 
 56   // ..........................................................
 57   // INTERNAL SUPPORT
 58   //
 59 
 60   /** @private
 61     Binds the button's selection state to the menu's visibility.
 62   */
 63   isActiveBinding: '*menu.isVisibleInWindow',
 64 
 65   /** @private
 66     Displays the menu.
 67 
 68     @param {SC.Event} evt
 69   */
 70   action: function(evt)
 71   {
 72     var menu = this.get('menu') ;
 73 
 74     if (!menu) {
 75       //@ if (debug)
 76       SC.Logger.warn("SC.PopupButton - Unable to show menu because the menu property is set to %@.".fmt(menu));
 77       //@ endif
 78       return NO ;
 79     }
 80 
 81     menu.popup(this, this.get('preferMatrix')) ;
 82     return YES;
 83   },
 84 
 85   /** @private
 86     On mouse down, we set the state of the button, save some state for further
 87     processing, then call the button's action method.
 88 
 89     @param {SC.Event} evt
 90     @returns {Boolean}
 91   */
 92   mouseDown: function(evt) {
 93     // If disabled, handle mouse down but ignore it.
 94     if (!this.get('isEnabled')) return YES ;
 95 
 96     this._isMouseDown = YES;
 97 
 98     this._action() ;
 99 
100     // Store the current timestamp. We register the timestamp at the end of
101     // the runloop so that the menu has been rendered, in case that operation
102     // takes more than a few hundred milliseconds.
103 
104     // One mouseUp, we'll use this value to determine how long the mouse was
105     // pressed.
106     this.invokeLast(this._recordMouseDownTimestamp);
107 
108     return YES ;
109   },
110 
111   /** @private
112     Records the current timestamp. This is invoked at the end of the runloop
113     by mouseDown. We use this value to determine the delay between mouseDown
114     and mouseUp.
115   */
116   _recordMouseDownTimestamp: function() {
117     this._menuRenderedTimestamp = new Date().getTime();
118   },
119 
120   /** @private
121     Because we responded YES to the mouseDown event, we have responsibility
122     for handling the corresponding mouseUp event.
123 
124     However, the user may click on this button, then drag the mouse down to a
125     menu item, and release the mouse over the menu item. We therefore need to
126     delegate any mouseUp events to the menu's menu item, if one is selected.
127 
128     We also need to differentiate between a single click and a click and hold.
129     If the user clicks and holds, we want to close the menu when they release.
130     Otherwise, we should wait until they click on the menu's modal pane before
131     removing our active state.
132 
133     @param {SC.Event} evt
134     @returns {Boolean}
135   */
136   mouseUp: function(evt) {
137     var timestamp = new Date().getTime(),
138         previousTimestamp = this._menuRenderedTimestamp,
139         menu = this.get('menu'),
140         touch = SC.platform.touch,
141         targetMenuItem;
142 
143     if (menu) {
144       // Get the menu item the user is currently hovering their mouse over
145       targetMenuItem = menu.getPath('rootMenu.targetMenuItem');
146 
147       if (targetMenuItem) {
148         // Have the menu item perform its action.
149         // If the menu returns NO, it had no action to
150         // perform, so we should close the menu immediately.
151         if (!targetMenuItem.performAction()) menu.remove();
152       } else {
153         // If the user waits more than certain amount of time between
154         // mouseDown and mouseUp, we can assume that they are clicking and
155         // dragging to the menu item, and we should close the menu if they
156         //mouseup anywhere not inside the menu.
157         if (!touch && (timestamp - previousTimestamp > SC.ButtonView.CLICK_AND_HOLD_DELAY)) {
158           menu.remove();
159         }
160       }
161     }
162 
163     // Reset state.
164     this._isMouseDown = NO;
165     sc_super();
166     return YES;
167   },
168 
169   /** @private
170     Overrides ButtonView's mouseExited method to remove the behavior where the
171     active state is removed on mouse exit. We want the button to remain active
172     as long as the menu is visible.
173 
174     @param {SC.Event} evt
175     @returns {Boolean}
176   */
177   mouseExited: function(evt) {
178     return YES;
179   },
180 
181   /** @private
182     Overrides performKeyEquivalent method to pass any keyboard shortcuts to
183     the menu.
184 
185     @param {String} charCode string corresponding to shortcut pressed (e.g.,
186     alt_shift_z)
187     @param {SC.Event} evt
188   */
189   performKeyEquivalent: function( charCode, evt )
190   {
191     if (!this.get('isEnabled')) return NO ;
192     var menu = this.get('menu') ;
193     return (!!menu && menu.performKeyEquivalent(charCode, evt)) ;
194   },
195 
196   /** @private */
197   acceptsFirstResponder: function() {
198     if(!SC.SAFARI_FOCUS_BEHAVIOR) return this.get('isEnabled');
199     else return NO;
200   }.property('isEnabled')
201 
202 });
