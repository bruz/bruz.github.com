 13 /** @class
 14 
 15   The RootResponder captures events coming from a web browser and routes them
 16   to the correct view in the view hierarchy.  Usually you do not work with a
 17   RootResponder directly.  Instead you will work with Pane objects, which
 18   register themselves with the RootResponder as needed to receive events.
 19 
 20   h1. RootResponder and Platforms
 21 
 22   RootResponder contains core functionality common among the different web
 23   platforms. You will likely be working with a subclass of RootResponder that
 24   implements functionality unique to that platform.
 25 
 26   The correct instance of RootResponder is detected at runtime and loaded
 27   transparently.
 28 
 29   h1. Event Types
 30 
 31   RootResponders can route four types of events:
 32 
 33   - Direct events, such as mouse and touch events.  These are routed to the
 34     nearest view managing the target DOM elment. RootResponder also handles
 35     multitouch events so that they are delegated to the correct views.
 36   - Keyboard events. These are sent to the keyPane, which will then send the
 37     event to the current firstResponder and up the responder chain.
 38   - Resize events. When the viewport resizes, these events will be sent to all
 39     panes.
 40   - Keyboard shortcuts. Shortcuts are sent to the keyPane first, which
 41     will go down its view hierarchy. Then they go to the mainPane, which will
 42     go down its view hierarchy.
 43   - Actions. Actions are generic messages that your application can send in
 44     response to user action or other events. You can either specify an
 45     explicit target, or allow the action to traverse the hierarchy until a
 46     view is found that handles it.
 47 */

168   /**
169     Makes the passed pane the new key pane.  If you pass null or if the pane
170     does not accept key focus, then key focus will transfer to the previous
171     key pane (if it is still attached), and so on down the stack.  This will
172     notify both the old pane and the new root View that key focus has changed.
173 
174     @param {SC.Pane} pane
175     @returns {SC.RootResponder} receiver
176   */
177   makeKeyPane: function(pane) {
178     // Was a pane specified?
179     var newKeyPane, previousKeyPane, previousKeyPanes ;
180 
181     if (pane) {
182       // Does the specified pane accept being the key pane?  If not, there's
183       // nothing to do.
184       if (!pane.get('acceptsKeyPane')) {
185         return this ;
186       }
187       else {
188         // It does accept key pane status?  Then push the current keyPane to
189         // the top of the stack and make the specified pane the new keyPane.
190         // First, though, do a sanity-check to make sure it's not already the
191         // key pane, in which case we have nothing to do.
192         previousKeyPane = this.get('keyPane') ;
193         if (previousKeyPane === pane) {
194           return this ;
195         }
196         else {
197           if (previousKeyPane) {
198             previousKeyPanes = this.get('previousKeyPanes') ;
199             previousKeyPanes.push(previousKeyPane) ;
200           }
201 
202           newKeyPane = pane ;
203         }
204       }
205     }
206     else {
207       // No pane was specified?  Then pop the previous key pane off the top of
208       // the stack and make it the new key pane, assuming that it's still
209       // attached and accepts key pane (its value for acceptsKeyPane might
210       // have changed in the meantime).  Otherwise, we'll keep going up the
211       // stack.
212       previousKeyPane = this.get('keyPane') ;
213       previousKeyPanes = this.get('previousKeyPanes') ;
214 
215       newKeyPane = null ;
216       while (previousKeyPanes.length > 0) {
217         var candidate = previousKeyPanes.pop();
218         if (candidate.get('isPaneAttached')  &&  candidate.get('acceptsKeyPane')) {
219           newKeyPane = candidate ;
220           break ;
221         }
222       }
223     }
224 
225 
226     // If we found an appropriate candidate, make it the new key pane.
227     // Otherwise, make the main pane the key pane (if it accepts it).
228     if (!newKeyPane) {
229       var mainPane = this.get('mainPane') ;
230       if (mainPane && mainPane.get('acceptsKeyPane')) newKeyPane = mainPane ;
231     }
232 
233     // now notify old and new key views of change after edit
234     if (previousKeyPane) previousKeyPane.willLoseKeyPaneTo(newKeyPane) ;
235     if (newKeyPane) newKeyPane.willBecomeKeyPaneFrom(previousKeyPane) ;
236 
237     this.set('keyPane', newKeyPane) ;
238 
239     if (newKeyPane) newKeyPane.didBecomeKeyPaneFrom(previousKeyPane) ;
240     if (previousKeyPane) previousKeyPane.didLoseKeyPaneTo(newKeyPane) ;
241 
242     return this ;
243   },

427   _responderFor: function(target, methodName) {
428     var defaultResponder = target ? target.get('defaultResponder') : null;
429 
430     if (target) {
431       target = target.get('firstResponder') || target;
432       do {
433         if (target.respondsTo(methodName)) return target ;
434       } while ((target = target.get('nextResponder'))) ;
435     }
436 
437     // HACK: Eventually we need to normalize the sendAction() method between
438     // this and the ResponderContext, but for the moment just look for a
439     // ResponderContext as the defaultResponder and return it if present.
440     if (typeof defaultResponder === SC.T_STRING) {
441       defaultResponder = SC.objectForPropertyPath(defaultResponder);
442     }
443 
444     if (!defaultResponder) return null;
445     else if (defaultResponder.isResponderContext) return defaultResponder;
446     else if (defaultResponder.respondsTo(methodName)) return defaultResponder;
447     else return null;
448   },

450   /**
451     Attempts to determine the initial target for a given action/target/sender
452     tuple.  This is the method used by sendAction() to try to determine the
453     correct target starting point for an action before trickling up the
454     responder chain.
455 
456     You send actions for user interface events and for menu actions.
457 
458     This method returns an object if a starting target was found or null if no
459     object could be found that responds to the target action.
460 
461     Passing an explicit target or pane constrains the target lookup to just
462     them; the defaultResponder and other panes are *not* searched.
463 
464     @param {Object|String} target or null if no target is specified
465     @param {String} method name for target
466     @param {Object} sender optional sender
467     @param {SC.Pane} optional pane
468     @returns {Object} target object or null if none found
469   */
470   targetForAction: function(methodName, target, sender, pane) {
471 
472     // 1. no action, no target...
473     if (!methodName || (SC.typeOf(methodName) !== SC.T_STRING)) {
474       return null ;
475     }
476 
477     // 2. an explicit target was passed...
478     if (target) {
479       if (SC.typeOf(target) === SC.T_STRING) {
480         target =  SC.objectForPropertyPath(target) || 
481                   SC.objectForPropertyPath(target, sender);
482       }
483 
484       if (target && !target.isResponderContext) {
485         if (target.respondsTo && !target.respondsTo(methodName)) {
486           target = null ;
487         } else if (SC.typeOf(target[methodName]) !== SC.T_FUNCTION) {
488           target = null ;
489         }
490       }
491 
492       return target ;
493     }
494 
495     // 3. an explicit pane was passed...
496     if (pane) {
497       return this._responderFor(pane, methodName) ;
498     }
499 
500     // 4. no target or pane passed... try to find target in the active panes
501     // and the defaultResponder
502     var keyPane = this.get('keyPane'), mainPane = this.get('mainPane') ;
503 
504     // ...check key and main panes first
505     if (keyPane && (keyPane !== pane)) {
506       target = this._responderFor(keyPane, methodName) ;
507     }
508     if (!target && mainPane && (mainPane !== keyPane)) {
509       target = this._responderFor(mainPane, methodName) ;
510     }
511 
512     // ...still no target? check the defaultResponder...
513     if (!target && (target = this.get('defaultResponder'))) {
514       if (SC.typeOf(target) === SC.T_STRING) {
515         target = SC.objectForPropertyPath(target) ;
516         if (target) this.set('defaultResponder', target) ; // cache if found
517       }
518       if (target) {
519         if (target.respondsTo && !target.respondsTo(methodName)) {
520           target = null ;
521         } else if (SC.typeOf(target[methodName]) !== SC.T_FUNCTION) {
522           target = null ;
523         }
524       }
525     }
526 
527     return target ;
528   },

598   /**
599     Called when the document is ready to begin handling events.  Setup event
600     listeners in this method that you are interested in observing for your
601     particular platform.  Be sure to call sc_super().
602 
603     @returns {void}
604   */
605   setup: function() {
606     // handle touch events
607     this.listenFor('touchstart touchmove touchend touchcancel'.w(), document);
608 
609     // handle basic events
610     this.listenFor('keydown keyup beforedeactivate mousedown mouseup click dblclick mouseout mouseover mousemove selectstart contextmenu'.w(), document)
611         .listenFor('resize'.w(), window);
612         
613     if(SC.browser.msie) this.listenFor('focusin focusout'.w(), document);
614     else this.listenFor('focus blur'.w(), window);
615 
616     // handle animation events
617     this.listenFor('webkitAnimationStart webkitAnimationIteration webkitAnimationEnd'.w(), document);
618     
619     // handle special case for keypress- you can't use normal listener to block the backspace key on Mozilla
620     if (this.keypress) {
621       if (SC.CAPTURE_BACKSPACE_KEY && SC.browser.mozilla) {
622         var responder = this ;
623         document.onkeypress = function(e) {
624           e = SC.Event.normalizeEvent(e);
625           return responder.keypress.call(responder, e);
626         };
627 
628         // SC.Event.add(window, 'unload', this, function() { document.onkeypress = null; }); // be sure to cleanup memory leaks
629 
630       // Otherwise, just add a normal event handler.
631       } else SC.Event.add(document, 'keypress', this, this.keypress);
632     }
633 
634     // handle these two events specially in IE
635     'drag selectstart'.w().forEach(function(keyName) {
636       var method = this[keyName] ;
637       if (method) {
638         if (SC.browser.msie) {
639           var responder = this ;
640           document.body['on' + keyName] = function(e) {
641             // return method.call(responder, SC.Event.normalizeEvent(e));
642             return method.call(responder, SC.Event.normalizeEvent(event || window.event)); // this is IE :(
643           };
644 
645           // be sure to cleanup memory leaks
646            SC.Event.add(window, 'unload', this, function() {
647             document.body['on' + keyName] = null;
648           });
649 
650         } else {
651           SC.Event.add(document, keyName, this, method);
652         }
653       }
654     }, this);
655 
656     // handle mousewheel specifically for FireFox
657     var mousewheel = SC.browser.mozilla ? 'DOMMouseScroll' : 'mousewheel';
658     SC.Event.add(document, mousewheel, this, this.mousewheel);
659 
660 
661 
662     // If the browser is identifying itself as a touch-enabled browser, but
663     // touch events are not present, assume this is a desktop browser doing
664     // user agent spoofing and simulate touch events automatically.
665     if (SC.browser && SC.platform && SC.browser.mobileSafari && !SC.platform.touch) {
666       SC.platform.simulateTouchEvents();
667     }
668 
669     // do some initial set
670     this.set('currentWindowSize', this.computeWindowSize()) ;
671     this.focus(); // assume the window is focused when you load.
672 
673     if (SC.browser.mobileSafari) {
674       // Monkey patch RunLoop if we're in MobileSafari
675       var f = SC.RunLoop.prototype.endRunLoop, patch;
676 
677       patch = function() {
678         // Call original endRunLoop implementation.
679         if (f) f.apply(this, arguments);
680 
681         // This is a workaround for a bug in MobileSafari.
682         // Specifically, if the target of a touchstart event is removed from the DOM,
683         // you will not receive future touchmove or touchend events. What we do is, at the
684         // end of every runloop, check to see if the target of any touches has been removed
685         // from the DOM. If so, we re-append it to the DOM and hide it. We then mark the target
686         // as having been moved, and it is de-allocated in the corresponding touchend event.
687         var touches = SC.RootResponder.responder._touches, touch, elem, target, textNode, view, found = NO;
688         if (touches) {
689           // Iterate through the touches we're currently tracking
690           for (touch in touches) {
691             if (touches[touch]._rescuedElement) continue; // only do once
692             
693             target = elem = touches[touch].target;
694 
695             // Travel up the hierarchy looking for the document body
696             while (elem && (elem = elem.parentNode) && !found) {
697               found = (elem === document.body);
698             }
699 
700             // If we aren't part of the body, move the element back
701             // but make sure we hide it from display.
702             if (!found && target) {
703 
704               // Actually clone this node and replace it in the original
705               // layer if needed
706               if (target.parentNode && target.cloneNode) {
707                 var clone = target.cloneNode(true);  
708                 target.parentNode.replaceChild(clone, target);
709                 target.swapNode = clone; // save for restore later

917   /**
918     The touch responder for any given touch is the view which will receive touch events
919     for that touch. Quite simple.
920 
921     makeTouchResponder takes a potential responder as an argument, and, by calling touchStart on each
922     nextResponder, finds the actual responder. As a side-effect of how it does this, touchStart is called
923     on the new responder before touchCancelled is called on the old one (touchStart has to accept the touch
924     before it can be considered cancelled).
925 
926     You usually don't have to think about this at all. However, if you don't want your view to,
927     for instance, prevent scrolling in a ScrollView, you need to make sure to transfer control
928     back to the previous responder:
929 
930     if (Math.abs(touch.pageY - touch.startY) > this.MAX_SWIPE) touch.restoreLastTouchResponder();
931 
932     You don't call makeTouchResponder on RootResponder directly. Instead, it gets called for you
933     when you return YES to captureTouch or touchStart.
934 
935     You do, however, use a form of makeTouchResponder to return to a previous touch responder. Consider
936     a button view inside a ScrollView: if the touch moves too much, the button should give control back
937     to the scroll view.
938 
939     if (Math.abs(touch.pageX - touch.startX) > 4) {
940       if (touch.nextTouchResponder) touch.makeTouchResponder(touch.nextTouchResponder);
941     }
942 
943     This will give control back to the containing view. Maybe you only want to do it if it is a ScrollView?
944 
945     if (Math.abs(touch.pageX - touch.startX) > 4 && touch.nextTouchResponder && touch.nextTouchResponder.isScrollable)
946       touch.makeTouchResponder(touch.nextTouchResponder);
947 
948     Possible gotcha: while you can do touch.nextTouchResponder, the responders are not chained in a linked list like
949     normal responders, because each touch has its own responder stack. To navigate through the stack (or, though
950     it is not recommended, change it), use touch.touchResponders (the raw stack array).
951 
952     makeTouchResponder is called with an event object. However, it usually triggers custom touchStart/touchCancelled
953     events on the views. The event object is passed so that functions such as stopPropagation may be called.
954   */
955   makeTouchResponder: function(touch, responder, shouldStack, upViewChain) {
956     var stack = touch.touchResponders, touchesForView;
957 
958     // find the actual responder (if any, I suppose)
959     // note that the pane's sendEvent function is slightly clever:
960     // if the target is already touch responder, it will just return it without calling touchStart
961     // we must do the same.
962     if (touch.touchResponder === responder) return;
963 
964     // send touchStart
965     // get the target pane
966     var pane;
967     if (responder) pane = responder.get('pane') ;
968     else pane = this.get('keyPane') || this.get('mainPane') ;
969 
970     // if the responder is not already in the stack...
971     
972     if (stack.indexOf(responder) < 0) {
973       // if we need to go up the view chain, do so
974       if (upViewChain) {
975         // if we found a valid pane, send the event to it
976         try {
977           responder = (pane) ? pane.sendEvent("touchStart", touch, responder) : null ;
978         } catch (e) {
979           SC.Logger.error("Error in touchStart: " + e);
980           responder = null;
981         }
982       } else {
983         
984         if ((responder.get ? responder.get("acceptsMultitouch") : responder.acceptsMultitouch) || !responder.hasTouch) {
985           if (!responder.touchStart(touch)) responder = null;
986         } else {
987           // do nothing; the responder is the responder, and may stay the responder, and all will be fine
988         }
989       }
990     }
991 
992     // if the item is in the stack, we will go to it (whether shouldStack is true or not)
993     // as it is already stacked
994     if (!shouldStack || (stack.indexOf(responder) > -1 && stack[stack.length - 1] !== responder)) {
995       // first, we should unassign the touch. Note that we only do this IF WE ARE removing
996       // the current touch responder. Otherwise we cause all sorts of headaches; why? Because,
997       // if we are not (suppose, for instance, that it is stacked), then the touch does not
998       // get passed back to the touch responder-- even while it continues to get events because
999       // the touchResponder is still set!
1000       this.unassignTouch(touch);
1001       
1002       // pop all other items
1003       var idx = stack.length - 1, last = stack[idx];
1004       while (last && last !== responder) {
1005         // unassign the touch
1006         touchesForView = this.touchesForView(last); // won't even exist if there are no touches
1007 
1008         // send touchCancelled (or, don't, if the view doesn't accept multitouch and it is not the last touch)
1009         if ((last.get ? last.get("acceptsMultitouch") : last.acceptsMultitouch) || !touchesForView) {
1010           if (last.touchCancelled) last.touchCancelled(touch);
1011         }
1012 
1013         // go to next (if < 0, it will be undefined, so lovely)
1014         idx--;
1015         last = stack[idx];
1016 
1017         // update responders (for consistency)
1018         stack.pop();
1019 
1020         touch.touchResponder = stack[idx];
1021         touch.nextTouchResponder = stack[idx - 1];
1022       }
1023 
1024     }
1025 
1026     // now that we've popped off, we can push on
1027     if (responder) {
1028       this.assignTouch(touch, responder);
1029 
1030       // keep in mind, it could be one we popped off _to_ above...
1031       if (responder !== touch.touchResponder) {
1032         stack.push(responder);
1033 
1034         // update responder helpers
1035         touch.touchResponder = responder;
1036         touch.nextTouchResponder = stack[stack.length - 2];
1037       }
1038     }

1041   /**
1042     captureTouch is used to find the view to handle a touch. It starts at the starting point and works down
1043     to the touch's target, looking for a view which captures the touch. If no view is found, it uses the target
1044     view.
1045 
1046     Then, it triggers a touchStart event starting at whatever the found view was; this propagates up the view chain
1047     until a view responds YES. This view becomes the touch's owner.
1048 
1049     You usually do not call captureTouch, and if you do call it, you'd call it on the touch itself:
1050     touch.captureTouch(startingPoint, shouldStack)
1051 
1052     If shouldStack is YES, the previous responder will be kept so that it may be returned to later.
1053   */
1054   captureTouch: function(touch, startingPoint, shouldStack) {
1055     if (!startingPoint) startingPoint = this;
1056 
1057     var target = touch.targetView, view = target,
1058         chain = [], idx, len;
1059 
1060     if (SC.LOG_TOUCH_EVENTS) {
1061       SC.Logger.info('  -- Received one touch on %@'.fmt(target.toString()));
1062     }
1063     // work up the chain until we get the root
1064     while (view && (view !== startingPoint)) {
1065       chain.unshift(view);
1066       view = view.get('nextResponder');
1067     }
1068 
1069     // work down the chain
1070     for (len = chain.length, idx = 0; idx < len; idx++) {
1071       view = chain[idx];
1072       if (SC.LOG_TOUCH_EVENTS) SC.Logger.info('  -- Checking %@ for captureTouch response…'.fmt(view.toString()));
1073 
1074       // see if it captured the touch
1075       if (view.tryToPerform('captureTouch', touch)) {
1076         if (SC.LOG_TOUCH_EVENTS) SC.Logger.info('   -- Making %@ touch responder because it returns YES to captureTouch'.fmt(view.toString()));
1077 
1078         // if so, make it the touch's responder
1079         this.makeTouchResponder(touch, view, shouldStack, YES); // triggers touchStart/Cancel/etc. event.
1080         return; // and that's all we need
1081       }
1082     }
1083 
1084     if (SC.LOG_TOUCH_EVENTS) SC.Logger.info("   -- Didn't find a view that returned YES to captureTouch, so we're calling touchStart");
1085     // if we did not capture the touch (obviously we didn't)
1086     // we need to figure out what view _will_
1087     // Thankfully, makeTouchResponder does exactly that: starts at the view it is supplied and keeps calling startTouch
1088     this.makeTouchResponder(touch, target, shouldStack, YES);
