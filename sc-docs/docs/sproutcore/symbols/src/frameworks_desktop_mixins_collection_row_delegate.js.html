 10 /** 
 11   @namespace
 12   
 13   CollectionRowDelegates are consulted by SC.ListView and SC.TableView to 
 14   control the height of rows, including specifying custom heights for 
 15   specific rows.  
 16   
 17   You can implement a custom row height in one of two ways.  
 18   
 19 */
 20 SC.CollectionRowDelegate = {
 21 
 22   /** walk like a duck */
 23   isCollectionRowDelegate: YES,
 24   
 25   /**
 26     Default row height.  Unless you implement some custom row height 
 27     support, this row height will be used for all items.
 28     
 29     @property
 30     @type Number
 31   */
 32   rowHeight: 18,
 33 
 34   /**
 35     Index set of rows that should have a custom row height.  If you need 
 36     certains rows to have a custom row height, then set this property to a 
 37     non-null value.  Otherwise leave it blank to disable custom row heights.
 38     
 39     @property
 40     @type SC.IndexSet
 41   */
 42   customRowHeightIndexes: null,
 43   
 44   /**
 45     Called for each index in the customRowHeightIndexes set to get the 
 46     actual row height for the index.  This method should return the default
 47     rowHeight if you don't want the row to have a custom height.
 48     
 49     The default implementation just returns the default rowHeight.
 50     
 51     @param {SC.CollectionView} view the calling view
 52     @param {Object} content the content array
 53     @param {Number} contentIndex the index 
 54     @returns {Number} row height
 55   */
 56   contentIndexRowHeight: function(view, content, contentIndex) {
 57     return this.get('rowHeight');    
 58   }
 59   
 60   
 61 };
