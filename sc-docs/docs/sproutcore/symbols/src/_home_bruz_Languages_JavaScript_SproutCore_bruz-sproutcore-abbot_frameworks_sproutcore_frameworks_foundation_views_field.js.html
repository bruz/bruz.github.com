 12 /** @class
 13 
 14   Base view for managing a view backed by an input element.  Since the web
 15   browser provides native support for editing input elements, this view
 16   provides basic support for listening to changes on these input elements and
 17   responding to them.
 18   
 19   Generally you will not work with a FieldView directly.  Instead, you should
 20   use one of the subclasses implemented by your target platform such as 
 21   SC.CheckboxView, SC.RadioView, SC.TextFieldView, and so on.
 22   
 23   @extends SC.View
 24   @extends SC.Control
 25   @extends SC.Validatable
 26   @since SproutCore 1.0
 27 */
 28 SC.FieldView = SC.View.extend(SC.Control, SC.Validatable,

 39   /**
 40     The raw value of the field itself.  This is computed from the 'value'
 41     property by passing it through any validator you might have set.  This is 
 42     the value that will be set on the field itself when the view is updated.
 43     
 44     @property {String}
 45   */  
 46   fieldValue: function() {
 47     var value = this.get('value');
 48     if (SC.typeOf(value) === SC.T_ERROR) value = value.get('errorValue');
 49     return this.fieldValueForObject(value);
 50   }.property('value', 'validator').cacheable(),

 56   /**
 57     Override to return an CoreQuery object that selects the input elements
 58     for the view.  If this method is defined, the field view will 
 59     automatically edit the attrbutes of the input element to reflect the 
 60     current isEnabled state among other things.
 61   */
 62   $input: function() { 
 63     if(this.get('isTextArea')){
 64       return this.$('textarea').andSelf().filter('textarea'); 
 65     }else{
 66       return this.$('input').andSelf().filter('input');
 67     }
 68   },

 70   /**
 71     Override to set the actual value of the field.
 72     
 73     The default implementation will simple copy the newValue to the value
 74     attribute of any input tags in the receiver view.  You can override this
 75     method to provide specific functionality needed by your view.
 76     
 77     @param {Object} newValue the value to display.
 78     @returns {SC.FieldView} receiver
 79   */
 80   setFieldValue: function(newValue) {
 81     if (SC.none(newValue)) newValue = '' ;
 82     var input = this.$input();
 83     
 84     // Don't needlessly set the element if it already has the value, because
 85     // doing so moves the cursor to the end in some browsers.
 86     if (input.val() !== newValue) {
 87       input.val(newValue);
 88     }
 89     return this ;
 90   },
 91   
 92   /**
 93     Override to retrieve the actual value of the field.
 94     
 95     The default implementation will simply retrieve the value attribute from
 96     the first input tag in the receiver view.
 97     
 98     @returns {String} value
 99   */
100   getFieldValue: function() {
101     return this.$input().val();
102   },
103   
104   _field_fieldValueDidChange: function(evt) {
105     SC.RunLoop.begin();
106     this.fieldValueDidChange(NO);
107     SC.RunLoop.end();  
108   },

117     partialChange parameter.  This will change the kind of validation done on
118     the value.  Otherwise, the validator may mark the field as having an error
119     when the user is still in mid-edit.
120   
121     @param partialChange (optional) YES if this is a partial change.
122     @returns {Boolean|SC.Error} result of validation.
123   */

126     var fieldValue = this.getFieldValue();
127     var value = this.objectForFieldValue(fieldValue, partialChange);

166   /** @private
167     invoked when the value property changes.  Sets the field value...
168   */
169   _field_valueDidChange: function() {
170     this.setFieldValue(this.get('fieldValue'));
171   }.observes('fieldValue'),
172 
173   /** @private
174     after the layer is created, set the field value and observe events
175   */
176   didCreateLayer: function() {
177     this.setFieldValue(this.get('fieldValue'));
178     SC.Event.add(this.$input(), 'change', this, this._field_fieldValueDidChange) ;
179   },
180 
181   /** @private
182     after the layer is append to the doc, set the field value and observe events
183     only for textarea.
184   */
185   didAppendToDocument: function() {
186     if (this.get('isTextArea')) {
187       this.setFieldValue(this.get('fieldValue'));
188       SC.Event.add(this.$input(), 'change', this, this._field_fieldValueDidChange) ;
189     }
190   },
191   
192   willDestroyLayer: function() {
193     SC.Event.remove(this.$input(), 'change', this, this._field_fieldValueDidChange); 
194   },

215   /**
216     Allow the browser to do its normal event handling for the mouse down
217     event.  But first, set isActive to YES.
218   */
219   mouseDown: function(evt) {  
220     this._field_isMouseDown = YES;
221     evt.allowDefault(); 
222     return YES; 
223   },
224   
225   /** @private
226     Remove the active class on mouseOut if mouse is down.
227   */  
228   mouseOut: function(evt) {
229     if (this._field_isMouseDown) this.set('isActive', NO);
230     evt.allowDefault();
231     return YES;
232   },
233   
234   /** @private
235     If mouse was down and we renter the button area, set the active state again.
236   */  
237   mouseOver: function(evt) {
238     this.set('isActive', this._field_isMouseDown);
239     evt.allowDefault();
240     return YES;
241   },
242   
243   /** @private
244     on mouse up, remove the isActive class and then allow the browser to do
245     its normal thing.
246   */  
247   mouseUp: function(evt) {
248     // track independently in case isEnabled has changed
249     if (this._field_isMouseDown) this.set('isActive', NO); 
250     this._field_isMouseDown = NO;
251     evt.allowDefault();
252     return YES ;
253   },
254   
255   /** @private
256     Simply allow keyDown & keyUp to pass through to the default web browser
257     implementation.
258   */
259   keyDown: function(evt) {
260 
261     // handle tab key
262     if (evt.which === 9) {
263       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
264       if (view) view.becomeFirstResponder();
265       else evt.allowDefault();
266       return YES ; // handled
267     }
268     
269     // validate keyDown...
270     if (this.performValidateKeyDown(evt)) {
271       this._isKeyDown = YES ;
272       evt.allowDefault(); 
273     } else {
274       evt.stop();
275     }
276     
277     return YES; 
278   },
279   
280   /** tied to the isEnabled state */
281   acceptsFirstResponder: function() {
282     if(!SC.SAFARI_FOCUS_BEHAVIOR) return this.get('isEnabled');
283     else return NO;
284   }.property('isEnabled'),
285   
286   willBecomeKeyResponderFrom: function(keyView) {
287     // focus the text field.
288     if (!this._isFocused) {
289       this._isFocused = YES ;
290       this.becomeFirstResponder();
291       if (this.get('isVisibleInWindow')) {
292         this.$input()[0].focus();
293       }
294     }
295   },
296   
297   willLoseKeyResponderTo: function(responder) {
298     if (this._isFocused) this._isFocused = NO ;
299   },
300     
301   // these methods use the validator to convert the raw field value returned
302   // by your subclass into an object and visa versa.
303   _field_setFieldValue: function(newValue) {
304     this.propertyWillChange('fieldValue');
305     if (this.fieldValueForObject) {
306       newValue = this.fieldValueForObject(newValue) ;
307     }
308     var ret = this.setFieldValue(newValue) ;
309     this.propertyDidChange('fieldValue');
310     return ret ;
311   },
312   
313   _field_getFieldValue: function() {
314     var ret = this.getFieldValue() ;
315     if (this.objectForFieldValue) ret = this.objectForFieldValue(ret);
316     return ret ;
317   }
