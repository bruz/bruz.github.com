  7 /**
  8   @class
  9 
 10   Displays a horizontal or vertical separator line.  Simply create one of 
 11   these views and configure the layout direction and layout frame.
 12   
 13   @extends SC.View
 14   @since SproutCore 1.0
 15 */
 16 SC.SeparatorView = SC.View.extend(
 17 /** @scope SC.SeparatorView.prototype */ {
 18 
 19   classNames: ['sc-separator-view'],
 20   tagName: 'span',
 21 
 22   /** 
 23     Select the direction of the separator line.  Must be one of SC.LAYOUT_VERTICAL or SC.LAYOUT_HORIZONTAL.
 24     
 25     @property {String}
 26   */
 27   layoutDirection: SC.LAYOUT_HORIZONTAL,
 28 
 29   render: function(context, firstTime) {
 30     if(firstTime) context.push('<span></span>');
 31 	  context.addClass(this.get('layoutDirection'));
 32   }
 33 
 34 
 35 
 36 });
