1 /** 
  2   An experimental CollectionView mixin that makes it extremely fast under
  3   certain circumstances, including for mobile devices.
  4 */
  5 SC.CollectionFastPath = {

  9   initMixin: function() {
 10     this._indexMap = {};

 13   /**
 14     Returns the pool for a given example view.
 15     
 16     The pool is calculated based on the guid for the example view class.
 17   */
 18   poolForExampleView: function(exampleView) {
 19     var poolKey = "_pool_" + SC.guidFor(exampleView);
 20     if (!this[poolKey]) this[poolKey] = [];
 21     return this[poolKey];
 22   },
 23   
 24   /**
 25     Creates an item view from a given example view, configuring it with basic settings
 26     and the supplied attributes.
 27   */
 28   createItemViewFromExampleView: function(exampleView, attrs) {
 29     // create the example view
 30     var ret = exampleView.create(attrs);
 31     
 32     // for our pooling, if it is poolable, mark the view as poolable and
 33     // give it a reference to its pool.
 34     if (ret.isPoolable) {
 35       ret.owningPool = this.poolForExampleView(exampleView);
 36     }
 37     
 38     // we will sometimes need to know what example view created the item view
 39     ret.createdFromExampleView = exampleView;
 40 
 41     // and now, return (duh)
 42     return ret;
 43   },
 44   
 45   configureItemView: function(itemView, attrs) {
 46     // set settings. Self explanatory.
 47     itemView.beginPropertyChanges();
 48     itemView.setIfChanged('content', attrs.content);
 49     itemView.setIfChanged('contentIndex', attrs.contentIndex);
 50     itemView.setIfChanged('parentView', attrs.parentView);
 51     itemView.setIfChanged('layerId', attrs.layerId);
 52     itemView.setIfChanged('isEnabled', attrs.isEnabled);
 53     itemView.setIfChanged('isSelected', attrs.isSelected);
 54     itemView.setIfChanged('outlineLevel', attrs.outlineLevel);
 55     itemView.setIfChanged('layout', attrs.layout);
 56     itemView.setIfChanged('disclosureState', attrs.disclosureState);
 57     itemView.setIfChanged('isVisibleInWindow', attrs.isVisibleInWindow);
 58     itemView.setIfChanged('isGroupView', attrs.isGroupView);
 59     itemView.setIfChanged('page', this.page);
 60     itemView.endPropertyChanges();
 61   },
 62   
 63   /**
 64     Configures a pooled view, calling .awakeFromPool if it is defined.
 65   */
 66   wakePooledView: function(itemView, attrs) {
 67     // configure
 68     this.configureItemView(itemView, attrs);
 69     
 70     // awake from the pool, etc.
 71     if (itemView.awakeFromPool) itemView.awakeFromPool(itemView.owningPool, this);
 72   },
 73   
 74   /**
 75     Gets an item view from an example view, from a pool if possible, and otherwise
 76     by generating it.
 77   */
 78   allocateItemView: function(exampleView, attrs) {
 79     // we will try to get it from a pool. This will fill ret. If ret is not
 80     // filled, then we'll know to generate one.
 81     var ret;
 82     
 83     // if it is poolable, we just grab from the pool.
 84     if (exampleView.prototype.isPoolable) {
 85       var pool = this.poolForExampleView(exampleView);
 86       if (pool.length > 0) {
 87         ret = pool.pop();
 88         this.wakePooledView(ret, attrs);
 89       }
 90     }
 91     
 92     if (!ret) {
 93       ret = this.createItemViewFromExampleView(exampleView, attrs);
 94     }
 95     
 96     return ret;
 97   },
 98   
 99   /**
100     Releases an item view. If the item view is pooled, it puts it into the pool;
101     otherwise, this calls .destroy().
102     
103     This is called for one of two purposes: to release a view that is no longer displaying,
104     or to release an older cached version of a view that needed to be replaced because the
105     example view changed.
106   */
107   releaseItemView: function(itemView) {
108     // if it is not poolable, there is not much we can do.
109     if (!itemView.isPoolable) {
110       itemView.destroy();
111       return;
112     }
113     
114     // otherwise, we need to return to view
115     var pool = itemView.owningPool;
116     pool.push(itemView);
117     if (itemView.hibernateInPool) itemView.hibernateInPool(pool, this);
118   },

120   /**
121     Returns YES if the item at the index is a group.
122     
123     @private
124   */
125   contentIndexIsGroup: function(index, contentObject) {
126     var contentDelegate = this.get("contentDelegate");
127     
128     // setup our properties
129     var groupIndexes = this.get('_contentGroupIndexes'), isGroupView = NO;
130     
131     // and do our checking
132     isGroupView = groupIndexes && groupIndexes.contains(index);
133     if (isGroupView) isGroupView = contentDelegate.contentIndexIsGroup(this, this.get("content"), index);
134     
135     // and return
136     return isGroupView;
137   },

139   /**
140     @private
141     Determines the example view for a content index. There are two optional parameters that will
142     speed things up: contentObject and isGroupView. If you don't supply them, they must be computed.
143   */
144   exampleViewForItem: function(item, index) {
145     var del = this.get('contentDelegate'),
146         groupIndexes = this.get('_contentGroupIndexes'),
147         key, ExampleView,
148         isGroupView = this.contentIndexIsGroup(index, item);
149 
150     if (isGroupView) {
151       // so, if it is indeed a group view, we go that route to get the example view
152       key = this.get('contentGroupExampleViewKey');
153       if (key && item) ExampleView = item.get(key);
154       if (!ExampleView) ExampleView = this.get('groupExampleView') || this.get('exampleView');
155     } else {
156       // otherwise, we go through the normal example view
157       key = this.get('contentExampleViewKey');
158       if (key && item) ExampleView = item.get(key);
159       if (!ExampleView) ExampleView = this.get('exampleView');
160     }
161     
162     return ExampleView;
163   },

165   /**
166     This may seem somewhat awkward, but it is for memory performance: this fills in a hash
167     YOU provide with the properties for the given content index.
168     
169     Properties include both the attributes given to the view and some CollectionView tracking
170     properties, most importantly the exampleView.
171     
172     
173     @private
174   */
175   setAttributesForItem: function(item, index, attrs) {
176     var del = this.get('contentDelegate'), 
177         isGroupView = this.contentIndexIsGroup(index),
178         ExampleView = this.exampleViewForItem(item, index),
179         content = this.get("content");
180     
181     // 
182     // FIGURE OUT "NORMAL" ATTRIBUTES
183     //
184     attrs.createdFromExampleView = ExampleView;
185     attrs.parentView = this.get('containerView') || this;
186     attrs.contentIndex = index;
187     attrs.owner = attrs.displayDelegate = this;
188     attrs.content = item;
189     attrs.page = this.page;
190     attrs.layerId = this.layerIdFor(index);
191     attrs.isEnabled = del.contentIndexIsEnabled(this, content, index);
192     attrs.isSelected = del.contentIndexIsSelected(this, content, index);
193     attrs.outlineLevel = del.contentIndexOutlineLevel(this, content, index);
194     attrs.disclosureState = del.contentIndexDisclosureState(this, content, index);
195     attrs.isVisibleInWindow = this.get('isVisibleInWindow');
196     attrs.isGroupView = isGroupView;
197     attrs.layout = this.layoutForContentIndex(index);
198     if (!attrs.layout) attrs.layout = ExampleView.prototype.layout;
199   },

208   /**
209     @private
210     Returns mapped item views for the supplied item.
211   */
212   mappedViewsForItem: function(item, map) {
213     if (!map) map = this._viewMap;
214     return map[SC.guidFor(item)];
215   },
216   
217   /**
218     @private
219     Returns the mapped view for an item at the specified index. 
220   */
221   mappedViewForItem: function(item, idx, map) {
222     if (!map) map = this._viewMap;
223     var m = map[SC.guidFor(item)];
224     if (!m) return undefined;
225     return m[idx];
226   },
227   
228   /**
229     @private
230     Maps a view to an item/index combination.
231   */
232   mapView: function(item, index, view, map) {
233     // get the default view map if a map was not supplied
234     if (!map) map = this._viewMap;
235     
236     // get the item map
237     var g = SC.guidFor(item),
238         imap = map[g];
239     if (!imap) imap = map[g] = {_length: 0};
240     
241     // fill in the index
242     imap[index] = view;
243     imap._length++;
244   },
245   
246   /**
247     Unmaps a view from an item/index combination.
248   */
249   unmapView: function(item, index, map) {
250     if (!map) map = this._viewMap;
251     var g = SC.guidFor(item),
252         imap = map[g];
253     
254     // return if there is nothing to do
255     if (!imap) return;
256     
257     // remove
258     if (imap[index]) {
259       var v = imap[index];
260       delete imap[index];
261       
262       imap._length--;
263       if (imap._length <= 0) delete map[g];
264     }
265   },
266   
267   /**
268     Returns the item view for the given content index.
269     NOTE: THIS WILL ADD THE VIEW TO DOM TEMPORARILY (it will be cleaned if
270           it is not used). As such, use sparingly.
271   */
272   itemViewForContentIndex: function(index) {
273     var view = this._indexMap[index];
274     if (!view) {
275       var content = this.get("content"), item;
276       item = content.objectAt(index);
277       if (!item) return null;
278       
279       // get example view and create item view
280       var exampleView = this.exampleViewForItem(item, index);
281       view = this.addItemView(exampleView, item, index);
282     }
283     return view;
284   },
285   
286   /**
287     @private
288     Returns the nearest item view index to the supplied index mapped to the item.
289   */
290   nearestMappedViewIndexForItem: function(item, index, map) {
291     var m = this.mappedViewsForItem(item, map);
292     if (!m) return null;
293     
294     // keep track of nearest and the nearest distance
295     var nearest = null, ndist = -1, dist = 0;
296     
297     // loop through
298     for (var idx in m) {
299       idx = parseInt(idx, 10);
300       if (isNaN(idx)) continue;
301       // get distance
302       dist = Math.abs(index - idx);
303       
304       // compare to nearest distance
305       if (ndist < 0 || dist < ndist) {
306         ndist = dist;
307         nearest = idx;
308       }
309     }
310     
311     return nearest;
312   },

314   /**
315     @private
316     Remaps the now showing views to their new indexes (if they have moved).
317   */
318   remapItemViews: function(nowShowing) {
319     // reset the view map, but keep the old for removing
320     var oldMap = this._viewMap || {},
321         newMap = (this._viewMap = {}),
322         indexMap = (this._indexMap = {}),
323         mayExist = [],
324         content = this.get("content"), item;
325 
326     if (!content) return;
327     var itemsToAdd = this._itemsToAdd;
328 
329     // first, find items which we can (that already exist, etc.)
330     nowShowing.forEach(function(idx) {
331       item = content.objectAt(idx);
332       
333       // determine if we have view(s) in the old map for the item
334       var possibleExistingViews = this.mappedViewsForItem(item, oldMap);
335       if (possibleExistingViews) {
336         
337         // if it is the same index, we just take it. End of story.
338         if (possibleExistingViews[idx]) {
339           var v = possibleExistingViews[idx];
340           this.unmapView(item, idx, oldMap);
341           this.mapView(item, idx, v, newMap);
342           indexMap[idx] = v;
343         } else {
344           // otherwise, we must investigate later
345           mayExist.push(idx);
346         }
347       } else {
348         // if it is in now showing but we didn't find a view, it needs to be created.
349         itemsToAdd.push(idx);
350       }
351     }, this);
352     
353     // now there are also some items which _could_ exist (but might not!)
354     for (var idx = 0, len = mayExist.length; idx < len; idx++) {
355       var newIdx = mayExist[idx];
356       item = content.objectAt(newIdx);
357       var nearestOldIndex = this.nearestMappedViewIndexForItem(item, newIdx, oldMap),
358           nearestView;
359       
360       if (!SC.none(nearestOldIndex)) {
361         nearestView = this.mappedViewForItem(item, nearestOldIndex, oldMap);
362         var newExampleView = this.exampleViewForItem(item, newIdx);
363         if (newExampleView === nearestView.createdFromExampleView) {
364           // if there is a near one, use it, and remove it from the map
365           this.unmapView(item, nearestOldIndex, oldMap);
366           this.mapView(item, newIdx, nearestView, newMap);
367           indexMap[newIdx] = nearestView;
368         } else {
369           itemsToAdd.push(newIdx);
370         }
371       } else {
372         // otherwise, we need to create it.
373         itemsToAdd.push(newIdx);
374       }
375     }
376     
377     return oldMap;
378   },

380   /**
381     Reloads.
382   */
383   reloadIfNeeded: function(nowShowing, scrollOnly) {
384     var content = this.get("content"), invalid;
385     
386     // we use the nowShowing to determine what should and should not be showing.
387     if (!nowShowing || !nowShowing.isIndexSet) nowShowing = this.get('nowShowing');
388     
389     // we only update if this is a non-scrolling update.
390     // don't worry: we'll actually update after the fact, and the invalid indexes should
391     // be queued up nicely.
392     if (!scrollOnly) {
393       invalid = this._invalidIndexes;
394       if (!invalid || !this.get('isVisibleInWindow')) return this;
395       this._invalidIndexes = NO; 
396       
397       // tell others we will be reloading
398       if (invalid.isIndexSet && invalid.contains(nowShowing)) invalid = YES ;
399       if (this.willReload) this.willReload(invalid === YES ? null : invalid);
400     }
401     
402     // get arrays of items to add/remove
403     var itemsToAdd = this._itemsToAdd || (this._itemsToAdd = []);
404     
405     // remap
406     var oldMap = this.remapItemViews(nowShowing);
407     
408     // The oldMap has the items to remove, so supply it to processRemovals
409     this.processRemovals(oldMap);
410     
411     // handle the invalid set (if it is present)
412     if (invalid) {
413       this.processUpdates(invalid === YES ? nowShowing : invalid);
414     }
415     
416     // process items to add
417     this.processAdds();
418     
419     // only clear the DOM pools if this is not during scrolling. Adding/removing is a
420     // bad idea while scrolling :)
421     if (!scrollOnly) this.clearDOMPools();
422     
423     // clear the lists
424     itemsToAdd.length = 0;
425     
426     // and if this is a full reload, we need to adjust layout
427     if (!scrollOnly) {
428       var layout = this.computeLayout();
429       if (layout) this.adjust(layout);
430       if (this.didReload) this.didReload(invalid === YES ? null : invalid);
431     }
432     
433     return this;
434   },
435   
436   /**
437     Loops through remove queue and removes.
438   */
439   processRemovals: function(oldMap) {
440     var content = this.get("content");
441     for (var guid in oldMap) {
442       var imap = oldMap[guid];
443       for (var itemIdx in imap) {
444         itemIdx = parseInt(itemIdx, 10);
445         if (isNaN(itemIdx)) continue;
446         
447         var view = imap[itemIdx];
448         
449         if (this._indexMap[itemIdx] === view) delete this._indexMap[itemIdx];
450         
451         view._isInCollection = NO;
452         this.removeItemView(view);
453       }
454     }
455   },
456   
457   /**
458     @private
459     Loops through update queue and... updates.
460   */
461   processUpdates: function(invalid) {
462     var u = this._itemsToUpdate, content = this.get("content"), item, view;
463     invalid.forEach(function(idx) {
464       item = content.objectAt(idx);
465       if (view = this.mappedViewForItem(item, idx)) {
466         if (!view._isInCollection) return;
467         var ex = this.exampleViewForItem(item, idx);
468         this.updateItemView(view, ex, item, idx);
469       }
470     }, this);
471   },
472   
473   /**
474     @private
475     Loops through add queue and, well, adds.
476   */
477   processAdds: function() {
478     var content = this.get("content");
479     
480     var add = this._itemsToAdd, idx, len = add.length, itemIdx, item;
481     for (idx = 0; idx < len; idx++) {
482       itemIdx = add[idx]; item = content.objectAt(itemIdx);
483       
484       // get example view and create item view
485       var exampleView = this.exampleViewForItem(item, itemIdx);
486       var view = this.addItemView(exampleView, item, itemIdx);
487     }
488   },

490   /**
491   @private
492     Clear all DOM pools.
493   */
494   clearDOMPools: function() {
495     var pools = this._domPools || (this._domPools = {});
496     for (var p in pools) {
497       this.clearDOMPool(pools[p]);
498     }
499   },
500   
501   domPoolSize: 10,
502   
503   /**
504   @private
505     Clears a specific DOM pool.
506   */
507   clearDOMPool: function(pool) {
508     var idx, len = pool.length, item;
509     
510     // we skip one because there is a buffer area of one while scrolling
511     for (idx = this.domPoolSize; idx < len; idx++) {
512       item = pool[idx];
513 
514       // remove from DOM
515       this.removeChild(item);
516       
517       // release the item
518       this.releaseItemView(item);
519     }
520     
521     // pool is cleared.
522     pool.length = Math.min(pool.length, this.domPoolSize);
523   },

525   /**
526   @private
527     Returns the DOM pool for the given exampleView.
528   */
529   domPoolForExampleView: function(exampleView) {
530     var pools = this._domPools || (this._domPools = {}), guid = SC.guidFor(exampleView);
531     var pool = pools[guid];
532     if (!pool) pool = pools[guid] = [];
533     return pool;
534   },

536   /**
537   @private
538     Tries to find an item for the given example view in a dom pool.
539     If one could not be found, returns null.
540   */
541   itemFromDOMPool: function(exampleView) {
542     var pool = this.domPoolForExampleView(exampleView);
543     if (pool.length < 1) return null;
544     var view = pool.shift();
545     if (view.wakeFromDOMPool) view.wakeFromDOMPool();
546     return view;
547   },
548   
549   /**
550     @private
551     Sends a view to a DOM pool.
552   */
553   sendToDOMPool: function(view) {
554     var pool = this.domPoolForExampleView(view.createdFromExampleView);
555     pool.push(view);
556     var f = view.get("frame");
557     view.adjust({ top: -f.height });
558     view.set("layerId", SC.guidFor(view));
559     if (view.sleepInDOMPool) view.sleepInDOMPool();
560   },
561   
562   /**
563   @private
564     Adds an item view (grabbing the actual item from one of the pools if possible).
565   */
566   addItemView: function(exampleView, object, index) {
567     var view, attrs = this._TMP_ATTRS || (this._TMP_ATTRS = {});
568     
569     // in any case, we need attributes
570     this.setAttributesForItem(object, index, attrs);
571     
572     // try to get from DOM pool first
573     if (view = this.itemFromDOMPool(exampleView)) {
574       // set attributes
575       this.configureItemView(view, attrs);
576       
577       // set that it is in the collection
578       view._isInCollection = YES;
579       
580       // add to view map (if not used, it will be removed)
581       this.mapView(object, index, view);
582       this._indexMap[index] = view;
583       
584       // and that should have repositioned too
585       return view;
586     }
587     
588     // otherwise, just allocate a view
589     view = this.allocateItemView(exampleView, attrs);
590     
591     // and then, add it
592     this.appendChild(view);
593     
594     // set that it is in the collection.
595     view._isInCollection = YES;
596     
597     // add to view map (if not used, it will be removed)
598     this.mapView(object, index, view);
599     this._indexMap[index] = view;
600     
601     return view;
602   },
603   
604   /**
605   @private
606     Removes an item view.
607   */
608   removeItemView: function(current) {
609     if (current.get("layerIsCacheable")) {
610       this.sendToDOMPool(current);
611     } else {
612       this.removeChild(current);
613     }
614     current._isInCollection = NO;
615   },
616   
617   /**
618     Updates the specified item view. If the view is not "layer cacheable" or the
619     example view has changed, it will be redrawn.
620     
621     Otherwise, nothing will happen.
622   */
623   updateItemView: function(current, exampleView, object, index) {
624     if (!current.get("layerIsCacheable") || current.createdFromExampleView !== exampleView) {
625       console.error("BAD UPDATE");
626       // unmap old and remove
627       this.unmapView(current, index);
628       delete this._indexMap[index];
629       this.removeItemView(current, object, index);
630       
631       // add new and map
632       var newView = this.addItemView(exampleView, object, index);
633     } else {
634       var attrs = this._TMP_ATTRS || (this._TMP_ATTRS = {});
635 
636       this.setAttributesForItem(object, index, attrs);
637       this.configureItemView(current, attrs);
638     }
639   },

642   /**
643     Tells ScrollView that this should receive live updates during touch scrolling.
644     We are so fast, aren't we?
645   */
646   _lastTopUpdate: 0,
647   _lastLeftUpdate: 0,
648   _tolerance: 100,

650   /**
651     The fast-path that computes a special 
652   */
653   touchScrollDidChange: function(left, top) {
654     var clippingFrame = this.get('clippingFrame');
655     
656     var cf = this._inScrollClippingFrame || (this._inScrollClippingFrame = {x: 0, y: 0, width: 0, height: 0});
657     cf.x = clippingFrame.x; cf.y = clippingFrame.y; cf.width = clippingFrame.width; cf.height = clippingFrame.height;
658     
659     // update
660     cf.x = left;
661     cf.y = top;
662     
663     var r = this.contentIndexesInRect(cf);
664     if (!r) return; // no rect, do nothing.
665     
666     var len = this.get('length'), 
667         max = r.get('max'), min = r.get('min');
668 
669     if (max > len || min < 0) {
670       r = r.copy();
671       r.remove(len, max-len).remove(min, 0-min).freeze();
672     }
673     
674     if (this._lastNowShowing) {
675       if (r.contains(this._lastNowShowing) && this._lastNowShowing.contains(r)) return;
676     }
677     this._lastNowShowing = r;
678     this.reloadIfNeeded(r, YES);
679   }
