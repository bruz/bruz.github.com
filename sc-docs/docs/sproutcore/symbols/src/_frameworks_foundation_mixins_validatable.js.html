  8 /**
  9   @namespace
 10 
 11   Views that include the Validatable mixin can be used with validators to 
 12   ensure their values are valid.  
 13   
 14 */
 15 SC.Validatable = {
 16   
 17   /** @private */
 18   initMixin: function() {
 19     this._validatable_validatorDidChange() ;
 20   },
 21   
 22   /**
 23     The validator for this field.  
 24   
 25     Set to a validator class or instance.  If this points to a class, it will 
 26     be instantiated when the validator is first used.
 27     
 28     @property {SC.Validator}
 29   */
 30   validator: null,
 31 
 32   /**
 33     This property must return the human readable name you want used when 
 34     describing an error condition.  For example, if set this property to
 35     "Your Email", then the returned error string might be something like
 36     "Your Email is not valid".
 37     
 38     You can return a loc string here if you like.  It will be localized when
 39     it is placed into the error string.
 40     
 41     @property {String}
 42   */
 43   errorLabel: null,
 44 
 45   /**
 46     YES if the receiver is currently valid.
 47     
 48     This property watches the value property by default.  You can override
 49     this property if you want to use some other method to calculate the
 50     current valid state.
 51     
 52     @property {Boolean}
 53   */
 54   isValid: function() { 
 55     return SC.typeOf(this.get('value')) !== SC.T_ERROR; 
 56   }.property('value'),
 57   
 58   /**
 59     The form that the view belongs to.  May be null if the view does not 
 60     belong to a form.  This property is usually set automatically by an 
 61     owner form view.
 62     
 63     @property {SC.View}
 64   */
 65   ownerForm: null,
 66   
 67   /**
 68     Attempts to validate the receiver. 
 69     
 70     Runs the validator and returns SC.VALIDATE_OK, SC.VALIDATE_NO_CHANGE,
 71     or an error object.  If no validator is installed, this method will
 72     always return SC.VALIDATE_OK.
 73 
 74     @param {Boolean} partialChange YES if this is a partial edit.
 75     @returns {String} SC.VALIDATE_OK, error, or SC.VALIDATE_NO_CHANGE
 76   */
 77   performValidate: function(partialChange) {
 78     var ret = SC.VALIDATE_OK ;
 79 
 80     if (this._validator) {
 81       var form = this.get('ownerForm') ;
 82       if (partialChange) {
 83         ret = this._validator.validatePartial(form,this) ;
 84 
 85         // if the partial returned NO_CHANGE, then check to see if the 
 86         // field is valid anyway.  If it is not valid, then don't update the
 87         // value.  This way the user can have partially constructed values 
 88         // without the validator trying to convert it to an object.
 89         if ((ret == SC.VALIDATE_NO_CHANGE) && (this._validator.validateChange(form, this) == SC.VALIDATE_OK)) {
 90           ret = SC.VALIDATE_OK; 
 91         }
 92       } else ret = this._validator.validateChange(form, this) ;
 93     }
 94     return ret ;
 95   },
 96 
 97   /**
 98     Runs validateSubmit.  You should use this in your implementation of 
 99     validateSubmit.  If no validator is installed, this always returns
100     SC.VALIDATE_OK
101     
102     @returns {String}
103   */
104   performValidateSubmit: function() {
105     return this._validator ? this._validator.validateSubmit(this.get('ownerForm'), this) : SC.VALIDATE_OK;
106   },
107   
108   /**
109     Runs a keypress validation.  Returns YES if the keypress should be 
110     allowed, NO otherwise.  If no validator is defined, always returns YES.
111     
112     @param {String} charStr the key string
113     @returns {Boolean}
114   */
115   performValidateKeyDown: function(evt) {
116     // ignore anything with ctrl or meta key press
117     var charStr = evt.getCharString();
118     if (!charStr) return YES ;
119     return this._validator ? this._validator.validateKeyDown(this.get('ownerForm'), this, charStr) : YES;
120   },
121   
122   /**
123     Returns the validator object, if one has been created.
124     
125     @property {SC.Validator}
126   */
127   validatorObject: function() {
128     return this._validator;
129   }.property(),
130   
131   /**
132     Invoked by the owner form just before submission.  Override with your 
133     own method to commit any final changes after you perform validation. 
134     
135     The default implementation simply calls performValidateSubmit() and 
136     returns that value.
137     
138     @property {Boolean}
139   */
140   validateSubmit: function() { return this.performValidateSubmit(); },
141   
142   /**
143     Convert the field value string into an object.
144     
145     This method will call the validators objectForFieldValue if it exists.
146     
147     @param {Object} fieldValue the raw value from the field.
148     @param {Boolean} partialChange
149     @returns {Object}
150   */
151   objectForFieldValue: function(fieldValue, partialChange) {
152     return this._validator ? this._validator.objectForFieldValue(fieldValue, this.get('ownerForm'), this) : fieldValue ;
153   },
154   
155   /**
156     Convert the object into a field value.
157     
158     This method will call the validator's fieldValueForObject if it exists.
159     
160     @param object {Object} the objec to convert
161     @returns {Object}
162   */
163   fieldValueForObject: function(object) {
164     return this._validator ? this._validator.fieldValueForObject(object, this.get('ownerForm'), this) : object ;
165   },
166   
167   _validatable_displayObserver: function() {
168     this.displayDidChange();
169   }.observes('isValid'),
170 
171   /** @private */
172   renderMixin: function(context) {
173     context.setClass('invalid', !this.get('isValid'));
174   },
175 
176   // invoked whenever the attached validator changes.
177   _validatable_validatorDidChange: function() {
178     var form = this.get('ownerForm') ;
179     var val = SC.Validator.findFor(form, this, this.get('validator')) ;
180     if (val != this._validator) {
181       this.propertyWillChange('validatorObject');
182       if (this._validator) this._validator.detachFrom(form, this) ;
183       this._validator = val;
184       if (this._validator) this._validator.attachTo(form, this) ;
185       this.propertyDidChange('validatorObject');
186     }  
187   }.observes('validator', 'ownerForm')
188       
189 };
