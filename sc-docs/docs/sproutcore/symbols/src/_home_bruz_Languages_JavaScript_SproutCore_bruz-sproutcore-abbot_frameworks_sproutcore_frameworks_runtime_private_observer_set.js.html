 12 /**
 13   @namespace
 14 
 15   This private class is used to store information about obversers on a 
 16   particular key.  Note that this object is not observable.  You create new
 17   instances by calling SC.beget(SC.ObserverSet) ;
 18 
 19   @since SproutCore 1.0
 20 */
 21 SC.ObserverSet = {
 22 
 23   /**
 24     the number of targets in the set.
 25   */
 26   targets: 0,
 27   
 28   _membersCacheIsValid: NO,
 29   
 30   /**
 31     Adds the named target/method observer to the set.  The method must be
 32     a function, not a string.
 33     
 34     Note that in debugging mode only, this method is overridden to also record
 35     the name of the object and function that resulted in the target/method
 36     being added.
 37   */
 38   add: function(target, method, context) {
 39     var targetGuid = (target) ? SC.guidFor(target) : "__this__";
 40     
 41     // get the set of methods
 42     var methods = this[targetGuid] ;
 43     if (!methods) {
 44       methods = this[targetGuid] = SC.CoreSet.create() ;
 45       methods.target = target ;
 46       methods.isTargetSet = YES ; // used for getMembers().
 47       this.targets++ ;
 48     }
 49     methods.add(method) ;
 50     
 51     // context is really useful sometimes but not used that often so this
 52     // implementation is intentionally lazy.
 53     if (context !== undefined) {
 54       var contexts = methods.contexts ;
 55       if (!contexts) context = methods.contexts = {} ;
 56       contexts[SC.guidFor(method)] = context ;
 57     }
 58     
 59     this._membersCacheIsValid = NO ;
 60   },
 61   
 62   /**
 63     removes the named target/method observer from the set.  If this is the
 64     last method for the named target, then the number of targets will also
 65     be reduced.
 66   
 67     returns YES if the items was removed, NO if it was not found.
 68   */
 69   remove: function(target, method) {
 70     var targetGuid = (target) ? SC.guidFor(target) : "__this__";
 71     
 72     // get the set of methods
 73     var methods = this[targetGuid] ;    
 74     if (!methods) return NO ;
 75     
 76     methods.remove(method) ;
 77     if (methods.length <= 0) {
 78       methods.target = null;
 79       methods.isTargetSet = NO ;
 80       methods.contexts = null ;
 81       delete this[targetGuid] ;
 82       this.targets-- ;
 83       
 84     } else if (methods.contexts) {
 85       delete methods.contexts[SC.guidFor(method)];
 86     }
 87 
 88     this._membersCacheIsValid = NO;
 89     
 90     return YES ;
 91   },
 92   
 93   /**
 94     Invokes the target/method pairs in the receiver.  Used by SC.RunLoop
 95     Note: does not support context
 96   */
 97   invokeMethods: function() {
 98     var key, value, idx, target, val;
 99     
100     // iterate through the set, look for sets.
101     for(key in this) {
102       if (!this.hasOwnProperty(key)) continue ;
103       value = this[key] ;
104       if (value && value.isTargetSet) {
105         idx = value.length;
106         target = value.target ;
107         while(--idx>=0) {
108           val = value[idx];
109           if(val) val.call(target);
110         }
111       }
112     }
113   },
114   
115   /**
116     Returns an array of target/method pairs.  This is cached.
117   */
118   getMembers: function() {
119     if (this._membersCacheIsValid) return this._members ;
120     
121     // need to recache, reset the array...
122     if (!this._members) {
123       this._members = [] ;
124     } else this._members.length = 0 ; // reset
125     var ret = this._members ;
126 
127     // iterate through the set, look for sets.
128     for(var key in this) {
129       if (!this.hasOwnProperty(key)) continue ;
130       var value = this[key] ;
131       if (value && value.isTargetSet) {
132         var idx = value.length;
133         var target = value.target ;
134         
135         // slightly slower - only do if we have contexts
136         var contexts = value.contexts ;
137         if (contexts) {
138           while(--idx>=0) {
139             var method = value[idx] ;
140             ret.push([target, method, contexts[SC.guidFor(method)]]) ;
141           }
142         } else {
143           while(--idx>=0) ret.push([target, value[idx]]);
144         }
145       }
146     }
147 
148     this._membersCacheIsValid = YES ;
149     return ret ;
150   },
151   
152   /**
153     Returns a new instance of the set with the contents cloned.
154   */
155   clone: function() {
156     var oldSet, newSet, key, ret = SC.ObserverSet.create() ;
157     for(key in this) {
158       if (!this.hasOwnProperty(key)) continue ;
159       oldSet = this[key];
160       if (oldSet && oldSet.isTargetSet) {
161         newSet = oldSet.clone();
162         newSet.target = oldSet.target ;
163         if (oldSet.contexts) newSet.contexts = SC.clone(oldSet.contexts);
164         ret[key] = newSet ;
165       }
166     }
167     ret.targets = this.targets ;
168     ret._membersCacheIsValid = NO ;
169     return ret ;
170   },
171   
172   /**
173     Creates a new instance of the observer set.
174   */
175   create: function() { return SC.beget(this); }
176   
177 } ;
178 
179 SC.ObserverSet.slice = SC.ObserverSet.clone ;
