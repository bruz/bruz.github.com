  9 /** @class
 10 
 11   A RangeObserver is used by Arrays to automatically observe all of the
 12   objects in a particular range on the array.  Whenever any property on one 
 13   of those objects changes, it will notify its delegate.  Likewise, whenever
 14   the contents of the array itself changes, it will notify its delegate and
 15   possibly update its own registration.
 16 
 17   This implementation uses only SC.Array methods.  It can be used on any 
 18   object that complies with SC.Array.  You may, however, choose to subclass
 19   this object in a way that is more optimized for your particular design.
 20   
 21   @since SproutCore 1.0
 22 */
 23 SC.RangeObserver = {
 24 
 25   /** 
 26     Walk like a duck.
 27     
 28     @property {Boolean}
 29   */
 30   isRangeObserver: YES,
 31   
 32   /** @private */
 33   toString: function() { 
 34     var base = this.indexes ? this.indexes.toString() : "SC.IndexSet<..>";
 35     return base.replace('IndexSet', 'RangeObserver(%@)'.fmt(SC.guidFor(this)));
 36   },
 37   
 38   /**
 39     Creates a new range observer owned by the source.  The indexSet you pass
 40     must identify the indexes you are interested in observing.  The passed
 41     target/method will be invoked whenever the observed range changes.
 42     
 43     Note that changes to a range are buffered until the end of a run loop
 44     unless a property on the record itself changes.
 45   
 46     @param {SC.Array} source the source array
 47     @param {SC.IndexSet} indexSet set of indexes to observer
 48     @param {Object} target the target
 49     @param {Function|String} method the method to invoke
 50     @param {Object} context optional context to include in callback
 51     @param {Boolean} isDeep if YES, observe property changes as well
 52     @returns {SC.RangeObserver} instance
 53   */
 54   create: function(source, indexSet, target, method, context, isDeep) {
 55     var ret = SC.beget(this);
 56     ret.source = source;
 57     ret.indexes = indexSet ? indexSet.frozenCopy() : null;
 58     ret.target = target;
 59     ret.method = method;
 60     ret.context = context ;
 61     ret.isDeep  = isDeep || NO ;
 62     ret.beginObserving();
 63     return ret ;
 64   },
 65 
 66   /**
 67     Create subclasses for the RangeObserver.  Pass one or more attribute
 68     hashes.  Use this to create customized RangeObservers if needed for your 
 69     classes.
 70     
 71     @param {Hash} attrs one or more attribute hashes
 72     @returns {SC.RangeObserver} extended range observer class
 73   */
 74   extend: function(attrs) {
 75     var ret = SC.beget(this), args = arguments, len = args.length, idx;
 76     for(idx=0;idx<len;idx++) SC.mixin(ret, args[idx]);
 77     return ret ;
 78   },
 79 
 80   /**
 81     Destroys an active ranger observer, cleaning up first.
 82     
 83     @param {SC.Array} source the source array
 84     @returns {SC.RangeObserver} receiver
 85   */
 86   destroy: function(source) { 
 87     this.endObserving(); 
 88     return this; 
 89   },
 90 
 91   /**
 92     Updates the set of indexes the range observer applies to.  This will 
 93     stop observing the old objects for changes and start observing the 
 94     new objects instead.
 95     
 96     @param {SC.Array} source the source array
 97     @returns {SC.RangeObserver} receiver
 98   */
 99   update: function(source, indexSet) {
100     if (this.indexes && this.indexes.isEqual(indexSet)) return this ;
101     
102     this.indexes = indexSet ? indexSet.frozenCopy() : null ;
103     this.endObserving().beginObserving();
104     return this;
105   },
106   
107   /**
108     Configures observing for each item in the current range.  Should update
109     the observing array with the list of observed objects so they can be
110     torn down later
111     
112     @returns {SC.RangeObserver} receiver
113   */
114   beginObserving: function() {
115     if (!this.isDeep) return this; // nothing to do
116     
117     var observing = this.observing;
118     if (!observing) observing = this.observing = SC.CoreSet.create();
119     
120     // cache iterator function to keep things fast
121     var func = this._beginObservingForEach;
122     if (!func) {
123       func = this._beginObservingForEach = function(idx) {
124         var obj = this.source.objectAt(idx);
125         if (obj && obj.addObserver) {
126           observing.push(obj);
127           obj._kvo_needsRangeObserver = YES ;
128         }
129       };
130     }
131     this.indexes.forEach(func,this);
132 
133     // add to pending range observers queue so that if any of these objects
134     // change we will have a chance to setup observing on them.
135     this.isObserving = NO ;
136     SC.Observers.addPendingRangeObserver(this);
137 
138     return this;
139   },
140   
141   /** @private
142     Called when an object that appears to need range observers has changed.
143     Check to see if the range observer contains this object in its list.  If
144     it does, go ahead and setup observers on all objects and remove ourself
145     from the queue.
146   */
147   setupPending: function(object) {
148     var observing = this.observing ;
149 
150     if (this.isObserving || !observing || (observing.get('length')===0)) {
151       return YES ;
152     } 
153     
154     if (observing.contains(object)) {
155       this.isObserving = YES ;
156 
157       // cache iterator function to keep things fast
158       var func = this._setupPendingForEach;
159       if (!func) {
160         var source = this.source,
161             method = this.objectPropertyDidChange;
162 
163         func = this._setupPendingForEach = function(idx) {
164           var obj = this.source.objectAt(idx),
165               guid = SC.guidFor(obj),
166               key ;
167               
168           if (obj && obj.addObserver) {
169             observing.push(obj);
170             obj.addObserver('*', this, method);
171             
172             // also save idx of object on range observer itself.  If there is
173             // more than one idx, convert to IndexSet.
174             key = this[guid];
175             if (key === undefined || key === null) {
176               this[guid] = idx ;
177             } else if (key.isIndexSet) {
178               key.add(idx);
179             } else {
180               key = this[guid] = SC.IndexSet.create(key).add(idx);
181             }
182             
183           }
184         };
185       }
186       this.indexes.forEach(func,this);
187       return YES ;
188       
189     } else return NO ;
190   },
191   
192   /**
193     Remove observers for any objects currently begin observed.  This is 
194     called whenever the observed range changes due to an array change or 
195     due to destroying the observer.
196     
197     @returns {SC.RangeObserver} receiver
198   */
199   endObserving: function() {
200     if (!this.isDeep) return this; // nothing to do
201     
202     var observing = this.observing;
203     
204     if (this.isObserving) {
205       var meth      = this.objectPropertyDidChange,
206           source    = this.source,
207           idx, lim, obj;
208 
209       if (observing) {
210         lim = observing.length;
211         for(idx=0;idx<lim;idx++) {
212           obj = observing[idx];
213           obj.removeObserver('*', this, meth);
214           this[SC.guidFor(obj)] = null;
215         }
216         observing.length = 0 ; // reset
217       } 
218       
219       this.isObserving = NO ;
220     }
221     
222     if (observing) observing.clear(); // empty set.
223     return this ;
224   },
225   
226   /**
227     Whenever the actual objects in the range changes, notify the delegate
228     then begin observing again.  Usually this method will be passed an 
229     IndexSet with the changed indexes.  The range observer will only notify
230     its delegate if the changed indexes include some of all of the indexes
231     this range observer is monitoring.
232     
233     @param {SC.IndexSet} changes optional set of changed indexes
234     @returns {SC.RangeObserver} receiver
235   */
236   rangeDidChange: function(changes) {
237     var indexes = this.indexes;
238     if (!changes || !indexes || indexes.intersects(changes)) {
239       this.endObserving(); // remove old observers
240       this.method.call(this.target, this.source, null, '[]', changes, this.context);
241       this.beginObserving(); // setup new ones
242     }
243     return this ;
244   },
245 
246   /**
247     Whenever an object changes, notify the delegate
248     
249     @param {Object} the object that changed
250     @param {String} key the property that changed
251     @returns {SC.RangeObserver} receiver
252   */
253   objectPropertyDidChange: function(object, key, value, rev) {
254     var context = this.context,
255         method  = this.method, 
256         guid    = SC.guidFor(object),
257         index   = this[guid];
258       
259     // lazily convert index to IndexSet.  
260     if (index && !index.isIndexSet) {
261       index = this[guid] = SC.IndexSet.create(index).freeze();
262     }
263     
264     if (context) {
265       method.call(this.target, this.source, object, key, index, context, rev);
266     } else {
267       method.call(this.target, this.source, object, key, index, rev);
268     }
269   }
270   
271 };
