  7 /**
  8   @class
  9 
 10   A ChildArray is used to map an array of ChildRecord
 11   
 12   @extends SC.Enumerable
 13   @extends SC.Array
 14   @since SproutCore 1.0
 15 */
 16 
 17 SC.ChildArray = SC.Object.extend(SC.Enumerable, SC.Array,
 18   /** @scope SC.ManyArray.prototype */ {
 19     
 20   /**
 21     If set, it is the default record recordType
 22   
 23     @property {SC.Record}
 24   */
 25   defaultRecordType: null,
 26   
 27   /**
 28     If set, the parent record will be notified whenever the array changes so that 
 29     it can change its own state
 30     
 31     @property {SC.Record}
 32   */
 33   record: null,
 34   
 35   /**
 36     If set will be used by the many array to get an editable version of the
 37     storeIds from the owner.
 38     
 39     @property {String}
 40   */
 41   propertyName: null,
 42   
 43   /**
 44     Actual references to the hashes
 45   */
 46   children: null,
 47   
 48   /**
 49     The store that owns this record array.  All record arrays must have a 
 50     store to function properly.
 51 
 52     @property {SC.Store}
 53   */
 54   store: function() {
 55     return this.get('record').get('store');
 56   }.property('record').cacheable(),
 57   
 58   /**
 59     The storeKey for the parent record of this many array.  Editing this 
 60     array will place the parent record into a READY_DIRTY state.
 61 
 62     @property {Number}
 63   */
 64   storeKey: function() {
 65     return this.get('record').get('storeKey');
 66   }.property('record').cacheable(),
 67   
 68   /**
 69     Returns the storeIds in read only mode.  Avoids modifying the record 
 70     unnecessarily.
 71     
 72     @property {SC.Array}
 73   */
 74   readOnlyChildren: function() {
 75     return this.get('record').readAttribute(this.get('propertyName'));
 76   }.property(),
 77   
 78   /**
 79     Returns an editable array of child hashes.  Marks the owner records as 
 80     modified. 
 81     
 82     @property {SC.Array}
 83   */
 84   editableChildren: function() {
 85     var store    = this.get('store'),
 86         storeKey = this.get('storeKey'),
 87         pname    = this.get('propertyName'),
 88         ret, hash;
 89         
 90     ret = store.readEditableProperty(storeKey, pname);    
 91     if (!ret) {
 92       hash = store.readEditableDataHash(storeKey);
 93       ret = hash[pname] = [];      
 94     }
 95     
 96     if (ret !== this._prevChildren) this.recordPropertyDidChange();
 97     return ret ;
 98   }.property(),
 99     
100   // ..........................................................
101   // ARRAY PRIMITIVES
102   // 
103 
104   /** @private
105     Returned length is a pass-through to the storeIds array.
106     
107     @property {Number}
108   */
109   length: function() {
110     var children = this.get('readOnlyChildren');
111     return children ? children.length : 0;
112   }.property('readOnlyChildren'),
113 
114   /** @private
115     Looks up the store id in the store ids array and materializes a
116     records.
117   */
118   objectAt: function(idx) {
119     var recs      = this._records, 
120         children = this.get('readOnlyChildren'),
121         hash, ret;
122     var len = children ? children.length : 0;
123     
124     if (!children) return undefined; // nothing to do
125     if (recs && (ret=recs[idx])) return ret ; // cached
126     if (!recs) this._records = recs = [] ; // create cache
127     
128     // If not a good index return undefined
129     if (idx >= len) return undefined;
130     hash = children.objectAt(idx);
131     if (!hash) return undefined;
132     
133     // not in cache, materialize
134     recs[idx] = ret = this._materializeChild(hash);
135     
136     return ret;
137   },
138 
139   /** @private
140     Pass through to the underlying array.  The passed in objects must be
141     records, which can be converted to storeIds.
142   */
143   replace: function(idx, amt, recs) {
144     var children = this.get('editableChildren'), 
145         len      = recs ? (recs.get ? recs.get('length') : recs.length) : 0,
146         record   = this.get('record'),
147         
148         pname    = this.get('propertyName'),
149         cr, recordType;  
150     children.replace(idx, amt, recs);
151     for(var i = idx; i <= idx+amt; i+=1){
152       this.objectAt(i);
153     }
154     // notify that the record did change...
155     record.recordDidChange(pname);
156     
157     return this;
158   },
159   
160   /*
161   calls normalize on each object in the array
162   */
163   normalize: function(){
164     this.forEach(function(child,id){
165       if(child.normalize) child.normalize();
166     });
167   },
168   
169   // ..........................................................
170   // INTERNAL SUPPORT
171   //  
172   
173   /** @private
174     Call to create an object from a hash
175   */
176   _materializeChild: function(hash){
177     var store = this.get('store'),
178         parentRecord = this.get('record'), 
179         recordType = this.get('defaultRecordType'),
180         id, ret, storeKey, pm;
181         
182     // Find the record type
183     if (!parentRecord) return undefined;
184     var nspace = parentRecord.get('childRecordNamespace');
185     // Get the record type.
186     if (hash.type && !SC.none(nspace)) {
187       recordType = nspace[hash.type];
188     }
189 
190     if (!recordType || SC.typeOf(recordType) !== SC.T_CLASS) {
191       throw 'ChildrenArray: Error during transform: Invalid record type.';
192     }
193     
194     pm = recordType.prototype.primaryKey || 'childRecordKey';
195     id = hash[pm];
196     storeKey = store.storeKeyExists(recordType, id);
197     if (storeKey){
198       ret = store.materializeRecord(storeKey);
199     } 
200     else {
201       ret = parentRecord.registerChildRecord(recordType, hash);
202     }
203     return ret;
204   },
205 
206   /** @private 
207     Invoked whenever the children array changes.  Observes changes.
208   */
209   recordPropertyDidChange: function(keys) {
210     
211     if (keys && !keys.contains(this.get('propertyName'))) return this;
212     
213     var children = this.get('readOnlyChildren');
214     var prev = this._prevChildren, f = this._childrenContentDidChange;
215     
216     if (children === prev) return this; // nothing to do
217     
218     if (prev) prev.removeObserver('[]', this, f);
219     this._prevChildren = children;
220     if (children) children.addObserver('[]', this, f);
221     
222     var rev = (children) ? children.propertyRevision : -1 ;
223     this._childrenContentDidChange(children, '[]', children, rev);
224   },
225 
226   /** @private
227     Invoked whenever the content of the children array changes.  This will
228     dump any cached record lookup and then notify that the enumerable content
229     has changed.
230   */
231   _childrenContentDidChange: function(target, key, value, rev) {
232     this._records = null ; // clear cache
233     this.enumerableContentDidChange();
234   },
235   
236   /** @private */
237   init: function() {
238     sc_super();
239     this.recordPropertyDidChange();
240   }
241   
242 }) ;
