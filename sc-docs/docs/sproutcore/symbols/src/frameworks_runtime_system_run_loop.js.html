 10 /**
 11   @class
 12   
 13   The run loop provides a universal system for coordinating events within
 14   your application.  The run loop processes timers as well as pending 
 15   observer notifications within your application.
 16   
 17   To use a RunLoop within your application, you should make sure your event
 18   handlers always begin and end with SC.RunLoop.begin() and SC.RunLoop.end()
 19   
 20   The RunLoop is important because bindings do not fire until the end of 
 21   your run loop is reached.  This improves the performance of your
 22   application.
 23   
 24   h2. Example
 25   
 26   This is how you could write your mouseup handler in jQuery:
 27   
 28   {{{
 29     $('#okButton').on('click', function() {
 30       SC.RunLoop.begin();
 31       
 32       // handle click event...
 33       
 34       SC.RunLoop.end(); // allows bindings to trigger...
 35     });
 36   }}}
 37   
 38   @extends SC.Object
 39   @since SproutCore 1.0
 40 */
 41 SC.RunLoop = SC.Object.extend(/** @scope SC.RunLoop.prototype */ {
 42   
 43   /**
 44     Call this method whenver you begin executing code.  
 45     
 46     This is typically invoked automatically for you from event handlers and 
 47     the timeout handler.  If you call setTimeout() or setInterval() yourself, 
 48     you may need to invoke this yourself.
 49     
 50     @returns {SC.RunLoop} receiver
 51   */
 52   beginRunLoop: function() {
 53     this._start = new Date().getTime() ; // can't use Date.now() in runtime
 54     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 55       console.log("-- SC.RunLoop.beginRunLoop at %@".fmt(this._start));
 56     } 
 57     this._runLoopInProgress = YES;
 58     return this ; 
 59   },
 60   
 61   /**
 62     YES when a run loop is in progress
 63   
 64     @property {Boolean}
 65   */
 66   isRunLoopInProgress: function() {
 67     return this._runLoopInProgress;
 68   }.property(),
 69   
 70   /**
 71     Call this method whenever you are done executing code.
 72     
 73     This is typically invoked automatically for you from event handlers and
 74     the timeout handler.  If you call setTimeout() or setInterval() yourself
 75     you may need to invoke this yourself.
 76     
 77     @returns {SC.RunLoop} receiver
 78   */
 79   endRunLoop: function() {
 80     // at the end of a runloop, flush all the delayed actions we may have 
 81     // stored up.  Note that if any of these queues actually run, we will 
 82     // step through all of them again.  This way any changes get flushed
 83     // out completely.
 84     var didChange ;
 85 
 86     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 87       console.log("-- SC.RunLoop.endRunLoop ~ flushing application queues");
 88     } 
 89     
 90     do {
 91       didChange = this.flushApplicationQueues() ;
 92       if (!didChange) didChange = this._flushinvokeLastQueue() ; 
 93     } while(didChange) ;
 94     this._start = null ;
 95 
 96     if (SC.LOG_BINDINGS || SC.LOG_OBSERVERS) {
 97       console.log("-- SC.RunLoop.endRunLoop ~ End");
 98     } 
 99     
100     SC.RunLoop.lastRunLoopEnd = Date.now();
101     this._runLoopInProgress = NO;
102     
103     return this ; 
104   },
105   
106   /**
107     Invokes the passed target/method pair once at the end of the runloop.
108     You can call this method as many times as you like and the method will
109     only be invoked once.  
110     
111     Usually you will not call this method directly but use invokeOnce() 
112     defined on SC.Object.
113     
114     Note that in development mode only, the object and method that call this
115     method will be recorded, for help in debugging scheduled code.
116     
117     @param {Object} target
118     @param {Function} method
119     @returns {SC.RunLoop} receiver
120   */
121   invokeOnce: function(target, method) {
122     // normalize
123     if (method === undefined) { 
124       method = target; target = this ;
125     }
126     
127     if (typeof method === "string") method = target[method];
128     if (!this._invokeQueue) this._invokeQueue = SC.ObserverSet.create();
129     this._invokeQueue.add(target, method);
130     return this ;
131   },
132   
133   /**
134     Invokes the passed target/method pair at the very end of the run loop,
135     once all other delayed invoke queues have been flushed.  Use this to 
136     schedule cleanup methods at the end of the run loop once all other work
137     (including rendering) has finished.
138 
139     If you call this with the same target/method pair multiple times it will
140     only invoke the pair only once at the end of the runloop.
141     
142     Usually you will not call this method directly but use invokeLast() 
143     defined on SC.Object.
144     
145     Note that in development mode only, the object and method that call this
146     method will be recorded, for help in debugging scheduled code.
147     
148     @param {Object} target
149     @param {Function} method
150     @returns {SC.RunLoop} receiver
151   */
152   invokeLast: function(target, method) {
153     // normalize
154     if (method === undefined) { 
155       method = target; target = this ;
156     }
157     
158     if (typeof method === "string") method = target[method];
159     if (!this._invokeLastQueue) this._invokeLastQueue = SC.ObserverSet.create();
160     this._invokeLastQueue.add(target, method);
161     return this ;
162   },
163   
164   /**
165     Executes any pending events at the end of the run loop.  This method is 
166     called automatically at the end of a run loop to flush any pending 
167     queue changes.
168     
169     The default method will invoke any one time methods and then sync any 
170     bindings that might have changed.  You can override this method in a 
171     subclass if you like to handle additional cleanup. 
172     
173     This method must return YES if it found any items pending in its queues
174     to take action on.  endRunLoop will invoke this method repeatedly until
175     the method returns NO.  This way if any if your final executing code
176     causes additional queues to trigger, then can be flushed again.
177     
178     @returns {Boolean} YES if items were found in any queue, NO otherwise
179   */
180   flushApplicationQueues: function() {
181     var hadContent = NO,
182         // execute any methods in the invokeQueue.
183         queue = this._invokeQueue;
184     if (queue && queue.targets > 0) {
185       this._invokeQueue = null; // reset so that a new queue will be created
186       hadContent = YES ; // needs to execute again
187       queue.invokeMethods();
188     }
189     
190     // flush any pending changed bindings.  This could actually trigger a 
191     // lot of code to execute.
192     return SC.Binding.flushPendingChanges() || hadContent ;
193   },
194   
195   _flushinvokeLastQueue: function() {
196     var queue = this._invokeLastQueue, hadContent = NO ;
197     if (queue && queue.targets > 0) {
198       this._invokeLastQueue = null; // reset queue.
199       hadContent = YES; // has targets!
200       if (hadContent) queue.invokeMethods();
201     }
202     return hadContent ;
203   }
204   
205 });
206 
207 /** 
208   The current run loop.  This is created automatically the first time you
209   call begin(). 
210   
211   @property {SC.RunLoop}
212 */
213 SC.RunLoop.currentRunLoop = null;
214 
215 /**
216   The default RunLoop class.  If you choose to extend the RunLoop, you can
217   set this property to make sure your class is used instead.
218   
219   @property {Class}
220 */
221 SC.RunLoop.runLoopClass = SC.RunLoop;
222 
223 /** 
224   Begins a new run loop on the currentRunLoop.  If you are already in a 
225   runloop, this method has no effect.
226   
227   @returns {SC.RunLoop} receiver
228 */
229 SC.RunLoop.begin = function() {    
230   var runLoop = this.currentRunLoop;
231   if (!runLoop) runLoop = this.currentRunLoop = this.runLoopClass.create();
232   runLoop.beginRunLoop();
233   return this ;
234 };
235 
236 /**
237   Ends the run loop on the currentRunLoop.  This will deliver any final 
238   pending notifications and schedule any additional necessary cleanup.
239   
240   @returns {SC.RunLoop} receiver
241 */
242 SC.RunLoop.end = function() {
243   var runLoop = this.currentRunLoop;
244   if (!runLoop) {
245     throw "SC.RunLoop.end() called outside of a runloop!";
246   }
247   runLoop.endRunLoop();
248   return this ;
249 } ;
250 
251 /**
252   Returns YES when a run loop is in progress
253 
254   @return {Boolean}
255 */
256 SC.RunLoop.isRunLoopInProgress = function() {
257   if(this.currentRunLoop) return this.currentRunLoop.get('isRunLoopInProgress');
258   return NO;
259 };
260 
261 /**
262   Helper method executes the passed function inside of a runloop.  Normally
263   not needed but useful for testing.
264   
265   @param {Function} callback callback to execute
266   @param {Object} target context for callback
267   @param {Boolean} if YES, does not start/end a new runloop if one is already running
268   @returns {SC} receiver
269 */
270 SC.run = function(callback, target, useExistingRunLoop) {
271   if(useExistingRunLoop) {
272     var alreadyRunning = SC.RunLoop.isRunLoopInProgress();
273     if(!alreadyRunning) SC.RunLoop.begin();
274     callback.call(target);
275     if(!alreadyRunning) SC.RunLoop.end();
276   } else {
277     SC.RunLoop.begin();
278     callback.call(target);
279     SC.RunLoop.end();
280   }
281 };
