314   /**
315     override this method to implement your own sorting of the menu. By
316     default, menu items are sorted using the value shown or the sortKey
317 
318     @param{SC.Array} objects the unsorted array of objects to display.
319     @returns sorted array of objects
320   */
321   sortObjects: function(objects) {
322     if(!this.get('disableSort')){
323       var nameKey = this.get('sortKey') || this.get('nameKey') ;
324       objects = objects.sort(function(a,b) {
325         if (nameKey) {
326           a = a.get ? a.get(nameKey) : a[nameKey] ;
327           b = b.get ? b.get(nameKey) : b[nameKey] ;
328         }
329         return (a<b) ? -1 : ((a>b) ? 1 : 0) ;
330       }) ;
331     }
332     return objects ;
333   },

340   render: function(context,firstTime) {
341     sc_super();
342     var layoutWidth, objects, len, nameKey, iconKey, valueKey, checkboxEnabled,
343       currentSelectedVal, shouldLocalize, separatorPostion, itemList, isChecked,
344       idx, name, icon, value, item, itemEnabled, isEnabledKey ;
345     layoutWidth = this.layout.width ;
346     if(firstTime && layoutWidth) {
347       this.adjust({ width: layoutWidth - this.SELECT_BUTTON_SPRITE_WIDTH }) ;
348     }
349 
350     objects = this.get('objects') ;
351     objects = this.sortObjects(objects) ;
352     len = objects.length ;
353 
354     //Get the namekey, iconKey and valueKey set by the user
355     nameKey = this.get('nameKey') ;
356     iconKey = this.get('iconKey') ;
357     valueKey = this.get('valueKey') ;
358     isEnabledKey = this.get('isEnabledKey') ;
359     checkboxEnabled = this.get('checkboxEnabled') ;
360 
361     //get the current selected value
362     currentSelectedVal = this.get('value') ;
363 
364     // get the localization flag.
365     shouldLocalize = this.get('localize') ;
366 
367     //get the separatorPostion
368     separatorPostion = this.get('separatorPostion') ;
369 
370     //itemList array to set the menu items
371     itemList = [] ;
372 
373     //to set the 'checkbox' property of menu items
374     isChecked = YES ;
375 
376     //index for finding the first item in the list
377     idx = 0 ;
378 
379     objects.forEach(function(object) {
380     if (object) {
381 
382       //Get the name value. If value key is not specified convert obj
383       //to string
384       name = nameKey ? (object.get ?
385         object.get(nameKey) : object[nameKey]) : object.toString() ;
386 
387       // localize name if specified.
388       name = shouldLocalize? name.loc() : name ;
389 
390       //Get the icon value
391       icon = iconKey ? (object.get ?
392         object.get(iconKey) : object[iconKey]) : null ;
393       if (SC.none(object[iconKey])) icon = null ;
394 
395       // get the value using the valueKey or the object
396         value = (valueKey) ? (object.get ?
397         object.get(valueKey) : object[valueKey]) : object ;
398 
399       if (!SC.none(currentSelectedVal) && !SC.none(value)){
400         if( currentSelectedVal === value ) {
401           this.set('title', name) ;
402           this.set('icon', icon) ;
403         }
404       }
405 
406       //Check if the item is currentSelectedItem or not
407       if(value === this.get('value')) {
408 
409         //set the itemIdx - To change the prefMatrix accordingly.
410         this.set('itemIdx', idx) ;
411         isChecked = !checkboxEnabled ? NO : YES ;
412       }
413       else {
414         isChecked = NO ;
415       }
416 
417       //Check if item is enabled
418       itemEnabled = (isEnabledKey) ? (object.get ?
419       object.get(isEnabledKey) : object[isEnabledKey]) : object ;
420       
421       if(NO !== itemEnabled) itemEnabled = YES ;
422 
423       //Set the first item from the list as default selected item
424       if (idx === 0) {
425         this._defaultVal = value ;
426         this._defaultTitle = name ;
427         this._defaultIcon = icon ;
428       }
429 
430       var item = SC.Object.create({
431         title: name,
432         icon: icon,
433         value: value,
434         isEnabled: itemEnabled,
435         checkbox: isChecked,
436         target: this,
437         action: 'displaySelectedItem'
438       }) ;
439 
440       //Set the items in the itemList array
441       itemList.push(item);
442     }
443 
444     idx += 1 ;
445 
446     // display the separator if specified by the user
447     if (separatorPostion && idx === (len-separatorPostion)) {
448       var separator = SC.Object.create({
449         separator: YES
450       }) ;
451       itemList.push(separator);
452     }
453 
454     this.set('itemList', itemList) ;
455     }, this ) ;
456 
457     if(firstTime) {
458       this.invokeLast(function() {
459         var value = this.get('value') ;
460         if(SC.none(value)) {
461           this.set('value', this._defaultVal) ;
462           this.set('title', this._defaultTitle) ;
463           this.set('icon', this._defaultIcon) ;
464         }
465       });
466     }
467 
468     //Set the preference matrix for the menu pane
469     this.changeSelectButtonPreferMatrix(this.itemIdx) ;

473   /**
474     Button action handler
475 
476     @private
477     @param {DOMMouseEvent} evt mouseup event that triggered the action
478   */
479   _action: function( evt )
480   {
481     var buttonLabel, menuWidth, scrollWidth, lastMenuWidth, offsetWidth,
482       items, elementOffsetWidth, largestMenuWidth, item, element, idx,
483       value, itemList, menuControlSize, menuHeightPadding, customView,
484       customMenuView, menu, itemsLength, dummyMenuItemView, menuItemViewEscapeHTML;
485 
486     buttonLabel = this.$('.sc-button-label')[0] ;
487 
488     var menuWidthOffset = SC.SelectButtonView.MENU_WIDTH_OFFSET ;
489     if(!this.get('isDefaultPosition')) {
490       switch (this.get('controlSize')) {
491         case SC.TINY_CONTROL_SIZE:
492           menuWidthOffset += SC.SelectButtonView.TINY_POPUP_MENU_WIDTH_OFFSET;
493           break;
494         case SC.SMALL_CONTROL_SIZE:
495           menuWidthOffset += SC.SelectButtonView.SMALL_POPUP_MENU_WIDTH_OFFSET;
496           break;
497         case SC.REGULAR_CONTROL_SIZE:
498           menuWidthOffset += SC.SelectButtonView.REGULAR_POPUP_MENU_WIDTH_OFFSET;
499           break;
500         case SC.LARGE_CONTROL_SIZE:
501           menuWidthOffset += SC.SelectButtonView.LARGE_POPUP_MENU_WIDTH_OFFSET;
502           break;
503         case SC.HUGE_CONTROL_SIZE:
504           menuWidthOffset += SC.SelectButtonView.HUGE_POPUP_MENU_WIDTH_OFFSET;
505           break;
506       }
507     }
508     // Get the length of the text on the button in pixels
509     menuWidth = this.get('layer').offsetWidth + menuWidthOffset ;
510     scrollWidth = buttonLabel.scrollWidth ;
511     lastMenuWidth = this.get('lastMenuWidth') ;
512     if(scrollWidth) {
513        // Get the original width of the label in the button
514        offsetWidth = buttonLabel.offsetWidth ;
515        if(scrollWidth && offsetWidth) {
516           menuWidth = menuWidth + scrollWidth - offsetWidth ;
517        }
518     }
519     if (!lastMenuWidth || (menuWidth > lastMenuWidth)) {
520       lastMenuWidth = menuWidth ;
521     }
522 
523     items = this.get('itemList') ;
524 
525     var customViewClassName = this.get('customViewClassName'),
526         customViewMenuOffsetWidth = this.get('customViewMenuOffsetWidth'),
527         className = 'sc-view sc-pane sc-panel sc-palette sc-picker sc-menu select-button sc-scroll-view sc-menu-scroll-view sc-container-view menuContainer sc-button-view sc-menu-item sc-regular-size' ;
528     className = customViewClassName ? (className + ' ' + customViewClassName) : className ;
529 
530     dummyMenuItemView = (this.get('customView') || SC.MenuItemView).create(); 
531     menuItemViewEscapeHTML = dummyMenuItemView.get('escapeHTML') ;
532     for (idx = 0, itemsLength = items.length; idx < itemsLength; ++idx) {
533       //getting the width of largest menu item
534       item = items.objectAt(idx) ;
535       element = document.createElement('div') ;
536       element.style.cssText = 'top:-10000px; left: -10000px;  position: absolute;' ;
537       element.className = className ;
538       element.innerHTML = menuItemViewEscapeHTML ? SC.RenderContext.escapeHTML(item.title) : item.title ;
539       document.body.appendChild(element) ;
540       elementOffsetWidth = element.offsetWidth + customViewMenuOffsetWidth;
541 
542       if (!largestMenuWidth || (elementOffsetWidth > largestMenuWidth)) {
543         largestMenuWidth = elementOffsetWidth ;
544       }
545       document.body.removeChild(element) ;
546     }
547     largestMenuWidth = (largestMenuWidth > lastMenuWidth) ?
548                       largestMenuWidth: lastMenuWidth ;
549 
550     // Get the window size width and compare with the lastMenuWidth.
551     // If it is greater than windows width then reduce the maxwidth by 25px
552     // so that the ellipsis property is enabled by default
553     var maxWidth = SC.RootResponder.responder.get('currentWindowSize').width;
554     if(largestMenuWidth > maxWidth) {
555       largestMenuWidth = (maxWidth - 25) ;
556     }
557 
558     this.set('lastMenuWidth',lastMenuWidth) ;
559     value = this.get('value') ;
560     itemList = this.get('itemList') ;
561     menuControlSize = this.get('controlSize') ;
562 
563     // get the user defined custom view
564     customView = this.get('customView') ;
565     customMenuView = customView ? customView : SC.MenuItemView ;
566 
567     menu  = SC.MenuPane.create({
568 
569       classNames: ['select-button'],
570 
571       items: itemList,
572 
573       exampleView: customMenuView,
574 
575       isEnabled: YES,
576       preferType: SC.PICKER_MENU,
577       itemHeightKey: 'height',
578       layout: { width: largestMenuWidth },
579       controlSize: menuControlSize,
580       itemWidth: lastMenuWidth,
581 
582       /**
583         PerformKeyEquivalent, for handling tab and shift + tab
584         Prevents the focus going to next fields when menu is open and you tab
585 
586         @param {String} keystring
587         @param {SC.Event} evt
588         @returns {Boolean}  YES if handled
589       */
590 
591       performKeyEquivalent: function( keystring, evt ) {
592         switch (keystring) {
593           case 'tab':
594           case 'shift_tab':
595             return YES ;
596           default:
597             return sc_super() ;
598         }
599       }
600     }) ;
601 
602     // no menu to toggle... bail...
603     if (!menu) return NO ;
604     menu.popup(this, this.preferMatrix) ;
605     this.set('menu', menu);
606 
607     customView = menu.menuItemViewForContentIndex(this.get('itemIdx'));
608     menu.set('currentMenuItem', customView) ;
609     customView.becomeFirstResponder();
610 
611     this.set('isActive', YES);
612     return YES ;
613   },

630   /**

632      position menu such that the selected item in the menu will be
633      place aligned to the item on the button when menu is opened.

635   changeSelectButtonPreferMatrix: function() {
636     var controlSizeTuning = 0, customMenuItemHeight = 0 ;
637     switch (this.get('controlSize')) {
638       case SC.TINY_CONTROL_SIZE:
639         controlSizeTuning = SC.SelectButtonView.TINY_OFFSET_Y;
640         customMenuItemHeight = SC.MenuPane.TINY_MENU_ITEM_HEIGHT;
641         break;
642       case SC.SMALL_CONTROL_SIZE:
643         controlSizeTuning = SC.SelectButtonView.SMALL_OFFSET_Y;
644         customMenuItemHeight = SC.MenuPane.SMALL_MENU_ITEM_HEIGHT;
645         break;
646       case SC.REGULAR_CONTROL_SIZE:
647         controlSizeTuning = SC.SelectButtonView.REGULAR_OFFSET_Y;
648         customMenuItemHeight = SC.MenuPane.REGULAR_MENU_ITEM_HEIGHT;
649         break;
650       case SC.LARGE_CONTROL_SIZE:
651         controlSizeTuning = SC.SelectButtonView.LARGE_OFFSET_Y;
652         customMenuItemHeight = SC.MenuPane.LARGE_MENU_ITEM_HEIGHT;
653         break;
654       case SC.HUGE_CONTROL_SIZE:
655         controlSizeTuning = SC.SelectButtonView.HUGE_OFFSET_Y;
656         customMenuItemHeight = SC.MenuPane.HUGE_MENU_ITEM_HEIGHT;
657         break;
658     }
659 
660     var preferMatrixAttributeTop = controlSizeTuning ,
661         itemIdx = this.get('itemIdx') ,
662         leftAlign = this.get('leftAlign'), defPreferMatrix, tempPreferMatrix ;
663 
664     if(this.get('isDefaultPosition')) {
665       defPreferMatrix = [1, 0, 3] ;
666       this.set('preferMatrix', defPreferMatrix) ;
667     }
668     else {
669       if(itemIdx) {
670         preferMatrixAttributeTop = itemIdx * customMenuItemHeight +
671           controlSizeTuning ;
672       }
673       tempPreferMatrix = [leftAlign, -preferMatrixAttributeTop, 2] ;
674       this.set('preferMatrix', tempPreferMatrix) ;
675     }

678   /**
679     @private
680 
681     Holding down the button should display the menu pane.
682   */
683   mouseDown: function(evt) {
684     if (!this.get('isEnabled')) return YES ; // handled event, but do nothing
685     this.set('isActive', YES);
686     this._isMouseDown = YES;
687     this.becomeFirstResponder() ;
688     this._action() ;
689 
690     // Store the current timestamp. We register the timestamp at the end of
691     // the runloop so that the menu has been rendered, in case that operation
692     // takes more than a few hundred milliseconds.
693 
694     // One mouseUp, we'll use this value to determine how long the mouse was
695     // pressed.
696     this.invokeLast(this._recordMouseDownTimestamp);
697     return YES ;

700   /** @private
701     Records the current timestamp. This is invoked at the end of the runloop
702     by mouseDown. We use this value to determine the delay between mouseDown
703     and mouseUp.
704   */
705   _recordMouseDownTimestamp: function() {
706     this._menuRenderedTimestamp = new Date().getTime();
707   },
708 
709   /** @private
710     Because we responded YES to the mouseDown event, we have responsibility
711     for handling the corresponding mouseUp event.
712 
713     However, the user may click on this button, then drag the mouse down to a
714     menu item, and release the mouse over the menu item. We therefore need to
715     delegate any mouseUp events to the menu's menu item, if one is selected.
716 
717     We also need to differentiate between a single click and a click and hold.
718     If the user clicks and holds, we want to close the menu when they release.
719     Otherwise, we should wait until they click on the menu's modal pane before
720     removing our active state.
721 
722     @param {SC.Event} evt
723     @returns {Boolean}
724   */
725   mouseUp: function(evt) {
726     var timestamp = new Date().getTime(),
727         previousTimestamp = this._menuRenderedTimestamp,
728         menu = this.get('menu'),
729         touch = SC.platform.touch,
730         targetMenuItem;
731 
732     if (menu) {
733       targetMenuItem = menu.getPath('rootMenu.targetMenuItem');
734 
735       if (targetMenuItem && targetMenuItem.get('mouseHasEntered')) {
736         // Have the menu item perform its action.
737         // If the menu returns NO, it had no action to
738         // perform, so we should close the menu immediately.
739         if (!targetMenuItem.performAction()) menu.remove();
740       } else if (!touch && (timestamp - previousTimestamp > SC.ButtonView.CLICK_AND_HOLD_DELAY)) {
741         // If the user waits more than a certain length of time between
742         // mouseDown and mouseUp, we can assume that they are clicking and
743         // dragging to the menu item, and we should close the menu if they
744         // mouseup anywhere not inside the menu.
745 
746         // As a special case, we should trigger an action on the currently
747         // selected menu item if the menu item is under the mouse and the user
748         // never moved their mouse before mouseup.
749         if (!menu.get('mouseHasEntered') && !this.get('isDefaultPosition')) {
750           targetMenuItem = menu.get('currentMenuItem');
751           if (targetMenuItem && !targetMenuItem.performAction()) {
752             menu.remove();
753           }
754         } else {
755           // Otherwise, just remove the menu because no selection
756           // has been made.
757           menu.remove();
758         }
759       }
760     }
761 
762 
763     // Reset state.
764     this._isMouseDown = NO;
765     return YES;
766   },

789   /**
790     @private
791 
792     Pressing the Up or Down arrow key should display the menu pane
793   */
794   interpretKeyEvents: function(event) {
795     if (event) {
796       if ((event.keyCode === 38 || event.keyCode === 40)) {
797         this._action() ;
798       }
799       else if (event.keyCode === 27) {
800         this.resignFirstResponder() ;
