643   /**
644     Called whenever the view needs to updates it's contentRangeObserver to

646     method yourself but you may override it if you need to provide some 
647     custom range observer behavior.

658     destroyRangeObserver() to cleanup any existing range observer.
659     
660     @returns {void}

662   updateContentRangeObserver: function() {
663     var nowShowing = this.get('nowShowing'),

667     if (!content) return ; // nothing to do
668     
669     if (observer) {
670       content.updateRangeObserver(observer, nowShowing);
671     } else {
672       var func = this.contentRangeDidChange;
673       observer = content.addRangeObserver(nowShowing, this, func, null);      
674       this._cv_contentRangeObserver = observer ;
675     }

712   /** @private
713     Whenever content property changes to a new value:
714     
715      - remove any old observers 
716      - setup new observers (maybe wait until end of runloop to do this?)
717      - recalc height/reload content
718      - set content as delegate if delegate was old content
719      - reset selection
720      
721     Whenever content array mutates:
722     
723      - possibly stop observing property changes on objects, observe new objs
724      - reload effected item views
725      - update layout for receiver
726   */
727   _cv_contentDidChange: function() {
728     var content = this.get('content'),
729         lfunc   = this.contentLengthDidChange ;
730 
731     if (content === this._content) return; // nothing to do
732 
733     // cleanup old content
734     this.removeContentRangeObserver();
735     if (this._content) {
736       this._content.removeObserver('length', this, lfunc);
737     }
738     
739     // cache
740     this._content = content;
741     
742     // add new observers - range observer will be added lazily
743     if (content) {
744       content.addObserver('length', this, lfunc);
745     }
746     
747     // notify all items changed
748     this.contentLengthDidChange();
749     this.contentRangeDidChange(content, null, '[]', null);
750     
751   }.observes('content'),

953           view = views[i];
954           isGroupView = view.get('isGroupView');
955           shouldReuse = isGroupView ? shouldReuseGroupViews : shouldReuseViews;
956           if (shouldReuse) {
957             viewPool = isGroupView ? this._GROUP_VIEW_POOL : this._VIEW_POOL;
958

969         }
970       }
971       
972

974       // We do this in order to maximize the change of re-use should the view

1455     will replace the selection with the passed value.  Otherwise the indexes
1456     will be added to the current selection.

1459     @param extend {Boolean} optionally extend the selection
1460     @returns {SC.CollectionView} receiver
1461   */
1462   select: function(indexes, extend) {
1463

1465         del     = this.get('selectionDelegate'),
1466         groupIndexes = this.get('_contentGroupIndexes'),
1467         sel;
1468         
1469     if(!this.get('isSelectable')) return this;
1470 
1471     // normalize
1472     if (SC.typeOf(indexes) === SC.T_NUMBER) {
1473       indexes = SC.IndexSet.create(indexes, 1);
1474     }
1475 
1476     // if we are passed an empty index set or null, clear the selection.
1477     if (indexes && indexes.get('length')>0) {
1478 
1479       // first remove any group indexes - these can never be selected
1480       if (groupIndexes && groupIndexes.get('length')>0) {
1481         indexes = indexes.copy().remove(groupIndexes);
1482       }

1484       // give the delegate a chance to alter the items

1487

1491     if (extend && (sel = this.get('selection'))) sel = sel.copy();

1495

1497       if (indexes.get('length')===1) {

1499         
1500       // otherwise select an index range
1501       } else sel.add(content, indexes);
1502       
1503     }
1504

1506     sel = del.collectionViewSelectionForProposedSelection(this, sel);
1507     if (!sel) sel = SC.SelectionSet.create(); // empty
1508     
1509     // if we're not extending the selection, clear the selection anchor
1510     this._selectionAnchor = null ;

1550    delegate. If a non-selectable item is found, the index is skipped. If
1551    no item is found, selection index is returned unmodified.

1554    selection index and the proposed index.

1558    @returns {Number} next selectable index. 
1559   */

1569     // fast path
1570     if (!groupIndexes && (del.collectionViewShouldSelectIndexes === this.collectionViewShouldSelectIndexes)) {
1571       return proposedIndex;
1572     }
1573 
1574     // loop forwards looking for an index that is allowed by delegate
1575     // we could alternatively just pass the whole range but this might be 
1576     // slow for the delegate

1583       }

1587     // if nothing was found, return top of selection

1589       sel = this.get('selection');
1590       bottom = sel ? sel.get('max') : -1 ;
1591     }
1592     return bottom ;

2730           isNextInDrag     = objects.indexes.contains(idx);
2731         } else {
2732           isPreviousInDrag = objects.indexes.contains(idx);
2733           isNextInDrag     = objects.indexes.contains(idx-1);

2818         
2819     // Get the correct insertion point, drop operation, etc.
2820     var state = this._computeDropOperationState(drag, null, op),
2821         idx   = state[0], dropOp = state[1], dragOp = state[2],
2822         del   = this.get('selectionDelegate'),
2823         performed, objects, data, content, shift, indexes;
2824         
2825     // The dragOp is the kinds of ops allowed.  The drag operation must 
2826     // be included in that set.
2827     if (dragOp & SC.DRAG_REORDER) {
2828       op = (op & SC.DRAG_MOVE) ? SC.DRAG_REORDER : SC.DRAG_NONE ;
2829     } else op = op & dragOp ;
2830     
2831     // If no allowed drag operation could be found, just return.
2832     if (op === SC.DRAG_NONE) return op;
2833     
2834     // Some operation is allowed through, give the delegate a chance to
2835     // handle it.
2836     performed = del.collectionViewPerformDragOperation(this, drag, op, idx, dropOp) ;
2837     
2838     // If the delegate did not handle the drag (i.e. returned SC.DRAG_NONE),
2839     // and the op type is REORDER, then do the reorder here.
2840     if ((performed === SC.DRAG_NONE) && (op & SC.DRAG_REORDER)) {
2841       
2842       data = drag.dataForType(this.get('reorderDataType')) ;
2843       if (!data) return SC.DRAG_NONE ;
2844       
2845       content = this.get('content') ;
2846       
2847       // check for special case - inserting BEFORE ourself...
2848       // in this case just pretend the move happened since it's a no-op 
2849       // anyway
2850       indexes = data.indexes;
2851       if (indexes.get('length')===1) {
2852         if (((dropOp === SC.DROP_BEFORE) || (dropOp === SC.DROP_AFTER)) &&
2853             (indexes.get('min')===idx)) return SC.DRAG_MOVE;
2854       }
2855       
2856       content.beginPropertyChanges(); // suspend notifications
2857 
2858       // get each object, then remove it from the content. they will be 
2859       // added again later.
2860       objects = [];
2861       shift = 0;
2862       data.indexes.forEach(function(i) {  
2863         objects.push(content.objectAt(i-shift));
2864         content.removeAt(i-shift);
2865         shift++;
2866         if (i < idx) idx--;
2867       }, this);
2868       
2869       // now insert objects into new insertion locaiton
2870       if (dropOp === SC.DROP_AFTER) idx++;
2871       content.replace(idx, 0, objects, dropOp);
2872       this.select(SC.IndexSet.create(idx, objects.length));
2873       content.endPropertyChanges(); // restart notifications
2874 
2875       // make the op into its actual value
2876       op = SC.DRAG_MOVE ;
2877     }
2878     
2879     return op;
