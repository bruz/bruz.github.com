 52 /**
 53   @namespace
 54   
 55   A Control is a view that also implements some basic state functionality.
 56   Apply this mixin to any view that you want to have standard control
 57   functionality including showing a selected state, enabled state, focus
 58   state, etc.
 59   
 60   h2. About Values and Content
 61   
 62   Controls typically are used to represent a single value, such as a number,
 63   boolean or string.  The value a control is managing is typically stored in
 64   a "value" property.  You will typically use the value property when working
 65   with controls such as buttons and text fields in a form.
 66   
 67   An alternative way of working with a control is to use it to manage some
 68   specific aspect of a content object.  For example, you might use a label
 69   view control to display the "name" property of a Contact record.  This 
 70   approach is often necessary when using the control as part of a collection
 71   view.
 72   
 73   You can use the content-approach to work with a control by setting the 
 74   "content" and "contentValueKey" properties of the control.  The 
 75   "content" property is the content object you want to manage, while the 
 76   "contentValueKey" is the name of the property on the content object 
 77   you want the control to display.
 78   
 79   The default implementation of the Control mixin will essentially map the
 80   contentValueKey of a content object to the value property of the 
 81   control.  Thus if you are writing a custom control yourself, you can simply
 82   work with the value property and the content object support will come for
 83   free.  Just write an observer for the value property and update your 
 84   view accordingly.
 85   
 86   If you are working with a control that needs to display multiple aspects
 87   of a single content object (for example showing an icon and label), then
 88   you can override the contentValueDidChange() method instead of observing
 89   the value property.  This method will be called anytime _any_ property 
 90   on the content object changes.  You should use this method to check the
 91   properties you care about on the content object and update your view if 
 92   anything you care about has changed.
 93   
 94   h2. Delegate Support
 95   
 96   Controls can optionally get the contentDisplayProperty from a 
 97   displayDelegate, if it is set.  The displayDelegate is often used to 
 98   delegate common display-related configurations such as which content value
 99   to show.  Anytime your control is shown as part of a collection view, the
100   collection view will be automatically set as its displayDelegate.
101   
102   @since SproutCore 1.0
103 */
104 SC.Control = {
105   
106   /** @private */
107   initMixin: function() {
108     this._control_contentDidChange() ; // setup content observing if needed.
109   },
110   
111   /** 
112     The selected state of this control.  Possible options are YES, NO or 
113     SC.MIXED_STATE.
114     
115     @property {Boolean}
116   */
117   isSelected: NO,
118   
119   /** @private */
120   isSelectedBindingDefault: SC.Binding.oneWay().bool(),
121   
122   /**
123     Set to YES when the item is currently active.  Usually this means the 
124     mouse is current pressed and hovering over the control, however the 
125     specific implementation my vary depending on the control.
126     
127     Changing this property value by default will cause the Control mixin to
128     add/remove an 'active' class name to the root element.
129     
130     @property {Boolean}
131   */
132   isActive: NO,
133   
134   /** @private */
135   isActiveBindingDefault: SC.Binding.oneWay().bool(),
136   
137   /**
138     The value represented by this control.
139     
140     Most controls represent a value of some type, such as a number, string
141     or image URL.  This property should hold that value.  It is bindable
142     and observable.  Changing this value will immediately change the
143     appearance of the control.  Likewise, editing the control 
144     will immediately change this value.
145     
146     If instead of setting a single value on a control, you would like to 
147     set a content object and have the control display a single property
148     of that control, then you should use the content property instead.
149 
150     @property {Object}
151   */
152   value: null,
153   
154   /**
155     The content object represented by this control.
156     
157     Often you need to use a control to display some single aspect of an 
158     object, especially if you are using the control as an item view in a
159     collection view.
160     
161     In those cases, you can set the content and contentValueKey for the
162     control.  This will cause the control to observe the content object for
163     changes to the value property and then set the value of that property 
164     on the "value" property of this object.
165     
166     Note that unless you are using this control as part of a form or 
167     collection view, then it would be better to instead bind the value of
168     the control directly to a controller property.
169     
170     @property {SC.Object}
171   */
172   content: null,
173   
174   /**
175     The property on the content object that would want to represent the 
176     value of this control.  This property should only be set before the
177     content object is first set.  If you have a displayDelegate, then
178     you can also use the contentValueKey of the displayDelegate.
179     
180     @property {String}
181   */
182   contentValueKey: null,
183   
184   /**
185     Invoked whenever any property on the content object changes.  
186     
187     The default implementation will update the value property of the view
188     if the contentValueKey property has changed.  You can override this
189     method to implement whatever additional changes you would like.
190     
191     The key will typically contain the name of the property that changed or 
192     '*' if the content object itself has changed.  You should generally do
193     a total reset of '*' is changed.
194     
195     @param {Object} target the content object
196     @param {String} key the property that changes
197     @returns {void}
198     @test in content
199   */
200   contentPropertyDidChange: function(target, key) {
201     return this.updatePropertyFromContent('value', key, 'contentValueKey');
202   },
203   
204   /**
205     Helper method you can use from your own implementation of 
206     contentPropertyDidChange().  This method will look up the content key to
207     extract a property and then update the property if needed.  If you do
208     not pass the content key or the content object, they will be computed 
209     for you.  It is more efficient, however, for you to compute these values
210     yourself if you expect this method to be called frequently.
211     
212     @param {String} prop local property to update
213     @param {String} key the contentproperty that changed
214     @param {String} contentKey the local property that contains the key
215     @param {Object} content
216     @returns {SC.Control} receiver
217   */
218   updatePropertyFromContent: function(prop, key, contentKey, content) {
219     var all = key === '*';
220     if (contentKey === undefined) {
221       contentKey = "content"+prop.capitalize()+"Key";
222     }
223     if (content === undefined) content = this.get('content');
224     
225     // get actual content key
226     contentKey = this[contentKey] ?
227       this.get(contentKey) :
228       this.getDelegateProperty(contentKey, this.displayDelegate) ;
229     
230     if (contentKey && (all || key === contentKey)) {
231       var v = (content) ?
232         (content.get ? content.get(contentKey) : content[contentKey]) :
233         null ;
234       this.set(prop, v) ;
235     }
236     return this ;
237   },
238   
239   /**
240     Relays changes to the value back to the content object if you are using
241     a content object.
242     
243     This observer is triggered whenever the value changes.  It will only do
244     something if it finds you are using the content property and
245     contentValueKey and the new value does not match the old value of the
246     content object.  
247     
248     If you are using contentValueKey in some other way than typically
249     implemented by this mixin, then you may want to override this method as
250     well.
251     
252     @returns {void}
253   */
254   updateContentWithValueObserver: function() {
255     var key = this.contentValueKey ?
256       this.get('contentValueKey') :
257       this.getDelegateProperty('contentValueKey', this.displayDelegate),
258       content = this.get('content') ;
259     if (!key || !content) return ; // do nothing if disabled
260     
261     // get value -- set on content if changed
262     var value = this.get('value');
263     if (typeof content.setIfChanged === SC.T_FUNCTION) {
264       content.setIfChanged(key, value);
265     } else {
266       // avoid re-writing inherited props
267       if (content[key] !== value) content[key] = value ;
268     }
269   }.observes('value'),
270   
271   /**
272     The name of the property this control should display if it is part of an
273     SC.FormView.
274     
275     If you add a control as part of an SC.FormView, then the form view will 
276     automatically bind the value to the property key you name here on the 
277     content object.
278     
279     @property {String}
280   */
281   fieldKey: null,
282   
283   /**
284     The human readable label you want shown for errors.  May be a loc string.
285     
286     If your field fails validation, then this is the name that will be shown
287     in the error explanation.  If you do not set this property, then the 
288     fieldKey or the class name will be used to generate a human readable name.
289     
290     @property {String}
291   */
292   fieldLabel: null,
293   
294   /**
295     The human readable label for this control for use in error strings.  This
296     property is computed dynamically using the following rules:
297     
298     If the fieldLabel is defined, that property is localized and returned.
299     Otherwise, if the keyField is defined, try to localize using the string 
300     "ErrorLabel.{fieldKeyName}".  If a localized name cannot be found, use a
301     humanized form of the fieldKey.
302     
303     Try to localize using the string "ErrorLabel.{ClassName}". Return a 
304     humanized form of the class name.
305     
306     @property {String}
307   */
308   errorLabel: function() {
309     var ret, fk, def ;
310     if (ret = this.get('fieldLabel')) return ret ;
311     
312     // if field label is not provided, compute something...
313     fk = this.get('fieldKey') || this.constructor.toString() ;
314     def = (fk || '').humanize().capitalize() ;
315     return "ErrorLabel."+fk
316       .locWithDefault(("FieldKey."+fk).locWithDefault(def)) ;
317       
318   }.property('fieldLabel','fieldKey').cacheable(),
319 
320   /**
321     The control size.  This will set a CSS style on the element that can be 
322     used by the current theme to vary the appearance of the control.
323     
324     @property {String}
325   */
326   controlSize: SC.REGULAR_CONTROL_SIZE,
327   
328   /** @private */
329   displayProperties: 'isEnabled isSelected isActive controlSize'.w(),
330   
331   /** @private */
332   _CONTROL_TMP_CLASSNAMES: {},
333   
334   /** @private
335     Invoke this method in your updateDisplay() method to update any basic 
336     control CSS classes.
337   */
338   renderMixin: function(context, firstTime) {
339     var sel = this.get('isSelected'), disabled = !this.get('isEnabled'),
340     // update the CSS classes for the control.  note we reuse the same hash
341     // to avoid consuming more memory
342         names = this._CONTROL_TMP_CLASSNAMES ; // temporary object
343     names.mixed = sel === SC.MIXED_STATE;
344     names.sel = sel && (sel !== SC.MIXED_STATE) ;
345     names.active = this.get('isActive') ;
346     context.setClass(names).addClass(this.get('controlSize'));
347   },
348   
349   /** @private
350     This should be null so that if content is also null, the
351     _contentDidChange won't do anything on init.
352   */
353   _control_content: null,
354   
355   /** @private
356     Observes when a content object has changed and handles notifying 
357     changes to the value of the content object.
358   */
359   _control_contentDidChange: function() {
360     var content = this.get('content') ;
361     if (this._control_content === content) return; // nothing changed
362     
363     var f = this.contentPropertyDidChange,
364     // remove an observer from the old content if necessary
365         old = this._control_content ;
366     if (old && old.removeObserver) old.removeObserver('*', this, f) ;
367     
368     // add observer to new content if necessary.
369     this._control_content = content ;
370     if (content && content.addObserver) content.addObserver('*', this, f) ;
371     
372     // notify that value did change.
373     this.contentPropertyDidChange(content, '*') ;
374     
375   }.observes('content')
376   
377 };
