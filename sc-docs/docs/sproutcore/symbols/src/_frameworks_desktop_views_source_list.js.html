  8 sc_require('views/list') ;
  9 sc_require('views/source_list_group');
 10 
 11 SC.BENCHMARK_SOURCE_LIST_VIEW = YES ;
 12 
 13 /** @class
 14   
 15   Displays a source list like the source list in iTunes.  SourceList views
 16   are very similar to ListView's but come preconfigured with the correct
 17   appearance and default behaviors of a source list.
 18   
 19   @extends SC.ListView
 20   @since SproutCore 1.0
 21 */
 22 SC.SourceListView = SC.ListView.extend(
 23 /** @scope SC.SourceListView.prototype */ {
 24 
 25   /**
 26     Add class name to HTML for styling.
 27   */
 28   classNames: ['sc-source-list'],
 29   
 30   /**
 31     Default row height for source list items is larger.
 32   */
 33   rowHeight: 32,
 34 
 35   /**
 36     By default source lists should not select on mouse down since you will
 37     often want to drag an item instead of selecting it.
 38   */
 39   selectOnMouseDown: NO,
 40   
 41   /**
 42     By default, SourceListView's trigger any action you set whenever the user
 43     clicks on an item.  This gives the SourceList a "menu" like behavior.
 44   */
 45   actOnSelect: YES
 46 
 47 });
