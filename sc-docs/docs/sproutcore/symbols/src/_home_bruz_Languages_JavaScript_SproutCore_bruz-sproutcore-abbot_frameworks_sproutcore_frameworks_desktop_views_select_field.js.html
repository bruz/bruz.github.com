  7 /**
  8   @class
  9 
 10   SelectFieldView displays browser-native popup menu.  To use this view,
 11   you should either bake into the HTML the preset list of options, or 
 12   you can set the -objects property to an array of items to show.  The
 13   value is current value of the select.
 14   
 15   @extends SC.FieldView
 16   @author Charles Jolley
 17   @author Mike Ball
 18   @since SproutCore 1.0
 19 */
 20 SC.SelectFieldView = SC.FieldView.extend(
 21 /** @scope SC.SelectFieldView.prototype */ {
 22 
 23   tagName: 'select',
 24   classNames: ['sc-select-field-view'],
 25  
 26   /**
 27     An array of items that will form the menu you want to show.
 28   */ 
 29   objects: [],
 30   
 31   /**
 32     Binding default for an array of objects
 33   */ 
 34   objectsBindingDefault: SC.Binding.multiple(),
 35 
 36   /**
 37     If you set this to a non-null value, then the name shown for each 
 38     menu item will be pulled from the object using the named property.
 39     if this is null, the collection objects themselves will be used.
 40   */
 41   nameKey: null,
 42 
 43   /**
 44    If you set this to a non-null value, then the value of this key will
 45    be used to sort the objects.  If this is not set, then nameKey will
 46    be used.
 47   */ 
 48   sortKey: null,
 49 
 50   /**
 51      Set this to a non-null value to use a key from the passed set of objects
 52      as the value for the options popup.  If you don't set this, then the
 53      objects themselves will be used as the value.
 54   */ 
 55   valueKey: null,
 56 
 57   /**
 58     set this to non-null to place an empty option at the top of the menu.   
 59   */
 60   emptyName: null,
 61 
 62   /**
 63     if true, the empty name will be localized.
 64   */
 65   localize: false,
 66   
 67   /**
 68     if true, it means that the nameKey, valueKey or objects changed
 69   */
 70   cpDidChange: YES,
 71   
 72   /**
 73     if true, it means that no sorting will occur, objects will appear 
 74     in the same order as in the array
 75   */
 76   disableSort: NO,
 77   
 78   
 79   
 80   /**
 81     override this to change the enabled/disabled state of menu items as they
 82     are built.  Return false if you want the menu item to be disabled.
 83     
 84     @param itemValue the value for the item to validate
 85     @param itemName the name of the menu item to validate
 86     @returns YES if the item should be enabled, NO otherwise
 87   */  
 88   validateMenuItem: function(itemValue, itemName) {
 89     return true ;
 90   },
 91 
 92   /**
 93     override this method to implement your own sorting of the menu. By
 94     default, menu items are sorted using the value shown or the sortKey
 95     
 96     @param objects the unsorted array of objects to display.
 97     @returns sorted array of objects
 98   */
 99   sortObjects: function(objects) {
100     if(!this.get('disableSort')){
101       var nameKey = this.get('sortKey') || this.get('nameKey') ;
102       objects = objects.sortProperty(nameKey);
103     }
104     return objects ;
105   },
106 
107   render: function(context, firstTime) {
108     if (this.get('cpDidChange')) {
109       this.set('cpDidChange', NO);
110       // get list of objects.
111       var nameKey = this.get('nameKey') ;
112       var valueKey = this.get('valueKey') ;
113       var objects = this.get('objects') ;
114       var fieldValue = this.get('value') ;
115       var el, selectElement;
116     
117       // get the localization flag.
118       var shouldLocalize = this.get('localize'); 
119    
120       // convert fieldValue to guid, if it is an object.
121       if (!valueKey && fieldValue) fieldValue = SC.guidFor(fieldValue) ;
122       if ((fieldValue === null) || (fieldValue === '')) fieldValue = '***' ;
123     
124       if (objects) {
125         objects = this.sortObjects(objects) ; // sort'em.
126         // var html = [] ;       
127         if(!firstTime){
128           selectElement=this.$input()[0];
129           selectElement.innerHTML='';
130         } 
131       
132         var emptyName = this.get('emptyName') ;
133         if (emptyName) {
134           if (shouldLocalize) emptyName = emptyName.loc() ;
135           if(firstTime){
136             context.push('<option value="***">'+emptyName+'</option>',
137                           '<option disabled="disabled"></option>') ;
138           }else{
139             el=document.createElement('option');
140             el.value="***";
141             el.innerHTML=emptyName;
142             selectElement.appendChild(el);
143             el=document.createElement('option');
144             el.disabled="disabled";
145             selectElement.appendChild(el);
146           }
147         }
148    
149           // generate option elements.
150         objects.forEach(function(object) {
151         if (object) {
152           // either get the name from the object or convert object to string.
153           var name = nameKey ? (object.get ? object.get(nameKey) : object[nameKey]) : object.toString() ;
154    
155           // localize name if specified.
156           if(shouldLocalize)
157           {
158             name = name.loc();
159           }
160    
161           // get the value using the valueKey or the object if no valueKey.
162           // then convert to a string or use _guid if one of available.
163           var value = (valueKey) ? (object.get ? object.get(valueKey) : object[valueKey]) : object ;
164           if (value) value = (SC.guidFor(value)) ? SC.guidFor(value) : value.toString() ;
165    
166           // render HTML
167           var disable = (this.validateMenuItem && this.validateMenuItem(value, name)) ? '' : 'disabled="disabled" ' ;
168           if(firstTime){
169             context.push('<option '+disable+'value="'+value+'">'+name+'</option>') ;
170           } else{
171             el=document.createElement('option');
172             el.value=value;
173             el.innerHTML=name;
174             if(disable.length>0) el.disable="disabled";
175             selectElement.appendChild(el);
176           }
177         // null value means separator.
178         } else {
179           if(firstTime){
180             context.push('<option disabled="disabled"></option>') ;
181           }else{
182             el=document.createElement('option');
183             el.disabled="disabled";
184             selectElement.appendChild(el);
185           }
186         }
187       }, this );
188    
189       this.setFieldValue(fieldValue);
190    
191       } else {
192         this.set('value',null);
193       }
194     }
195   },
196   
197   displayProperties: ['objects','nameKey','valueKey'],
198 
199   _objectsObserver: function() {
200     this.set('cpDidChange', YES);
201   }.observes('objects'),
202    
203   _nameKeyObserver: function() {
204     this.set('cpDidChange', YES);
205   }.observes('nameKey'),
206    
207   _valueKeyObserver: function() {
208     this.set('cpDidChange', YES);
209   }.observes('valueKey'),
210     
211   
212    
213   // .......................................
214   // PRIVATE
215   //
216    
217   $input: function() { return this.$(); },
218    
219   /* @private */
220   mouseDown: function(evt) {
221     
222     if (!this.get('isEnabled')) {
223       evt.stop();
224       return YES;
225     } else return sc_super();
226   },
227    
228   // when fetching the raw value, convert back to an object if needed...
229   /** @private */
230   getFieldValue: function() {
231     var value = sc_super(); // get raw value... 
232     var valueKey = this.get('valueKey') ;
233     var objects = this.get('objects') ;
234     var found, object;
235     
236     // Handle empty selection.
237     if (value == '***') {
238       value = null ;
239     
240     // If no value key was set and there are objects then match back to an
241     // object.
242     } else if (value && objects) {
243       // objects = Array.from(objects) ;
244       
245       var loc = (SC.typeOf(objects.length) === SC.T_FUNCTION) ? objects.length() : objects.length;
246       
247       found = null ; // matching object goes here.
248       while(!found && (--loc >= 0)) {
249         object = objects.objectAt? objects.objectAt(loc) : objects[loc] ;
250         if (!object) continue; // null means placeholder; just skip
251       
252         // get value using valueKey if there is one or use object
253         // map to _guid or toString.
254         if (valueKey) object = (object.get) ? object.get(valueKey) : object[valueKey] ;
255         var ov = (object) ? (SC.guidFor(object) ? SC.guidFor(object) : object.toString()) : null ;
256       
257         // use this object value if it matches.
258         if (value == ov) found = object ;
259       }
260     }
261     
262     return (valueKey || found) ? found : value;
263   },
264   
265   setFieldValue: function(newValue) {
266     if (SC.none(newValue)) { newValue = '' ; }
267     else {
268       newValue = ((newValue) ? (SC.guidFor(newValue) ? SC.guidFor(newValue) : newValue.toString()) : null );
269     }
270     this.$input().val(newValue);
271     return this ;
272   },
273   
274   
275   
276  
277   
278   fieldDidFocus: function() {
279     var isFocused = this.get('isFocused');
280     if (!isFocused) this.set('isFocused', true);
281   },
282 
283   fieldDidBlur: function() {
284     var isFocused = this.get('isFocused');
285     if (isFocused) this.set('isFocused', false);
286   },
287 
288 
289   
290   _isFocusedObserver: function() {
291     this.$().setClass('focus', this.get('isFocused'));
292   }.observes('isFocused'),
293 
294   didCreateLayer: function() {
295     var input = this.$input();
296     if (this.get('isEnabled') === false) this.$()[0].disabled = true;
297     SC.Event.add(input, 'blur', this, this.fieldDidBlur);
298     SC.Event.add(input, 'focus',this, this.fieldDidFocus);
299     return sc_super();
300   },
301   
302   willDestroyLayer: function() {
303     var input = this.$input();
304     SC.Event.remove(input, 'focus', this, this.fieldDidFocus);
305     SC.Event.remove(input, 'blur', this, this.fieldDidBlur);
306     return sc_super();
307   }
308  
309 });
