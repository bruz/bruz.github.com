  8 /**
  9   @class
 10   
 11   A simple object representing the selection inside a text field.  Each
 12   object is frozen and contains exactly three properties:
 13   
 14     *  start
 15     *  end
 16     *  length
 17   
 18   Important note:  In Internet Explorer, newlines in textara elements are
 19   considered two characters.  SproutCore does not currently try to hide this from you.
 20   
 21   @extends SC.Object
 22   @extends SC.Copyable
 23   @extends SC.Freezable
 24   @since SproutCore 1.0
 25 */
 26 
 27 SC.TextSelection = SC.Object.extend(SC.Copyable, SC.Freezable,
 28 /** @scope SC.TextSelection.prototype */ {  
 29 
 30   /**
 31     The number of characters appearing to the left of the beginning of the
 32     selection, starting at 0.
 33     
 34     @type {Number}
 35   */
 36   start: -1,
 37   
 38   
 39   /**
 40     The number of characters appearing to the left of the end of the
 41     selection.
 42 
 43     This will have the same value as 'start' if there is no selection and
 44     instead there is only a caret.
 45     
 46     @type {Number}
 47   */
 48   end: -1,
 49  
 50    
 51   /**
 52     The length of the selection.  This is equivalent to (end - start) and
 53     exists mainly as a convenience.
 54     
 55     @property {Number}
 56   */
 57   length: function() {
 58     var start = this.get('start') ;
 59     var end   = this.get('end') ;
 60     if ((start) === -1  ||  (end === -1)) {
 61       return -1 ;
 62     }
 63     else {
 64       return end - start ;
 65     }
 66   }.property('start', 'end').cacheable(),
 67   
 68   
 69   
 70   // ..........................................................
 71   // INTERNAL SUPPORT
 72   //
 73   
 74   init: function() {
 75     sc_super();
 76     this.freeze();
 77   },
 78   
 79   
 80   copy: function() {
 81     return SC.TextSelection.create({
 82       start: this.get('start'),
 83       end:   this.get('end')
 84     });
 85   },
 86   
 87   
 88   toString: function() {
 89     var length = this.get('length');
 90     if (length  &&  length > 0) {
 91       if (length === 1) {
 92         return "[%@ character selected: {%@, %@}]".fmt(length, this.get('start'), this.get('end'));
 93       }
 94       else {
 95         return "[%@ characters selected: {%@, %@}]".fmt(length, this.get('start'), this.get('end'));
 96       }
 97     }
 98     else {
 99       return "[no text selected; caret at %@]".fmt(this.get('start'));
100     }
101   }
102 
103 }) ;
