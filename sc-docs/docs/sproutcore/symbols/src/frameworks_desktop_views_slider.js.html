  8 /** @class
  9 
 10   A SliderView shows a horizontal slider control that you can use to set 
 11   variable values.
 12 
 13   You can use a slider view much like you would any other control.  Simply
 14   set the value or content/contentValueKey to whatever value you want to 
 15   display.  You can also set the maximumValue and minValue properties to 
 16   determine the mapping of the control to its children.
 17   
 18   @extends SC.View
 19   @extends SC.Control
 20   @since SproutCore 1.0
 21   @test in progress
 22 */
 23 SC.SliderView = SC.View.extend(SC.Control,
 24 /** @scope SC.SliderView.prototype */ {
 25   
 26   classNames: 'sc-slider-view',
 27   
 28   /** 
 29     The DOM element that displays the handle.  This element will have its
 30     top-left corner updated to reflect the current state of the slider.  Use
 31     margin-offsets to properly position your handle over this location.
 32   */
 33   handleSelector: 'img.sc-handle',
 34   
 35   /**
 36     Bind this to the current value of the progress bar.  Note that by default 
 37     an empty value will disable the progress bar and a multiple value too make 
 38     it indeterminate.
 39   */
 40   value: 0.50,
 41   valueBindingDefault: SC.Binding.single().notEmpty(),
 42   
 43   /**
 44     The minimum value of the progress.
 45   */ 
 46   minimum: 0,
 47   minimumBindingDefault: SC.Binding.single().notEmpty(),
 48 
 49   /**
 50     Optionally specify the key used to extract the minimum progress value 
 51     from the content object.  If this is set to null then the minimum value
 52     will not be derived from the content object.
 53     
 54     @property {String}
 55   */
 56   contentMinimumKey: null,
 57   
 58   /**
 59     The maximum value of the progress bar.
 60   */
 61   maximum: 1.0,
 62   maximumBindingDefault: SC.Binding.single().notEmpty(),
 63 
 64   /**
 65     Optionally specify the key used to extract the maximum progress value 
 66     from the content object.  If this is set to null then the maximum value
 67     will not be derived from the content object.
 68     
 69     @property {String}
 70   */
 71   contentMaximumKey: null,
 72   
 73   /**
 74     Optionally set to the minimum step size allowed.
 75     
 76     All values will be rounded to this step size when displayed.
 77   */
 78   step: 0.1,
 79 
 80   // ..........................................................
 81   // INTERNAL PROPERTIES
 82   // 
 83   
 84   displayProperties: 'value minimum maximum'.w(),
 85   
 86   render: function(context, firstTime) {
 87     sc_super();
 88     
 89     var min = this.get('minimum'),
 90         max = this.get('maximum'),
 91         value = this.get('value'),
 92         step = this.get('step');
 93 
 94     // determine the constrained value.  Must fit within min & max
 95     value = Math.min(Math.max(value, min), max);
 96 
 97     // limit to step value
 98     if (!SC.none(step) && step !== 0) {
 99       value = Math.round(value / step) * step;
100     }
101     
102     // determine the percent across
103     if(value!==0) value = Math.floor((value - min) / (max - min) * 100);
104     
105     if(firstTime) {
106       var blankImage = SC.BLANK_IMAGE_URL;
107       context.push('<span class="sc-inner">',
108                     '<span class="sc-leftcap"></span>',
109                     '<span class="sc-rightcap"></span>',
110                     '<img src="', blankImage, 
111                     '" class="sc-handle" style="left: ', value, '%" />',
112                     '</span>');
113     }
114     else {
115       this.$(this.get('handleSelector')).css('left', value + "%");
116     }
117     
118   },
119   
120   _isMouseDown: NO,
121   
122   mouseDown: function(evt) {
123     if (!this.get('isEnabled')) return YES; // nothing to do...
124     this.set('isActive', YES);
125     this._isMouseDown = YES ;
126     return this._triggerHandle(evt, true);
127   },
128   
129   // mouseDragged uses same technique as mouseDown.
130   mouseDragged: function(evt) { 
131     return this._isMouseDown ? this._triggerHandle(evt) : YES; 
132   },
133   
134   // remove active class
135   mouseUp: function(evt) {
136     if (this._isMouseDown) this.set('isActive', NO);
137     var ret = this._isMouseDown ? this._triggerHandle(evt) : YES ;
138     this._isMouseDown = NO;
139     return ret ;
140   },
141   
142   mouseWheel: function(evt) {
143     if (!this.get('isEnabled')) return YES;
144     var min = this.get('minimum'),
145         max = this.get('maximum'),
146         newVal = this.get('value')+((evt.wheelDeltaX+evt.wheelDeltaY)*0.01),
147         step = this.get('step'),
148         value = Math.round(newVal / step) * step ;
149     if (newVal< min) this.setIfChanged('value', min);
150     else if (newVal> max) this.setIfChanged('value', max);
151     else this.setIfChanged('value', newVal);
152     return YES ;  
153   },
154   
155   touchStart: function(evt){
156     return this.mouseDown(evt);
157   },
158   
159   touchEnd: function(evt){
160     return this.mouseUp(evt);
161   },
162   
163   touchesDragged: function(evt){
164     return this.mouseDragged(evt);
165   },
166   
167   /** @private
168     Updates the handle based on the mouse location of the handle in the
169     event.
170   */
171   _triggerHandle: function(evt, firstEvent) {
172     var width = this.get('frame').width,
173         min = this.get('minimum'), max=this.get('maximum'),  
174         step = this.get('step'), v=this.get('value'), loc;
175         
176     if(firstEvent){    
177       loc = this.convertFrameFromView({ x: evt.pageX }).x;
178       this._evtDiff = evt.pageX - loc;
179     }else{
180       loc = evt.pageX-this._evtDiff;
181     }
182     
183     // constrain loc to 8px on either side (left to allow knob overhang)
184     loc = Math.max(Math.min(loc,width-8), 8) - 8;
185     width -= 16 ; // reduce by margin
186     
187     // convert to percentage
188     loc = loc / width ;
189     
190     // convert to value using minimum/maximum then constrain to steps
191     loc = min + ((max-min)*loc);
192     if (step !== 0) loc = Math.round(loc / step) * step ;
193 
194     // if changes by more than a rounding amount, set v.
195     if (Math.abs(v-loc)>=0.01) this.set('value', loc); // adjust 
196     return YES ;
197   },
198   
199   /** tied to the isEnabled state */
200   acceptsFirstResponder: function() {
201     if(!SC.SAFARI_FOCUS_BEHAVIOR) return this.get('isEnabled');
202     else return NO;
203   }.property('isEnabled'),
204   
205   willBecomeKeyResponderFrom: function(keyView) {
206     // focus the text field.
207     if (!this._isFocused) {
208       this._isFocused = YES ;
209       this.becomeFirstResponder();
210       if (this.get('isVisibleInWindow')) {
211         this.$()[0].focus();
212       }
213     }
214   },
215   
216   willLoseKeyResponderTo: function(responder) {
217     if (this._isFocused) this._isFocused = NO ;
218   },
219   
220   keyDown: function(evt) {
221 
222      // handle tab key
223      if (evt.which === 9) {
224        var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
225        if(view) view.becomeFirstResponder();
226        else evt.allowDefault(); 
227        return YES ; // handled
228      }
229      
230      if (evt.which === 37 || evt.which === 38 || evt.which === 39 || evt.which === 40){
231        var min = this.get('minimum'),max=this.get('maximum'),
232           step = this.get('step'),
233           size = max-min, val=0, calculateStep;
234      
235        if (evt.which === 37 || evt.which === 38 ){
236          if(step === 0){
237            if(size<100){
238              val = this.get('value')-1;
239            }else{
240              calculateStep = Math.abs(size/100);
241              if(calculateStep<2) calculateStep =2;
242              val = this.get('value')-Math.abs(size/100);
243            }
244          }else{
245            val = this.get('value')-step;
246          }
247        }
248        if (evt.which === 39 || evt.which === 40 ){
249            if(step === 0){
250               if(size<100){
251                 val = this.get('value') + 2;
252               }else{
253                 calculateStep = Math.abs(size/100);
254                 if(calculateStep<2) calculateStep =2;
255                 val = this.get('value')+calculateStep;
256               }
257             }else{
258               val = this.get('value')+step;
259             }       
260        }
261        if(val>=min && val<=max) this.set('value', val);
262      }
263      //handle arrows
264 
265      // validate keyDown...
266      // if (this.performValidateKeyDown(evt)) {
267      //    this._isKeyDown = YES ;
268      //    evt.allowDefault(); 
269      //  } else {
270      //    evt.stop();
271      //  }
272      SC.RunLoop.begin().end();
273      return YES; 
274    },
275 
276   contentPropertyDidChange: function(target, key) {
277     var content = this.get('content');
278     this.beginPropertyChanges()
279       .updatePropertyFromContent('value', key, 'contentValueKey', content)
280       .updatePropertyFromContent('minimum', key, 'contentMinimumKey', content)
281       .updatePropertyFromContent('maximum', key, 'contentMaximumKey', content)
282       .updatePropertyFromContent('isIndeterminate', key, 'contentIsIndeterminateKey', content)
283     .endPropertyChanges();
284   }  
285   
286 });
