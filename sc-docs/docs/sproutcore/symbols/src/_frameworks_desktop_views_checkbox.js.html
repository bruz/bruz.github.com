  8 /** @class
  9 
 10   Renders a checkbox button view specifically.
 11   
 12   This view is basically a button view preconfigured to generate the correct
 13   HTML and to set to use a TOGGLE_BEHAVIOR for its buttons.
 14   
 15   This view renders a simulated checkbox that can display a mixed state and 
 16   has other features not found in platform-native controls.  
 17   
 18   @extends SC.FieldView
 19   @since SproutCore 1.0
 20 */
 21 SC.CheckboxView = SC.ButtonView.extend(SC.StaticLayout, SC.Button,
 22   /** @scope SC.CheckboxView.prototype */ {
 23 
 24   classNames: ['sc-checkbox-view'],
 25   tagName: 'label',
 26 
 27   /* Ellipsis is disabled by default to allow multiline text */
 28   needsEllipsis: NO,
 29 
 30   render: function(context, firstTime) {
 31     var dt, elem,
 32         value = this.get('value'),
 33         ariaValue = value === SC.MIXED_MODE ? 
 34                 'mixed' : (value === this.get('toggleOnValue') ? 
 35                     'true': 'false');
 36     
 37     // add checkbox -- set name to view guid to separate it from others
 38     if (firstTime) {
 39       var blank = SC.BLANK_IMAGE_URL,
 40           disabled = this.get('isEnabled') ? '' : 'disabled="disabled"',
 41           guid = SC.guidFor(this);
 42       
 43       context.attr('role', 'checkbox');
 44       dt = this._field_currentDisplayTitle = this.get('displayTitle');
 45 
 46       if(SC.browser.msie) context.attr('for', guid);
 47       context.push('<span class="button" ></span>');
 48       if(this.get('needsEllipsis')){
 49         context.push('<span class="label ellipsis">', dt, '</span>');
 50       }else{
 51         context.push('<span class="label">', dt, '</span>');  
 52       }
 53       context.attr('name', guid);
 54 
 55     // since we don't want to regenerate the contents each time 
 56     // actually search for and update the displayTitle.
 57     } else {
 58       
 59       dt = this.get('displayTitle');
 60       if (dt !== this._field_currentDisplayTitle) {
 61         this._field_currentDisplayTitle = dt;
 62         this.$('span.label').text(dt);
 63       }
 64     }
 65     context.attr('aria-checked', ariaValue);
 66   },
 67   
 68   acceptsFirstResponder: function() {
 69     if(!SC.SAFARI_FOCUS_BEHAVIOR) return this.get('isEnabled');
 70     else return NO;
 71   }.property('isEnabled'),
 72   
 73   
 74   mouseDown: function(evt) {
 75     if(!this.get('isEnabled')) return YES;
 76     this.set('isActive', YES);
 77     this._isMouseDown = YES;
 78     // even if radiobuttons are not set to get firstResponder, allow default 
 79     // action, that way textfields loose focus as expected.
 80     if (evt) evt.allowDefault();
 81     return YES;
 82   },
 83   
 84   mouseUp: function(evt) {
 85     if(!this.get('isEnabled') || 
 86       (evt && evt.target && !this.$().within(evt.target))) {
 87       return YES;
 88     }
 89     var val = this.get('value');
 90     if (val === this.get('toggleOnValue')) {
 91       this.$().attr('aria-checked', 'false');
 92       this.set('value', this.get('toggleOffValue'));
 93     }
 94     else {
 95       this.$().attr('aria-checked', 'true');
 96       this.set('value', this.get('toggleOnValue'));
 97     }
 98     this.set('isActive', NO);
 99     this._isMouseDown = NO;
100     return YES;
101   },
102   
103   
104   touchStart: function(evt) {
105     return this.mouseDown(evt);
106   },
107   
108   touchEnd: function(evt) {
109     return this.mouseUp(evt);
110   }
111     
112 }) ;
