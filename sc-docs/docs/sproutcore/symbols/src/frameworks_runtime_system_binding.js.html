 57 /**
 58   @namespace 
 59   
 60   A binding simply connects the properties of two objects so that whenever the
 61   value of one property changes, the other property will be changed also.  You
 62   do not usually work with Binding objects directly but instead describe
 63   bindings in your class definition using something like:
 64   
 65   {{{
 66     valueBinding: "MyApp.someController.title"
 67   }}}
 68     
 69   This will create a binding from "MyApp.someController.title" to the "value"
 70   property of your object instance automatically.  Now the two values will be
 71   kept in sync.
 72   
 73   h2. Customizing Your Bindings
 74   
 75   In addition to synchronizing values, bindings can also perform some basic 
 76   transforms on values.  These transforms can help to make sure the data fed 
 77   into one object always meets the expectations of that object regardless of
 78   what the other object outputs.
 79   
 80   To customize a binding, you can use one of the many helper methods defined 
 81   on SC.Binding like so:
 82   
 83   {{{
 84     valueBinding: SC.Binding.single("MyApp.someController.title") 
 85   }}}
 86     
 87   This will create a binding just like the example above, except that now the
 88   binding will convert the value of MyApp.someController.title to a single 
 89   object (removing any arrays) before applying it to the "value" property of
 90   your object.
 91   
 92   You can also chain helper methods to build custom bindings like so:
 93   
 94   {{{
 95     valueBinding: SC.Binding.single("MyApp.someController.title").notEmpty("(EMPTY)")
 96   }}}
 97     
 98   This will force the value of MyApp.someController.title to be a single value
 99   and then check to see if the value is "empty" (null, undefined, empty array,
100   or an empty string).  If it is empty, the value will be set to the string 
101   "(EMPTY)".
102   
103   h2. One Way Bindings
104   
105   One especially useful binding customization you can use is the oneWay() 
106   helper.  This helper tells SproutCore that you are only interested in 
107   receiving changes on the object you are binding from.  For example, if you
108   are binding to a preference and you want to be notified if the preference 
109   has changed, but your object will not be changing the preference itself, you
110   could do:
111   
112   {{{
113     bigTitlesBinding: SC.Binding.oneWay("MyApp.preferencesController.bigTitles")
114   }}}
115     
116   This way if the value of MyApp.preferencesController.bigTitles changes the
117   "bigTitles" property of your object will change also.  However, if you 
118   change the value of your "bigTitles" property, it will not update the 
119   preferencesController.
120   
121   One way bindings are almost twice as fast to setup and twice as fast to 
122   execute because the binding only has to worry about changes to one side.
123   
124   You should consider using one way bindings anytime you have an object that 
125   may be created frequently and you do not intend to change a property; only 
126   to monitor it for changes. (such as in the example above).
127       
128   h2. Adding Custom Transforms
129   
130   In addition to using the standard helpers provided by SproutCore, you can 
131   also defined your own custom transform functions which will be used to 
132   convert the value.  To do this, just define your transform function and add
133   it to the binding with the transform() helper.  The following example will 
134   not allow Integers less than ten.  Note that it checks the value of the 
135   bindings and allows all other values to pass:
136   
137   {{{
138     valueBinding: SC.Binding.transform(function(value, binding) {
139       return ((SC.typeOf(value) === SC.T_NUMBER) && (value < 10)) ? 10 : value;      
140     }).from("MyApp.someController.value")
141   }}}
142   
143   If you would like to instead use this transform on a number of bindings,
144   you can also optionally add your own helper method to SC.Binding.  This
145   method should simply return the value of this.transform(). The example 
146   below adds a new helper called notLessThan() which will limit the value to
147   be not less than the passed minimum:
148   
149   {{{
150     SC.Binding.notLessThan = function(minValue) {
151       return this.transform(function(value, binding) {
152         return ((SC.typeOf(value) === SC.T_NUMBER) && (value < minValue)) ? minValue : value ;
153       }) ;
154     } ;
155   }}}
156   
157   You could specify this in your core.js file, for example.  Then anywhere in 
158   your application you can use it to define bindings like so:
159   
160   {{{
161     valueBinding: SC.Binding.from("MyApp.someController.value").notLessThan(10)
162   }}}
163   
164   Also, remember that helpers are chained so you can use your helper along with
165   any other helpers.  The example below will create a one way binding that 
166   does not allow empty values or values less than 10:
167   
168   {{{
169     valueBinding: SC.Binding.oneWay("MyApp.someController.value").notEmpty().notLessThan(10)
170   }}}
171   
172   Note that the built in helper methods all allow you to pass a "from" 
173   property path so you don't have to use the from() helper to set the path.  
174   You can do the same thing with your own helper methods if you like, but it 
175   is not required.
176   
177   h2. Creating Custom Binding Templates
178   
179   Another way you can customize bindings is to create a binding template.  A
180   template is simply a binding that is already partially or completely 
181   configured.  You can specify this template anywhere in your app and then use 
182   it instead of designating your own custom bindings.  This is a bit faster on
183   app startup but it is mostly useful in making your code less verbose.
184   
185   For example, let's say you will be frequently creating one way, not empty 
186   bindings that allow values greater than 10 throughout your app.  You could
187   create a binding template in your core.js like this:
188   
189   {{{
190     MyApp.LimitBinding = SC.Binding.oneWay().notEmpty().notLessThan(10);
191   }}}
192   
193   Then anywhere you want to use this binding, just refer to the template like 
194   so:
195 
196   {{{
197     valueBinding: MyApp.LimitBinding.beget("MyApp.someController.value")
198   }}}
199     
200   Note that when you use binding templates, it is very important that you 
201   always start by using beget() to extend the template.  If you do not do 
202   this, you will end up using the same binding instance throughout your app 
203   which will lead to erratic behavior.
204   
205   h2. How to Manually Activate a Binding
206 
207   All of the examples above show you how to configure a custom binding, but 
208   the result of these customizations will be a binding template, not a fully 
209   active binding.  The binding will actually become active only when you 
210   instantiate the object the binding belongs to.  It is useful however, to 
211   understand what actually happens when the binding is activated.
212   
213   For a binding to function it must have at least a "from" property and a "to"
214   property.  The from property path points to the object/key that you want to
215   bind from while the to path points to the object/key you want to bind to.  
216   
217   When you define a custom binding, you are usually describing the property 
218   you want to bind from (such as "MyApp.someController.value" in the examples
219   above).  When your object is created, it will automatically assign the value
220   you want to bind "to" based on the name of your binding key.  In the 
221   examples above, during init, SproutCore objects will effectively call 
222   something like this on your binding:
223   
224   {{{
225     binding = this.valueBinding.beget().to("value", this) ;
226   }}}
227     
228   This creates a new binding instance based on the template you provide, and 
229   sets the to path to the "value" property of the new object.  Now that the 
230   binding is fully configured with a "from" and a "to", it simply needs to be
231   connected to become active.  This is done through the connect() method:
232   
233   {{{
234     binding.connect() ;
235   }}}
236     
237   Now that the binding is connected, it will observe both the from and to side 
238   and relay changes.
239   
240   If you ever needed to do so (you almost never will, but it is useful to 
241   understand this anyway), you could manually create an active binding by 
242   doing the following:
243   
244   {{{
245     SC.Binding.from("MyApp.someController.value")
246      .to("MyApp.anotherObject.value")
247      .connect();
248   }}}
249      
250   You could also use the bind() helper method provided by SC.Object. (This is 
251   the same method used by SC.Object.init() to setup your bindings):
252   
253   {{{
254     MyApp.anotherObject.bind("value", "MyApp.someController.value") ;
255   }}}
256 
257   Both of these code fragments have the same effect as doing the most friendly
258   form of binding creation like so:
259   
260   {{{
261     MyApp.anotherObject = SC.Object.create({
262       valueBinding: "MyApp.someController.value",
263       
264       // OTHER CODE FOR THIS OBJECT...
265       
266     }) ;
267   }}}
268   
269   SproutCore's built in binding creation method make it easy to automatically
270   create bindings for you.  You should always use the highest-level APIs 
271   available, even if you understand how to it works underneath.
272   
273   @since SproutCore 1.0
274 */

277   /**
278     This is the core method you use to create a new binding instance.  The
279     binding instance will have the receiver instance as its parent which means
280     any configuration you have there will be inherited.  
281     
282     The returned instance will also have its parentBinding property set to the 
283     receiver.
284 
285     @param fromPath {String} optional from path.
286     @returns {SC.Binding} new binding instance
287   */
288   beget: function(fromPath) {
289     var ret = SC.beget(this) ;
290     ret.parentBinding = this;
291     if (fromPath !== undefined) ret = ret.from(fromPath) ;
292     return ret ;
293   },

295   /**
296     Returns a builder function for compatibility.  
297   */
298   builder: function() {
299     var binding = this,
300         ret = function(fromProperty) { return binding.beget().from(fromProperty); };
301     ret.beget = function() { return binding.beget(); } ;
302     return ret ;
303   },

305   /**
306     This will set "from" property path to the specified value.  It will not
307     attempt to resolve this property path to an actual object/property tuple
308     until you connect the binding.
309 
310     The binding will search for the property path starting at the root level 
311     unless you specify an alternate root object as the second paramter to this 
312     method.  Alternatively, you can begin your property path with either "." or
313     "*", which will use the root object of the to side be default.  This special
314     behavior is used to support the high-level API provided by SC.Object.
315     
316     @param propertyPath {String|Tuple} A property path or tuple
317     @param root {Object} optional root object to use when resolving the path.
318     @returns {SC.Binding} this
319   */
320   from: function(propertyPath, root) {
321     
322     // if the propertyPath is null/undefined, return this.  This allows the
323     // method to be called from other methods when the fromPath might be 
324     // optional. (cf single(), multiple())
325     if (!propertyPath) return this ;
326     
327     // beget if needed.
328     var binding = (this === SC.Binding) ? this.beget() : this ;
329     binding._fromPropertyPath = propertyPath ;
330     binding._fromRoot = root ;
331     binding._fromTuple = null ;
332     return binding ;
333   },
334   
335   /**
336    This will set the "to" property path to the specified value.  It will not 
337    attempt to reoslve this property path to an actual object/property tuple
338    until you connect the binding.
339     
340     @param propertyPath {String|Tuple} A property path or tuple
341     @param root {Object} optional root object to use when resolving the path.
342     @returns {SC.Binding} this
343   */
344   to: function(propertyPath, root) {
345     // beget if needed.
346     var binding = (this === SC.Binding) ? this.beget() : this ;
347     binding._toPropertyPath = propertyPath ;
348     binding._toRoot = root ;
349     binding._toTuple = null ; // clear out any existing one.
350     return binding ;
351   },
352 
353   /**
354     Attempts to connect this binding instance so that it can receive and relay
355     changes.  This method will raise an exception if you have not set the 
356     from/to properties yet.
357     
358     @returns {SC.Binding} this
359   */
360   connect: function() {
361     // If the binding is already connected, do nothing.
362     if (this.isConnected) return this ;
363     this.isConnected = YES ;
364     this._connectionPending = YES ; // its connected but not really...    
365     this._syncOnConnect = YES ;
366     SC.Binding._connectQueue.add(this) ;
367     return this; 
368   },
369   
370   /** @private
371     Actually connects the binding.  This is done at the end of the runloop
372     to give you time to setup your entire object graph before the bindings 
373     try to activate.
374   */
375   _connect: function() {
376     if (!this._connectionPending) return; //nothing to do
377     this._connectionPending = NO ;
378 
379     var path, root,
380         bench = SC.BENCHMARK_BINDING_SETUP;
381 
382     if (bench) SC.Benchmark.start("SC.Binding.connect()");
383     
384     // try to connect the from side.
385     // as a special behavior, if the from property path begins with either a 
386     // . or * and the fromRoot is null, use the toRoot instead.  This allows 
387     // for support for the SC.Object shorthand:
388     //
389     // contentBinding: "*owner.value"
390     //
391     path = this._fromPropertyPath; root = this._fromRoot ;
392     
393     if (typeof path === "string") {
394       
395       // if the first character is a '.', this is a static path.  make the 
396       // toRoot the default root.
397       if (path.indexOf('.') === 0) {
398         path = path.slice(1);
399         if (!root) root = this._toRoot ;
400         
401       // if the first character is a '*', then setup a tuple since this is a 
402       // chained path.
403       } else if (path.indexOf('*') === 0) {
404         path = [this._fromRoot || this._toRoot, path.slice(1)] ;
405         root = null ;
406       }
407     }
408     this._fromObserverData = [path, this, this.fromPropertyDidChange, root];
409     SC.Observers.addObserver.apply(SC.Observers, this._fromObserverData);
410     
411     // try to connect the to side
412     if (!this._oneWay) {
413       path = this._toPropertyPath; root = this._toRoot ;
414       this._toObserverData = [path, this, this.toPropertyDidChange, root];
415       SC.Observers.addObserver.apply(SC.Observers, this._toObserverData);
416     }
417 
418     if (bench) SC.Benchmark.end("SC.Binding.connect()");
419 
420     // now try to sync if needed
421     if (this._syncOnConnect) {
422       this._syncOnConnect = NO ;
423       if (bench) SC.Benchmark.start("SC.Binding.connect().sync");
424       this.sync();
425       if (bench) SC.Benchmark.end("SC.Binding.connect().sync");
426     }
427   },
428   
429   /**
430     Disconnects the binding instance.  Changes will no longer be relayed.  You
431     will not usually need to call this method.
432     
433     @returns {SC.Binding} this
434   */
435   disconnect: function() {
436     if (!this.isConnected) return this; // nothing to do.
437     
438     // if connection is still pending, just cancel
439     if (this._connectionPending) {
440       this._connectionPending = NO ;
441       
442     // connection is completed, disconnect.
443     } else {
444       SC.Observers.removeObserver.apply(SC.Observers, this._fromObserverData);
445       if (!this._oneWay) {
446         SC.Observers.removeObserver.apply(SC.Observers, this._toObserverData);
447       }
448     }
449     
450     this.isConnected = NO ;
451     return this ;  
452   },

454   /**
455     Invoked whenever the value of the "from" property changes.  This will mark
456     the binding as dirty if the value has changed.
457   */
458   fromPropertyDidChange: function(target, key) {
459     var v = target ? target.get(key) : null;
460 
461     //console.log("fromPropertyDidChange: %@ v = %@".fmt(this, v)) ;
462     
463     // if the new value is different from the current binding value, then 
464     // schedule to register an update.
465     if (v !== this._bindingValue || key === '[]') {
466 
467       this._setBindingValue(target, key) ;
468       this._changePending = YES ;
469       SC.Binding._changeQueue.add(this) ; // save for later.  
470     }
471   },
472 
473   /**
474     Invoked whenever the value of the "to" property changes.  This will mark the
475     binding as dirty only if:
476     
477     - the binding is not one way
478     - the value does not match the stored transformedBindingValue
479     
480     if the value does not match the transformedBindingValue, then it will 
481     become the new bindingValue. 
482   */
483   toPropertyDidChange: function(target, key) {
484     if (this._oneWay) return; // nothing to do
485     
486     var v = target.get(key) ;
487     
488     // if the new value is different from the current binding value, then 
489     // schedule to register an update.
490     if (v !== this._transformedBindingValue) {
491       this._setBindingValue(target, key) ;
492       this._changePending = YES ;
493       SC.Binding._changeQueue.add(this) ; // save for later.  
494     }
495   },
496   
497   /** @private
498     Saves the source location for the binding value.  This will be used later
499     to actually update the binding value.
500   */
501   _setBindingValue: function(source, key) {
502     this._bindingSource = source;
503     this._bindingKey    = key;
504   },
505   
506   /** @private 
507     Updates the binding value from the current binding source if needed.  This
508     should be called just before using this._bindingValue.
509   */
510   _computeBindingValue: function() {
511     var source = this._bindingSource,
512         key    = this._bindingKey,
513         v, idx;
514     
515     this._bindingValue = v = (source ? source.getPath(key) : null);
516     
517     // apply any transforms to get the to property value also
518     var transforms = this._transforms;
519     if (transforms) {
520       var len = transforms.length,
521           transform;
522       for(idx=0;idx<len;idx++) {
523         transform = transforms[idx] ;
524         v = transform(v, this) ;
525       }
526     }
527 
528     // if error objects are not allowed, and the value is an error, then
529     // change it to null.
530     if (this._noError && SC.typeOf(v) === SC.T_ERROR) v = null ;
531     
532     this._transformedBindingValue = v;
533   },

535   _connectQueue: SC.CoreSet.create(),
536   _alternateConnectQueue: SC.CoreSet.create(),
537   _changeQueue: SC.CoreSet.create(),
538   _alternateChangeQueue: SC.CoreSet.create(),
539   _changePending: NO,

541   /**
542     Call this method on SC.Binding to flush all bindings with changed pending.
543     
544     @returns {Boolean} YES if changes were flushed.
545   */
546   flushPendingChanges: function() {
547     
548     // don't allow flushing more than one at a time
549     if (this._isFlushing) return NO; 
550     this._isFlushing = YES ;
551     SC.Observers.suspendPropertyObserving();
552 
553     var didFlush = NO,
554         log = SC.LOG_BINDINGS,
555         // connect any bindings
556         queue, binding ;
557     while((queue = this._connectQueue).length >0) {
558       this._connectQueue = this._alternateConnectQueue ;
559       this._alternateConnectQueue = queue ;
560       while(binding = queue.pop()) binding._connect() ;
561     }
562     
563     // loop through the changed queue...
564     while ((queue = this._changeQueue).length > 0) {
565       if (log) console.log("Begin: Trigger changed bindings") ;
566       
567       didFlush = YES ;
568       
569       // first, swap the change queues.  This way any binding changes that
570       // happen while we flush the current queue can be queued up.
571       this._changeQueue = this._alternateChangeQueue ;
572       this._alternateChangeQueue = queue ;
573       
574       // next, apply any bindings in the current queue.  This may cause 
575       // additional bindings to trigger, which will end up in the new active 
576       // queue.
577       while(binding = queue.pop()) binding.applyBindingValue() ;
578       
579       // now loop back and see if there are additional changes pending in the
580       // active queue.  Repeat this until all bindings that need to trigger 
581       // have triggered.
582       if (log) console.log("End: Trigger changed bindings") ;
583     }
584     
585     // clean up
586     this._isFlushing = NO ;
587     SC.Observers.resumePropertyObserving();
588 
589     return didFlush ;
590   },

592   /**
593     This method is called at the end of the Run Loop to relay the changed 
594     binding value from one side to the other.
595   */
596   applyBindingValue: function() {
597     this._changePending = NO ;
598 
599     // compute the binding targets if needed.
600     this._computeBindingTargets() ;
601     this._computeBindingValue();
602     
603     var v = this._bindingValue,
604         tv = this._transformedBindingValue,
605         bench = SC.BENCHMARK_BINDING_NOTIFICATIONS,
606         log = SC.LOG_BINDINGS ; 
607     
608     // the from property value will always be the binding value, update if 
609     // needed.
610     if (!this._oneWay && this._fromTarget) {
611       if (log) console.log("%@: %@ -> %@".fmt(this, v, tv)) ;
612       if (bench) SC.Benchmark.start(this.toString() + "->") ;
613       this._fromTarget.setPathIfChanged(this._fromPropertyKey, v) ;
614       if (bench) SC.Benchmark.end(this.toString() + "->") ;
615     }
616     
617     // update the to value with the transformed value if needed.
618     if (this._toTarget) {
619       if (log) console.log("%@: %@ <- %@".fmt(this, v, tv)) ;
620       if (bench) SC.Benchmark.start(this.toString() + "<-") ;
621       this._toTarget.setPathIfChanged(this._toPropertyKey, tv) ;
622       if (bench) SC.Benchmark.start(this.toString() + "<-") ;
623     }
624   },
625 
626   /**
627     Calling this method on a binding will cause it to check the value of the 
628     from side of the binding matches the current expected value of the 
629     binding. If not, it will relay the change as if the from side's value has 
630     just changed.
631     
632     This method is useful when you are dynamically connecting bindings to a 
633     network of objects that may have already been initialized.
634   */
635   sync: function() {
636 
637     // do nothing if not connected
638     if (!this.isConnected) return this;
639     
640     // connection is pending, just note that we should sync also
641     if (this._connectionPending) {
642       this._syncOnConnect = YES ;
643       
644     // we are connected, go ahead and sync
645     } else {
646       this._computeBindingTargets() ;
647       var target = this._fromTarget,
648           key = this._fromPropertyKey ;
649       if (!target || !key) return this ; // nothing to do
650 
651       // get the new value
652       var v = target.getPath(key) ;
653 
654       // if the new value is different from the current binding value, then 
655       // schedule to register an update.
656       if (v !== this._bindingValue || key === '[]') {
657         this._setBindingValue(target, key) ;
658         this._changePending = YES ;
659         SC.Binding._changeQueue.add(this) ; // save for later.  
660       }
661     }
662     
663     return this ;
664   },
665   
666   // set if you call sync() when the binding connection is still pending.
667   _syncOnConnect: NO,
668   
669   _computeBindingTargets: function() {
670     if (!this._fromTarget) {
671 
672       var path, root, tuple ;
673       
674       // if the fromPropertyPath begins with a . or * then we may use the 
675       // toRoot as the root object.  Similar code exists in connect() so if 
676       // you make a change to one be sure to update the other.
677       path = this._fromPropertyPath; root = this._fromRoot ;
678       if (typeof path === "string") {
679         
680         // static path beginning with the toRoot
681         if (path.indexOf('.') === 0) {
682           path = path.slice(1) ; // remove the .
683           if (!root) root = this._toRoot; // use the toRoot optionally
684           
685         // chained path beginning with toRoot.  Setup a tuple
686         } else if (path.indexOf('*') === 0) {
687           path = [root || this._toRoot, path.slice(1)];
688           root = null ;
689         }
690       }
691       
692       tuple = SC.tupleForPropertyPath(path, root) ;
693       if (tuple) {
694         this._fromTarget = tuple[0]; this._fromPropertyKey = tuple[1] ;
695       }
696     }
697 
698     if (!this._toTarget) {
699       path = this._toPropertyPath; root = this._toRoot ;
700       tuple = SC.tupleForPropertyPath(path, root) ;
701       if (tuple) {
702         this._toTarget = tuple[0]; this._toPropertyKey = tuple[1] ;
703       }
704     }
705   },

707   /**
708     Configures the binding as one way.  A one-way binding will relay changes
709     on the "from" side to the "to" side, but not the other way around.  This
710     means that if you change the "to" side directly, the "from" side may have 
711     a different value.
712     
713     @param fromPath {String} optional from path to connect.
714     @param aFlag {Boolean} Optionally pass NO to set the binding back to two-way
715     @returns {SC.Binding} this
716   */
717   oneWay: function(fromPath, aFlag) {
718     
719     // If fromPath is a bool but aFlag is undefined, swap.
720     if ((aFlag === undefined) && (SC.typeOf(fromPath) === SC.T_BOOL)) {
721       aFlag = fromPath; fromPath = null ;
722     }
723     
724     // beget if needed.
725     var binding = this.from(fromPath) ;
726     if (binding === SC.Binding) binding = binding.beget() ;
727     binding._oneWay = (aFlag === undefined) ? YES : aFlag ;
728     return binding ;
729   },
730   
731   /**
732     Adds the specified transform function to the array of transform functions.
733     
734     The function you pass must have the following signature:
735     
736     {{{
737       function(value) {} ;
738     }}}
739     
740     It must return either the transformed value or an error object.  
741         
742     Transform functions are chained, so they are called in order.  If you are
743     extending a binding and want to reset the transforms, you can call
744     resetTransform() first.
745     
746     @param transformFunc {Function} the transform function.
747     @returns {SC.Binding} this
748   */
749   transform: function(transformFunc) {
750     var binding = (this === SC.Binding) ? this.beget() : this ;
751     var t = binding._transforms ;
752     
753     // clone the transform array if this comes from the parent
754     if (t && (t === binding.parentBinding._transform)) {
755       t = binding._transforms = t.slice() ;
756     }
757     
758     // create the transform array if needed.
759     if (!t) t = binding._transforms = [] ;
760     
761     // add the transform function
762     t.push(transformFunc) ;
763     return binding;
764   },
765   
766   /**
767     Resets the transforms for the binding.  After calling this method the 
768     binding will no longer transform values.  You can then add new transforms
769     as needed.
770   
771     @returns {SC.Binding} this
772   */
773   resetTransforms: function() {
774     var binding = (this === SC.Binding) ? this.beget() : this ;
775     binding._transforms = null ; return binding ;
776   },

778   /**
779     Specifies that the binding should not return error objects.  If the value
780     of a binding is an Error object, it will be transformed to a null value
781     instead.
782     
783     Note that this is not a transform function since it will be called at the
784     end of the transform chain.
785     
786     @param fromPath {String} optional from path to connect.
787     @param aFlag {Boolean} optionally pass NO to allow error objects again.
788     @returns {SC.Binding} this
789   */
790   noError: function(fromPath, aFlag) {
791     // If fromPath is a bool but aFlag is undefined, swap.
792     if ((aFlag === undefined) && (SC.typeOf(fromPath) === SC.T_BOOL)) {
793       aFlag = fromPath; fromPath = null ;
794     }
795     
796     // beget if needed.
797     var binding = this.from(fromPath) ;
798     if (binding === SC.Binding) binding = binding.beget() ;
799     binding._noError = (aFlag === undefined) ? YES : aFlag ;
800     return binding ;
801   },

803   /**
804     Adds a transform to the chain that will allow only single values to pass.
805     This will allow single values, nulls, and error values to pass through.  If
806     you pass an array, it will be mapped as so:
807     
808     {{{
809       [] => null
810       [a] => a
811       [a,b,c] => Multiple Placeholder
812     }}}
813     
814     You can pass in an optional multiple placeholder or it will use the 
815     default.
816     
817     Note that this transform will only happen on forwarded valued.  Reverse
818     values are send unchanged.
819     
820     @param fromPath {String} from path or null
821     @param placeholder {Object} optional placeholder value.
822     @returns {SC.Binding} this
823   */
824   single: function(fromPath, placeholder) {
825     if (placeholder === undefined) {
826       placeholder = SC.MULTIPLE_PLACEHOLDER ;
827     }
828     return this.from(fromPath).transform(function(value, isForward) {
829       if (value && value.isEnumerable) {
830         var len = value.get('length');
831         value = (len>1) ? placeholder : (len<=0) ? null : value.firstObject();
832       }
833       return value ;
834     }) ;
835   },

837   /** 
838     Adds a transform that will return the placeholder value if the value is 
839     null, undefined, an empty array or an empty string.  See also notNull().
840     
841     @param fromPath {String} from path or null
842     @param placeholder {Object} optional placeholder.
843     @returns {SC.Binding} this
844   */
845   notEmpty: function(fromPath, placeholder) {
846     if (placeholder === undefined) placeholder = SC.EMPTY_PLACEHOLDER ;
847     return this.from(fromPath).transform(function(value, isForward) {
848       if (SC.none(value) || (value === '') || (SC.isArray(value) && value.length === 0)) {
849         value = placeholder ;
850       }
851       return value ;
852     }) ;
853   },

856     Adds a transform that will return the placeholder value if the value is
857     null.  Otherwise it will passthrough untouched.  See also notEmpty().
858     
859     @param fromPath {String} from path or null
860     @param placeholder {Object} optional placeholder;
861     @returns {SC.Binding} this
862   */
863   notNull: function(fromPath, placeholder) {
864     if (placeholder === undefined) placeholder = SC.EMPTY_PLACEHOLDER ;
865     return this.from(fromPath).transform(function(value, isForward) {
866       if (SC.none(value)) value = placeholder ;
867       return value ;
868     }) ;
869   },

871   /** 
872     Adds a transform that will convert the passed value to an array.  If 
873     the value is null or undefined, it will be converted to an empty array.
874 
875     @param fromPath {String} optional from path
876     @returns {SC.Binding} this
877   */
878   multiple: function(fromPath) {

885   /**
886     Adds a transform to convert the value to a bool value.  If the value is
887     an array it will return YES if array is not empty.  If the value is a string
888     it will return YES if the string is not empty.
889   
890     @param fromPath {String} optional from path
891     @returns {SC.Binding} this
892   */
893   bool: function(fromPath) {
894     return this.from(fromPath).transform(function(v) {
895       var t = SC.typeOf(v) ;
896       if (t === SC.T_ERROR) return v ;
897       return (t == SC.T_ARRAY) ? (v.length > 0) : (v === '') ? NO : !!v ;
898     }) ;

901   /**
902     Adds a transform that forwards the logical 'AND' of values at 'pathA' and
903     'pathB' whenever either source changes.  Note that the transform acts strictly
904     as a one-way binding, working only in the direction
905     
906       'pathA' AND 'pathB' --> value  (value returned is the result of ('pathA' && 'pathB'))
907 
908     Usage example where a delete button's 'isEnabled' value is determined by whether
909     something is selected in a list and whether the current user is allowed to delete:
910     
911       deleteButton: SC.ButtonView.design({
912         isEnabledBinding: SC.Binding.and('MyApp.itemsController.hasSelection', 'MyApp.userController.canDelete')
913       })
914 
915   */
916   and: function(pathA, pathB) {
917 
918     // create an object to do the logical computation
919     var gate = SC.Object.create({
920       valueABinding: pathA,
921       valueBBinding: pathB,
922 
923       and: function() {
924         return (this.get('valueA') && this.get('valueB'));
925       }.property('valueA', 'valueB').cacheable()
926     });
927 
928     // add a transform that depends on the result of that computation.
929     return this.from('and', gate).oneWay();
930   },
931 
932   /**
933     Adds a transform that forwards the 'OR' of values at 'pathA' and
934     'pathB' whenever either source changes.  Note that the transform acts strictly
935     as a one-way binding, working only in the direction
936 
937       'pathA' AND 'pathB' --> value  (value returned is the result of ('pathA' || 'pathB'))
938 
939   */
940   or: function(pathA, pathB) {
941     
942     // create an object to the logical computation
943     var gate = SC.Object.create({
944       valueABinding: pathA,
945       valueBBinding: pathB,
946       
947       or: function() {
948         return (this.get('valueA') || this.get('valueB'));
949       }.property('valueA', 'valueB').cacheable()
950     });
951     
952     return this.from('or', gate).oneWay();
953   },

955   /**
956     Adds a transform to convert the value to the inverse of a bool value.  This
957     uses the same transform as bool() but inverts it.
958     
959     @param fromPath {String} optional from path
960     @returns {SC.Binding} this
961   */
962   not: function(fromPath) {
963     return this.from(fromPath).transform(function(v) {
964       var t = SC.typeOf(v) ;
965       if (t === SC.T_ERROR) return v ;
966       return !((t == SC.T_ARRAY) ? (v.length > 0) : (v === '') ? NO : !!v) ;
967     }) ;
968   },

973     @returns {SC.Binding} this
974   */
975   isNull: function(fromPath) {

983     var from = this._fromRoot ? "<%@>:%@".fmt(this._fromRoot,this._fromPropertyPath) : this._fromPropertyPath;
984 
985     var to = this._toRoot ? "<%@>:%@".fmt(this._toRoot,this._toPropertyPath) : this._toPropertyPath;

987     var oneWay = this._oneWay ? '[oneWay]' : '';
988     return "SC.Binding%@(%@ -> %@)%@".fmt(SC.guidFor(this), from, to, oneWay);
