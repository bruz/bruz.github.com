  8 /**
  9   @class
 10   
 11   The inline text editor is used to display an editable area for controls 
 12   that are not always editable such as label views and source list views.
 13   
 14   You generally will not use the inline editor directly but instead will
 15   invoke beginEditing() and endEditing() on the views you are 
 16   editing. If you would like to use the inline editor for your own views, 
 17   you can do that also by using the editing API described here.
 18   
 19   h2. Using the Inline Editor in Your Own Views
 20 
 21   If you need to use the inline editor with custom views you have written,
 22   you will need to work with the class methods to begin, commit, and discard
 23   editing changes.
 24   
 25   h3. Starting the Editor
 26   
 27   The inline editor works by positioning itself over the top of your view 
 28   with the same offset, width, and font information.  As the user types, the
 29   field will automatically resize vertically to make room for the user's text.
 30   
 31   To activate the inline editor you must call beginEdition() with at least 
 32   the target view you want the editor to position itself on top of:
 33   
 34   {{{
 35     SC.InlineTextFieldView.beginEditing({
 36       target: view, validator: validator
 37     }) ;
 38   }}}
 39 
 40   You can pass a variety of options to this method to configure the inline
 41   editor behavior, including:
 42 
 43   - *frame* The editors initial frame in viewport coordinates. (REQ)
 44   - *delegate* Delegate to receive update notices. (REQ)
 45   - *value* Initial value of the edit field.
 46   - *exampleElement* A DOM element to use when copying styles.
 47   - *multiline* If YES then the hitting return will add to the value instead of exiting the inline editor.
 48   - *selectedRange* The range of text that should be selected.  If omitted, then the insertion point will be placed at the end of the value.
 49   - *commitOnBlur* If YES then blurring will commit the value, otherwise it will discard the current value.  Defaults to YES.
 50   - *validator* Validator to be attached to the field.
 51   
 52   If the inline editor is currently in use elsewhere, it will automatically
 53   close itself over there and begin editing for your view instead.  The 
 54   editor expects your source view to implement the InlineTextFieldViewDelegate
 55   protocol.
 56 
 57   h2. Commiting or Discarding Changes
 58   
 59   Normally the editor will automatically commit or discard its changes 
 60   whenever the user exits the edit mode.  If you need to force the editor to
 61   end editing, you can do so by calling commitEditing() or discardEditing():
 62   
 63   {{{
 64     SC.InlineTextFieldView.commitEditing();
 65     SC.InlineTextFieldView.discardEditing();
 66   }}}
 67   
 68   Both methods will try to end the editing context and will call the 
 69   relevent delegate methods on the delegate you passed to beginEditing().
 70   
 71   Note that it is possible an editor may not be able to commit editing 
 72   changes because either the delegate disallowed it or because its validator
 73   failed.  In this case commitEditing() will return NO.  If you want to
 74   end editing anyway, you can discard the editing changes instead by calling
 75   discardEditing().  This method will generally suceed unless your delegate
 76   refuses it as well.
 77   
 78   @extends SC.TextFieldView
 79   @extends SC.DelegateSupport
 80   @since SproutCore 1.0
 81 */

 90   /**
 91     Invoked by the class method to begin editing on an inline editor.
 92     
 93     You generally should call the class method beginEditing() instead of
 94     this one since it will make sure to create and use the shared editor
 95     instance.
 96 
 97     @params options {Hash} hash of options for editing
 98     @returns {SC.InlineTextFieldView|Boolean} this if editor began editing, 
 99       NO if it failed.
100   */
101   beginEditing: function(options) {
102 
103     // options are required
104     //@if(debug)
105     if (!options) throw "InlineTextField.beginEditing() requires options";
106     //@end
107 
108     // can't begin editing again if already editing
109     if (this.get('isEditing')) return NO ;
110     
111     var layout={}, pane, delLayout, paneElem, del;
112 
113     del = this._delegate = options.delegate ;
114     this.set('delegate', this._delegate);
115     
116     // continue only if the delegate allows it
117     if (!this.invokeDelegateMethod(del, 'inlineEditorShouldBeginEditing', this)) {
118       //@if(debug)
119       SC.Logger.warn('InlineTextField.beginEditing() cannot begin without inlineEditorShouldBeginEditing() on the delegate.');
120       //@end
121       return NO;
122     }
123     this.beginPropertyChanges();
124     
125     this.set('isEditing', YES) ;
126     this.set('escapeHTML', options.escapeHTML) ;
127     
128     this._optframe = options.frame ;
129     this._optIsCollection = options.isCollection;
130     this._exampleElement = options.exampleElement ;
131 
132     if (!this._optframe || !this._delegate) {
133       throw "At least frame and delegate options are required for inline editor";
134     }
135     
136     this._originalValue = options.value || '' ;
137     this._multiline = (options.multiline !== undefined) ? options.multiline : NO ;
138     if (this._multiline) {
139       this.set('isTextArea', YES);
140     } else {
141       this.set('isTextArea', NO);
142     }
143     this._commitOnBlur =  (options.commitOnBlur !== undefined) ? options.commitOnBlur : YES ;
144 
145     // set field values
146     this.set('validator', options.validator) ;
147     this.set('value', this._originalValue) ;
148     //this.set('selectedRange', options.selectedRange || { start: this._originalValue.length, length: 0 }) ;
149     
150     // add to window.
151     
152     pane = this._delegate.pane();
153 
154     layout.height = this._optframe.height;
155     layout.width=this._optframe.width;
156     delLayout = this._delegate.get('layout');
157     paneElem = pane.$()[0];
158     if (this._optIsCollection && delLayout.left) {
159       layout.left=this._optframe.x-delLayout.left-paneElem.offsetLeft-1;
160       if(SC.browser.msie==7) layout.left--;
161     } else {
162       layout.left=this._optframe.x-paneElem.offsetLeft-1;
163       if(SC.browser.msie==7) layout.left--;
164     }
165     
166     if (this._optIsCollection && delLayout.top) {
167       layout.top=this._optframe.y-delLayout.top-paneElem.offsetTop;
168       if(SC.browser.msie==7) layout.top=layout.top-2;
169     } else {
170       layout.top=this._optframe.y-paneElem.offsetTop;
171       if(SC.browser.msie==7) layout.top=layout.top-2;  
172     }
173 
174     this.set('layout', layout);
175   
176     this.set('parentNode', pane);
177     // get style for view.
178    
179     pane.appendChild(this);
180     
181     del = this._delegate ;
182 
183     this._className = this.getDelegateProperty(del,"inlineEditorClassName");
184     if(this._className && !this.hasClassName(this._className)) {
185       this.setClassName(this._className,true);
186     }
187     
188     this.invokeDelegateMethod(del, 'inlineEditorWillBeginEditing', this) ;
189     // this.resizeToFit(this.getFieldValue()) ;
190 
191     this._previousFirstResponder = pane ? pane.get('firstResponder') : null;
192     this.becomeFirstResponder();
193     this.endPropertyChanges() ;
194     
195     // TODO: remove? if(SC.browser.mozilla)this.invokeOnce(this.becomeFirstResponder) ;
196       
197     // Become first responder and notify the delegate after run loop completes
198     this.invokeLast(function() {
199       this.invokeDelegateMethod(del, 'inlineEditorDidBeginEditing', this);
200     });
201     
202     return this;
203   },

242   /** @private
243     Called by commitEditing and discardEditing to actually end editing.
244     
245     @param {String} finalValue that will be set as value
246     @param {Event} evt that triggered the end editing to occur
247     @param {Boolean} didDiscard if called from discardEditing
248     @returns {Boolean} NO if editing did not exit
249   */
250   _endEditing: function(finalValue, evt, didDiscard) {
251     if (!this.get('isEditing')) return YES ;
252     
253     // get permission from the delegate.
254     var del = this._delegate ;
255     if (!this.invokeDelegateMethod(del, 'inlineEditorShouldEndEditing', this, finalValue, evt, didDiscard)) {
256       //@if(debug)
257       SC.Logger.warn('InlineTextField._endEditing() cannot end without inlineEditorShouldEndEditing() on the delegate.');
258       //@end
259       return NO;
260     }
261     // OK, we are allowed to end editing.  Notify delegate of final value
262     // and clean up.
263     this.invokeDelegateMethod(del, 'inlineEditorDidEndEditing', this, finalValue, evt, didDiscard) ;
264 
265     // If the delegate set a class name, let's clean it up:
266     if(this._className) this.setClassName(this._className, false);
267     
268     // cleanup cached values
269     this._originalValue = this._delegate = this._exampleElement =  this._optframe = this._className = null ;
270     this.set('isEditing', NO) ;
271 
272     // resign first responder if not done already.  This may call us in a 
273     // loop but since isEditing is already NO, nothing will happen.
274     if (this.get('isFirstResponder')) {
275       var pane = this.get('pane');
276       if (pane && this._previousFirstResponder) {
277         pane.makeFirstResponder(this._previousFirstResponder);
278       } else this.resignFirstResponder();
279     }
280     this._previousFirstResponder = null ; // clearout no matter what
281     
282     if (this.get('parentNode')) this.removeFromParent() ;  
283     
284     return YES ;
285   },

362   willLoseFirstResponder: function(responder, evt) {
363     if (responder !== this) return;

365     // if we're about to lose first responder for any reason other than
366     // ending editing, make sure we clear the previous first responder so

368     this._previousFirstResponder = null;

371     // it can be passed to the delegate
372     this._origEvent = evt;
373

375     // too.

377     return this.blurEditor(evt) ;

440     }
441     return YES ;

462   /** Call this method to make the inline editor begin editing for your view.
463       
464       If the inline editor is already being used for another value it will
465       try to dismiss itself from the other editor and attach itself to the
466       new view instead.  If this process fails for some reason (for example
467       if the other view did not allow the view to end editing) then this
468       method will return false.
469 
470       You should pass a set of options that at least includes the target
471       view.  See class definition for options.
472       
473       @params options {Hash} hash of options for editing
474       @returns {Boolean} YES if editor began editing, NO if it failed.
475   */
476   beginEditing: function(options) {
477     this._exampleElement = options.exampleElement ;
478     
479     // If exampleInlineTextFieldView is set, load this class otherwise use
480     // the default, this.
481     var klass = options.exampleInlineTextFieldView 
482               ? options.exampleInlineTextFieldView : this,
483         layout = options.delegate.get('layout'),
484         s = this.updateViewStyle(),
485         p = this.updateViewPaddingStyle();
486     
487     var str= ".inline-editor input{"+s+"} ";
488     str= str+".inline-editor textarea{"+s+"} .inline-editor .padding{"+p+"}";
489     var pa= document.getElementsByTagName('head')[0],
490     el= document.createElement('style');
491     el.type= 'text/css';
492     el.media= 'screen';
493     if(el.styleSheet) el.styleSheet.cssText= str;// IE method
494     else el.appendChild(document.createTextNode(str));// others
495     pa.appendChild(el);
496     
497     this.editor = klass.create({ classNames: 'inline-editor', layout: layout}) ;
498     return this.editor.beginEditing(options) ;
499     
500   },
501   
502   /** Save the current value of the inline editor and exit edit mode.
503   
504     If the inline editor is being used it will try to end the editing and
505     close.  If the inline editor could not end for some reason (for example
506     if the delegate did not allow the editing to end) then this method will
507     return NO.
508     
509     @returns {Boolean} YES if the inline editor ended or no edit was in 
510       progress.
511   */
512   commitEditing: function() {
513     return this.editor ? this.editor.commitEditing() : YES ;
514   },
515 
516   /** Discard the current value of the inline editor and exit edit mode.
517   
518     If the inline editor is in use, this method will try to end the editing,
519     restoring the original value of the target view.  If the inline editor
520     could not end for some reason (for example if the delegate did not 
521     allow editing to end) then this method will return NO.
522     
523     @returns {Boolean} YES if the inline editor ended or no edit was in progress.
524   */
525   discardEditing: function() {
526     return this.editor ? this.editor.discardEditing() : YES ;  
527   },

529   /** @private */
530   updateViewStyle: function() {
531     var el = this._exampleElement[0],
532         styles = '',
533         s=SC.getStyle(el,'font-size');
534     if(s && s.length>0) styles = styles + "font-size: "+ s + " !important; ";
535     s=SC.getStyle(el,'font-family');
536     if(s && s.length>0) styles = styles + "font-family: " + s + " !important; ";
537     s=SC.getStyle(el,'font-weight');
538     if(s && s.length>0) styles = styles + "font-weight: " + s + " !important; ";
539     s=SC.getStyle(el,'z-index');
540     if(s && s.length>0) styles = styles + "z-index: " + s + " !important; ";
541     s=SC.getStyle(el,'line-height');
542     if(s && s.length>0) styles = styles + "line-height: " + s + " !important; ";
543     s=SC.getStyle(el,'text-align');
544     if(s && s.length>0) styles = styles + "text-align: " + s + " !important; ";
545     s=SC.getStyle(el,'top-margin');
546     if(s && s.length>0) styles = styles + "top-margin: " + s + " !important; ";
547     s=SC.getStyle(el,'bottom-margin');
548     if(s && s.length>0) styles = styles + "bottom-margin: " + s + " !important; ";
549     s=SC.getStyle(el,'left-margin');
550     if(s && s.length>0) styles = styles + "left-margin: " + s + " !important; ";
551     s=SC.getStyle(el,'right-margin');
552     if(s && s.length>0) styles = styles + "right-margin: " + s + " !important; ";
553     
554     return styles;
555   },

559     var el = this._exampleElement[0] ;   
560     var styles = '';
561     var s=SC.getStyle(el,'padding-top');
562     if(s && s.length>0) styles = styles + "top: "+ s + " !important; ";
563     s=SC.getStyle(el,'padding-bottom');
564     if(s && s.length>0) styles = styles + "bottom: " + s + " !important; ";
565     s=SC.getStyle(el,'padding-left');
566     if(s && s.length>0) styles = styles + "left: " + s + " !important; ";
567     s=SC.getStyle(el,'padding-right');
568     if(s && s.length>0) styles = styles + "right: " + s + " !important; ";
