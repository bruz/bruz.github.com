  8 /**
  9   This platform object allows you to conditionally support certain HTML5
 10   features.
 11 
 12   Rather than relying on the user agent, it detects whether the given elements
 13   and events are supported by the browser, allowing you to create much more
 14   robust apps.
 15 */
 16 
 17 SC.platform = {
 18   /**
 19     YES if the current device supports touch events, NO otherwise.
 20 
 21     @property {Boolean}
 22   */
 23   touch: ('createTouch' in document),
 24   
 25   bounceOnScroll: (/iPhone|iPad|iPod/).test(navigator.platform),
 26   pinchToZoom: (/iPhone|iPad|iPod/).test(navigator.platform),
 27 
 28   /**
 29     A hash that contains properties that indicate support for new HTML5
 30     input attributes.
 31 
 32     For example, to test to see if the placeholder attribute is supported,
 33     you would verify that SC.platform.input.placeholder is YES.
 34   */
 35   input: (function(attributes) {
 36     var ret = {},
 37         len = attributes.length,
 38         elem = document.createElement('input'),
 39         attr, idx;
 40 
 41     for (idx=0; idx < len; idx++) {
 42       attr = attributes[idx];
 43 
 44       ret[attr] = !!(attr in elem);
 45     }
 46 
 47     return ret;
 48   })(('autocomplete readonly list size required multiple maxlength '
 49     +'pattern min max step placeholder').w()),
 50 
 51   /**
 52     YES if the application is currently running as a standalone application.
 53 
 54     For example, if the user has saved your web application to their home
 55     screen on an iPhone OS-based device, this property will be true.
 56     @property {Boolean}
 57   */
 58   standalone: navigator.standalone,
 59 
 60   /**
 61     Prefix for browser specific CSS attributes. Calculated later.
 62   */
 63   cssPrefix: null,
 64 
 65   /**
 66     Prefix for browsew specific CSS attributes when used in the DOM. Calculated later.
 67   */
 68   domCSSPrefix: null,
 69 
 70   /** @private
 71     Removes event listeners from the document.
 72 
 73     @param {Array} events Array of strings representing the events to remove
 74   */
 75   removeEvents: function(events) {
 76     var idx, len = events.length, key;
 77     for (idx = 0; idx < len; idx++) {
 78       key = events[idx];
 79       SC.Event.remove(document, key, SC.RootResponder.responder, SC.RootResponder.responder[key]);
 80     }
 81   },
 82 
 83   /** @private
 84     Replaces an event listener with another.
 85 
 86     @param {String} evt The event to replace
 87     @param {Function} replacement The method that should be called instead
 88   */
 89   replaceEvent: function(evt, replacement) {
 90     SC.Event.remove(document, evt, SC.RootResponder.responder, SC.RootResponder.responder[evt]);
 91     SC.Event.add(document, evt, this, replacement);
 92   },
 93 
 94 
 95   /**
 96     Whether the browser supports CSS transitions. Calculated later.
 97   */
 98   supportsCSSTransitions: NO,
 99 
100   /**
101     Whether the browser supports 2D CSS transforms. Calculated later.
102   */
103   supportsCSSTransforms: NO,
104 
105   /**
106     Whether the browser understands 3D CSS transforms.
107     This does not guarantee that the browser properly handles them.
108     Calculated later.
109   */
110   understandsCSS3DTransforms: NO,
111 
112   /**
113     Whether the browser can properly handle 3D CSS transforms. Calculated later.
114   */
115   supportsCSS3DTransforms: NO
116 
117 };

119 /* Calculate CSS Prefixes */
120 
121 (function(){
122   var userAgent = navigator.userAgent.toLowerCase();
123   if ((/webkit/).test(userAgent)) {
124     SC.platform.cssPrefix = 'webkit';
125     SC.platform.domCSSPrefix = 'Webkit';
126   } else if((/opera/).test( userAgent )) {
127     SC.platform.cssPrefix = 'opera';
128     SC.platform.domCSSPrefix = 'O';
129   } else if((/msie/).test( userAgent ) && !(/opera/).test( userAgent )) {
130     SC.platform.cssPrefix = 'ms';
131     SC.platform.domCSSPrefix = 'ms';
132   } else if((/mozilla/).test( userAgent ) && !(/(compatible|webkit)/).test( userAgent )) {
133     SC.platform.cssPrefix = 'moz';
134     SC.platform.domCSSPrefix = 'Moz';
135   };
136 })();

138 /* Calculate transform support */
139 
140 (function(){
141   // a test element
142   var el = document.createElement("div");
143 
144   // the css and javascript to test
145   var css_browsers = ["-moz-", "-moz-", "-o-", "-ms-", "-webkit-"];
146   var test_browsers = ["moz", "Moz", "o", "ms", "webkit"];
147 
148   // prepare css
149   var css = "", i = null;
150   for (i = 0; i < css_browsers.length; i++) {
151     css += css_browsers[i] + "transition:all 1s linear;";
152     css += css_browsers[i] + "transform: translate(1px, 1px);";
153     css += css_browsers[i] + "perspective: 500px;";
154   }
155 
156   // set css text
157   el.style.cssText = css;
158 
159   // test
160   for (i = 0; i < test_browsers.length; i++)
161   {
162     if (el.style[test_browsers[i] + "TransitionProperty"] !== undefined) SC.platform.supportsCSSTransitions = YES;
163     if (el.style[test_browsers[i] + "Transform"] !== undefined) SC.platform.supportsCSSTransforms = YES;
164     if (el.style[test_browsers[i] + "Perspective"] !== undefined || el.style[test_browsers[i] + "PerspectiveProperty"] !== undefined) {
165       SC.platform.understandsCSS3DTransforms = YES;
166       SC.platform.supportsCSS3DTransforms = YES;
167     }
168   }
169 
170   // unfortunately, we need a bit more to know FOR SURE that 3D is allowed
171   if (window.media && window.media.matchMedium) {
172     if (!window.media.matchMedium('(-webkit-transform-3d)')) SC.platform.supportsCSS3DTransforms = NO;
173   } else if(window.styleMedia && window.styleMedia.matchMedium) {
174     if (!window.styleMedia.matchMedium('(-webkit-transform-3d)')) SC.platform.supportsCSS3DTransforms = NO;    
175   }
176 
177 })();
