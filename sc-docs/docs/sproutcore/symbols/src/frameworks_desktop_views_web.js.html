  7 /** @class
  8 
  9   Used to display an iframe. The source, (specified by the value property) of 
 10   the iFrame should be from the same domain. (i.e. the src / value should be 
 11   from the same domain) if you want to access the contents of the iframe.
 12 
 13   @extends SC.View
 14   @extends SC.Control
 15   @since SproutCore 1.0
 16 */
 17 SC.WebView = SC.View.extend(SC.Control, {
 18 /** @scope SC.WebView.prototype */
 19   classNames: 'sc-web-view',
 20   displayProperties: ['value', 'shouldAutoResize'],
 21 
 22   /**
 23   The content of the iframe can be bigger than the size specifed when creating
 24   the view. If you want the view to be auto-resized to the dimensions of the 
 25   iframe, then set the value of this property to YES.
 26   The web view can be auto resized only if the contents are from the same
 27   domain as the parent domain.
 28   @property{Boolean}
 29   */
 30   shouldAutoResize: NO,
 31 
 32   render: function(context, firstTime) {
 33     var src = this.get('value');
 34     if (firstTime) {
 35       context.push('<iframe src="' + src + 
 36       '" style="position: absolute; width: 100%; height: 100%; border: 0px; margin: 0px; padding: 0p;"></iframe>');
 37     } else {
 38       var iframe = this.$('iframe');
 39       // clear out the previous src, to force a reload
 40       iframe.attr('src', 'javascript:;');
 41       iframe.attr('src', src);
 42     }
 43   },
 44 
 45   /**
 46   Called when the layer gets created. 
 47   */
 48   didCreateLayer: function() {
 49     var f = this.$('iframe');
 50     // Attach an onload event to the iframe.
 51     SC.Event.add(f, 'load', this, this.iframeDidLoad);
 52   },
 53 
 54 
 55   /** 
 56   Called when iframe onload event is fired.
 57   1. Resizes the view to fit the contents of the iframe using the 
 58   scroll width and scroll height of the contents of the iframe
 59   
 60   The iframe contents can be accessed only when the src is from the same
 61   domain as the parent document
 62   @returns {void}
 63   */
 64   iframeDidLoad: function() {
 65     //fit the iframe to size of the contents.
 66     if (this.get('shouldAutoResize') === YES){
 67       var contentWindow;
 68       var iframeElt = this.$('iframe')[0];
 69       if(iframeElt && iframeElt.contentWindow){
 70         contentWindow = iframeElt.contentWindow;
 71         if(contentWindow && contentWindow.document && contentWindow.document.documentElement){
 72           var docElement = contentWindow.document.documentElement;
 73           // setting the width before the height gives more accurate results.. 
 74           // atleast for the test iframe content i'm using.
 75           //TODO: try out document flows other than top to bottom.
 76           if (!SC.browser.isIE){
 77             this.$().width(docElement.scrollWidth);
 78             this.$().height(docElement.scrollHeight);          
 79           } else {
 80             this.$().width(docElement.scrollWidth + 12);
 81             this.$().height(docElement.scrollHeight + 5);          
 82           }
 83         }
 84       }
 85     }
 86   }
 87 });
