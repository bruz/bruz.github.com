 11 /**
 12   @class
 13 
 14   This permits you to perform queries on your data store,
 15   written in a SQL-like language. Here is a simple example:
 16     
 17   {{{
 18     q = SC.Query.create({
 19       conditions: "firstName = 'Jonny' AND lastName = 'Cash'"
 20     })
 21   }}}
 22     
 23   You can check if a certain record matches the query by calling:
 24 
 25   {{{
 26     q.contains(record)
 27   }}}
 28   
 29   To find all records of your store, that match query q, use findAll with
 30   query q as argument:
 31   
 32   {{{
 33     r = MyApp.store.findAll(q)
 34   }}}
 35   
 36   r will be a record array containing all matching records.
 37   To limit the query to a record type of MyApp.MyModel,
 38   you can specify the type as a property of the query like this:
 39   
 40   {{{
 41     q = SC.Query.create({ 
 42       conditions: "firstName = 'Jonny' AND lastName = 'Cash'",
 43       recordType: MyApp.MyModel 
 44     })
 45   }}}
 46   
 47   Calling find() like above will now return only records of type t.
 48   It is recommended to limit your query to a record type, since the query will
 49   have to look for matching records in the whole store, if no record type
 50   is given.
 51   
 52   You can give an order, which the resulting records should follow, like this:
 53   
 54   {{{
 55     q = SC.Query.create({ 
 56       conditions: "firstName = 'Jonny' AND lastName = 'Cash'",
 57       recordType: MyApp.MyModel,
 58       orderBy: "lastName, year DESC" 
 59     });
 60   }}}
 61   
 62   The default order direction is ascending. You can change it to descending
 63   by writing DESC behind the property name like in the example above.
 64   If no order is given, or records are equal in respect to a given order,
 65   records will be ordered by guid.
 66 
 67   h2. SproutCore Query Language
 68   
 69   Features of the query language:
 70   
 71   h4. Primitives:
 72 
 73   - record properties
 74   - null, undefined
 75   - true, false
 76   - numbers (integers and floats)
 77   - strings (double or single quoted)
 78   
 79   h4. Parameters:
 80 
 81   - %@ (wild card)
 82   - {parameterName} (named parameter)
 83 
 84   Wild cards are used to identify parameters by the order in which they appear 
 85   in the query string. Named parameters can be used when tracking the order 
 86   becomes difficult. Both types of parameters can be used by giving the 
 87   parameters as a property to your query object:
 88   
 89   {{{
 90     yourQuery.parameters = yourParameters
 91   }}}
 92   
 93   where yourParameters should have one of the following formats:
 94 
 95     for wild cards: [firstParam, secondParam, thirdParam]
 96     for named params: {name1: param1, mane2: parma2}
 97 
 98   You cannot use both types of parameters in a single query!
 99   
100   h4. Operators:
101   
102   - =
103   - !=
104   - <
105   - <=
106   - >
107   - >=
108   - BEGINS_WITH (checks if a string starts with another one)
109   - ENDS_WITH   (checks if a string ends with another one)
110   - CONTAINS    (checks if a string contains another one, or if an object is in an array)
111   - MATCHES     (checks if a string is matched by a regexp,
112                 you will have to use a parameter to insert the regexp)
113   - ANY         (checks if the thing on its left is contained in the array
114                 on its right, you will have to use a parameter
115                 to insert the array)
116   - TYPE_IS     (unary operator expecting a string containing the name 
117                 of a Model class on its right side, only records of this type
118                 will match)
119     
120   h4. Boolean Operators:
121   
122   - AND
123   - OR
124   - NOT
125   
126   h4. Parenthesis for grouping:
127   
128   - ( and )
129 
130 
131   h2. Adding Your Own Query Handlers
132 
133   You can extend the query language with your own operators by calling:
134 
135   {{{
136     SC.Query.registerQueryExtension('your_operator', your_operator_definition);
137   }}}
138 
139   See details below. As well you can provide your own comparison functions
140   to control ordering of specific record properties like this:
141 
142   {{{
143     SC.Query.registerComparison(property_name, comparison_for_this_property);
144   }}}

323   /** 
324     Returns YES if record is matched by the query, NO otherwise.  This is 
325     used when computing a query locally.  
326  
327     @param {SC.Record} record the record to check
328     @param {Hash} parameters optional override parameters
329     @returns {Boolean} YES if record belongs, NO otherwise
330   */ 
331   contains: function(record, parameters) {
332 
333     // check the recordType if specified
334     var rtype, ret = YES ;    
335     if (rtype = this.get('recordTypes')) { // plural form
336       ret = rtype.find(function(t) { return SC.kindOf(record, t); });
337     } else if (rtype = this.get('recordType')) { // singular
338       ret = SC.kindOf(record, rtype);
339     }
340     
341     if (!ret) return NO ; // if either did not pass, does not contain
342 
343     // if we have a scope - check for that as well
344     var scope = this.get('scope');
345     if (scope && !scope.contains(record)) return NO ;
346     
347     // now try parsing
348     if (!this._isReady) this.parse(); // prepare the query if needed
349     if (!this._isReady) return NO ;
350     if (parameters === undefined) parameters = this.parameters || this;
351     
352     // if parsing worked we check if record is contained
353     // if parsing failed no record will be contained
354     return this._tokenTree.evaluate(record, parameters);
355   },

357   /**
358     Returns YES if the query matches one or more of the record types in the
359     passed set.
360     
361     @param {SC.Set} types set of record types
362     @returns {Boolean} YES if record types match
363   */
364   containsRecordTypes: function(types) {
365     var rtype = this.get('recordType');
366     if (rtype) {
367       return !!types.find(function(t) { return SC.kindOf(t, rtype); });
368     
369     } else if (rtype = this.get('recordTypes')) {
370       return !!rtype.find(function(t) { 
371         return !!types.find(function(t2) { return SC.kindOf(t2,t); });
372       });
373       
374     } else return YES; // allow anything through
375   },

377   /**
378     Returns the sort order of the two passed records, taking into account the
379     orderBy property set on this query.  This method does not verify that the
380     two records actually belong in the query set or not; this is checked using
381     contains().
382  
383     @param {SC.Record} record1 the first record
384     @param {SC.Record} record2 the second record
385     @returns {Number} -1 if record1 < record2, 
386                       +1 if record1 > record2,
387                       0 if equal
388   */
389   compare: function(record1, record2) {
390     // IMPORTANT:  THIS CODE IS ALSO INLINED INSIDE OF THE 'compareStoreKeys'
391     //             CLASS METHOD.  IF YOU CHANGE THIS IMPLEMENTATION, BE SURE
392     //             TO UPDATE IT THERE, TOO.
393     //
394     // (Any clients overriding this method will have their version called,
395     // however.  That's why we'll keep this here; clients might want to
396     // override it and call sc_super()).
397 
398     var result = 0, 
399         propertyName, order, len, i;
400 
401     // fast cases go here
402     if (record1 === record2) return 0;
403     
404     // if called for the first time we have to build the order array
405     if (!this._isReady) this.parse();
406     if (!this._isReady) { // can't parse. guid is wrong but consistent
407       return SC.compare(record1.get('id'),record2.get('id'));
408     }
409     
410     // For every property specified in orderBy until non-eql result is found.
411     // Or, if orderBy is a comparison function, simply invoke it with the
412     // records.
413     order = this._order;
414     if (SC.typeOf(order) === SC.T_FUNCTION) {
415       result = order.call(null, record1, record2);
416     }
417     else {
418       len   = order ? order.length : 0;
419       for (i=0; result===0 && (i < len); i++) {
420         propertyName = order[i].propertyName;
421         // if this property has a registered comparison use that
422         if (SC.Query.comparisons[propertyName]) {
423           result = SC.Query.comparisons[propertyName](
424                     record1.get(propertyName),record2.get(propertyName));
425                   
426         // if not use default SC.compare()
427         } else {
428           result = SC.compare(
429                     record1.get(propertyName), record2.get(propertyName) );
430         }
431       
432         if ((result!==0) && order[i].descending) result = (-1) * result;
433       }
434     }
435 
436     // return result or compare by guid
437     if (result !== 0) return result ;
438     else return SC.compare(record1.get('id'),record2.get('id'));
439   },

441   /** @private 
442       Becomes YES once the query has been successfully parsed 
443   */
444   _isReady:     NO,
445   
446   /**
447     This method has to be called before the query object can be used.
448     You will normaly not have to do this, it will be called automatically
449     if you try to evaluate a query.
450     You can however use this function for testing your queries.
451  
452     @returns {Boolean} true if parsing succeeded, false otherwise
453   */
454   parse: function() {
455     var conditions = this.get('conditions'),
456         lang       = this.get('queryLanguage'),
457         tokens, tree;
458         
459     tokens = this._tokenList = this.tokenizeString(conditions, lang);
460     tree = this._tokenTree = this.buildTokenTree(tokens, lang);
461     this._order = this.buildOrder(this.get('orderBy'));
462     
463     this._isReady = !!tree && !tree.error;
464     if (tree && tree.error) throw tree.error;
465     return this._isReady;
466   },
467   
468   /**
469     Returns the same query but with the scope set to the passed record array.
470     This will copy the receiver.  It also stores these queries in a cache to
471     reuse them if possible.
472     
473     @param {SC.RecordArray} recordArray the scope
474     @returns {SC.Query} new query
475   */
476   queryWithScope: function(recordArray) {
477     // look for a cached query on record array.
478     var key = SC.keyFor('__query__', SC.guidFor(this)),
479         ret = recordArray[key];
480         
481     if (!ret) {
482       recordArray[key] = ret = this.copy();
483       ret.set('scope', recordArray);
484       ret.freeze();
485     }
486     
487     return ret ;
488   },

503     Implement the Copyable API to clone a query object once it has been

506   copy: function() {
507     var opts = {}, 
508         keys = this.get('copyKeys'),
509         loc  = keys ? keys.length : 0,
510         key, value, ret;
511         
512     while(--loc >= 0) {
513       key = keys[loc];
514       value = this.get(key);
515       if (value !== undefined) opts[key] = value ;
516     }
517     
518     ret = this.constructor.create(opts);
519     opts = null;
520     return ret ;
521   },

880   /**
881     Takes a string and tokenizes it based on the grammar definition
882     provided. Called by parse().
883     
884     @param {String} inputString the string to tokenize
885     @param {Object} grammar the grammar definition (normally queryLanguage)
886     @returns {Array} list of tokens
887   */
888   tokenizeString: function (inputString, grammar) {
889 	
890 	
891     var tokenList           = [],
892         c                   = null,
893         t                   = null,
894         token               = null,
895         tokenType           = null,
896         currentToken        = null,
897         currentTokenType    = null,
898         currentTokenValue   = null,
899         currentDelimeter    = null,
900         endOfString         = false,
901         endOfToken          = false,
902         belongsToToken      = false,
903         skipThisCharacter   = false,
904         rememberCount       = {};
905   
906   
907     // helper function that adds tokens to the tokenList
908   
909     function addToken (tokenType, tokenValue) {
910       t = grammar[tokenType];
911       //tokenType = t.tokenType;
912       
913       // handling of special cases
914       // check format
915       if (t.format && !t.format.test(tokenValue)) tokenType = "UNKNOWN";
916       // delimeted token (e.g. by ")
917       if (t.delimeted) skipThisCharacter = true;
918       
919       // reserved words
920       if ( !t.delimeted ) {
921         for ( var anotherToken in grammar ) {
922           if ( grammar[anotherToken].reservedWord
923                && anotherToken == tokenValue ) {
924             tokenType = anotherToken;
925           }
926         }
927       }
928       
929       // reset t
930       t = grammar[tokenType];
931       // remembering count type
932       if ( t && t.rememberCount ) {
933         if (!rememberCount[tokenType]) rememberCount[tokenType] = 0;
934         tokenValue = rememberCount[tokenType];
935         rememberCount[tokenType] += 1;
936       }
937 
938       // push token to list
939       tokenList.push( {tokenType: tokenType, tokenValue: tokenValue} );
940 
941       // and clean up currentToken
942       currentToken      = null;
943       currentTokenType  = null;
944       currentTokenValue = null;
945     }
946   
947   
948     // stepping through the string:
949     
950     if (!inputString) return [];
951     
952     var iStLength = inputString.length;
953     
954     for (var i=0; i < iStLength; i++) {
955       
956       // end reached?
957       endOfString = (i===iStLength-1);
958       
959       // current character
960       c = inputString.charAt(i);
961     
962       // set true after end of delimeted token so that
963       // final delimeter is not catched again
964       skipThisCharacter = false;
965         
966     
967       // if currently inside a token
968     
969       if ( currentToken ) {
970       
971         // some helpers
972         t = grammar[currentToken];
973         endOfToken = t.delimeted ? c===currentDelimeter : t.notAllowed.test(c);
974       
975         // if still in token
976         if ( !endOfToken ) currentTokenValue += c;
977       
978         // if end of token reached
979         if (endOfToken || endOfString) {
980           addToken(currentToken, currentTokenValue);
981         }
982       
983         // if end of string don't check again
984         if ( endOfString && !endOfToken ) skipThisCharacter = true;
985       }
986     
987       // if not inside a token, look for next one
988     
989       if ( !currentToken && !skipThisCharacter ) {
990         // look for matching tokenType
991         for ( token in grammar ) {
992           t = grammar[token];
993           if (t.firstCharacter && t.firstCharacter.test(c)) {
994             currentToken = token;
995           }
996         }
997 
998         // if tokenType found
999         if ( currentToken ) {
1000           t = grammar[currentToken];
1001           currentTokenValue = c;
1002           // handling of special cases
1003           if ( t.delimeted ) {
1004             currentTokenValue = "";
1005             if ( t.lastCharacter ) currentDelimeter = t.lastCharacter;
1006             else currentDelimeter = c;
1007           }
1008 
1009           if ( t.singleCharacter || endOfString ) {
1010             addToken(currentToken, currentTokenValue);
1011           }
1012         }
1013       }
1014     }
1015     
1016     return tokenList;
1017   },

1025   /**
1026     Takes an array of tokens and returns a tree, depending on the
1027     specified tree logic. The returned object will have an error property
1028     if building of the tree failed. Check it to get some information
1029     about what happend.
1030     If everything worked the tree can be evaluated by calling:
1031     
1032       tree.evaluate(record, parameters)
1033     
1034     If tokenList is empty, a single token will be returned which will
1035     evaluate to true for all records.
1036     
1037     @param {Array} tokenList the list of tokens
1038     @param {Object} treeLogic the logic definition (normally queryLanguage)
1039     @returns {Object} token tree
1040   */
1041   buildTokenTree: function (tokenList, treeLogic) {
1042   
1043     var l                    = tokenList.slice();
1044     var i                    = 0;
1045     var openParenthesisStack = [];
1046     var shouldCheckAgain     = false;
1047     var error                = [];
1048     
1049   
1050     // empty tokenList is a special case
1051     if (!tokenList || tokenList.length === 0) {
1052       return { evaluate: function(){ return true; } };
1053     }
1054   
1055   
1056     // some helper functions
1057   
1058     function tokenLogic (position) {
1059       var p = position;
1060       if ( p < 0 ) return false;
1061       
1062       var tl = treeLogic[l[p].tokenType];
1063       
1064       if ( ! tl ) {
1065         error.push("logic for token '"+l[p].tokenType+"' is not defined");
1066         return false;
1067       }
1068 
1069       // save evaluate in token, so that we don't have
1070       // to look it up again when evaluating the tree
1071       l[p].evaluate = tl.evaluate;
1072       return tl;
1073     }
1074   
1075     function expectedType (side, position) {
1076       var p = position;
1077       var tl = tokenLogic(p);
1078       if ( !tl )            return false;
1079       if (side == 'left')   return tl.leftType;
1080       if (side == 'right')  return tl.rightType;
1081     }
1082   
1083     function evalType (position) {
1084       var p = position;
1085       var tl = tokenLogic(p);
1086       if ( !tl )  return false;
1087       else        return tl.evalType;
1088     }
1089   
1090     function removeToken (position) {
1091       l.splice(position, 1);
1092       if ( position <= i ) i--;
1093     }
1094   
1095     function preceedingTokenExists (position) {
1096       var p = position || i;
1097       if ( p > 0 )  return true;
1098       else          return false;
1099     }
1100   
1101     function tokenIsMissingChilds (position) {
1102       var p = position;
1103       if ( p < 0 )  return true;
1104       return (expectedType('left',p) && !l[p].leftSide)
1105           || (expectedType('right',p) && !l[p].rightSide);
1106     }
1107   
1108     function typesAreMatching (parent, child) {
1109       var side = (child < parent) ? 'left' : 'right';
1110       if ( parent < 0 || child < 0 )                      return false;
1111       if ( !expectedType(side,parent) )                   return false;
1112       if ( !evalType(child) )                             return false;
1113       if ( expectedType(side,parent) == evalType(child) ) return true;
1114       else                                                return false;
1115     }
1116   
1117     function preceedingTokenCanBeMadeChild (position) {
1118       var p = position;
1119       if ( !tokenIsMissingChilds(p) )   return false;
1120       if ( !preceedingTokenExists(p) )  return false;
1121       if ( typesAreMatching(p,p-1) )    return true;
1122       else                              return false;
1123     }
1124   
1125     function preceedingTokenCanBeMadeParent (position) {
1126       var p = position;
1127       if ( tokenIsMissingChilds(p) )    return false;
1128       if ( !preceedingTokenExists(p) )  return false;
1129       if ( !tokenIsMissingChilds(p-1) ) return false;
1130       if ( typesAreMatching(p-1,p) )    return true;
1131       else                              return false;
1132     }
1133   
1134     function makeChild (position) {
1135       var p = position;
1136       if (p<1) return false;
1137       l[p].leftSide = l[p-1];
1138       removeToken(p-1);
1139     }
1140   
1141     function makeParent (position) {
1142       var p = position;
1143       if (p<1) return false;
1144       l[p-1].rightSide = l[p];
1145       removeToken(p);
1146     }
1147   
1148     function removeParenthesesPair (position) {
1149       removeToken(position);
1150       removeToken(openParenthesisStack.pop());
1151     }
1152   
1153     // step through the tokenList
1154   
1155     for (i=0; i < l.length; i++) {
1156       shouldCheckAgain = false;
1157     
1158       if ( l[i].tokenType == 'UNKNOWN' ) {
1159         error.push('found unknown token: '+l[i].tokenValue);
1160       }
1161       
1162       if ( l[i].tokenType == 'OPEN_PAREN' ) openParenthesisStack.push(i);
1163       if ( l[i].tokenType == 'CLOSE_PAREN' ) removeParenthesesPair(i);
1164       
1165       if ( preceedingTokenCanBeMadeChild(i) ) makeChild(i);
1166       
1167       if ( preceedingTokenCanBeMadeParent(i) ){
1168         makeParent(i);
1169         shouldCheckAgain = true;
1170       } 
1171       
1172       if ( shouldCheckAgain ) i--;
1173     
1174     }
1175   
1176     // error if tokenList l is not a single token now
1177     if (l.length == 1) l = l[0];
1178     else error.push('string did not resolve to a single tree');
1179   
1180     // error?
1181     if (error.length > 0) return {error: error.join(',\n'), tree: l};
1182     // everything fine - token list is now a tree and can be returned
1183     else return l;
1184   
1185   },

1324         tempQueries = K._TMP_QUERIES,

1329         record2     = store.materializeRecord(storeKey2);
1330 
1331     // If the query implements a custom 'compare' function, then use it.
1332     // Otherwise, we have the logic from the standard version inlined here.
1333     if (compareFunc !== K.prototype.compare) {
1334       return compareFunc.call(query, record1, record2);
1335     }
1336     else {
1337       // THIS CODE IS THE SAME AS THE 'compare' METHOD, EXCEPT THAT 'this' HAS
1338       // BEEN CHANGED TO 'query'.
1339       //
1340       // It is inlined here to avoid the extra method invocation in the
1341       // typical case where the client does not supply a custom 'compare'
1342       // function.

1344       var result = 0, 
1345           propertyName, order, len, i;
1346

1362       }

1366           propertyName = order[i].propertyName;
1367           // if query property has a registered comparison use that
1368           if (SC.Query.comparisons[propertyName]) {
1369             result = SC.Query.comparisons[propertyName](
1370                       record1.get(propertyName),record2.get(propertyName));
1371

1373           } else {

1375                       record1.get(propertyName), record2.get(propertyName) );
1376           }
1377       
1378           if ((result!==0) && order[i].descending) result = (-1) * result;
1379         }
1380       }
1381

1384       else return SC.compare(record1.get('id'),record2.get('id'));

1388   /**

1391     calls to this method will return the same instance.  This is not possible 
1392     however, when you pass custom parameters or set ordering. All returned 
1393     queries are frozen.
1394     
1395     Usually you will not call this method directly.  Instead use the more
1396     convenient SC.Query.local() and SC.Query.remote().
1397     
1398     h2. Examples
1399     
1400     There are a number of different ways you can call this method.  
1401

1403     type or types, including any subclasses:

1416     
1417     {{{
1418       var married = SC.Query.local(Ab.Person, "isMarried=YES");
1419       var married = SC.Query.local(Ab.Person, "isMarried=%@", [YES]);
1420       var married = SC.Query.local(Ab.Person, "isMarried={married}", {
1421         married: YES
1422       });
1423     }}}

1426     how you specify an order, for example:
1427     
1428     {{{
1429       var orderedPeople = SC.Query.local(Ab.Person, { orderBy: "firstName" });
1430     }}}
1431     
1432     @param {String} location the query location.
1433     @param {SC.Record|Array} recordType the record type or types.
1434     @param {String} conditions optional conditions
1435     @param {Hash} params optional params. or pass multiple args.
1436     @returns {SC.Query}
1437   */

1439     
1440     var opts = null,
1441         ret, cache, key, tmp;
1442     
1443     // fast case for query objects.
1444     if (recordType && recordType.isQuery) { 
1445       if (recordType.get('location') === location) return recordType;
1446       else return recordType.copy().set('location', location).freeze();
1447     }
1448     
1449     // normalize recordType
1450     if (typeof recordType === SC.T_STRING) {
1451       ret = SC.objectForPropertyPath(recordType);
1452       if (!ret) throw "%@ did not resolve to a class".fmt(recordType);
1453       recordType = ret ;
1454     } else if (recordType && recordType.isEnumerable) {
1455       ret = [];
1456       recordType.forEach(function(t) {
1457         if (typeof t === SC.T_STRING) t = SC.objectForPropertyPath(t);
1458         if (!t) throw "cannot resolve record types: %@".fmt(recordType);

1465     if (conditions === undefined) conditions = null;
1466 
1467     // normalize other params. if conditions is just a hash, treat as opts
1468     if (!params && (typeof conditions !== SC.T_STRING)) {
1469       opts = conditions;
1470       conditions = null ;
1471     }
1472     
1473     // special case - easy to cache.
1474     if (!params && !opts) {
1475 
1476       tmp = SC.Query._scq_recordTypeCache;
1477       if (!tmp) tmp = SC.Query._scq_recordTypeCache = {};
1478       cache = tmp[location];
1479       if (!cache) cache = tmp[location] = {}; 
1480       
1481       if (recordType.isEnumerable) {
1482         key = recordType.map(function(k) { return SC.guidFor(k); });
1483         key = key.sort().join(':');
1484       } else key = SC.guidFor(recordType);
1485       
1486       if (conditions) key = [key, conditions].join('::');
1487       
1488       ret = cache[key];
1489       if (!ret) {
1490         if (recordType.isEnumerable) {
1491           opts = { recordTypes: recordType.copy() };
1492         } else opts = { recordType: recordType };
1493         
1494         opts.location = location ;
1495         opts.conditions = conditions ;
1496         ret = cache[key] = SC.Query.create(opts).freeze();
1497       }
1498     // otherwise parse extra conditions and handle them
1499     } else {
1500 
1501       if (!opts) opts = {};
1502       if (!opts.location) opts.location = location ; // allow override
1503 
1504       // pass one or more recordTypes.
1505       if (recordType && recordType.isEnumerable) {
1506         opts.recordsTypes = recordType;
1507       } else opts.recordType = recordType;
1508 
1509       // set conditions and params if needed
1510       if (conditions) opts.conditions = conditions;
1511       if (params) opts.parameters = params;
1512 
1513       ret = SC.Query.create(opts).freeze();
1514     }
1515     
1516     return ret ;
1517   },
