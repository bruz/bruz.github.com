  7 /** @class
  8 
  9   A RadioView is used to create a group of radio buttons.  The user can use
 10   these buttons to pick from a choice of options.
 11   
 12   This view renders simulated radio buttons that can display a mixed state and 
 13   has other features not found in platform-native controls.
 14   
 15   The radio buttons themselves are designed to be styled using CSS classes with
 16   the following structure:
 17   
 18   <label class="sc-radio-button">
 19   <img class="button" src="some_image.gif"/>
 20   <input type="radio" name="<sc-guid>" value=""/>
 21   <span class="sc-button-label">Label for button1</span>
 22   </label>
 23   
 24   Setting up a RadioView accepts a number of properties, for example:
 25   {
 26     items: [{ title: "Red", 
 27               value: "red", 
 28               enabled: YES, 
 29               icon: "button_red" },
 30             { title: "Green", 
 31               value: "green", 
 32               enabled: YES, 
 33               icon: 'button_green' }],
 34     value: 'red',
 35     itemTitleKey: 'title',
 36     itemValueKey: 'value',
 37     itemIconKey: 'icon',
 38     itemIsEnabledKey: 'enabled',
 39     isEnabled: YES,
 40     layoutDirection: SC.LAYOUT_HORIZONTAL
 41   }
 42   
 43   Default layoutDirection is vertical. 
 44   Default isEnabled is YES.
 45   
 46   The value property can be either a string, as above, or an array of strings
 47   for pre-checking multiple values.
 48   
 49   The items array can contain either strings, or as in the example above a 
 50   hash. When using a hash, make sure to also specify the itemTitleKey
 51   and itemValueKey you are using. Similarly, you will have to provide 
 52   itemIconKey if you are using icons radio buttons. The individual items 
 53   enabled property is YES by default, and the icon is optional.
 54   
 55   @extends SC.FieldView
 56   @since SproutCore 1.0
 57 */
 58 SC.RadioView = SC.View.extend(SC.Control,
 59 /** @scope SC.RadioView.prototype */
 60 {
 61 
 62   // HTML design options
 63   classNames: ['sc-radio-view'],
 64 
 65   /**
 66     The value of the currently selected item, and which will be checked in the 
 67     UI. This can be either a string or an array with strings for checking 
 68     multiple values.
 69   */
 70   value: null,
 71 
 72   /**
 73     This property indicates how the radio buttons are arranged.
 74   */
 75   layoutDirection: SC.LAYOUT_VERTICAL,
 76 
 77   // escape the HTML in label text
 78   escapeHTML: YES,
 79 
 80   /** 
 81     The items property can be either an array with strings, or a
 82     hash. When using a hash, make sure to also specify the appropriate
 83     itemTitleKey, itemValueKey, itemIsEnabledKey and itemIconKey.
 84   */
 85   items: [],
 86 
 87   /** 
 88     If items property is a hash, specify which property will function as
 89     the title with this itemTitleKey property.
 90   */
 91   itemTitleKey: null,
 92 
 93   /** 
 94     If items property is a hash, specify which property will function as
 95     the value with this itemValueKey property.
 96   */
 97   itemValueKey: null,
 98 
 99   /** 
100     If items property is a hash, specify which property will function as
101     the value with this itemIsEnabledKey property.
102   */
103   itemIsEnabledKey: null,
104 
105   /** 
106     If items property is a hash, specify which property will function as
107     the value with this itemIconKey property.
108   */
109   itemIconKey: null,
110 
111   /** 
112     If the items array itself changes, add/remove observer on item... 
113   */
114   itemsDidChange: function() {
115     if (this._items) {
116       this._items.removeObserver('[]', this, this.itemContentDidChange);
117     }
118     this._items = this.get('items');
119     if (this._items) {
120       this._items.addObserver('[]', this, this.itemContentDidChange);
121     }
122     this.itemContentDidChange();
123   }.observes('items'),
124 
125   /** 
126     Invoked whenever the item array or an item in the array is changed.
127     This method will regenerate the list of items.
128   */
129   itemContentDidChange: function() {
130     // Force regeneration of buttons
131     this._renderAsFirstTime = YES;
132   
133     this.notifyPropertyChange('_displayItems');
134   },
135 
136   // ..........................................................
137   // PRIVATE SUPPORT
138   // 
139   /** 
140     The display properties for radio buttons are the value and _displayItems.
141   */
142   displayProperties: ['value', '_displayItems'],
143 
144   render: function(context, firstTime) {
145     var item, idx, icon, name, itemsLength, url, className, disabled, sel, labelText, selectionState, selectionStateClassNames, items = this.get('_displayItems'),
146     value = this.get('value'),
147     isArray = SC.isArray(value);
148 
149     context.addClass(this.get('layoutDirection'));
150 
151     // isArray is set only when there are two active checkboxes 
152     // which can only happen with mixed state
153     if (isArray && value.length <= 0) {
154       value = value[0];
155       isArray = NO;
156     }
157 
158     // if necessary, regenerate the radio buttons
159     if (this._renderAsFirstTime) {
160       firstTime = YES;
161       this._renderAsFirstTime = NO;
162     }
163 
164     if (firstTime) {
165       context.attr('role', 'radiogroup');
166       // generate tags from this.
167       name = SC.guidFor(this); // name for this group
168       itemsLength = items.length;
169       for (idx = 0; idx < itemsLength; idx++) {
170         item = items[idx];
171 
172         // get the icon from the item, if one exists...
173         icon = item[3];
174         if (icon) {
175           url = (icon.indexOf('/') >= 0) ? icon: SC.BLANK_IMAGE_URL;
176           className = (url === icon) ? '': icon;
177           icon = '<img src="' + url + '" class="icon ' + className + '" alt="" />';
178         } else icon = '';
179 
180         if (item) {
181           sel = (isArray) ? (value.indexOf(item[1]) >= 0) : (value === item[1]);
182         } else {
183           sel = NO;
184         }
185         selectionStateClassNames = this._getSelectionStateClassNames(item, sel, value, isArray, false);
186 
187         labelText = this.escapeHTML ? SC.RenderContext.escapeHTML(item[0]) : item[0];
188         
189         context.push('<div class="sc-radio-button ', 
190                     selectionStateClassNames, '" ', 
191                     'aria-checked="', sel ? 'true':'false','" ',
192                     'role="radio"' , ' index="', idx,'">',
193                     '<span class="button"></span>',
194                     '<span class="sc-button-label">', 
195                     icon, labelText, '</span></div>');
196       }
197 
198     } else {
199       // update the selection state on all of the DOM elements.  The options are
200       // sel or mixed.  These are used to display the proper setting...
201       this.$('.sc-radio-button').forEach(function(button) {
202 
203         button = this.$(button);
204         idx = parseInt(button.attr('index'), 0);
205         item = (idx >= 0) ? items[idx] : null;
206 
207         if (item) {
208           sel = (isArray) ? (value.indexOf(item[1]) >= 0) : (value === item[1]);
209         } else {
210           sel = NO;
211         }
212         selectionState = this._getSelectionStateClassNames(item, sel, value, isArray, true);
213         button.attr('aria-checked', sel ? 'true': 'false');
214         // set class of label
215         button.setClass(selectionState);
216 
217         // avoid memory leaks
218         idx = selectionState = null;
219       },
220       this);
221     }
222   },
223 
224   /** @private - 
225     Will iterate the items property to return an array with items that is 
226     indexed in the following structure:
227       [0] => Title (or label)
228       [1] => Value
229       [2] => Enabled (YES default)
230       [3] => Icon (image URL)
231   */
232   _displayItems: function() {
233     var items = this.get('items'), 
234         loc = this.get('localize'),
235         titleKey = this.get('itemTitleKey'), valueKey = this.get('itemValueKey'),
236         isEnabledKey = this.get('itemIsEnabledKey'), 
237         iconKey = this.get('itemIconKey'),
238         ret = [], max = (items)? items.get('length') : 0,
239         item, title, value, idx, isArray, isEnabled, icon;
240     
241     for(idx=0;idx<max;idx++) {
242       item = items.objectAt(idx); 
243       
244       // if item is an array, just use the items...
245       if (SC.typeOf(item) === SC.T_ARRAY) {
246         title = item[0];
247         value = item[1];
248 
249         // otherwise, possibly use titleKey,etc.
250       } else if (item) {
251         // get title.  either use titleKey or try to convert the value to a 
252         // string.
253         if (titleKey) {
254           title = item.get ? item.get(titleKey) : item[titleKey];
255         } else title = (item.toString) ? item.toString() : null;
256 
257         if (valueKey) {
258           value = item.get ? item.get(valueKey) : item[valueKey];
259         } else value = item;
260 
261         if (isEnabledKey) {
262           isEnabled = item.get ? item.get(isEnabledKey) : item[isEnabledKey];
263         } else isEnabled = YES;
264 
265         if (iconKey) {
266           icon = item.get ? item.get(iconKey) : item[iconKey];
267         } else icon = null;
268 
269         // if item is nil, use somedefaults...
270       } else {
271         title = value = icon = null;
272         isEnabled = NO;
273       }
274 
275       // localize title if needed
276       if (loc) title = title.loc();
277       ret.push([title, value, isEnabled, icon]);
278     }
279 
280     return ret; // done!
281   }.property('items', 'itemTitleKey', 'itemValueKey', 'itemIsEnabledKey', 'localize', 'itemIconKey').cacheable(),
282 
283   /** @private - 
284     Will figure out what class names to assign each radio button.
285     This method can be invoked either as part of render() either when:
286     1. firstTime is set and we need to assign the class names as a string
287     2. we already have the DOM rendered but we just need to update class names
288        assigned to the the input field parent
289   */
290   _getSelectionStateClassNames: function(item, sel, value, isArray, shouldReturnObject) {
291     var classNames, key;
292 
293     // now set class names
294     classNames = {
295       sel: (sel && !isArray),
296       mixed: (sel && isArray),
297       disabled: (!item[2])
298     };
299 
300     if (shouldReturnObject) {
301       return classNames;
302     } else {
303       // convert object values to string
304       var classNameArray = [];
305       for (key in classNames) {
306         if (!classNames.hasOwnProperty(key)) continue;
307         if (classNames[key]) classNameArray.push(key);
308       }
309       return classNameArray.join(" ");
310     }
311   },
312 
313   /**
314     If the user clicks on of the items mark it as active on mouseDown unless
315     is disabled.
316 
317     Save the element that was clicked on so we can remove the active state on
318     mouseUp.
319   */
320   mouseDown: function(evt) {
321     if (!this.get('isEnabled')) return YES;
322     var target = evt.target;
323     while (target) {
324       if (target.className && target.className.indexOf('sc-radio-button') > -1) break;
325       target = target.parentNode;
326     }
327     if (!target) return NO;
328 
329     target = this.$(target);
330     if (target.hasClass('disabled')) return YES;
331     target.addClass('active');
332     this._activeRadioButton = target;
333     // even if radiobuttons are not set to get firstResponder, allow default 
334     // action, that way textfields loose focus as expected.
335     evt.allowDefault();
336     return YES;
337   },
338 
339   /**
340     If we have a radio element that was clicked on previously, make sure we
341     remove the active state. Then update the value if the item clicked is 
342     enabled.
343   */
344   mouseUp: function(evt) {
345     if (!this.get('isEnabled')) return YES;
346     var active = this._activeRadioButton,
347     target = evt.target,
348     items = this.get('_displayItems'),
349     index,
350     item;
351 
352     if (active) {
353       active.removeClass('active');
354       this._activeRadioButton = null;
355     } else return YES;
356 
357     while (target) {
358       if (target.className && target.className.indexOf('sc-radio-button') > -1) break;
359       target = target.parentNode;
360     }
361     target = this.$(target);
362     if (target[0] !== active[0] || target.hasClass('disabled')) return YES;
363 
364     index = parseInt(target.attr('index'), 0);
365     item = items[index];
366     this.set('value', item[1]);
367   },
368 
369   touchStart: function(evt) {
370     return this.mouseDown(evt);
371   },
372 
373   touchEnd: function(evt) {
374     return this.mouseUp(evt);
375   }
376 });
