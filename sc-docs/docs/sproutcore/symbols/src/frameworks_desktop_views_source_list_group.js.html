  9 sc_require('views/disclosure');

 11 /**
 12   @class
 13   
 14   Displays a group view in a source list.  Handles displaying a disclosure
 15   triangle which can be used to show/hide children.
 16   
 17   @extends SC.View
 18   @extends SC.Control
 19   @author Charles Jolley
 20   @author Erich Ocean
 21   @version 1.0
 22   @since 0.9
 23 */
 24 
 25 SC.SourceListGroupView = SC.View.extend(SC.Control, SC.CollectionGroup,
 26 /** @scope SC.SourceListGroupView.prototoype */ {
 27   
 28   classNames: ['sc-source-list-group'],
 29   
 30   // ..........................................................
 31   // KEY PROPERTIES
 32   // 
 33   
 34   /**
 35     The content object the source list group will display.
 36     
 37     @type SC.Object
 38   */
 39   content: null,
 40   
 41   /**
 42     The current group visibility.  Used by the source list to determine the 
 43     layout size of the group.
 44     
 45     @type Boolean
 46   */
 47   isGroupVisible: YES,
 48   
 49   /** 
 50     YES if group is showing its titlebar.
 51     
 52     Group views will typically hide their header if the content is set to 
 53     null.  You can also override this method to always hide the header if 
 54     you want and the SourceListView will not leave room for it.
 55     
 56     @type Boolean
 57   */
 58   hasGroupTitle: YES,
 59   
 60   /**
 61     The content property key to use as the group view's title.
 62     
 63     @type String
 64   */
 65   groupTitleKey: null,
 66   
 67   /**
 68     The content property key to use to determine if the group's children are 
 69     visible or not.
 70     
 71     @type String
 72   */
 73   groupVisibleKey: null,
 74   
 75   render: function(context, firstTime) {
 76     context.push('<div role="button" class="sc-source-list-label sc-disclosure-view sc-button-view button disclosure no-disclosure">',
 77               '<img src="'+SC.BLANK_IMAGE_URL+'" class="button" />',
 78               '<span class="label"></span></div>') ;
 79   },
 80   
 81   /** @private */
 82   createChildViews: function() {
 83     
 84   },
 85   
 86   /** @private */
 87   contentPropertyDidChange: function(target, key) {
 88     var content = this.get('content') ;
 89     var labelView = this.outlet('labelView') ;
 90     
 91     // hide labelView if content is null.
 92     if (content === null) {
 93       labelView.setIfChanged('isVisible', NO) ;
 94       this.setIfChanged('hasGroupTitle', NO) ;
 95       return ;
 96     } else {
 97       labelView.setIfChanged('isVisible', YES) ;
 98       this.setIfChanged('hasGroupTitle', YES) ;
 99     }
100     
101    // set the title if that changed.
102     var groupTitleKey = this.getDelegateProperty('groupTitleKey', this.displayDelegate) ;
103     if ((key == '*') || (groupTitleKey && (key == groupTitleKey))) {
104       var title = (content && content.get && groupTitleKey) ? content.get(groupTitleKey) : content;
105       if (title != this._title) {
106         this._title = title ;
107         if (title) title = title.capitalize() ;
108         labelView.set('title', title) ;
109       }
110     }
111     
112     // set the group visibility if changed
113     var groupVisibleKey = this.getDelegateProperty('groupVisibleKey', this.displayDelegate) ;
114     if ((key == '*') || (groupVisibleKey && (key == groupVisibleKey))) {
115       if (groupVisibleKey) {
116         labelView.removeClassName('no-disclosure') ;
117         
118         var isVisible = (content && content.get) ?
119           !!content.get(groupVisibleKey) :
120           YES ;
121         if (isVisible != this.get('isGroupVisible')) {
122           this.set('isGroupVisible', isVisible) ;
123           labelView.set('value', isVisible) ;
124         }
125       } else labelView.addClassName('no-disclosure') ;
126     }
127   },
128   
129   /** @private
130     Called when the user clicks on the disclosure triangle
131   */
132   disclosureValueDidChange: function(newValue) {
133     if (newValue == this.get('isGroupVisible')) return; // nothing to do
134     
135     // update group if necessary
136     var group = this.get('content') ;
137     var groupVisibleKey = this.getDelegateProperty('groupVisibleKey', this.displayDelegate) ;
138     if (group && group.set && groupVisibleKey) {
139       group.set(groupVisibleKey, newValue) ;
140     }
141     
142     // update my own value and then update my collection view.
143     this.set('isGroupVisible', newValue) ;
144     if (this.owner && this.owner.updateChildren) this.owner.updateChildren(true) ;
145     
146   },
147   
148   /** @private */
149   labelView: SC.DisclosureView.extend({
150     
151     /** 
152       Always default to open disclosures.
153       
154       @type Boolean
155     */
156     value: YES,
157     
158     /** @private
159       If the disclosure value changes, call the owner's method.  Note
160       normally you would do this with a binding, but since this is a semi-
161       private class anyway, there is no reason to go to all that trouble.
162     */
163     _valueObserver: function() {
164       if (this.owner) this.owner.disclosureValueDidChange(this.get('value')) ;
165     }.observes('value')
166     
167   })
168   
169 });
