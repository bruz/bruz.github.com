  8 /**
  9   @class
 10 
 11   A ThumbView works in concert with SC.SplitView to adjust the divider 
 12   position from an arbitrary subview of the SplitView. Simply make an
 13   instance of ThumbView a child somewhere in the childViews (or 
 14   descendants) of the split view and add the path to the ThumbView to the
 15   SplitView's thumbViews array.
 16   
 17   SplitView will automatically set the splitView property of the views in
 18   its thumbViews array.
 19 
 20   @extends SC.View
 21   @author Erich Ocean
 22   @test in split
 23 */
 24 SC.ThumbView = SC.View.extend(
 25 /** @scope SC.ThumbView.prototype */ {
 26 
 27   classNames: ['sc-thumb-view'],
 28   
 29   /**
 30     Enable this thumb view to control its parent split view.
 31   */
 32   isEnabled: YES,
 33   isEnabledBindingDefault: SC.Binding.bool(),
 34   
 35   /** @private */
 36   prepareContext: function(context, firstTime) {
 37     var splitView = this.get('splitView') ;
 38     if (splitView) this.set('cursor', splitView.get('thumbViewCursor')) ;
 39     return sc_super() ;
 40   },
 41   
 42   mouseDown: function(evt) {
 43     if (!this.get('isEnabled')) return NO ;
 44     
 45     var splitView = this.get('splitView');
 46     return (splitView) ? splitView.mouseDownInThumbView(evt, this) : sc_super();
 47   },
 48   
 49   touchStart: function(evt) {
 50     return this.mouseDown(evt);
 51   }
 52     
 53 });
