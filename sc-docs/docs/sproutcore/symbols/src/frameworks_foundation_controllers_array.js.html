 10 /**
 11   @class
 12 
 13   An ArrayController provides a way for you to publish an array of objects
 14   for CollectionView or other controllers to work with.  To work with an 
 15   ArrayController, set the content property to the array you want the 
 16   controller to manage.  Then work directly with the controller object as if
 17   it were the array itself.
 18   
 19   When you want to display an array of objects in a CollectionView, bind the
 20   "arrangedObjects" of the array controller to the CollectionView's "content"
 21   property.  This will automatically display the array in the collection view.
 22 
 23   @extends SC.Controller
 24   @extends SC.Array
 25   @extends SC.SelectionSupport
 26   @author Charles Jolley
 27   @since SproutCore 1.0
 28 */
 29 SC.ArrayController = SC.Controller.extend(SC.Array, SC.SelectionSupport,
 30 /** @scope SC.ArrayController.prototype */ {
 31 
 32   // ..........................................................
 33   // PROPERTIES
 34   // 
 35   
 36   /**
 37     The content array managed by this controller.  
 38     
 39     You can set the content of the ArrayController to any object that 
 40     implements SC.Array or SC.Enumerable.  If you set the content to an object
 41     that implements SC.Enumerable only, you must also set the orderBy property
 42     so that the ArrayController can order the enumerable for you.
 43     
 44     If you set the content to a non-enumerable and non-array object, then the
 45     ArrayController will wrap the item in an array in an attempt to normalize
 46     the result.
 47     
 48     @property {SC.Array}
 49   */
 50   content: null,
 51 
 52   /**
 53     Makes the array editable or not.  If this is set to NO, then any attempts
 54     at changing the array content itself will throw an exception.
 55     
 56     @property {Boolean}
 57   */
 58   isEditable: YES,
 59   
 60   /**
 61     Used to sort the array.
 62     
 63     If you set this property to a key name, array of key names, or a function,
 64     then then ArrayController will automatically reorder your content array
 65     to match the sort order.  (If you set a function, the function will be
 66     used to sort).
 67 
 68     Normally, you should only use this property if you set the content of the
 69     controller to an unordered enumerable such as SC.Set or SC.SelectionSet.
 70     In this case the orderBy property is required in order for the controller
 71     to property order the content for display.
 72     
 73     If you set the content to an array, it is usually best to maintain the 
 74     array in the proper order that you want to display things rather than 
 75     using this method to order the array since it requires an extra processing
 76     step.  You can use this orderBy property, however, for displaying smaller 
 77     arrays of content.
 78     
 79     Note that you can only to use addObject() to insert new objects into an
 80     array that is ordered.  You cannot manually reorder or insert new objects
 81     into specific locations because the order is managed by this property 
 82     instead.
 83     
 84     If you pass a function, it should be suitable for use in compare().
 85     
 86     @property {String|Array|Function}
 87   */
 88   orderBy: null,
 89     
 90   /**
 91     Set to YES if you want the controller to wrap non-enumerable content    
 92     in an array and publish it.  Otherwise, it will treat single content like 
 93     null content.
 94     
 95     @property {Boolean}
 96   */
 97   allowsSingleContent: YES,
 98   
 99   /**
100     Set to YES if you want objects removed from the array to also be
101     deleted.  This is a convenient way to manage lists of items owned
102     by a parent record object.
103     
104     Note that even if this is set to NO, calling destroyObject() instead of
105     removeObject() will still destroy the object in question as well as 
106     removing it from the parent array.
107     
108     @property {Boolean}
109   */
110   destroyOnRemoval: NO,
111 
112   /**
113     Returns an SC.Array object suitable for use in a CollectionView.  
114     Depending on how you have your ArrayController configured, this property
115     may be one of several different values.  
116     
117     @property {SC.Array}
118   */
119   arrangedObjects: function() {
120     return this;
121   }.property().cacheable(),
122   
123   /**
124     Computed property indicates whether or not the array controller can 
125     remove content.  You can delete content only if the content is not single
126     content and isEditable is YES.
127     
128     @property {Boolean}
129   */
130   canRemoveContent: function() {
131     var content = this.get('content'), ret;
132     ret = !!content && this.get('isEditable') && this.get('hasContent');
133     if (ret) {
134       return !content.isEnumerable || 
135              (SC.typeOf(content.removeObject) === SC.T_FUNCTION);
136     } else return NO ;
137   }.property('content', 'isEditable', 'hasContent'),
138   
139   /**
140     Computed property indicates whether you can reorder content.  You can
141     reorder content as long a the controller isEditable and the content is a
142     real SC.Array-like object.  You cannot reorder content when orderBy is
143     non-null.
144     
145     @property {Boolean}
146   */
147   canReorderContent: function() {
148     var content = this.get('content'), ret;
149     ret = !!content && this.get('isEditable') && !this.get('orderBy');
150     return ret && !!content.isSCArray;
151   }.property('content', 'isEditable', 'orderBy'),
152   
153   /**
154     Computed property insides whether you can add content.  You can add 
155     content as long as the controller isEditable and the content is not a 
156     single object.
157     
158     Note that the only way to simply add object to an ArrayController is to
159     use the addObject() or pushObject() methods.  All other methods imply 
160     reordering and will fail.
161     
162     @property {Boolean}
163   */
164   canAddContent: function() {
165     var content = this.get('content'), ret ;
166     ret = content && this.get('isEditable') && content.isEnumerable;
167     if (ret) {
168       return (SC.typeOf(content.addObject) === SC.T_FUNCTION) || 
169              (SC.typeOf(content.pushObject) === SC.T_FUNCTION); 
170     } else return NO ;
171   }.property('content', 'isEditable'),
172   
173   /**
174     Set to YES if the controller has valid content that can be displayed,
175     even an empty array.  Returns NO if the content is null or not enumerable
176     and allowsSingleContent is NO.
177     
178     @property {Boolean}
179   */
180   hasContent: function() {
181     var content = this.get('content');
182     return !!content && 
183            (!!content.isEnumerable || !!this.get('allowsSingleContent'));
184   }.property('content', 'allowSingleContent'),
185 
186   /**
187     Returns the current status property for the content.  If the content does
188     not have a status property, returns SC.Record.READY.
189     
190     @property {Number}
191   */
192   status: function() {
193     var content = this.get('content'),
194         ret = content ? content.get('status') : null;
195     return ret ? ret : SC.Record.READY;
196   }.property().cacheable(),
197   
198   // ..........................................................
199   // METHODS
200   // 
201   
202   /**
203     Adds an object to the array.  If the content is ordered, this will add the 
204     object to the end of the content array.  The content is not ordered, the
205     location depends on the implementation of the content.
206     
207     If the source content does not support adding an object, then this method 
208     will throw an exception.
209     
210     @param {Object} object the object to add
211     @returns {SC.ArrayController} receiver
212   */
213   addObject: function(object) {
214     if (!this.get('canAddContent')) throw "%@ cannot add content".fmt(this);
215     
216     var content = this.get('content');
217     if (content.isSCArray) content.pushObject(object);
218     else if (content.addObject) content.addObject(object);
219     else throw "%@.content does not support addObject".fmt(this);
220     
221     return this;
222   },
223   
224   /**
225     Removes the passed object from the array.  If the underyling content 
226     is a single object, then this simply sets the content to null.  Otherwise
227     it will call removeObject() on the content.
228     
229     Also, if destroyOnRemoval is YES, this will actually destroy the object.
230     
231     @param {Object} object the object to remove
232     @returns {SC.ArrayController} receiver
233   */
234   removeObject: function(object) {
235     if (!this.get('canRemoveContent')) {
236       throw "%@ cannot remove content".fmt(this);
237     }
238     
239     var content = this.get('content');
240     if (content.isEnumerable) content.removeObject(object);
241     else {
242       this.set('content', null);
243     }
244     
245     if (this.get('destroyOnRemoval') && object.destroy) object.destroy();
246     return this; 
247   },

253   /**
254     Compute the length of the array based on the observable content
255     
256     @property {Number}
257   */
258   length: function() {
259     var content = this._scac_observableContent();
260     return content ? content.get('length') : 0;
261   }.property().cacheable(),
262 
263   /** @private
264     Returns the object at the specified index based on the observable content
265   */
266   objectAt: function(idx) {
267     var content = this._scac_observableContent();
268     return content ? content.objectAt(idx) : undefined ;    
269   },
270   
271   /** @private
272     Forwards a replace on to the content, but only if reordering is allowed.
273   */
274   replace: function(start, amt, objects) {
275     // check for various conditions before a replace is allowed
276     if (!objects || objects.get('length')===0) {
277       if (!this.get('canRemoveContent')) {
278         throw "%@ cannot remove objects from the current content".fmt(this);
279       }
280     } else if (!this.get('canReorderContent')) {
281       throw "%@ cannot add or reorder the current content".fmt(this);
282     }    
283     
284     // if we can do this, then just forward the change.  This should fire
285     // updates back up the stack, updating rangeObservers, etc.
286     var content = this.get('content'); // note: use content, not observable
287     var objsToDestroy = [], i, objsLen;
288     if (this.get('destroyOnRemoval')){
289       for(i=0; i<amt; i++){
290         objsToDestroy.push(content.objectAt(i+start));
291       }
292     }
293     
294     if (content) content.replace(start, amt, objects);
295     for(i=0, objsLen = objsToDestroy.length; i<objsLen; i++){
296       
297       objsToDestroy[i].destroy();
298     }
299     objsToDestroy = null;
300     
301     return this; 
302   },
303 
304   indexOf: function(object, startAt) {
305     var content = this._scac_observableContent();
306     return content ? content.indexOf(object, startAt) : -1;
307   },

313   /** @private */
314   init: function() {
315     sc_super();
316     this._scac_contentDidChange();
317   },
318   
319   /** @private
320     Cached observable content property.  Set to NO to indicate cache is 
321     invalid.
322   */
323   _scac_cached: NO,
324   
325   /**
326     @private
327     
328     Returns the current array this controller is actually managing.  Usually
329     this should be the same as the content property, but sometimes we need to
330     generate something different because the content is not a regular array.
331     
332     Passing YES to the force parameter will force this value to be recomputed.
333   
334     @returns {SC.Array} observable or null
335   */
336   _scac_observableContent: function() {
337     var ret = this._scac_cached;
338     if (ret !== NO) return ret;
339     
340     var content = this.get('content'),
341         orderBy, func, t, len;
342     
343     // empty content
344     if (SC.none(content)) return this._scac_cached = [];
345 
346     // wrap non-enumerables
347     if (!content.isEnumerable) {
348       ret = this.get('allowsSingleContent') ? [content] : [];
349       return (this._scac_cached = ret);
350     } 
351     
352     // no-wrap
353     orderBy = this.get('orderBy');
354     if (!orderBy) {
355       if (content.isSCArray) return (this._scac_cached = content) ;
356       else throw "%@.orderBy is required for unordered content".fmt(this);     
357     }
358     
359     // all remaining enumerables must be sorted.
360     
361     // build array - then sort it
362     switch(SC.typeOf(orderBy)) {
363     case SC.T_STRING:
364       orderBy = [orderBy];
365       break;
366     case SC.T_FUNCTION:
367       func = orderBy ;
368       break;
369     case SC.T_ARRAY:
370       break;
371     default:
372       throw "%@.orderBy must be Array, String, or Function".fmt(this);
373     }
374         
375     // generate comparison function if needed - use orderBy
376     if (!func) {  
377       len = orderBy.get('length');
378       func = function(a,b) {
379         var idx=0, status=0, key, aValue, bValue, descending;
380         for(idx=0;(idx<len)&&(status===0);idx++) {
381           key = orderBy.objectAt(idx);
382           descending = NO;
383           
384           if (key.indexOf('ASC') > -1) {
385             key = key.split('ASC ')[1];
386           } else if (key.indexOf('DESC') > -1) {
387             key = key.split('DESC ')[1];
388             descending = YES;
389           }
390         
391           if (!a) aValue = a ;
392           else if (a.isObservable) aValue = a.get(key);
393           else aValue = a[key];
394 
395           if (!b) bValue = b ;
396           else if (b.isObservable) bValue = b.get(key);
397           else bValue = b[key];
398         
399           status = SC.compare(aValue, bValue);
400           if (descending) status = (-1) * status;
401         }
402         return status ;
403       };
404     }
405 
406     ret = [];
407     content.forEach(function(o) { ret.push(o); });
408     ret.sort(func);
409     
410     func = null ; // avoid memory leaks
411     return (this._scac_cached = ret) ;
412   },

414   /** @private
415     Whenever content changes, setup and teardown observers on the content
416     as needed.
417   */
418   _scac_contentDidChange: function() {
419 
420     this._scac_cached = NO; // invalidate observable content
421     
422     var cur    = this.get('content'),
423         orders = !!this.get('orderBy'),
424         last   = this._scac_content,
425         oldlen = this._scac_length || 0,
426         ro     = this._scac_rangeObserver,
427         func   = this._scac_rangeDidChange,
428         efunc  = this._scac_enumerableDidChange,
429         sfunc  = this._scac_contentStatusDidChange,
430         newlen;
431         
432     if (last === cur) return this; // nothing to do
433 
434     // teardown old observer
435     if (last) {
436       if (ro && last.isSCArray) last.removeRangeObserver(ro);
437       else if (last.isEnumerable) last.removeObserver('[]', this, efunc);
438       last.removeObserver('status', this, sfunc);
439     }
440     
441     ro = null;
442     
443     // save new cached values 
444     this._scac_cached = NO;
445     this._scac_content = cur ;
446     
447     // setup new observers
448     // also, calculate new length.  do it manually instead of using 
449     // get(length) because we want to avoid computed an ordered array.
450     if (cur) {
451       if (!orders && cur.isSCArray) ro = cur.addRangeObserver(null,this,func);
452       else if (cur.isEnumerable) cur.addObserver('[]', this, efunc);
453       newlen = cur.isEnumerable ? cur.get('length') : 1; 
454       cur.addObserver('status', this, sfunc);
455       
456     } else newlen = SC.none(cur) ? 0 : 1;
457 
458     this._scac_rangeObserver = ro;
459     
460 
461     // finally, notify enumerable content has changed.
462     this._scac_length = newlen;
463     this._scac_contentStatusDidChange();
464     this.enumerableContentDidChange(0, newlen, newlen - oldlen);
465     this.updateSelectionAfterContentChange();
466   }.observes('content'),
467   
468   /** @private
469     Whenever enumerable content changes, need to regenerate the 
470     observableContent and notify that the range has changed.  
471     
472     This is called whenever the content enumerable changes or whenever orderBy
473     changes.
474   */
475   _scac_enumerableDidChange: function() {
476     var content = this.get('content'), // use content directly
477         newlen  = content ? content.get('length') : 0,
478         oldlen  = this._scac_length;
479         
480     this._scac_length = newlen;
481     this.beginPropertyChanges();
482     this._scac_cached = NO; // invalidate
483     this.enumerableContentDidChange(0, newlen, newlen-oldlen);
484     this.endPropertyChanges();
485     this.updateSelectionAfterContentChange();
486   }.observes('orderBy'),
487   
488   /** @private
489     Whenever array content changes, need to simply forward notification.
490     
491     Assumes that content is not null and is SC.Array.
492   */
493   _scac_rangeDidChange: function(array, objects, key, indexes) {
494     if (key !== '[]') return ; // nothing to do
495     
496     var content = this.get('content');
497     this._scac_length = content.get('length');
498     this._scac_cached = NO; // invalidate
499     
500     // if array length has changed, just notify every index from min up
501     if (indexes) {
502       this.beginPropertyChanges();
503       indexes.forEachRange(function(start, length) {
504         this.enumerableContentDidChange(start, length, 0);
505       }, this);
506       this.endPropertyChanges();
507       this.updateSelectionAfterContentChange();
508     }
509   },
510   
511   /** @private
512     Whenver the content "status" property changes, relay out.
513   */
514   _scac_contentStatusDidChange: function() {
515     this.notifyPropertyChange('status');
516   }
