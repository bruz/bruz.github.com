 30 /** @class
 31   Displays a preformatted modal alert pane.
 32   
 33   Alert panes are a simple way to provide modal messaging that otherwise 
 34   blocks the user's interaction with your application.  Alert panes are 
 35   useful for showing important error messages and confirmation dialogs.  They
 36   provide a better user experience than using the OS-level alert dialogs.
 37   
 38   h1. Displaying an Alert Pane
 39   
 40   The easiest way to display an alert pane is to use one of the various 
 41   class methods defined on SC.AlertPane, passing the message and an optional
 42   detailed description and caption.  
 43   
 44   There are four variations of this method can you can invoke:  
 45   
 46   - *warn()* - displays an alert pane with a warning icon to the left.
 47   - *error()* - displays an alert with an error icon to the left
 48   - *info()* - displays an alert with an info icon to the left
 49   - *plain()* - displays an alert w/o any icon
 50   - *show()* - displays an alert with a customizable icon to the left
 51   
 52   In addition to passing a message, description and caption, you can also customize
 53   the title of the button 1 (OK) and add an optional button 2 and 3 (Cancel or Extra).  Just
 54   pass these titles of these buttons to enable them or null to disable then.
 55   
 56   Additionally, you can pass a delegate object as the last parameter.  This
 57   delegate's 'alertPaneDidDismiss()' method will be called when the pane
 58   is dismissed, passing the pane instance and a key indicating which 
 59   button was pressed.
 60   
 61   h1. Examples
 62   
 63   Show a simple AlertPane with an OK button:
 64   
 65   {{{
 66     SC.AlertPane.warn("Could not load calendar", "Your internet connection may be unavailable or our servers may be down.  Try again in a few minutes.");
 67   }}}
 68   
 69   Show an AlertPane with a customized OK title (title will be 'Try Again'):
 70   
 71   {{{
 72     SC.AlertPane.warn("Could not load calendar", "Your internet connection may be unavailable or our servers may be down.  Try again in a few minutes.", "Try Again");
 73   }}}
 74   
 75   Show an AlertPane with a custom OK, a Cancel button and an Extra button, 
 76   each with custom titles.  Also, pass a delegate that will be invoked when
 77   the user's dismisses the dialog.
 78   
 79   {{{
 80 
 81     MyApp.calendarController = SC.Object.create({
 82       alertPaneDidDismiss: function(pane, status) {
 83         switch(status) {
 84           case SC.BUTTON1_STATUS:
 85             this.tryAgain();
 86             break;
 87             
 88           case SC.BUTTON2_STATUS:
 89             // do nothing
 90             break;
 91             
 92           case SC.BUTTON3_STATUS:
 93             this.showMoreInfo();
 94             break;
 95         }
 96       },
 97       
 98       ...
 99     });
100     
101     SC.AlertPane.warn("Could not load calendar", "Your internet connection may be unavailable or our servers may be down.  Try again in a few minutes.", "Try Again", "Cancel", "More Info...", MyApp.calendarController);
102   }}}
103   
104   @extends SC.PanelPane
105   @since SproutCore 1.0
106 */
107 SC.AlertPane = SC.PanelPane.extend({
108   
109   classNames: 'sc-alert',
110   
111   /**
112     The delegate to notify when the pane is dismissed.  If you set a 
113     delegate, it should respond to the method:
114     
115     {{{
116       alertPaneDidDismiss: function(pane, status)
117     }}}
118     
119     The status will be on of SC.BUTTON1_STATUS, SC.BUTTON2_STATUS or SC.BUTTON3_STATUS
120     depending on which button was clicked.
121     
122     @property {Object}
123   */
124   delegate: null,
125 
126   /**
127     The icon URL or class name.  If you do not set this, an alert icon will
128     be shown instead.
129     
130     @property {String}
131   */
132   icon: 'sc-icon-alert-48',
133 
134   /**
135     The primary message to display.  This message will appear in large bold
136     type at the top of the alert.
137     
138     @property {String}
139   */
140   message: "",
141 
142   /**
143     An optional detailed decription.  Use this string to provide further 
144     explanation of the condition and, optionally, ways the user can resolve
145     the problem.
146     
147     @property {String}
148   */
149   description: "",
150   
151   displayDescription: function() {
152     var desc = this.get('description');
153     if (!desc || desc.length === 0) return desc ;
154     
155     desc = SC.RenderContext.escapeHTML(desc); // remove HTML
156     return '<p class="description">' + desc.split('\n').join('</p><p class="description">') + '</p>';
157   }.property('description').cacheable(),
158 
159   /**
160     An optional detailed caption.  Use this string to provide further 
161     fine print explanation of the condition and, optionally, ways the user can resolve
162     the problem.
163     
164     @property {String}
165   */
166   caption: "",
167   
168   displayCaption: function() {
169     var caption = this.get('caption');
170     if (!caption || caption.length === 0) return caption ;
171     
172     caption = SC.RenderContext.escapeHTML(caption); // remove HTML
173     return '<p class="caption">' + caption.split('\n').join('</p><p class="caption">') + '</p>';
174   }.property('caption').cacheable(),
175   
176   /**
177     The button view for the button 1 (OK).
178     
179     @property {SC.ButtonView}
180   */
181   buttonOne: SC.outlet('contentView.childViews.1.childViews.1'),
182 
183   /**
184     The button view for the button 2 (Cancel).
185     
186     @property {SC.ButtonView}
187   */
188   buttonTwo: SC.outlet('contentView.childViews.1.childViews.0'),
189 
190   /**
191     The button view for the button 3 (Extra).
192     
193     @property {SC.ButtonView}
194   */
195   buttonThree: SC.outlet('contentView.childViews.2.childViews.0'),
196 
197   /**
198     The view for the button 3 (Extra) wrapper.
199     
200     @property {SC.View}
201   */
202   buttonThreeWrapper: SC.outlet('contentView.childViews.2'),
203   
204   layout: { top : 0.3, centerX: 0, width: 500 },
205 
206   /** @private - internal view that is actually displayed */
207   contentView: SC.View.extend({
208     
209     useStaticLayout: YES,
210     
211     layout: { left: 0, right: 0, top: 0, height: "auto" },
212 		
213     childViews: [
214       SC.View.extend(SC.StaticLayout, {
215         classNames: ['info'],
216 
217         render: function(context, firstTime) {
218           var pane = this.get('pane');
219           var blank = SC.BLANK_IMAGE_URL ;
220           if(pane.get('icon') == 'blank') context.addClass('plain');
221           context.push('<img src="'+blank+'" class="icon '+pane.get('icon')+'" />');
222           context.begin('h1').attr('class', 'header').text(pane.get('message') || '').end();
223           context.push(pane.get('displayDescription') || '');
224           context.push(pane.get('displayCaption') || '');
225           context.push('<div class="separator"></div>');
226         }
227       }),
228 
229       SC.View.extend({
230         layout: { bottom: 13, height: 24, right: 18, width: 466 },
231         childViews: ['cancelButton', 'okButton'],
232         classNames: ['text-align-right'],
233         cancelButton : SC.ButtonView.extend({
234             useStaticLayout: YES,
235             actionKey: SC.BUTTON2_STATUS,
236             localize: YES,
237             titleMinWidth: 64,
238             layout: { right: 5, height: 'auto', width: 'auto', bottom: 0 },
239             theme: 'capsule',
240             title: "Cancel", 
241             isCancel: YES,
242             action: "dismiss",
243             isVisible: NO
244           }),
245 
246         okButton : SC.ButtonView.extend({
247             useStaticLayout: YES,
248             actionKey: SC.BUTTON1_STATUS,
249             localize: YES,
250             titleMinWidth: 64,
251             layout: { left: 0, height: 'auto', width: 'auto', bottom: 0 },
252             theme: 'capsule',
253             title: "OK", 
254             isDefault: YES,
255             action: "dismiss"
256           })
257       }),
258       
259       SC.View.extend({
260         layout: { bottom: 13, height: 24, left: 18, width: 150 },
261         isVisible: NO,
262         childViews: [
263           SC.ButtonView.extend({
264             useStaticLayout: YES,
265             actionKey: SC.BUTTON3_STATUS,
266             localize: YES,
267             titleMinWidth: 64,
268             layout: { left: 0, height: 'auto', width: 'auto', bottom: 0 },
269             theme: 'capsule',
270             title: "Extra", 
271             action: "dismiss",
272             isVisible: NO
273           })]
274       })]
275   }),
276 
277   /**
278     Action triggered whenever any button is pressed.  Notifies any delegate
279     and then hides the alert pane.
280   */
281   dismiss: function(sender) {
282     var del = this.delegate;
283     if (del && del.alertPaneDidDismiss) {
284       del.alertPaneDidDismiss(this, sender.get('actionKey'));
285     }
286     this.remove(); // hide alert
287   },
288   
289   /** @private 
290     Executes whenever one of the icon, message, description or caption is changed.
291     This simply causes the UI to refresh.
292   */
293   alertInfoDidChange: function() {
294     var v = this.getPath('contentView.childViews.0');
295     if (v) v.displayDidChange(); // re-render message
296   }.observes('icon', 'message', 'displayDescription', 'displayCaption')
297 });

299 /** @private
300   internal method normalizes arguments for processing by helper methods.
301 */
302 SC.AlertPane._normalizeArguments = function(args) {
303   args = SC.A(args); // convert to real array
304   var len = args.length, delegate = args[len-1];
305   if (SC.typeOf(delegate) !== SC.T_STRING) {
306     args[len-1] = null;
307   } else delegate = null ;
308   args[7] = delegate ;
309   return args ;
310 };

312 /**
313   Displays a new alert pane according to the passed parameters.  Every 
314   parameter except for the message is optional.  You can always pass the 
315   delegate as the last parameter and it will be used, even if you omit items
316   in between.
317   
318   If you need to pass other parameters but you want to omit some others 
319   in between, pass null and the related UI item will be hidden
320   
321   Note that if you pass an icon, it should be 48 x 48 in size.
322   
323   @param {String} message the primary message
324   @param {String} description an optional detailed description
325   @param {String} caption an optional detailed fine print caption
326   @param {String} button1Title optional unlocalized title for button 1 (OK)
327   @param {String} button2Title optional unlocalized title for button 2 (Cancel)
328   @param {String} button3Title optional unlocalized title for button 3 (extra)
329   @param {String} iconUrl optional URL or class name for icon.
330   @param {Object} delegate optional delegate to notify when pane is dismissed
331   @returns {SC.AlertPane} new alert pane
332 */
333 SC.AlertPane.show = function(message, description, caption, button1Title, button2Title, button3Title, iconUrl, delegate) {
334   
335   // get the delegate and normalize the rest of the params
336   var args = this._normalizeArguments(arguments);
337   
338   // create basic AlertPane
339   var ret = this.create({
340     message: args[0] || '',
341     description: args[1] || null,
342     caption: args[2] || null,
343     icon: args[6] || 'sc-icon-alert-48',
344     delegate: args[7]
345   });
346   
347   // customize buttons as needed
348   var buttonKeys = 'buttonOne buttonTwo buttonThree'.w(), button, title;
349   for(var idx=0;idx<3;idx++) {
350     button = ret.get(buttonKeys[idx]);
351     title = args[idx + 3];
352     if (title) {
353       button.set('title', title).set('isVisible', YES);
354       if(title=='?') button.set('titleMinWidth', 0);
355       if (idx==2) {
356         var button_wrapper = ret.get('buttonThreeWrapper');
357         button_wrapper.set('isVisible', YES);
358       }
359     }
360   }
361   var show = ret.append() ; // make visible.
362   ret.adjust('height', ret.childViews[0].$().height()) ;
363   ret.updateLayout() ;
364   return show ;
365 };

367 /**
368   Displays a warning alert pane.  See SC.AlertPane.show() for complete details. 
369   
370   @returns {SC.AlertPane} the pane
371 */
372 SC.AlertPane.warn = function(message, description, caption, button1Title, button2Title, button3Title, delegate) {
373   var args = this._normalizeArguments(arguments);
374   args[6] = 'sc-icon-alert-48';
375   return this.show.apply(this, args);
376 };
