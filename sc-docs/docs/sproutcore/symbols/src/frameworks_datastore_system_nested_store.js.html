 10 /**
 11   @class
 12 
 13   A nested store can buffer changes to a parent store and then commit them
 14   all at once.  You usually will use a NestedStore as part of store chaining
 15   to stage changes to your object graph before sharing them with the rest of
 16   the application.
 17   
 18   Normally you will not create a nested store directly.  Instead, you can 
 19   retrieve a nested store by using the chain() method.  When you are finished
 20   working with the nested store, destroy() will dispose of it.
 21   
 22   @extends SC.Store
 23   @since SproutCore 1.0
 24 */
 25 SC.NestedStore = SC.Store.extend(
 26 /** @scope SC.NestedStore.prototype */ {
 27 
 28   /**
 29     This is set to YES when there are changes that have not been committed 
 30     yet.
 31 
 32     @property {Boolean}
 33     @default NO
 34   */
 35   hasChanges: NO,
 36 
 37   /**
 38     The parent store this nested store is chained to.  Nested stores must have
 39     a parent store in order to function properly.  Normally, you create a 
 40     nested store using the SC.Store#chain() method and this property will be
 41     set for you.
 42     
 43     @property {SC.Store}
 44   */
 45   parentStore: null,
 46 
 47   /**
 48     YES if the view is nested. Walk like a duck
 49     
 50     @property {Boolean}
 51   */
 52   isNested: YES,
 53 
 54   /**
 55     If YES, then the attribute hash state will be locked when you first 
 56     read the data hash or status.  This means that if you retrieve a record
 57     then change the record in the parent store, the changes will not be 
 58     visible to your nested store until you commit or discard changes.
 59     
 60     If NO, then the attribute hash will lock only when you write data.
 61     
 62     Normally you want to lock your attribute hash the first time you read it.
 63     This will make your nested store behave most consistently.  However, if
 64     you are using multiple sibling nested stores at one time, you may want
 65     to turn off this property so that changes from one store will be reflected
 66     in the other one immediately.  In this case you will be responsible for
 67     ensuring that the sibling stores do not edit the same part of the object
 68     graph at the same time.
 69     
 70     @property {Boolean} 
 71   */
 72   lockOnRead: YES,
 73 
 74   /** @private
 75     Array contains the base revision for an attribute hash when it was first
 76     cloned from the parent store.  If the attribute hash is edited and 
 77     commited, the commit will fail if the parent attributes hash has been 
 78     edited since.
 79     
 80     This is a form of optimistic locking, hence the name.
 81     
 82     Each store gets its own array of locks, which are selectively populated
 83     as needed.
 84     
 85     Note that this is kept as an array because it will be stored as a dense 
 86     array on some browsers, making it faster.
 87     
 88     @property {Array}
 89   */
 90   locks: null,
 91 
 92   /** @private
 93     An array that includes the store keys that have changed since the store
 94     was last committed.  This array is used to sync data hash changes between
 95     chained stores.  For a log changes that may actually be committed back to
 96     the server see the changelog property.
 97     
 98     @property {SC.Set}
 99   */
100   chainedChanges: null,
101

106   /**
107     find() cannot accept REMOTE queries in a nested store.  This override will
108     verify that condition for you.  See SC.Store#find() for info on using this
109     method.
110     
111     @returns {SC.Record|SC.RecordArray}
112   */
113   find: function(query) {
114     if (query && query.isQuery && query.get('location') !== SC.Query.LOCAL) {
115       throw "SC.Store#find() can only accept LOCAL queries in nested stores";
116     }
117     return sc_super();
118   },

120   /**
121     Propagate this store's changes to its parent.  If the store does not 
122     have a parent, this has no effect other than to clear the change set.
123 
124     @param {Boolean} force if YES, does not check for conflicts first
125     @returns {SC.Store} receiver
126   */
127   commitChanges: function(force) {
128     if (this.get('hasChanges')) {
129       var pstore = this.get('parentStore');
130       pstore.commitChangesFromNestedStore(this, this.chainedChanges, force);
131     }
132 
133     // clear out custom changes - even if there is nothing to commit.
134     this.reset();
135     return this ;
136   },
137 
138   /**
139     Discard the changes made to this store and reset the store.
140     
141     @returns {SC.Store} receiver
142   */
143   discardChanges: function() {
144     // any locked records whose rev or lock rev differs from parent need to
145     // be notified.
146     var records, locks;
147     if ((records = this.records) && (locks = this.locks)) {
148       var pstore = this.get('parentStore'), psRevisions = pstore.revisions;
149       var revisions = this.revisions, storeKey, lock, rev;
150       for (storeKey in records) {
151         if (!records.hasOwnProperty(storeKey)) continue ;
152         if (!(lock = locks[storeKey])) continue; // not locked.
153 
154         rev = psRevisions[storeKey];
155         if ((rev !== lock) || (revisions[storeKey] > rev)) {
156           this._notifyRecordPropertyChange(parseInt(storeKey, 10));
157         }
158       }
159     }   
160     
161     this.reset();
162     this.flush();
163     return this ;
164   },
165   
166   /**
167     When you are finished working with a chained store, call this method to 
168     tear it down.  This will also discard any pending changes.
169     
170     @returns {SC.Store} receiver
171   */
172   destroy: function() {
173     this.discardChanges();
174     
175     var parentStore = this.get('parentStore');
176     if (parentStore) parentStore.willDestroyNestedStore(this);
177     
178     sc_super();  
179     return this ;
180   },

182   /**
183     Resets a store's data hash contents to match its parent.
184     
185     @returns {SC.Store} receiver
186   */
187   reset: function() {
188 
189     // requires a pstore to reset
190     var parentStore = this.get('parentStore');
191     if (!parentStore) throw SC.Store.NO_PARENT_STORE_ERROR;
192     
193     // inherit data store from parent store.
194     this.dataHashes = SC.beget(parentStore.dataHashes);
195     this.revisions  = SC.beget(parentStore.revisions);
196     this.statuses   = SC.beget(parentStore.statuses);
197     
198     // also, reset private temporary objects
199     this.chainedChanges = this.locks = this.editables = null;
200     this.changelog = null ;
201 
202     // TODO: Notify record instances
203     
204     this.set('hasChanges', NO);
205   },

246   // CORE ATTRIBUTE API
247   // 
248   // The methods in this layer work on data hashes in the store.  They do not
249   // perform any changes that can impact records.  Usually you will not need 
250   // to use these methods.

252   /**
253     Returns the current edit status of a storekey.  May be one of INHERITED,
254     EDITABLE, and LOCKED.  Used mostly for unit testing.
255     
256     @param {Number} storeKey the store key
257     @returns {Number} edit status
258   */
259   storeKeyEditState: function(storeKey) {
260     var editables = this.editables, locks = this.locks;
261     return (editables && editables[storeKey]) ? SC.Store.EDITABLE : (locks && locks[storeKey]) ? SC.Store.LOCKED : SC.Store.INHERITED ;
262   },
263    
264   /**  @private
265     Locks the data hash so that it iterates independently from the parent 
266     store.
267   */
268   _lock: function(storeKey) {
269     var locks = this.locks, rev, editables;
270     
271     // already locked -- nothing to do
272     if (locks && locks[storeKey]) return this;
273 
274     // create locks if needed
275     if (!locks) locks = this.locks = [];
276 
277     // fixup editables
278     editables = this.editables;
279     if (editables) editables[storeKey] = 0;
280     
281     
282     // if the data hash in the parent store is editable, then clone the hash
283     // for our own use.  Otherwise, just copy a reference to the data hash
284     // in the parent store. -- find first non-inherited state
285     var pstore = this.get('parentStore'), editState;
286     while(pstore && (editState=pstore.storeKeyEditState(storeKey)) === SC.Store.INHERITED) {
287       pstore = pstore.get('parentStore');
288     }
289     
290     if (pstore && editState === SC.Store.EDITABLE) {
291       this.dataHashes[storeKey] = SC.clone(pstore.dataHashes[storeKey]);
292       if (!editables) editables = this.editables = [];
293       editables[storeKey] = 1 ; // mark as editable
294       
295     } else this.dataHashes[storeKey] = this.dataHashes[storeKey];
296     
297     // also copy the status + revision
298     this.statuses[storeKey] = this.statuses[storeKey];
299     rev = this.revisions[storeKey] = this.revisions[storeKey];
300     
301     // save a lock and make it not editable
302     locks[storeKey] = rev || 1;    
303     
304     return this ;
305   },
306   
307   /** @private - adds chaining support */
308   readDataHash: function(storeKey) {
309     if (this.get('lockOnRead')) this._lock(storeKey);
310     return this.dataHashes[storeKey];
311   },

313   /** @private - adds chaining support */
314   readEditableDataHash: function(storeKey) {
315 
316     // lock the data hash if needed
317     this._lock(storeKey);
318     
319     return sc_super();
320   },

322   /** @private - adds chaining support - 
323     Does not call sc_super because the implementation of the method vary too
324     much. 
325   */
326   writeDataHash: function(storeKey, hash, status) {
327     var locks = this.locks, didLock = NO, rev ;
328 
329     // Update our dataHash and/or status, depending on what was passed in.
330     // Note that if no new hash was passed in, we'll lock the storeKey to
331     // properly fork our dataHash from our parent store.  Similarly, if no
332     // status was passed in, we'll save our own copy of the value.
333     if (hash) {
334       this.dataHashes[storeKey] = hash;
335     }
336     else {
337       this._lock(storeKey);
338       didLock = YES;
339     }
340 
341     if (status) {
342       this.statuses[storeKey] = status;
343     }
344     else {
345       if (!didLock) this.statuses[storeKey] = (this.statuses[storeKey] || SC.Record.READY_NEW);
346     }
347 
348     if (!didLock) {
349       rev = this.revisions[storeKey] = this.revisions[storeKey]; // copy ref
350     
351       // make sure we lock if needed.
352       if (!locks) locks = this.locks = [];
353       if (!locks[storeKey]) locks[storeKey] = rev || 1;
354     }
355     
356     // Also note that this hash is now editable.  (Even if we locked it,
357     // above, it may not have been marked as editable.)
358     var editables = this.editables;
359     if (!editables) editables = this.editables = [];
360     editables[storeKey] = 1 ; // use number for dense array support
361     
362     return this ;
363   },
364 
365   /** @private - adds chaining support */
366   removeDataHash: function(storeKey, status) {
367     
368     // record optimistic lock revision
369     var locks = this.locks;
370     if (!locks) locks = this.locks = [];
371     if (!locks[storeKey]) locks[storeKey] = this.revisions[storeKey] || 1;
372 
373     return sc_super();
374   },

376   /** @private - book-keeping for a single data hash. */
377   dataHashDidChange: function(storeKeys, rev, statusOnly, key) {
378     
379     // update the revision for storeKey.  Use generateStoreKey() because that
380     // gaurantees a universally (to this store hierarchy anyway) unique 
381     // key value.
382     if (!rev) rev = SC.Store.generateStoreKey();
383     var isArray, len, idx, storeKey;
384     
385     isArray = SC.typeOf(storeKeys) === SC.T_ARRAY;
386     if (isArray) {
387       len = storeKeys.length;
388     } else {
389       len = 1;
390       storeKey = storeKeys;
391     }
392 
393     var changes = this.chainedChanges;
394     if (!changes) changes = this.chainedChanges = SC.Set.create();
395     
396     for(idx=0;idx<len;idx++) {
397       if (isArray) storeKey = storeKeys[idx];
398       this._lock(storeKey);
399       this.revisions[storeKey] = rev;
400       changes.add(storeKey);
401       this._notifyRecordPropertyChange(storeKey, statusOnly, key);
402     }
403 
404     this.setIfChanged('hasChanges', YES);
405     return this ;
406   },

412   /** @private - adapt for nested store */
413   commitChangesFromNestedStore: function(nestedStore, changes, force) {
414 
415     sc_super();
416     
417     // save a lock for each store key if it does not have one already
418     // also add each storeKey to my own changes set.
419     var pstore = this.get('parentStore'), psRevisions = pstore.revisions, i;
420     var myLocks = this.locks, myChanges = this.chainedChanges,len,storeKey;
421     if (!myLocks) myLocks = this.locks = [];
422     if (!myChanges) myChanges = this.chainedChanges = SC.Set.create();
423 
424     len = changes.length ;
425     for(i=0;i<len;i++) {
426       storeKey = changes[i];
427       if (!myLocks[storeKey]) myLocks[storeKey] = psRevisions[storeKey]||1;
428       myChanges.add(storeKey);
429     }
430     
431     // Finally, mark store as dirty if we have changes
432     this.setIfChanged('hasChanges', myChanges.get('length')>0);
433     this.flush();
434     
435     return this ;
436   },

460   /** @private - adapt for nested store
461   
462     Unlike for the main store, for nested stores if isRefresh=YES, we'll throw
463     an error if the record is dirty.  We'll otherwise avoid setting our status
464     because that can disconnect us from upper and/or lower stores.
465   */
466   retrieveRecords: function(recordTypes, ids, storeKeys, isRefresh) {
467     var pstore = this.get('parentStore'), idx, storeKey, newStatus,
468       len = (!storeKeys) ? ids.length : storeKeys.length,
469       K = SC.Record, status;
470 
471     // Is this a refresh?
472     if (isRefresh) {
473       for(idx=0;idx<len;idx++) {
474         storeKey = !storeKeys ? pstore.storeKeyFor(recordTypes, ids[idx]) : storeKeys[idx];
475         status   = this.peekStatus(storeKey);
476         
477         // We won't allow calling retrieve on a dirty record in a nested store
478         // (although we do allow it in the main store).  This is because doing
479         // so would involve writing a unique status, and that would break the
480         // status hierarchy, so even though lower stores would complete the
481         // retrieval, the upper layers would never inherit the new statuses.
482         if (status & K.DIRTY) {
483           throw SC.Store.NESTED_STORE_RETRIEVE_DIRTY_ERROR;
484         }
485         else {
486           // Not dirty?  Then abandon any status we had set (to re-establish
487           // any prototype linkage breakage) before asking our parent store to
488           // perform the retrieve.
489           var dataHashes = this.dataHashes,
490               revisions  = this.revisions,
491               statuses   = this.statuses,
492               editables  = this.editables,
493               locks      = this.locks;
494 
495           var changed    = NO;
496           var statusOnly = NO;
497   
498           if (dataHashes  &&  dataHashes.hasOwnProperty(storeKey)) {
499             delete dataHashes[storeKey];
500             changed = YES;
501           }
502           if (revisions   &&  revisions.hasOwnProperty(storeKey)) {
503             delete revisions[storeKey];
504             changed = YES;
505           }
506           if (editables) delete editables[storeKey];
507           if (locks) delete locks[storeKey];
508 
509           if (statuses  &&  statuses.hasOwnProperty(storeKey)) {
510             delete statuses[storeKey];
511             if (!changed) statusOnly = YES;
512             changed = YES;
513           }
514           
515           if (changed) this._notifyRecordPropertyChange(storeKey, statusOnly);
516         }
517       }
518     }
519     
520     return pstore.retrieveRecords(recordTypes, ids, storeKeys, isRefresh);
521   },
