  8 /**
  9   @namespace
 10 
 11   The SC.ContentDisplay mixin makes it easy to automatically update your view
 12   display whenever relevant properties on a content object change.  To use
 13   this mixin, include it in your view and then add the names of the 
 14   properties on the content object you want to trigger a displayDidChange() 
 15   method on your view. Your updateDisplay() method will then be called at the 
 16   end of the run loop.
 17   
 18   h2. Example
 19   
 20   {{{
 21     MyApp.MyViewClass = SC.View.extend(SC.ContentDisplay, { 
 22       contentDisplayProperties: 'title isEnabled hasChildren'.w(),
 23       ...
 24     });
 25   }}}
 26   
 27   @since SproutCore 1.0
 28 */
 29 SC.ContentDisplay = {
 30   
 31   /** @private */
 32   concatenatedProperties: 'contentDisplayProperties',
 33 
 34   /** @private */
 35   displayProperties: ['content'],
 36   
 37   /** 
 38     Add an array with the names of any property on the content object that
 39     should trigger an update of the display for your view.  Changes to the
 40     content object will only invoke your display method once per runloop.
 41     
 42     @property {Array}
 43   */
 44   contentDisplayProperties: [],
 45   
 46   /** @private */
 47   _display_contentDidChange: function(target, key, value) {
 48     // handle changes to the content...
 49     if ((value = this.get('content')) != this._display_content) {
 50 
 51       // get the handler method
 52       var f = this._display_contentPropertyDidChange ;
 53       
 54       // stop listening to old content.
 55       var content = this._display_content;
 56       if (content) {
 57         if (SC.isArray(content)) {
 58           content.invoke('removeObserver', '*', this, f) ;
 59         } else if (content.removeObserver) {
 60           content.removeObserver('*', this, f) ;
 61         }
 62       }
 63       
 64       // start listening for changes on the new content object.
 65       content = this._display_content = value ; 
 66       if (content) {
 67         if (SC.isArray(content)) {
 68           content.invoke('addObserver', '*', this, f) ;
 69         } else if (content.addObserver) {
 70           content.addObserver('*', this, f) ;
 71         }
 72       }
 73 
 74       // notify everyone that everything is different now.
 75       this.allPropertiesDidChange() ;
 76       this.endPropertyChanges() ;
 77     }
 78   }.observes('content', 'contentDisplayProperties'),
 79   
 80   /** @private Invoked when properties on the content object change. */
 81   _display_contentPropertyDidChange: function(target, key, value, propertyRevision) {
 82     if (key === '*') {
 83       this.displayDidChange() ;
 84     } else {
 85       // only update if a displayProperty actually changed...s
 86       var ary = this.get('contentDisplayProperties') ;
 87       if (ary && ary.indexOf(key)>=0) this.displayDidChange();
 88     }
 89   }
 90   
 91 } ;
