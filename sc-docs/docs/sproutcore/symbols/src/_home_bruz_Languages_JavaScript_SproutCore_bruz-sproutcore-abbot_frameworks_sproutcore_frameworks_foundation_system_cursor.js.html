 26 /**
 27   @class SC.Cursor
 28 
 29   A Cursor object is used to sychronize the cursor used by multiple views at 
 30   the same time. For example, thumb views within a split view acquire a cursor
 31   instance from the split view and set it as their cursor. The split view is 
 32   able to update its cursor object to reflect the state of the split view.
 33   Because cursor objects are implemented internally with CSS, this is a very 
 34   efficient way to update the same cursor for a group of view objects.
 35   
 36   Note: This object creates an anonymous CSS class to represent the cursor. 
 37   The anonymous CSS class is automatically added by SproutCore to views that
 38   have the cursor object set as "their" cursor. Thus, all objects attached to 
 39   the same cursor object will have their cursors updated simultaneously with a
 40   single DOM call.
 41   
 42   @extends SC.Object
 43 */
 44 SC.Cursor = SC.Object.extend(
 45 /** @scope SC.Cursor.prototype */ {
 46   
 47   /** @private */
 48   init: function() {
 49     sc_super() ;
 50     
 51     // create a unique style rule and add it to the shared cursor style sheet
 52     var cursorStyle = this.get('cursorStyle') || SC.DEFAULT_CURSOR ,
 53         ss = this.constructor.sharedStyleSheet(),
 54         guid = SC.guidFor(this);
 55     
 56     if (ss.insertRule) { // WC3
 57       ss.insertRule(
 58         '.'+guid+' {cursor: '+cursorStyle+';}',
 59         ss.cssRules ? ss.cssRules.length : 0
 60       ) ;
 61     } else if (ss.addRule) { // IE
 62       ss.addRule('.'+guid, 'cursor: '+cursorStyle) ;
 63     }
 64     
 65     this.cursorStyle = cursorStyle ;
 66     this.className = guid ; // used by cursor clients...
 67     return this ;
 68   },
 69   
 70   /**
 71     This property is the connection between cursors and views. The default
 72     SC.View behavior is to add this className to a view's layer if it has
 73     its cursor property defined.
 74     
 75     @readOnly
 76     @property {String} the css class name updated by this cursor
 77   */
 78   className: null,
 79   
 80   /**
 81     @property {String} the cursor value, can be 'url("path/to/cursor")'
 82   */
 83   cursorStyle: SC.DEFAULT_CURSOR,
 84   
 85   /** @private */
 86   cursorStyleDidChange: function() {
 87     var cursorStyle, rule, selector, ss, rules, idx, len;
 88     cursorStyle = this.get('cursorStyle') || SC.DEFAULT_CURSOR;
 89     rule = this._rule;
 90     if (rule) {
 91       rule.style.cursor = cursorStyle ; // fast path
 92       return ;
 93     }
 94     
 95     // slow path, taken only once
 96     selector = '.'+this.get('className') ;
 97     ss = this.constructor.sharedStyleSheet() ;
 98     rules = (ss.cssRules ? ss.cssRules : ss.rules) || [] ;
 99     
100     // find our rule, cache it, and update the cursor style property
101     for (idx=0, len = rules.length; idx<len; ++idx) {
102       rule = rules[idx] ;
103       if (rule.selectorText === selector) {
104         this._rule = rule ; // cache for next time
105         rule.style.cursor = cursorStyle ; // update the cursor
106         break ;
107       }
108     }
109   }.observes('cursorStyle')
110   
111   // TODO implement destroy
112   
113 });
114 
115 /** @private */
116 SC.Cursor.sharedStyleSheet = function() {
117   var head, ss = this._styleSheet ;
118   if (!ss) {
119     // create the stylesheet object the hard way (works everywhere)
120     ss = document.createElement('style') ;
121     ss.type = 'text/css' ;
122     head = document.getElementsByTagName('head')[0] ;
123     if (!head) head = document.documentElement ; // fix for Opera
124     head.appendChild(ss) ;
125     
126     // get the actual stylesheet object, not the DOM element
127     ss = document.styleSheets[document.styleSheets.length-1] ;
128     this._styleSheet = ss ;
129   }
130   return ss ;
131 };
