 11 /** @class
 12   
 13   A list view renders vertical lists of items.  It is a specialized form of
 14   collection view that is simpler than the table view, but more refined than
 15   a generic collection.
 16   
 17   You can use a list view just like a collection view, except that often you
 18   also should provide a default rowHeight.  Setting this value will allow 
 19   the ListView to optimize its rendering.
 20   
 21   h2. Variable Row Heights
 22 
 23   Normally you set the row height through the rowHeight property.  You can 
 24   also support custom row heights by implementing the 
 25   contentCustomRowHeightIndexes property to return an index set.
 26   
 27   h2. Using ListView with Very Large Data Sets
 28   
 29   ListView implements incremental rendering, which means it will only render
 30   HTML for the items that are current visible on the screen.  You can use it
 31   to efficiently render lists with 100K+ items very efficiently.  
 32   
 33   If you need to work with very large lists of items, however, be aware that
 34   calculate variable rows heights can become very expensive since the list 
 35   view will essentially have to iterate over every item in the collection to
 36   collect its row height.  
 37   
 38   To work with very large lists, you should consider making your row heights
 39   uniform.  This will allow the list view to efficiently render content 
 40   without worrying about the overall performance.
 41   
 42   Alternatively, you may want to consider overriding the 
 43   offsetForRowAtContentIndex() and heightForRowAtContentIndex() methods to 
 44   perform some faster calculations that do not require inspecting every 
 45   item in the collection.
 46   
 47   Note that row heights and offsets are cached so once they are calculated
 48   the list view will be able to display very quickly.
 49   
 50   (Can we also have an 'estimate row heights' property that will simply 
 51   cheat for very long data sets to make rendering more efficient?)
 52   
 53   @extends SC.CollectionView
 54   @extends SC.CollectionRowDelegate
 55   @since SproutCore 1.0
 56 */

 88   /**
 89     Returns the current collectionRowDelegate.  This property will recompute
 90     everytime the content changes.
 91   */
 92   rowDelegate: function() {
 93     var del     = this.delegate,
 94         content = this.get('content');
 95     return this.delegateFor('isCollectionRowDelegate', del, content);
 96   }.property('delegate', 'content').cacheable(),

 98   /** @private 
 99     Whenever the rowDelegate changes, begin observing important properties
100   */
101   _sclv_rowDelegateDidChange: function() {
102     var last = this._sclv_rowDelegate,
103         del  = this.get('rowDelegate'),
104         func = this._sclv_rowHeightDidChange,
105         func2 = this._sclv_customRowHeightIndexesDidChange;
106         
107     if (last === del) return this; // nothing to do
108     this._sclv_rowDelegate = del; 
109 
110     // last may be null on a new object
111     if (last) {
112       last.removeObserver('rowHeight', this, func);
113       last.removeObserver('customRowHeightIndexes', this, func2);
114     }
115     
116     if (!del) {
117       throw "Internal Inconsistancy: ListView must always have CollectionRowDelegate";
118     }
119     
120     del.addObserver('rowHeight', this, func);
121     del.addObserver('customRowHeightIndexes', this, func2);
122     this._sclv_rowHeightDidChange()._sclv_customRowHeightIndexesDidChange();
123     return this ;
124   }.observes('rowDelegate'),
125 
126   /** @private 
127     called whenever the rowHeight changes.  If the property actually changed
128     then invalidate all row heights.
129   */
130   _sclv_rowHeightDidChange: function() {
131     var del = this.get('rowDelegate'),
132         height = del.get('rowHeight'), 
133         indexes;
134         
135     if (height === this._sclv_rowHeight) return this; // nothing to do
136     this._sclv_rowHeight = height;
137 
138     indexes = SC.IndexSet.create(0, this.get('length'));
139     this.rowHeightDidChangeForIndexes(indexes);
140     return this ;
141   },
142 
143   /** @private 
144     called whenever the customRowHeightIndexes changes.  If the property 
145     actually changed then invalidate affected row heights.
146   */
147   _sclv_customRowHeightIndexesDidChange: function() {
148     var del     = this.get('rowDelegate'),
149         indexes = del.get('customRowHeightIndexes'), 
150         last    = this._sclv_customRowHeightIndexes,
151         func    = this._sclv_customRowHeightIndexesContentDidChange;
152         
153     // nothing to do
154     if ((indexes===last) || (last && last.isEqual(indexes))) return this;
155 
156     // if we were observing the last index set, then remove observer
157     if (last && this._sclv_isObservingCustomRowHeightIndexes) {
158       last.removeObserver('[]', this, func);
159     }
160     
161     // only observe new index set if it exists and it is not frozen.
162     if (this._sclv_isObservingCustomRowHeightIndexes = indexes && !indexes.get('isFrozen')) {
163       indexes.addObserver('[]', this, func);
164     }
165     
166     this._sclv_customRowHeightIndexesContentDidChange();
167     return this ;
168   },
169 
170   /** @private
171     Called whenever the customRowHeightIndexes set is modified.
172   */
173   _sclv_customRowHeightIndexesContentDidChange: function() {
174     var del     = this.get('rowDelegate'),
175         indexes = del.get('customRowHeightIndexes'), 
176         last    = this._sclv_customRowHeightIndexes, 
177         changed;
178 
179     // compute the set to invalidate.  the union of cur and last set
180     if (indexes && last) {
181       changed = indexes.copy().add(last);
182     } else changed = indexes || last ;
183     this._sclv_customRowHeightIndexes = indexes ? indexes.frozenCopy() : null; 
184 
185     // invalidate
186     this.rowHeightDidChangeForIndexes(changed);
187     return this ;
188   },

194   /**
195     Returns the top offset for the specified content index.  This will take
196     into account any custom row heights and group views.
197     
198     @param {Number} idx the content index
199     @returns {Number} the row offset
200   */
201   rowOffsetForContentIndex: function(idx) {
202     if (idx === 0) return 0 ; // fastpath
203 
204     var del       = this.get('rowDelegate'),
205         rowHeight = del.get('rowHeight'),
206         rowSpacing, ret, custom, cache, delta, max, content ;
207         
208     ret = idx * rowHeight;
209 
210     rowSpacing = this.get('rowSpacing');
211 		if(rowSpacing){ 
212       ret += idx * rowSpacing; 
213     } 
214 
215     if (del.customRowHeightIndexes && (custom=del.get('customRowHeightIndexes'))) {
216       
217       // prefill the cache with custom rows.
218       cache = this._sclv_offsetCache;
219       if (!cache) {
220         cache = this._sclv_offsetCache = [];
221         delta = max = 0 ;
222         custom.forEach(function(idx) {
223           delta += this.rowHeightForContentIndex(idx)-rowHeight;
224           cache[idx+1] = delta;
225           max = idx ;
226         }, this);
227         this._sclv_max = max+1;
228       }
229       
230       // now just get the delta for the last custom row before the current 
231       // idx.
232       delta = cache[idx];
233       if (delta === undefined) {
234         delta = cache[idx] = cache[idx-1];
235         if (delta === undefined) {
236           max = this._sclv_max;
237           if (idx < max) max = custom.indexBefore(idx)+1;
238           delta = cache[idx] = cache[max] || 0;
239         }
240       }
241 
242       ret += delta ;
243     }
244     
245     return ret ;
246   },
247   
248   /**
249     Returns the row height for the specified content index.  This will take
250     into account custom row heights and group rows.
251     
252     @param {Number} idx content index
253     @returns {Number} the row height
254   */
255   rowHeightForContentIndex: function(idx) {
256     var del = this.get('rowDelegate'),
257         ret, cache, content, indexes;
258     
259     if (del.customRowHeightIndexes && (indexes=del.get('customRowHeightIndexes'))) {
260       cache = this._sclv_heightCache ;
261       if (!cache) {
262         cache = this._sclv_heightCache = [];
263         content = this.get('content');
264         indexes.forEach(function(idx) {
265           cache[idx] = del.contentIndexRowHeight(this, content, idx);
266         }, this);
267       }
268       
269       ret = cache[idx];
270       if (ret === undefined) ret = del.get('rowHeight');
271     } else ret = del.get('rowHeight');
272     
273     return ret ;
274   },
275   
276   /**
277     Call this method whenever a row height has changed in one or more indexes.
278     This will invalidate the row height cache and reload the content indexes.
279     Pass either an index set or a single index number.
280 
281     This method is called automatically whenever you change the rowHeight
282     or customRowHeightIndexes properties on the collectionRowDelegate.
283     
284     @param {SC.IndexSet|Number} indexes 
285     @returns {SC.ListView} receiver
286   */  
287   rowHeightDidChangeForIndexes: function(indexes) {
288     var len     = this.get('length');
289 
290     // clear any cached offsets
291     this._sclv_heightCache = this._sclv_offsetCache = null;
292     
293     // find the smallest index changed; invalidate everything past it
294     if (indexes && indexes.isIndexSet) indexes = indexes.get('min');
295     this.reload(SC.IndexSet.create(indexes, len-indexes));
296     return this ;
297   },

303   /**
304     The layout for a ListView is computed from the total number of rows 
305     along with any custom row heights.
306   */
307   computeLayout: function() {
308     // default layout
309     var ret = this._sclv_layout;
310     if (!ret) ret = this._sclv_layout = {};
311     ret.minHeight = this.rowOffsetForContentIndex(this.get('length'));
312     this.set('calculatedHeight',ret.minHeight);
313     return ret ;
314   },

316   /**
317   
318     Computes the layout for a specific content index by combining the current
319     row heights.
320   
321   */
322   layoutForContentIndex: function(contentIndex) {
323     return {
324       top:    this.rowOffsetForContentIndex(contentIndex),
325       height: this.rowHeightForContentIndex(contentIndex),
326       left:   0, 
327       right:  0
328     };
329   },

331   /**
332     Override to return an IndexSet with the indexes that are at least 
333     partially visible in the passed rectangle.  This method is used by the 
334     default implementation of computeNowShowing() to determine the new 
335     nowShowing range after a scroll.
336     
337     Override this method to implement incremental rendering.
338     
339     The default simply returns the current content length.
340     
341     @param {Rect} rect the visible rect or a point
342     @returns {SC.IndexSet} now showing indexes
343   */
344   contentIndexesInRect: function(rect) {
345     var rowHeight = this.get('rowDelegate').get('rowHeight'),
346         top       = SC.minY(rect),
347         bottom    = SC.maxY(rect),
348         height    = rect.height || 0,
349         len       = this.get('length'),
350         offset, start, end;
351 
352     // estimate the starting row and then get actual offsets until we are 
353     // right.
354     start = (top - (top % rowHeight)) / rowHeight;
355     offset = this.rowOffsetForContentIndex(start);
356     
357     // go backwards until top of row is before top edge
358     while(start>0 && offset>top) {
359       start--;
360       offset -= this.rowHeightForContentIndex(start);
361     }
362     
363     // go forwards until bottom of row is after top edge
364     offset += this.rowHeightForContentIndex(start);
365     while(start<len && offset<=top) {
366       start++;
367       offset += this.rowHeightForContentIndex(start);
368     }
369     if (start<0) start = 0;
370     if (start>=len) start=len;
371     
372     
373     // estimate the final row and then get the actual offsets until we are 
374     // right. - look at the offset of the _following_ row
375     end = start + ((height - (height % rowHeight)) / rowHeight) ;
376     if (end > len) end = len;
377     offset = this.rowOffsetForContentIndex(end);
378     
379     // walk backwards until top of row is before or at bottom edge
380     while(end>=start && offset>=bottom) {
381       end--;
382       offset -= this.rowHeightForContentIndex(end);
383     }
384     
385     // go forwards until bottom of row is after bottom edge
386     offset += this.rowHeightForContentIndex(end);
387     while(end<len && offset<bottom) {
388       end++;
389       offset += this.rowHeightForContentIndex(end);
390     }
391     
392     end++; // end should be after start
393     
394     // if height is greater than 0, on some platforms we should just render
395     // to specific windows in order to minimize render time.
396     // if (height > 0 && !SC.browser.msie) {
397     //   start = start - (start % 50);
398     //   if (start < 0) start = 0 ;
399     //   end   = end - (end % 50) + 50;
400     // }
401     
402     if (end<start) end = start;
403     if (end>len) end = len ;
404     
405     // convert to IndexSet and return
406     return SC.IndexSet.create(start, end-start);
407   },

432   /**
433     Default implementation will show an insertion point
434   */
435   showInsertionPoint: function(itemView, dropOperation) {
436     var view = this._insertionPointView;
437     if (!view) {
438       view = this._insertionPointView 
439            = this.get('insertionPointView').create();
440     }
441     
442     var index  = itemView.get('contentIndex'),
443         len    = this.get('length'),
444         layout = SC.clone(itemView.get('layout')),
445         level  = itemView.get('outlineLevel'),
446         indent = itemView.get('outlineIndent') || 0,
447         group;
448 
449     // show item indented if we are inserting at the end and the last item
450     // is a group item.  This is a special case that should really be 
451     // converted into a more general protocol.
452     if ((index >= len) && index>0) {
453       group = this.itemViewForContentIndex(len-1);
454       if (group.get('isGroupView')) {
455         level = 1;
456         indent = group.get('outlineIndent');
457       }
458     }
459     
460     if (SC.none(level)) level = -1;
461     
462     if (dropOperation & SC.DROP_ON) {
463       this.hideInsertionPoint();
464       itemView.set('isSelected', YES);
465       this._lastDropOnView = itemView;
466     } else {
467 
468       if (this._lastDropOnView) {
469         this._lastDropOnView.set('isSelected', NO);
470         this._lastDropOnView = null;
471       }
472       
473       if (dropOperation & SC.DROP_AFTER) layout.top += layout.height;
474       
475       layout.height = 2;
476       layout.right  = 0;
477       layout.left   = ((level+1) * indent) + 12;
478       delete layout.width;
479 
480       view.set('layout', layout);
481       this.appendChild(view);
482     }
483   },
484   
485   hideInsertionPoint: function() {
486     if (this._lastDropOnView) {
487       this._lastDropOnView.set('isSelected', NO);
488       this._lastDropOnView = null;
489     }
490     
491     var view = this._insertionPointView;
492     if (view) view.removeFromParent().destroy();
493     this._insertionPointView = null;
494   },

496   /**
497     Compute the insertion index for the passed location.  The location is 
498     a point, relative to the top/left corner of the receiver view.  The return
499     value is an index plus a dropOperation, which is computed as such:
500     
501     - if outlining is not used and you are within 5px of an edge, DROP_BEFORE
502       the item after the edge.
503       
504     - if outlining is used and you are within 5px of an edge and the previous
505       item has a different outline level then the next item, then DROP_AFTER
506       the previous item if you are closer to that outline level.
507       
508     - if dropOperation = SC.DROP_ON and you are over the middle of a row, then
509       use DROP_ON.
510   */
511   insertionIndexForLocation: function(loc, dropOperation) {
512     var locRect = {x:loc.x, y:loc.y, width:1, height:1},
513         indexes = this.contentIndexesInRect(locRect),
514         index   = indexes.get('min'),
515         len     = this.get('length'),
516         min, max, diff, clevel, cindent, plevel, pindent, itemView, pgroup;
517 
518     // if there are no indexes in the rect, then we need to either insert
519     // before the top item or after the last item.  Figure that out by 
520     // computing both.
521     if (SC.none(index) || index<0) {
522       if ((len===0) || (loc.y <= this.rowOffsetForContentIndex(0))) index = 0;
523       else if (loc.y >= this.rowOffsetForContentIndex(len)) index = len;
524     }
525 
526     // figure the range of the row the location must be within.
527     min = this.rowOffsetForContentIndex(index);
528     max = min + this.rowHeightForContentIndex(index);
529     
530     // now we know which index we are in.  if dropOperation is DROP_ON, figure
531     // if we can drop on or not.
532     if (dropOperation == SC.DROP_ON) {
533       // editable size - reduce height by a bit to handle dropping
534       if (this.get('isEditable')) diff=Math.min(Math.floor((max-min)*0.2),5);
535       else diff = 0;
536       
537       // if we're inside the range, then DROP_ON
538       if (loc.y >= (min+diff) || loc.y <= (max+diff)) {
539         return [index, SC.DROP_ON];
540       }
541     }
542     
543     
544     
545     // ok, now if we are in last 10px, go to next item.
546     if ((index<len) && (loc.y >= max-10)) index++;
547     
548     // finally, let's decide if we want to actually insert before/after.  Only
549     // matters if we are using outlining.
550     if (index>0) {
551 
552       itemView = this.itemViewForContentIndex(index-1);
553       pindent  = (itemView ? itemView.get('outlineIndent') : 0) || 0;
554       plevel   = itemView ? itemView.get('outlineLevel') : 0;
555       
556       if (index<len) {
557         itemView = this.itemViewForContentIndex(index);
558         clevel   = itemView ? itemView.get('outlineLevel') : 0;
559         cindent  = (itemView ? itemView.get('outlineIndent') : 0) || 0;
560         cindent  *= clevel;
561       } else {
562         clevel = itemView.get('isGroupView') ? 1 : 0; // special case...
563         cindent = pindent * clevel;  
564       }
565 
566       pindent  *= plevel;
567 
568       // if indent levels are different, then try to figure out which level 
569       // it should be on.
570       if ((clevel !== plevel) && (cindent !== pindent)) {
571         
572         // use most inner indent as boundary
573         if (pindent > cindent) {
574           index--;
575           dropOperation = SC.DROP_AFTER;
576         }
577       }
578     }
579 
580     // we do not support dropping before a group item.  If dropping before 
581     // a group item, always try to instead drop after the previous item.  If
582     // the previous item is also a group then, well, dropping is just not 
583     // allowed.  Note also that dropping at 0, first item must not be group
584     // and dropping at length, last item must not be a group
585     //
586     if (dropOperation === SC.DROP_BEFORE) {
587       itemView = (index<len) ? this.itemViewForContentIndex(index) : null;
588       if (!itemView || itemView.get('isGroupView')) {
589         if (index>0) {
590           itemView = this.itemViewForContentIndex(index-1);
591           
592           // don't allow a drop if the previous item is a group view and we're
593           // insert before the end.  For the end, allow the drop if the 
594           // previous item is a group view but OPEN.
595           if (!itemView.get('isGroupView') || (itemView.get('disclosureState') === SC.BRANCH_OPEN)) {
596             index = index-1;
597             dropOperation = SC.DROP_AFTER;
598           } else index = -1;
599 
600         } else index = -1;
601       }
602       
603       if (index<0) dropOperation = SC.DRAG_NONE ;
604     } 
605     
606     // return whatever we came up with
607     return [index, dropOperation];
608   },
