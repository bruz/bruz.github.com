  8 /**
  9   @class
 10 
 11   A ManyArray is used to map an array of record ids back to their 
 12   record objects which will be materialized from the owner store on demand.
 13   
 14   Whenever you create a toMany() relationship, the value returned from the 
 15   property will be an instance of ManyArray.  You can generally customize the
 16   behavior of ManyArray by passing settings to the toMany() helper.
 17   
 18   @extends SC.Enumerable
 19   @extends SC.Array
 20   @since SproutCore 1.0
 21 */
 22 
 23 SC.ManyArray = SC.Object.extend(SC.Enumerable, SC.Array,
 24   /** @scope SC.ManyArray.prototype */ {
 25 
 26   /**
 27     recordType will tell what type to transform the record to when
 28     materializing the record.
 29 
 30     @property {String}
 31   */
 32   recordType: null,
 33   
 34   /**
 35     If set, the record will be notified whenever the array changes so that 
 36     it can change its own state
 37     
 38     @property {SC.Record}
 39   */
 40   record: null,
 41   
 42   /**
 43     If set will be used by the many array to get an editable version of the
 44     storeIds from the owner.
 45     
 46     @property {String}
 47   */
 48   propertyName: null,
 49   
 50   
 51   /**
 52     The ManyAttribute that created this array.
 53   
 54     @property {SC.ManyAttribute}
 55   */
 56   manyAttribute: null,
 57   
 58   /**
 59     The store that owns this record array.  All record arrays must have a 
 60     store to function properly.
 61 
 62     @property {SC.Store}
 63   */
 64   store: function() {
 65     return this.get('record').get('store');
 66   }.property('record').cacheable(),
 67   
 68   /**
 69     The storeKey for the parent record of this many array.  Editing this 
 70     array will place the parent record into a READY_DIRTY state.
 71 
 72     @property {Number}
 73   */
 74   storeKey: function() {
 75     return this.get('record').get('storeKey');
 76   }.property('record').cacheable(),
 77 
 78 
 79   /**
 80     Returns the storeIds in read only mode.  Avoids modifying the record 
 81     unnecessarily.
 82     
 83     @property {SC.Array}
 84   */
 85   readOnlyStoreIds: function() {
 86     return this.get('record').readAttribute(this.get('propertyName'));
 87   }.property(),
 88   
 89   
 90   /**
 91     Returns an editable array of storeIds.  Marks the owner records as 
 92     modified. 
 93     
 94     @property {SC.Array}
 95   */
 96   editableStoreIds: function() {
 97     var store    = this.get('store'),
 98         storeKey = this.get('storeKey'),
 99         pname    = this.get('propertyName'),
100         ret, hash;
101         
102     ret = store.readEditableProperty(storeKey, pname);    
103     if (!ret) {
104       hash = store.readEditableDataHash(storeKey);
105       ret = hash[pname] = [];      
106     }
107     
108     if (ret !== this._prevStoreIds) this.recordPropertyDidChange();
109     return ret ;
110   }.property(),
111

123     // NOTE: can't use get() b/c manyAttribute looks like a computed prop
124     var attr = this.manyAttribute;
125     return attr ? attr.get('isEditable') : NO;

151     Computed from owner many attribute

156     // NOTE: can't use get() b/c manyAttribute looks like a computed prop

158     return attr ? attr.get('orderBy') : null;

162   // ARRAY PRIMITIVES

164 
165   /** @private
166     Returned length is a pass-through to the storeIds array.
167     
168     @property {Number}
169   */
170   length: function() {
171     var storeIds = this.get('readOnlyStoreIds');
172     return storeIds ? storeIds.get('length') : 0;
173   }.property('readOnlyStoreIds'),
174 
175   /** @private
176     Looks up the store id in the store ids array and materializes a
177     records.
178   */
179   objectAt: function(idx) {
180     var recs      = this._records, 
181         storeIds  = this.get('readOnlyStoreIds'),
182         store     = this.get('store'),
183         recordType = this.get('recordType'),
184         storeKey, ret, storeId ;
185         
186     if (!storeIds || !store) return undefined; // nothing to do
187     if (recs && (ret=recs[idx])) return ret ; // cached
188 
189     // not in cache, materialize
190     if (!recs) this._records = recs = [] ; // create cache
191     storeId = storeIds.objectAt(idx);
192     if (storeId) {
193 
194       // if record is not loaded already, then ask the data source to 
195       // retrieve it
196       storeKey = store.storeKeyFor(recordType, storeId);
197       
198       if (store.readStatus(storeKey) === SC.Record.EMPTY) {
199         store.retrieveRecord(recordType, null, storeKey);
200       }
201       
202       recs[idx] = ret = store.materializeRecord(storeKey);
203     }
204     return ret ;
205   },
206 
207   /** @private
208     Pass through to the underlying array.  The passed in objects must be
209     records, which can be converted to storeIds.
210   */
211   replace: function(idx, amt, recs) {
212     
213     if (!this.get('isEditable')) {
214       throw "%@.%@[] is not editable".fmt(this.get('record'), this.get('propertyName'));
215     }
216     
217     var storeIds = this.get('editableStoreIds'), 
218         len      = recs ? (recs.get ? recs.get('length') : recs.length) : 0,
219         record   = this.get('record'),
220         pname    = this.get('propertyName'),
221         i, keys, ids, toRemove, inverse, attr, inverseRecord;
222 
223     // map to store keys
224     ids = [] ;
225     for(i=0;i<len;i++) ids[i] = recs.objectAt(i).get('id');
226 
227     // if we have an inverse - collect the list of records we are about to 
228     // remove
229     inverse = this.get('inverse');
230     if (inverse && amt>0) {
231       toRemove = SC.ManyArray._toRemove;
232       if (toRemove) SC.ManyArray._toRemove = null; // reuse if possible
233       else toRemove = [];
234       
235       for(i=0;i<amt;i++) toRemove[i] = this.objectAt(i);
236     }
237     
238     // pass along - if allowed, this should trigger the content observer 
239     storeIds.replace(idx, amt, ids);
240 
241     // ok, notify records that were removed then added; this way reordered
242     // objects are added and removed
243     if (inverse) {
244       
245       // notive removals
246       for(i=0;i<amt;i++) {
247         inverseRecord = toRemove[i];
248         attr = inverseRecord ? inverseRecord[inverse] : null;
249         if (attr && attr.inverseDidRemoveRecord) {
250           attr.inverseDidRemoveRecord(inverseRecord, inverse, record, pname);
251         }
252       }
253 
254       if (toRemove) {
255         toRemove.length = 0; // cleanup
256         if (!SC.ManyArray._toRemove) SC.ManyArray._toRemove = toRemove;
257       }
258 
259       // notify additions
260       for(i=0;i<len;i++) {
261         inverseRecord = recs.objectAt(i);
262         attr = inverseRecord ? inverseRecord[inverse] : null;
263         if (attr && attr.inverseDidAddRecord) {
264           attr.inverseDidAddRecord(inverseRecord, inverse, record, pname);
265         }
266       }
267       
268     }
269 
270     // only mark record dirty if there is no inverse or we are master
271     if (record && (!inverse || this.get('isMaster'))) {
272       record.recordDidChange(pname);
273     } 
274     
275     return this;
276   },

279   // INVERSE SUPPORT

282   /**
283     Called by the ManyAttribute whenever a record is removed on the inverse
284     of the relationship.
285     
286     @param {SC.Record} inverseRecord the record that was removed
287     @returns {SC.ManyArray} receiver
288   */
289   removeInverseRecord: function(inverseRecord) {
290     
291     if (!inverseRecord) return this; // nothing to do
292     var id = inverseRecord.get('id'),
293         storeIds = this.get('editableStoreIds'),
294         idx      = (storeIds && id) ? storeIds.indexOf(id) : -1,
295         record;
296     
297     if (idx >= 0) {
298       storeIds.removeAt(idx);
299       if (this.get('isMaster') && (record = this.get('record'))) {
300         record.recordDidChange(this.get('propertyName'));
301       }
302     }
303     
304     return this;
305   },
306 
307   /**
308     Called by the ManyAttribute whenever a record is added on the inverse
309     of the relationship.
310     
311     @param {SC.Record} record the record this array is a part of
312     @param {String} key the key this array represents
313     @param {SC.Record} inverseRecord the record that was removed
314     @param {String} inverseKey the name of inverse that was changed
315     @returns {SC.ManyArray} receiver
316   */
317   addInverseRecord: function(inverseRecord) {
318     
319     if (!inverseRecord) return this;
320     var id = inverseRecord.get('id'),
321         storeIds = this.get('editableStoreIds'),
322         orderBy  = this.get('orderBy'),
323         len      = storeIds.get('length'),
324         idx, record;
325         
326     // find idx to insert at.
327     if (orderBy) {
328       idx = this._findInsertionLocation(inverseRecord, 0, len, orderBy);
329     } else idx = len;
330     
331     storeIds.insertAt(idx, inverseRecord.get('id'));
332     if (this.get('isMaster') && (record = this.get('record'))) {
333       record.recordDidChange(this.get('propertyName'));
334     }
335     
336     return this;
337   },
338   
339   // binary search to find insertion location
340   _findInsertionLocation: function(rec, min, max, orderBy) {
341     var idx   = min+Math.floor((max-min)/2),
342         cur   = this.objectAt(idx),
343         order = this._compare(rec, cur, orderBy);
344     if (order < 0) {
345       if (idx===0) return idx;
346       else return this._findInsertionLocation(rec, 0, idx, orderBy);
347     } else if (order > 0) {
348       if (idx >= max) return idx;
349       else return this._findInsertionLocation(rec, idx, max, orderBy);
350     } else return idx;
351   },
352 
353   _compare: function(a, b, orderBy) {
354     var t = SC.typeOf(orderBy),
355         ret, idx, len;
356         
357     if (t === SC.T_FUNCTION) ret = orderBy(a, b);
358     else if (t === SC.T_STRING) ret = SC.compare(a,b);
359     else {
360       len = orderBy.get('length');
361       ret = 0;
362       for(idx=0;(ret===0) && (idx<len);idx++) ret = SC.compare(a,b);
363     }
364 
365     return ret ;
366   },

372   /** @private 
373     Invoked whenever the storeIds array changes.  Observes changes.
374   */
375   recordPropertyDidChange: function(keys) {
376     
377     if (keys && !keys.contains(this.get('propertyName'))) return this;
378     
379     var storeIds = this.get('readOnlyStoreIds');
380     var prev = this._prevStoreIds, f = this._storeIdsContentDidChange;
381 
382     if (storeIds === prev) return this; // nothing to do
383 
384     if (prev) prev.removeObserver('[]', this, f);
385     this._prevStoreIds = storeIds;
386     if (storeIds) storeIds.addObserver('[]', this, f);
387 
388     var rev = (storeIds) ? storeIds.propertyRevision : -1 ;
389     this._storeIdsContentDidChange(storeIds, '[]', storeIds, rev);
390     
391   },
392 
393   /** @private
394     Invoked whenever the content of the storeIds array changes.  This will
395     dump any cached record lookup and then notify that the enumerable content
396     has changed.
397   */
398   _storeIdsContentDidChange: function(target, key, value, rev) {
399     this._records = null ; // clear cache
400     this.enumerableContentDidChange();
401   },
402   
403   /** @private */
404   unknownProperty: function(key, value) {
405     var ret = this.reducedProperty(key, value);
406     return ret === undefined ? sc_super() : ret;
407   },
408 
409   /** @private */
410   init: function() {
411     sc_super();
412     this.recordPropertyDidChange();
413   }
