 10 /**
 11   @class
 12 
 13   SC.MenuPane allows you to display a standard menu. Menus appear over other
 14   panes, and block input to other views until a selection is made or the pane
 15   is dismissed by clicking outside of its bounds.
 16 
 17   You can create a menu pane and manage it yourself, or you can use the
 18   SC.SelectButtonView and SC.PopupButtonView controls to manage the menu for
 19   you.
 20 
 21   h2. Specifying Menu Items
 22 
 23   The menu pane examines the @items@ property to determine what menu items
 24   should be presented to the user.
 25 
 26   In its most simple form, you can provide an array of strings. Every item
 27   will be converted into a menu item whose title is the string.
 28 
 29   If you need more control over the menu items, such as specifying a keyboard
 30   shortcut, enabled state, custom height, or submenu, you can provide an array
 31   of content objects.
 32 
 33   Out of the box, the menu pane has some default keys it uses to get
 34   information from the objects. For example, to find out the title of the menu
 35   item, the menu pane will ask your object for its @title@ property. If you
 36   need to change this key, you can set the @itemTitleKey@ property on the pane
 37   itself.
 38 
 39   {{{
 40     var menuItems = [
 41       { title: 'Menu Item', keyEquivalent: 'ctrl_shift_n' },
 42       { title: 'Checked Menu Item', isChecked: YES, keyEquivalent: 'ctrl_a' },
 43       { title: 'Selected Menu Item', keyEquivalent: 'backspace' },
 44       { isSeparator: YES },
 45       { title: 'Menu Item with Icon', icon: 'inbox', keyEquivalent: 'ctrl_m' },
 46       { title: 'Menu Item with Icon', icon: 'folder', keyEquivalent: 'ctrl_p' }
 47     ];
 48 
 49     var menu = SC.MenuPane.create({
 50       items: menuItems
 51     });
 52   }}}
 53 
 54   h2. Observing User Selections
 55 
 56   To determine when a user clicks on a menu item, you can observe the
 57   @selectedItem@ property for changes.
 58 
 59   @extends SC.PickerPane
 60   @since SproutCore 1.0
 61 */

621   /**
622     The array of child menu item views that compose the menu.
623 
624     This computed property parses @displayItems@ and constructs an SC.MenuItemView (or whatever class you have set as the @exampleView@) for every item.
625 
626     @property
627     @type Array
628     @readOnly
629   */
630   menuItemViews: function() {
631     var views = [], items = this.get('displayItems'),
632         exampleView = this.get('exampleView'), item, view,
633         height, heightKey, separatorKey, defaultHeight, separatorHeight,
634         menuHeight, menuHeightPadding, keyEquivalentKey, keyEquivalent,
635         keyArray, idx,
636         len;
637 
638     if (!items) return views; // return an empty array
639 
640     heightKey = this.get('itemHeightKey');
641     separatorKey = this.get('itemSeparatorKey');
642     defaultHeight = this.get('itemHeight');
643     keyEquivalentKey = this.get('itemKeyEquivalentKey');
644     separatorHeight = this.get('itemSeparatorHeight');
645 
646     menuHeightPadding = Math.floor(this.get('menuHeightPadding')/2);
647     menuHeight = menuHeightPadding;
648 
649     keyArray = this.menuItemKeys.map(SC._menu_fetchKeys, this);
650 
651     len = items.get('length');
652     for (idx = 0; idx < len; idx++) {
653       item = items[idx];
654       height = item.get(heightKey);
655       if (!height) {
656         height = item.get(separatorKey) ? separatorHeight : defaultHeight;
657       }
658       view = this._menuView.createChildView(exampleView, {
659         layout: { height: height, top: menuHeight },
660         contentDisplayProperties: keyArray,
661         content: item,
662         parentMenu: this
663       });
664       views[idx] = view;
665       menuHeight += height;
666 
667       keyEquivalent = item.get(keyEquivalentKey);
668       if (keyEquivalent) {
669         this._keyEquivalents[keyEquivalent] = view;
670       }
671     }
672 
673     this.set('menuHeight', menuHeight+menuHeightPadding);
674     return views;
675   }.property('displayItems').cacheable(),

722   /**
723     Returns an array of normalized display items.
724 
725     Because the items property can be provided as either an array of strings,
726     or an object with key-value pairs, or an exotic mish-mash of both, we need
727     to normalize it for our display logic.
728 
729     If an @items@ member is an object, we can assume it is formatted properly
730     and leave it as-is.
731 
732     If an @items@ member is a string, we create a hash with the title value
733     set to that string, and some sensible defaults for the other properties.
734 
735     As a last resort, if an @items@ member is an array, we have a legacy
736     handler that converts the array into a hash. This behavior is deprecated
737     and is not guaranteed to be supported in the future.
738 
739     A side effect of running this computed property is that the menuHeight
740     property is updated.
741 
742     @displayItems@ should never be set directly; instead, set @items@ and
743     @displayItems@ will update automatically.
744 
745     @property
746     @type Array
747     @isReadOnly
748   */
749   displayItems: function() {
750     var items = this.get('items'), localize = this.get('localize'),
751         itemHeight = this.get('itemHeight'), len,
752         ret = [], idx, item, itemType;
753 
754     if (!items) return null;
755 
756     len = items.get('length');
757 
758     // Loop through the items property and transmute as needed, then
759     // copy the new objects into the ret array.
760     for (idx = 0; idx < len; idx++) {
761       item = items.objectAt(idx) ;
762 
763       // fast track out if we can't do anything with this item
764       if (!item) continue;
765 
766       itemType = SC.typeOf(item);
767       if (itemType === SC.T_STRING) {
768         item = SC.Object.create({ title: item,
769                                   value: item,
770                                   isEnabled: YES
771                                });
772       } else if (itemType === SC.T_HASH) {
773         item = SC.Object.create(item);
774       } else if (itemType === SC.T_ARRAY) {
775         item = this.convertArrayMenuItemToObject(item);
776       }
777       item.contentIndex = idx;
778 
779       ret.push(item);
780     }
781 
782     return ret;
783   }.property('items').cacheable(),

981   insertText: function(chr, evt) {
982     var timer = this._timer, keyBuffer = this._keyBuffer;
983 
984     if (timer) {
985       timer.invalidate();
986     }
987     timer = this._timer = SC.Timer.schedule({
988       target: this,
989       action: 'clearKeyBuffer',
990       interval: 500,
991       isPooled: NO
992     });
993 
994     keyBuffer = keyBuffer || '';
995     keyBuffer += chr.toUpperCase();
996 
997     this.selectMenuItemForString(keyBuffer);
998     this._keyBuffer = keyBuffer;

1003   performKeyEquivalent: function(keyEquivalent) {
1004     //If menu is not visible
1005     if (!this.get('isVisibleInWindow')) return NO;
1006     
1007     // Look for menu item that has this key equivalent
1008     var menuItem = this._keyEquivalents[keyEquivalent];
1009 
1010     // If found, have it perform its action
1011     if (menuItem) {
1012       menuItem.performAction(YES);
1013       return YES;
1014     }
1015 
1016     // If escape key or the enter key was pressed and no menu item handled it,
1017     // close the menu pane and return YES that the event was handled
1018     if (keyEquivalent === 'escape'|| keyEquivalent === 'return') {
1019       this.remove();
1020       return YES;
1021     }
1022 
1023     return NO;
1024

1027   selectMenuItemForString: function(buffer) {
1028     var items = this.get('menuItemViews'), item, title, idx, len, bufferLength;
1029     if (!items) return;
1030 
1031     bufferLength = buffer.length;
1032     len = items.get('length');
1033     for (idx = 0; idx < len; idx++) {
1034       item = items.objectAt(idx);
1035       title = item.get('title');
1036 
1037       if (!title) continue;
1038 
1039       title = title.replace(/ /g,'').substr(0,bufferLength).toUpperCase();
1040       if (title === buffer) {
1041         this.set('currentMenuItem', item);
1042         item.becomeFirstResponder();
1043         break;
1044       }
1045     }
