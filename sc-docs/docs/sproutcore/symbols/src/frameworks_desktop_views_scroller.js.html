  8 /** @class
  9 
 10   Displays a horizontal or vertical scroller.  You will not usually need to
 11   work with scroller views directly, but you may override this class to
 12   implement your own custom scrollers.
 13 
 14   Because the scroller uses the dimensions of its constituent elements to
 15   calculate layout, you may need to override the default display metrics.
 16 
 17   You can either create a subclass of ScrollerView with the new values, or
 18   provide your own in your theme:
 19 
 20 {{{
 21   SC.mixin(SC.ScrollerView.prototype, {
 22     scrollbarThickness: 14,
 23     capLength: 18,
 24     capOverlap: 14,
 25     buttonOverlap: 11,
 26     buttonLength: 41
 27   });
 28 }}}
 29 
 30   You can change whether scroll buttons are displayed by setting the
 31   hasButtons property.
 32

190   /**
191     Generates the HTML that gets displayed to the user.
192 
193     The first time render is called, the HTML will be output to the DOM.
194     Successive calls will reposition the thumb based on the value property.
195 
196     @param {SC.RenderContext} context the render context
197     @param {Boolean} firstTime YES if this is creating a layer
198     @private

200   render: function(context, firstTime) {
201     var classNames = [],
202         buttons = '',
203         thumbPosition, thumbLength, thumbCenterLength, thumbElement,
204         value, max, scrollerLength, length, pct;
205 
206     // We set a class name depending on the layout direction so that we can
207     // style them differently using CSS.
208     switch (this.get('layoutDirection')) {
209       case SC.LAYOUT_VERTICAL:
210         classNames.push('sc-vertical');
211         break;
212       case SC.LAYOUT_HORIZONTAL:
213         classNames.push('sc-horizontal');
214         break;
215     }
216 
217     // The appearance of the scroller changes if disabled
218     if (!this.get('isEnabled')) classNames.push('disabled');
219     // Whether to hide the thumb and buttons
220     if (this.get('controlsHidden')) classNames.push('controls-hidden');
221 
222     // Change the class names of the DOM element all at once to improve
223     // performance
224     context.addClass(classNames);
225 
226     // Calculate the position and size of the thumb
227     thumbLength = this.get('thumbLength');
228     thumbPosition = this.get('thumbPosition');
229 
230     // If this is the first time, generate the actual HTML
231     if (firstTime) {
232       if (this.get('hasButtons')) {
233         buttons = '<div class="button-bottom"></div><div class="button-top"></div>';
234       } else {
235         buttons = '<div class="endcap"></div>';
236       }
237 
238       switch (this.get('layoutDirection')) {
239         case SC.LAYOUT_VERTICAL:
240         context.push('<div class="track"></div>',
241                       '<div class="cap"></div>',
242                       buttons,
243                       '<div class="thumb" style="height: '+thumbLength+'px;">',
244                       '<div class="thumb-center"></div>',
245                       '<div class="thumb-top"></div>',
246                       '<div class="thumb-bottom"></div></div>');
247         break;
248         case SC.LAYOUT_HORIZONTAL:
249         context.push('<div class="track"></div>',
250                       '<div class="cap"></div>',
251                       buttons,
252                       '<div class="thumb" style="width: '+thumbLength+'px;">',
253                       '<div class="thumb-center"></div>',
254                       '<div class="thumb-top"></div>',
255                       '<div class="thumb-bottom"></div></div>');
256       }
257     } else {
258       // The HTML has already been generated, so all we have to do is
259       // reposition and resize the thumb
260 
261       // If we aren't displaying controls don't bother
262       if (this.get('controlsHidden')) return;
263 
264       thumbElement = this.$('.thumb');
265 
266       this.adjustThumb(thumbElement, thumbPosition, thumbLength);
267     }

338   /**
339     Returns the total length of the track in which the thumb sits.
340 
341     The length of the track is the height or width of the scroller, less the
342     cap length and the button length. This property is used to calculate the
343     position of the thumb relative to the view.
344 
345     @property
346     @private
347   */

353     // Subtract the size of the scroll buttons, or the end cap if they are
354     // not shown.

357     return scrollerLength;

381     length of the track times the content proportion.

395   /**
396     The position of the thumb in the track.

398     @property {Number}

432   /**
433     Returns the value for a position within the scroller's frame.
434     
435     @private
436   */
437   valueForPosition: function(pos) {
438     var max = this.get('maximum'),
439         trackLength = this.get('trackLength'),
440         thumbLength = this.get('thumbLength'),
441         capLength = this.get('capLength'),
442         capOverlap = this.get('capOverlap'), value; 
443     
444     value = pos - (capLength - capOverlap);
445     value = value / (trackLength - thumbLength);
446     value = value * max;
447     return value;
448   },

450   /**
451     Handles mouse down events and adjusts the value property depending where
452     the user clicked.
453 
454     If the control is disabled, we ignore all mouse input.
455 
456     If the user clicks the thumb, we note the position of the mouse event but
457     do not take further action until they begin to drag.
458 
459     If the user clicks the track, we adjust the value a page at a time, unless
460     alt is pressed, in which case we scroll to that position.
461 
462     If the user clicks the buttons, we adjust the value by a fixed amount, unless
463     alt is pressed, in which case we adjust by a page.
464 
465     If the user clicks and holds on either the track or buttons, those actions
466     are repeated until they release the mouse button.
467 
468     @param evt {SC.Event} the mousedown event
469     @private
470   */
471   mouseDown: function(evt) {
472     if (!this.get('isEnabled')) return NO;
473     
474     // keep note of altIsDown for later.
475     this._altIsDown = evt.altKey;
476     this._shiftIsDown = evt.shiftKey;
477 
478     var target = evt.target,
479         thumbPosition = this.get('thumbPosition'),
480         value, clickLocation, clickOffset,
481         scrollerLength = this.get('scrollerLength');
482         
483     // Determine the subcontrol that was clicked
484     if (target.className.indexOf('thumb') >= 0) {
485       // Convert the mouseDown coordinates to the view's coordinates
486       clickLocation = this.convertFrameFromView({ x: evt.pageX, y: evt.pageY });
487 
488       clickLocation.x -= thumbPosition;
489       clickLocation.y -= thumbPosition;
490 
491       // Store the starting state so we know how much to adjust the
492       // thumb when the user drags
493       this._thumbDragging = YES;
494       this._thumbOffset = clickLocation;
495       this._mouseDownLocation = { x: evt.pageX, y: evt.pageY };
496       this._thumbPositionAtDragStart = this.get('thumbPosition');
497       this._valueAtDragStart = this.get("value");
498     } else if (target.className.indexOf('button-top') >= 0) {
499       // User clicked the up/left button
500       // Decrement the value by a fixed amount or page size
501       this.decrementProperty('value', (this._altIsDown ? scrollerLength : 30));
502       this.makeButtonActive('.button-top');
503       // start a timer that will continue to fire until mouseUp is called
504       this.startMouseDownTimer('scrollUp');
505       this._isScrollingUp = YES;
506     } else if (target.className.indexOf('button-bottom') >= 0) {
507       // User clicked the down/right button
508       // Increment the value by a fixed amount
509       this.incrementProperty('value', (this._altIsDown ? scrollerLength : 30));
510       this.makeButtonActive('.button-bottom');
511       // start a timer that will continue to fire until mouseUp is called
512       this.startMouseDownTimer('scrollDown');
513       this._isScrollingDown = YES;
514     } else {
515       // User clicked in the track
516       var scrollToClick = this.get("shouldScrollToClick");
517       if (evt.altKey) scrollToClick = !scrollToClick;
518       
519       var trackLength = this.get('trackLength'),
520           thumbLength = this.get('thumbLength'),
521           frame = this.convertFrameFromView({ x: evt.pageX, y: evt.pageY }),
522           mousePosition;
523 
524       switch (this.get('layoutDirection')) {
525         case SC.LAYOUT_VERTICAL:
526           this._mouseDownLocation = mousePosition = frame.y;
527           break;
528         case SC.LAYOUT_HORIZONTAL:
529           this._mouseDownLocation = mousePosition = frame.x;
530           break;
531       }
532       
533       if (scrollToClick) {
534         this.set('value', this.valueForPosition(mousePosition - (thumbLength / 2)));
535         
536         // and start a normal mouse down
537         thumbPosition = this.get('thumbPosition');
538         
539         this._thumbDragging = YES;
540         this._thumbOffset = {x: frame.x - thumbPosition, y: frame.y - thumbPosition };
541         this._mouseDownLocation = {x:evt.pageX, y:evt.pageY};
542         this._thumbPositionAtDragStart = thumbPosition;
543         this._valueAtDragStart = this.get("value");
544       } else {
545         // Move the thumb up or down a page depending on whether the click
546         // was above or below the thumb
547         if (mousePosition < thumbPosition) {
548           this.decrementProperty('value',scrollerLength);
549           this.startMouseDownTimer('page');
550         } else {
551           this.incrementProperty('value', scrollerLength);
552           this.startMouseDownTimer('page');
553         }
554       }
555       
556     }
557 
558     return YES;
559   },

561   /**
562     When the user releases the mouse button, remove any active
563     state from the button controls, and cancel any outstanding
564     timers.
565 
566     @param evt {SC.Event} the mousedown event
567     @private
568   */
569   mouseUp: function(evt) {
570     var active = this._scs_buttonActive, ret = NO, timer;
571 
572     // If we have an element that was set as active in mouseDown,
573     // remove its active state
574     if (active) {
575       active.removeClass('active');
576       ret = YES;
577     }
578 
579     // Stop firing repeating events after mouseup
580     timer = this._mouseDownTimer;
581     if (timer) {
582       timer.invalidate();
583       this._mouseDownTimer = null;
584     }
585 
586     this._thumbDragging = NO;
587     this._isScrollingDown = NO;
588     this._isScrollingUp = NO;
589 
590     return ret;
591   },

593   /**
594     If the user began the drag on the thumb, we calculate the difference
595     between the mouse position at click and where it is now.  We then
596     offset the thumb by that amount, within the bounds of the track.
597     
598     If the user began scrolling up/down using the buttons, this will track
599     what component they are currently over, changing the scroll direction.
600 
601     @param evt {SC.Event} the mousedragged event
602     @private
603   */
604   mouseDragged: function(evt) {
605     var value, length, delta, thumbPosition,
606         target = evt.target,
607         thumbPositionAtDragStart = this._thumbPositionAtDragStart,
608         isScrollingUp = this._isScrollingUp,
609         isScrollingDown = this._isScrollingDown,
610         active = this._scs_buttonActive,
611         timer;
612 
613     // Only move the thumb if the user clicked on the thumb during mouseDown
614     if (this._thumbDragging) {
615       
616       switch (this.get('layoutDirection')) {
617         case SC.LAYOUT_VERTICAL:
618           delta = (evt.pageY - this._mouseDownLocation.y);
619           break;
620         case SC.LAYOUT_HORIZONTAL:
621           delta = (evt.pageX - this._mouseDownLocation.x);
622           break;
623       }
624       
625       // if we are in alt now, but were not before, update the old thumb position to the new one
626       if (evt.altKey) {
627         if (!this._altIsDown || (this._shiftIsDown !== evt.shiftKey)) {
628           thumbPositionAtDragStart = this._thumbPositionAtDragStart = thumbPositionAtDragStart+delta;
629           delta = 0;
630           this._mouseDownLocation = { x: evt.pageX, y: evt.pageY };
631           this._valueAtDragStart = this.get("value");
632         }
633         
634         // because I feel like it. Probably almost no one will find this tiny, buried feature.
635         // Too bad.
636         if (evt.shiftKey) delta = -delta;
637         
638         this.set('value', Math.round(this._valueAtDragStart + delta * 2));
639       } else {
640         thumbPosition = thumbPositionAtDragStart + delta;
641         length = this.get('trackLength') - this.get('thumbLength');
642         this.set('value', Math.round( (thumbPosition/length) * this.get('maximum')));
643       }
644     
645     } else if (isScrollingUp || isScrollingDown) {
646       var nowScrollingUp = NO, nowScrollingDown = NO;
647       
648       var topButtonRect = this.$('.button-top')[0].getBoundingClientRect();
649       var bottomButtonRect = this.$('.button-bottom')[0].getBoundingClientRect();
650       
651       switch (this.get('layoutDirection')) {
652         case SC.LAYOUT_VERTICAL:
653           if (evt.pageY < topButtonRect.bottom) nowScrollingUp = YES;
654           else nowScrollingDown = YES;
655           break;
656         case SC.LAYOUT_HORIZONTAL:
657           if (evt.pageX < topButtonRect.right) nowScrollingUp = YES;
658           else nowScrollingDown = YES;
659           break;
660       }
661       
662       if ((nowScrollingUp || nowScrollingDown) && nowScrollingUp !== isScrollingUp){
663         //
664         // STOP OLD
665         //
666         
667         // If we have an element that was set as active in mouseDown,
668         // remove its active state
669         if (active) {
670          active.removeClass('active');
671         }
672 
673         // Stop firing repeating events after mouseup
674         this._mouseDownTimerAction = nowScrollingUp ? "scrollUp" : "scrollDown";
675         
676         if (nowScrollingUp) {
677           this.makeButtonActive('.button-top');
678         } else if (nowScrollingDown) {
679           this.makeButtonActive('.button-bottom');
680         }
681         
682          this._isScrollingUp = nowScrollingUp;
683          this._isScrollingDown = nowScrollingDown;
684       }
685     }
686     
687     
688     this._altIsDown = evt.altKey;
689     this._shiftIsDown = evt.shiftKey;
690     return YES;
691   },

693   /**
694     Starts a timer that fires after 300ms.  This is called when the user
695     clicks a button or inside the track to move a page at a time. If they
696     continue holding the mouse button down, we want to repeat that action
697     after a small delay.  This timer will be invalidated in mouseUp.
698     
699     Specify "immediate" as YES if it should not wait.
700 
701     @private
702   */
703   startMouseDownTimer: function(action, immediate) {
704     var timer;
705 
706     this._mouseDownTimerAction = action;
707     this._mouseDownTimer = SC.Timer.schedule({
708       target: this, action: this.mouseDownTimerDidFire, interval: immediate ? 0 : 300
709     });
710   },
711 
712   /**
713     Called by the mousedown timer.  This method determines the initial
714     user action and repeats it until the timer is invalidated in mouseUp.
715 
716     @private
717   */
718   mouseDownTimerDidFire: function() {
719     var scrollerLength = this.get('scrollerLength'),
720         mouseLocation = SC.device.get('mouseLocation'),
721         thumbPosition = this.get('thumbPosition'),
722         thumbLength = this.get('thumbLength'),
723         timerInterval = 50;
724 
725     switch (this.get('layoutDirection')) {
726       case SC.LAYOUT_VERTICAL:
727         mouseLocation = this.convertFrameFromView(mouseLocation).y;
728         break;
729       case SC.LAYOUT_HORIZONTAL:
730         mouseLocation = this.convertFrameFromView(mouseLocation).x;
731         break;
732     }
733 
734     switch (this._mouseDownTimerAction) {
735       case 'scrollDown':
736         this.incrementProperty('value', this._altIsDown ? scrollerLength : 30);
737         break;
738       case 'scrollUp':
739         this.decrementProperty('value', this._altIsDown ? scrollerLength : 30);
740         break;
741       case 'page':
742         timerInterval = 150;
743         if (mouseLocation < thumbPosition) {
744           this.decrementProperty('value', scrollerLength);
745         } else if (mouseLocation > thumbPosition+thumbLength) {
746           this.incrementProperty('value', scrollerLength);
747         }
748     }
749 
750     this._mouseDownTimer = SC.Timer.schedule({
751       target: this, action: this.mouseDownTimerDidFire, interval: timerInterval
752     });
753   },

755   /**

777   adjustThumb: function(thumb, position, length) {
778     var thumbInner = this.$('.thumb-inner');
779     var max = this.get("scrollerLength") - this.capLength, min = this.get("minimum") + this.capLength;
780     
781     if (position + length > max) {
782       position = Math.min(max - 20, position);
783       length = max - position;
784     }
785     
786     if (position < min) {
787       length -= min - position;
788       position = min;
789     }
790     
791     switch (this.get('layoutDirection')) {
792       case SC.LAYOUT_VERTICAL:
793         if (this._thumbPosition !== position) thumb.css('-webkit-transform', 'translate3d(0px,' + position + 'px,0px)');
794         if (this._thumbSize !== length) {
795           thumbInner.css('-webkit-transform', 'translate3d(0px,' + Math.round(length - 1044) + 'px,0px)');
796         }
797         break;
798       case SC.LAYOUT_HORIZONTAL:
799         if (this._thumbPosition !== position) thumb.css('-webkit-transform', 'translate3d(' + position + 'px,0px,0px)');
800         if (this._thumbSize !== length) {
801           thumbInner.css('-webkit-transform', 'translate3d(' + Math.round(length - 1044) + 'px,0px,0px)');
802         }
803         break;
804     }
805 
806     this._thumbPosition = position;
807     this._thumbSize = length;
808   },

810   render: function(context, firstTime) {
811     var classNames = [],
812         buttons = '',
813         thumbPosition, thumbLength, thumbCenterLength, thumbElement,
814         value, max, scrollerLength, length, pct;

816     // We set a class name depending on the layout direction so that we can

819       case SC.LAYOUT_VERTICAL:
820         classNames.push('sc-vertical');

822       case SC.LAYOUT_HORIZONTAL:
823         classNames.push('sc-horizontal');
824         break;
825     }
826 
827     // The appearance of the scroller changes if disabled
828     if (!this.get('isEnabled')) classNames.push('disabled');
829     // Whether to hide the thumb and buttons
830     if (this.get('controlsHidden')) classNames.push('controls-hidden');
831 
832     // Change the class names of the DOM element all at once to improve
833     // performance
834     context.addClass(classNames);
835 
836     // Calculate the position and size of the thumb
837     thumbLength = this.get('thumbLength');
838     thumbPosition = this.get('thumbPosition');
839 
840     // If this is the first time, generate the actual HTML

857                       '<div class="thumb-center"></div>',
