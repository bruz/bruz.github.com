 93 /** 
 94   @class
 95   
 96   Base class for managing a view.  Views provide two functions:
 97   
 98   1. They translate state and events into drawing instructions for the 
 99      web browser and
100   
101   2. They act as first responders for incoming keyboard, mouse, and 
102      touch events.
103   
104   h2. View Initialization
105   
106   When a view is setup, there are several methods you can override that 
107   will be called at different times depending on how your view is created.
108   Here is a guide to which method you want to override and when:
109   
110   - *init:* override this method for any general object setup (such as 
111     observers, starting timers and animations, etc) that you need to happen 
112     everytime the view is created, regardless of whether or not its layer 
113     exists yet.
114     
115   - *render:* override this method to generate or update your HTML to reflect
116     the current state of your view.  This method is called both when your view
117     is first created and later anytime it needs to be updated.
118 
119   - *didCreateLayer:* the render() method is used to generate new HTML.  
120     Override this method to perform any additional setup on the DOM you might
121     need to do after creating the view.  For example, if you need to listen
122     for events.
123     
124   - *willDestroyLayer:* if you implement didCreateLayer() to setup event 
125     listeners, you should implement this method as well to remove the same 
126     just before the DOM for your view is destroyed.
127     
128   - *updateLayer:* Normally, when a view needs to update its content, it will
129     re-render the view using the render() method.  If you would like to 
130     override this behavior with your own custom updating code, you can 
131     replace updateLayer() with your own implementation instead.
132     
133   - *didAppendToDocument:* in theory all DOM setup could be done
134     in didCreateLayer() as you already have a DOM element instantiated. 
135     However there is cases where the element has to be first appended to the
136     Document because there is either a bug on the browser or you are using 
137     plugins which objects are not instantiated until you actually append the
138     element to the DOM. This will allow you to do things like registering 
139     DOM events on flash or quicktime objects.
140   
141   @extends SC.Responder
142   @extends SC.DelegateSupport
143   @since SproutCore 1.0
144 */

284   /**
285     Recomputes the isVisibleInWindow property based on the visibility of the 
286     view and its parent.  If the recomputed value differs from the current 
287     isVisibleInWindow state, this method will also call 
288     recomputIsVisibleInWindow() on its child views as well.  As an optional 
289     optimization, you can pass the isVisibleInWindow state of the parentView 
290     if you already know it.
291     
292     You will not generally need to call or override this method yourself. It 
293     is used by the SC.View hierarchy to relay window visibility changes up 
294     and down the chain.
295     
296     @property {Boolean} parentViewIsVisible
297     @returns {SC.View} receiver 
298   */
299   recomputeIsVisibleInWindow: function(parentViewIsVisible) {
300     var previous = this.get('isVisibleInWindow'),
301         current  = this.get('isVisible'),
302         parentView;
303     
304     // isVisibleInWindow = isVisible && parentView.isVisibleInWindow
305     // this approach only goes up to the parentView if necessary.
306     if (current) {
307       // If we weren't passed in 'parentViewIsVisible' (we generally aren't;
308       // it's an optimization), then calculate it.
309       if (parentViewIsVisible === undefined) {
310         parentView = this.get('parentView');
311         parentViewIsVisible = parentView ? parentView.get('isVisibleInWindow') : NO;
312       }
313       current = current && parentViewIsVisible;
314     }
315     
316     // If our visibility has changed, then set the new value and notify our
317     // child views to update their value.
318     //if (previous !== current) {
319       this.set('isVisibleInWindow', current) ;
320       
321       var childViews = this.get('childViews'), len = childViews.length, idx;
322       for(idx=0;idx<len;idx++) {
323         childViews[idx].recomputeIsVisibleInWindow(current);
324       }
325 
326 
327       // We'll also kick off some necessary work when the visibility changes.
328       // This more appropriately belongs in a 'isVisibleInWindow' observer or
329       // some such helper method because this work is not strictly related to
330       // computing the visibility, but view performance is critical, so
331       // avoiding the extra observer is worthwhile.
332       if (current) {
333         // If we just became visible, update layer + layout if needed...
334         this.displayDidChange();
335 
336         if (this.get('childViewsNeedLayout')) {
337           this.invokeOnce(this.layoutChildViewsIfNeeded);
338         }
339       }
340       else {
341         // We need to set some internal state to indicate that we went from
342         // visible to invisible.
343         //
344         // (Typically, invisible views are not updated for performance
345         // reasons, but if we went from visible to invisible, we need to
346         // update once.)
347         this._forceLayerUpdateDueToVisibilityChange = YES;
348         this.displayDidChange();
349 
350         // Also, if we were previously 'firstResponder', resign it.
351         if (this.get('isFirstResponder')) this.resignFirstResponder();
352       }
353     //}

383   /**
384     Insert the view into the the receiver's childNodes array.
385     
386     The view will be added to the childNodes array before the beforeView.  If 
387     beforeView is null, then the view will be added to the end of the array.  
388     This will also add the view's rootElement DOM node to the receivers 
389     containerElement DOM node as a child.
390     
391     If the specified view already belongs to another parent, it will be 
392     removed from that view first.
393     
394     @param {SC.View} view
395     @param {SC.View} beforeView
396     @returns {SC.View} the receiver
397   */
398   insertBefore: function(view, beforeView) { 
399     view.beginPropertyChanges(); // limit notifications
400     
401     // remove view from old parent if needed.  Also notify views.
402     if (view.get('parentView')) view.removeFromParent() ;
403     if (this.willAddChild) this.willAddChild(view, beforeView) ;
404     if (view.willAddToParent) view.willAddToParent(this, beforeView) ;
405     
406     // set parentView of child
407     view.set('parentView', this);
408     
409     // add to childView's array.
410     var idx, childViews = this.get('childViews') ;
411     if (childViews.needsClone) this.set(childViews = []);
412     idx = (beforeView) ? childViews.indexOf(beforeView) : childViews.length;
413     if (idx<0) idx = childViews.length ;
414     childViews.insertAt(idx, view) ;
415     
416     // The DOM will need some fixing up, note this on the view.
417     view.parentViewDidChange() ;
418     view.layoutDidChange() ;
419     var pane = view.get('pane');
420     if(pane && pane.get('isPaneAttached')) {
421       view._notifyDidAppendToDocument();
422     }
423     
424     // notify views
425     if (this.didAddChild) this.didAddChild(view, beforeView) ;
426     if (view.didAddToParent) view.didAddToParent(this, beforeView) ;
427     
428     view.endPropertyChanges();
429

445     if (view.willRemoveFromParent) view.willRemoveFromParent() ;

453         idx = childViews.indexOf(view) ;

461     if (view.didRemoveFromParent) view.didRemoveFromParent(this) ;

696   /**
697     Attempts to discover the layer in the parent layer.  The default 
698     implementation looks for an element with an ID of layerId (or the view's
699     guid if layerId is null).  You can override this method to provide your
700     own form of lookup.  For example, if you want to discover your layer using
701     a CSS class name instead of an ID.
702     
703     @param {DOMElement} parentLayer the parent's DOM layer
704     @returns {DOMElement} the discovered layer
705   */
706   findLayerInParentLayer: function(parentLayer) {
707     var layerId = this.get('layerId'),
708         node, i, ilen,found, elem;
709     
710     // first, let's try the fast path...
711     if(parentLayer.getElementById) elem = parentLayer.getElementById(layerId) ;
712     else elem = document.getElementById(layerId) ;
713     
714     // TODO: use code generation to only really do this check on IE
715     if (SC.browser.msie && elem && elem.id !== layerId) elem = null ;
716     
717     // if browser supports querySelector use that.
718     if (!elem && parentLayer.querySelector) {
719       // TODO: make querySelector work on all platforms...
720       elem = parentLayer.querySelector('#' + layerId);
721     }
722     
723     // if no element was found the fast way, search down the parentLayer for
724     // the element.  This code should not be invoked very often.  Usually a
725     // DOM element will be discovered by the first method above.
726     // This code uses a BFS algorithm as is expected to find the layer right 
727     // below the parent.
