  8 sc_require('models/record');
  9 sc_require('models/record_attribute');
 10 
 11 /** @class
 12   
 13   SingleAttribute is a subclass of RecordAttribute and handles to-one
 14   relationships.
 15 
 16   There are many ways you can configure a SingleAttribute:
 17   
 18   {{{
 19     group: SC.Record.toOne('MyApp.Group', { 
 20       inverse: 'contacts', // set the key used to represent the inverse 
 21       isMaster: YES|NO, // indicate whether changing this should dirty
 22       transform: function(), // transforms value <=> storeKey,
 23       isEditable: YES|NO, make editable or not
 24     });
 25   }}}
 26   
 27   @extends SC.RecordAttribute
 28   @since SproutCore 1.0
 29 */
 30 SC.SingleAttribute = SC.RecordAttribute.extend(
 31   /** @scope SC.SingleAttribute.prototype */ {
 32 
 33   /**
 34     Specifies the property on the member record that represents the inverse
 35     of the current relationship.  If set, then modifying this relationship
 36     will also alter the opposite side of the relationship.
 37     
 38     @property {String}
 39   */
 40   inverse: null,
 41   
 42   /**
 43     If set, determines that when an inverse relationship changes whether this
 44     record should become dirty also or not.
 45     
 46     @property {Boolean}
 47   */
 48   isMaster: YES,
 49   
 50   
 51   /**
 52     @private - implements support for handling inverse relationships.
 53   */
 54   call: function(record, key, newRec) {
 55     var attrKey = this.get('key') || key,
 56         inverseKey, isMaster, oldRec, attr, ret, nvalue;
 57     
 58     // WRITE
 59     if (newRec !== undefined) {
 60 
 61       // can only take other records or null
 62       if (newRec && !SC.kindOf(newRec, SC.Record)) {
 63         throw "%@ is not an instance of SC.Record".fmt(newRec);
 64       }
 65 
 66       inverseKey = this.get('inverse');
 67       if (inverseKey) oldRec = this._scsa_call(record, key);
 68 
 69       // careful: don't overwrite value here.  we want the return value to 
 70       // cache.
 71       nvalue = this.fromType(record, key, newRec) ; // convert to attribute.
 72       record.writeAttribute(attrKey, nvalue, !this.get('isMaster')); 
 73       ret = newRec ;
 74 
 75       // ok, now if we have an inverse relationship, get the inverse 
 76       // relationship and notify it of what is happening.  This will allow it
 77       // to update itself as needed.  The callbacks implemented here are 
 78       // supported by both SingleAttribute and ManyAttribute.
 79       //
 80       if (inverseKey && (oldRec !== newRec)) {
 81         if (oldRec && (attr = oldRec[inverseKey])) {
 82           attr.inverseDidRemoveRecord(oldRec, inverseKey, record, key);
 83         }
 84 
 85         if (newRec && (attr = newRec[inverseKey])) {
 86           attr.inverseDidAddRecord(newRec, inverseKey, record, key);
 87         }
 88       }
 89       
 90     // READ 
 91     } else ret = this._scsa_call(record, key, newRec);
 92 
 93     return ret ;
 94   },
 95   
 96   /** @private - save original call() impl */
 97   _scsa_call: SC.RecordAttribute.prototype.call,
 98   
 99   /**
100     Called by an inverse relationship whenever the receiver is no longer part
101     of the relationship.  If this matches the inverse setting of the attribute
102     then it will update itself accordingly.
103     
104     @param {SC.Record} the record owning this attribute
105     @param {String} key the key for this attribute
106     @param {SC.Record} inverseRecord record that was removed from inverse
107     @param {String} key key on inverse that was modified
108     @returns {void}
109   */
110   inverseDidRemoveRecord: function(record, key, inverseRecord, inverseKey) {
111 
112     var myInverseKey  = this.get('inverse'),
113         curRec   = this._scsa_call(record, key),
114         isMaster = this.get('isMaster'), attr;
115 
116     // ok, you removed me, I'll remove you...  if isMaster, notify change.
117     record.writeAttribute(key, null, !isMaster);
118     record.notifyPropertyChange(key);
119 
120     // if we have another value, notify them as well...
121     if ((curRec !== inverseRecord) || (inverseKey !== myInverseKey)) {
122       if (curRec && (attr = curRec[myInverseKey])) {
123         attr.inverseDidRemoveRecord(curRec, myInverseKey, record, key);
124       }
125     }
126   },
127   
128   /**
129     Called by an inverse relationship whenever the receiver is added to the 
130     inverse relationship.  This will set the value of this inverse record to 
131     the new record.
132     
133     @param {SC.Record} the record owning this attribute
134     @param {String} key the key for this attribute
135     @param {SC.Record} inverseRecord record that was added to inverse
136     @param {String} key key on inverse that was modified
137     @returns {void}
138   */
139   inverseDidAddRecord: function(record, key, inverseRecord, inverseKey) {
140     
141     var myInverseKey  = this.get('inverse'),
142         curRec   = this._scsa_call(record, key),
143         isMaster = this.get('isMaster'), 
144         attr, nvalue; 
145 
146     // ok, replace myself with the new value...
147     nvalue = this.fromType(record, key, inverseRecord); // convert to attr.
148     record.writeAttribute(key, nvalue, !isMaster);
149     record.notifyPropertyChange(key);
150 
151     // if we have another value, notify them as well...
152     if ((curRec !== inverseRecord) || (inverseKey !== myInverseKey)) {
153       if (curRec && (attr = curRec[myInverseKey])) {
154         attr.inverseDidRemoveRecord(curRec, myInverseKey, record, key);
155       }
156     }
157   }
158 
159 });
