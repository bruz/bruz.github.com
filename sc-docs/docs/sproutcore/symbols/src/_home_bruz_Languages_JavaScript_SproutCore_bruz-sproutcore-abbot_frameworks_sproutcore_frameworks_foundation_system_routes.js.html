  8 /**
  9   SC.routes manages the browser location. You can change the hash part of the
 10   current location. The following code
 11   
 12   {{{
 13     SC.routes.set('location', 'notes/edit/4');
 14   }}}
 15   
 16   will change the location to http://domain.tld/my_app#notes/edit/4. Adding
 17   routes will register a handler that will be called whenever the location
 18   changes and matches the route:
 19   
 20   {{{
 21     SC.routes.add(':controller/:action/:id', MyApp, MyApp.route);
 22   }}}
 23   
 24   You can pass additional parameters in the location hash that will be relayed
 25   to the route handler:
 26   
 27   {{{
 28     SC.routes.set('location', 'notes/show/4?format=xml&language=fr');
 29   }}}
 30   
 31   The syntax for the location hash is described in the location property
 32   documentation, and the syntax for adding handlers is described in the
 33   add method documentation.
 34   
 35   Browsers keep track of the locations in their history, so when the user
 36   presses the 'back' or 'forward' button, the location is changed, SC.route
 37   catches it and calls your handler. Except for Internet Explorer versions 7
 38   and earlier, which do not modify the history stack when the location hash
 39   changes.
 40   
 41   @since SproutCore 1.1
 42 */
 43 SC.routes = SC.Object.create({
 44     
 45   /** @private
 46     A boolean value indicating whether or the ping method has been called
 47     to setup the SC.routes.
 48   
 49     @property
 50     @type {Boolean}
 51   */
 52   _didSetup: NO,
 53   
 54   
 55   /** @private
 56     Internal representation of the current location hash.
 57   
 58     @property
 59     @type {String}
 60   */
 61   _location: null,
 62   
 63   /** @private
 64     Routes are stored in a tree structure, this is the root node.
 65   
 66     @property
 67     @type {SC.routes._Route}
 68   */
 69   _firstRoute: null,
 70   
 71   /** @private
 72     Internal method used to extract and merge the parameters of a URL.
 73     
 74     @returns {Hash}
 75   */
 76   _extractParametersAndRoute: function(obj) {
 77     var params = {},
 78         route = obj.route || '',
 79         separator, parts, i, len, crumbs, key;
 80     
 81     separator = (route.indexOf('?') < 0 && route.indexOf('&') >= 0) ? '&' : '?';
 82     parts = route.split(separator);
 83     route = parts[0];
 84     if (parts.length === 1) {
 85       parts = [];
 86     } else if (parts.length === 2) {
 87       parts = parts[1].split('&');
 88     } else if (parts.length > 2) {
 89       parts.shift();
 90     }
 91     
 92     // extract the parameters from the route string
 93     len = parts.length;
 94     for (i = 0; i < len; ++i) {
 95       crumbs = parts[i].split('=');
 96       params[crumbs[0]] = crumbs[1];
 97     }
 98     
 99     // overlay any parameter passed in obj
100     for (key in obj) {
101       if (obj.hasOwnProperty(key) && key !== 'route') {
102         params[key] = '' + obj[key];
103       }
104     }
105     
106     // build the route
107     parts = [];
108     for (key in params) {
109       parts.push([key, params[key]].join('='));
110     }
111     params.params = separator + parts.join('&');
112     params.route = route;
113     
114     return params;
115   },
116   
117   /**
118     The current location hash. It is the part in the browser's location after
119     the '#' mark.
120     
121     The following code
122     
123     {{{
124       SC.routes.set('location', 'notes/edit/4');
125     }}}
126     
127     will change the location to http://domain.tld/my_app#notes/edit/4 and call
128     the correct route handler if it has been registered with the add method.
129     
130     You can also pass additional parameters. They will be relayed to the route
131     handler. For example, the following code
132     
133     {{{
134       SC.routes.add(':controller/:action/:id', MyApp, MyApp.route);
135       SC.routes.set('location', 'notes/show/4?format=xml&language=fr');
136     }}}
137     
138     will change the location to 
139     http://domain.tld/my_app#notes/show/4?format=xml&language=fr and call the
140     MyApp.route method with the following argument:
141     
142     {{{
143       { route: 'notes/show/4',
144         params: '?format=xml&language=fr',
145         controller: 'notes',
146         action: 'show',
147         id: '4',
148         format: 'xml',
149         language: 'fr' }
150     }}}
151     
152     The location can also be set with a hash, the following code
153     
154     {{{
155       SC.routes.set('location',
156         { route: 'notes/edit/4', format: 'xml', language: 'fr' });
157     }}}
158     
159     will change the location to
160     http://domain.tld/my_app#notes/show/4?format=xml&language=fr.
161     
162     The 'notes/show/4&format=xml&language=fr' syntax for passing parameters,
163     using a '&' instead of a '?', as used in SproutCore 1.0 is still supported.
164     
165     @property
166     @type {String}
167   */
168   location: function(key, value) {
169     var crumbs;
170     
171     if (value !== undefined) {
172       if (value === null) {
173         value = '';
174       }
175       
176       if (typeof(value) === 'object') {
177         crumbs = this._extractParametersAndRoute(value);
178         value = crumbs.route + crumbs.params;
179       }
180       
181       if(!SC.empty(value) || (this._location && this._location !== value)) {
182         window.location.hash = encodeURI(value);
183       }
184       this._location = value;
185       
186       return this;
187     }
188     return this._location;
189   }.property(),
190   
191   /**
192     You usually don't need to call this method. It is done automatically after
193     the application has been initialized.
194     
195     It registers for the hashchange event if available. If not, it creates a
196     timer that looks for location changes every 150ms.
197   */
198   ping: function() {
199     var that;
200     
201     if (!this._didSetup) {
202       this._didSetup = YES;
203       
204       if (this.get('supportsOnHashChange')) {
205         this.hashChange();
206         SC.Event.add(window, 'hashchange', this, this.hashChange);
207       
208       } else {
209         // we don't use a SC.Timer because we don't want
210         // a run loop to be triggered at each ping
211         that = this;
212         this._invokeHashChange = function() {
213           that.hashChange();
214           setTimeout(that._invokeHashChange, 100);
215         };
216         this._invokeHashChange();
217       }
218     }
219   },
220   
221   /**
222     Whether the browser supports HTML5 history management.
223     
224     Code copied from Google's closure library which is under the Apache 2
225     license.
226     
227     {@link http://closure-library.googlecode.com/svn/trunk/closure/goog/history/history.js}
228     {@link http://www.w3.org/TR/html5/history.html}
229     @property
230     @returns {Boolean}
231   */
232   supportsOnHashChange: function() {
233     return SC.browser.isSafari && SC.browser.compareVersion(532,1) >= 0 ||
234            SC.browser.isMozilla && SC.browser.compareVersion(1,9,2) >= 0 ||
235            SC.browser.isIE && document.documentMode >= 8;
236   }.property().cacheable(),
237   
238   /**
239     Event handler for the hashchange event. Called automatically by the browser
240     if it supports the hashchange event, or by our timer if not.
241   */
242   hashChange: function(event) {
243     var loc = window.location.hash;
244     
245     loc = (loc && loc.length > 0) ? loc.slice(1, loc.length) : '';
246     if (!SC.browser.isMozilla) {
247       // because of bug https://bugzilla.mozilla.org/show_bug.cgi?id=483304
248       loc = decodeURI(loc);
249     }
250     
251     if (this.get('location') !== loc) {
252       SC.RunLoop.begin();
253       this.set('location', loc);
254       SC.RunLoop.end();
255     }
256   },
257   
258   /**
259     Adds a route handler. Routes have the following format:
260       - 'users/show/5' is a static route and only matches this exact string,
261       - ':action/:controller/:id' is a dynamic route and the handler will be
262         called with the 'action', 'controller' and 'id' parameters passed in a
263         hash,
264       - '*url' is a wildcard route, it matches the whole route and the handler
265         will be called with the 'url' parameter passed in a hash.
266     
267     Route types can be combined, the following are valid routes:
268       - 'users/:action/:id'
269       - ':controller/show/:id'
270       - ':controller/ *url' (ignore the space, because of jslint)
271     
272     @param {String} route the route to be registered
273     @param {Object} target the object on which the method will be called, or
274       directly the function to be called to handle the route
275     @param {Function} method the method to be called on target to handle the
276       route, can be a function or a string
277   */
278   add: function(route, target, method) {
279     if (!this._didSetup) {
280       this.invokeLast(this.ping);
281     }
282     
283     if (method === undefined && SC.typeOf(target) === SC.T_FUNCTION) {
284       method = target;
285       target = null;
286     } else if (SC.typeOf(method) === SC.T_STRING) {
287       method = target[method];
288     }
289     
290     if (!this._firstRoute) this._firstRoute = this._Route.create();
291     this._firstRoute.add(route.split('/'), target, method);
292     
293     return this;
294   },
295   
296   /**
297     Observer of the 'location' property that calls the correct route handler
298     when the location changes.
299   */
300   locationDidChange: function() {
301     this.trigger();
302   }.observes('location'),
303   
304   /**
305     Triggers a route even if already in that route (does change the location, if it
306     is not already changed, as well).
307     
308     If the location is not the same as the supplied location, this simply lets "location"
309     handle it (which ends up coming back to here).
310   */
311   trigger: function() {
312     var firstRoute = this._firstRoute,
313         location = this.get('location'),
314         params, route;
315     
316     if (firstRoute) {
317       params = this._extractParametersAndRoute({ route: location });
318       location = params.route;
319       delete params.route;
320       delete params.params;
321       route = firstRoute.routeForParts(location.split('/'), params);
322       if (route && route.target && route.method) {
323         route.method.call(route.target, params);
324       }
325     }
326   },
327   
328   /**
329     @private
330     @class
331 
332     SC.routes._Route is a class used internally by SC.routes. The routes defined by your
333     application are stored in a tree structure, and this is the class for the
334     nodes.
335   */
336   _Route: SC.Object.extend(
337   /** @scope SC.routes._Route.prototype */ {
338 
339     target: null,
340 
341     method: null,
342 
343     staticRoutes: null,
344 
345     dynamicRoutes: null,
346 
347     wildcardRoutes: null,
348 
349     add: function(parts, target, method) {
350       var part, nextRoute;
351 
352       // clone the parts array because we are going to alter it
353       parts = SC.clone(parts);
354 
355       if (!parts || parts.length === 0) {
356         this.target = target;
357         this.method = method;
358 
359       } else {
360         part = parts.shift();
361 
362         // there are 3 types of routes
363         switch (part.slice(0, 1)) {
364 
365         // 1. dynamic routes
366         case ':':
367           part = part.slice(1, part.length);
368           if (!this.dynamicRoutes) this.dynamicRoutes = {};
369           if (!this.dynamicRoutes[part]) this.dynamicRoutes[part] = this.constructor.create();
370           nextRoute = this.dynamicRoutes[part];
371           break;
372 
373         // 2. wildcard routes
374         case '*':
375           part = part.slice(1, part.length);
376           if (!this.wildcardRoutes) this.wildcardRoutes = {};
377           nextRoute = this.wildcardRoutes[part] = this.constructor.create();
378           break;
379 
380         // 3. static routes
381         default:
382           if (!this.staticRoutes) this.staticRoutes = {};
383           if (!this.staticRoutes[part]) this.staticRoutes[part] = this.constructor.create();
384           nextRoute = this.staticRoutes[part];
385         }
386 
387         // recursively add the rest of the route
388         if (nextRoute) nextRoute.add(parts, target, method);
389       }
390     },
391 
392     routeForParts: function(parts, params) {
393       var part, key, route;
394 
395       // clone the parts array because we are going to alter it
396       parts = SC.clone(parts);
397 
398       // if parts is empty, we are done
399       if (!parts || parts.length === 0) {
400         return this.method ? this : null;
401 
402       } else {
403         part = parts.shift();
404 
405         // try to match a static route
406         if (this.staticRoutes && this.staticRoutes[part]) {
407           return this.staticRoutes[part].routeForParts(parts, params);
408 
409         } else {
410 
411           // else, try to match a dynamic route
412           for (key in this.dynamicRoutes) {
413             route = this.dynamicRoutes[key].routeForParts(parts, params);
414             if (route) {
415               params[key] = part;
416               return route;
417             }
418           }
419 
420           // else, try to match a wilcard route
421           for (key in this.wildcardRoutes) {
422             parts.unshift(part);
423             params[key] = parts.join('/');
424             return this.wildcardRoutes[key].routeForParts(null, params);
425           }
426 
427           // if nothing was found, it means that there is no match
428           return null;
429         }
430       }
431     }
432 
433   })
434   
435 });
