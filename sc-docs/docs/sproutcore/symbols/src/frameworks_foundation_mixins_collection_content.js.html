  9 /**
 10   Used for contentIndexDisclosureState().  Indicates open branch node.
 11   
 12   @property {Number}
 13 */
 14 SC.BRANCH_OPEN = 0x0011;
 15 
 16 /**
 17   Used for contentIndexDisclosureState().  Indicates closed branch node.
 18   
 19   @property {Number}
 20 */
 21 SC.BRANCH_CLOSED = 0x0012;
 22 
 23 /**
 24   Used for contentIndexDisclosureState().  Indicates leaf node.
 25   
 26   @property {Number}
 27 */
 28 SC.LEAF_NODE = 0x0020;
 29 
 30 /**
 31   @namespace
 32 
 33   This mixin provides standard methods used by a CollectionView to provide
 34   additional meta-data about content in a collection view such as selection
 35   or enabled state.
 36   
 37   You can apply this mixin to a class that you set as a delegate or to the
 38   object you set as content.  SC.ArrayControllers automatically implement
 39   this mixin.
 40   
 41   @since SproutCore 1.0
 42 */
 43 SC.CollectionContent = {
 44 
 45   /**
 46     Used to detect the mixin by SC.CollectionView
 47 
 48     @property {Boolean}
 49   */
 50   isCollectionContent: YES,
 51   
 52   /**
 53     Return YES if the content index should be selected.  Default behavior 
 54     looks at the selection property on the view.
 55     
 56     @param {SC.CollectionView} view the collection view
 57     @param {SC.Array} content the content object
 58     @param {Number} idx the content index
 59     @returns {Boolean} YES, NO, or SC.MIXED_STATE
 60   */
 61   contentIndexIsSelected: function(view, content, idx) {
 62     var sel = view.get('selection');
 63     return sel ? sel.contains(content, idx) : NO ;
 64   },
 65   
 66   /**
 67     Returns YES if the content index should be enabled.  Default looks at the
 68     isEnabled state of the collection view.
 69     looks at the selection property on the view.
 70     
 71     @param {SC.CollectionView} view the collection view
 72     @param {SC.Array} content the content object
 73     @param {Number} idx the content index
 74     @returns {Boolean} YES, NO, or SC.MIXED_STATE
 75   */
 76   contentIndexIsEnabled: function(view, content, idx) {
 77     return view.get('isEnabled');
 78   },
 79   
 80   // ..........................................................
 81   // GROUPING
 82   // 
 83   
 84   /**
 85     Optionally return an index set containing the indexes that may be group
 86     views.  For each group view, the delegate will actually be asked to 
 87     confirm the view is a group using the contentIndexIsGroup() method.
 88     
 89     If grouping is not enabled, return null.
 90     
 91     @param {SC.CollectionView} view the calling view
 92     @param {SC.Array} content the content object
 93     @return {SC.IndexSet} 
 94   */
 95   contentGroupIndexes: function(view, content) {
 96     return null;
 97   },
 98   
 99   /**
100     Returns YES if the item at the specified content index should be rendered
101     using the groupExampleView instead of the regular exampleView.  Note that
102     a group view is different from a branch/leaf view.  Group views often 
103     appear with different layout and a different look and feel.
104 
105     Default always returns NO.
106     
107     @param {SC.CollectionView} view the collection view
108     @param {SC.Array} content the content object
109     @param {Number} idx the content index
110     @returns {Boolean} YES, NO, or SC.MIXED_STATE
111   */
112   contentIndexIsGroup: function(view, content, idx) {
113     return NO ;
114   },
115   
116   // ..........................................................
117   // OUTLINE VIEWS
118   // 
119   
120   /**
121     Returns the outline level for the item at the specified index.  Can be 
122     used to display hierarchical lists.
123     
124     Default always returns -1 (no outline).
125     
126     @param {SC.CollectionView} view the collection view
127     @param {SC.Array} content the content object
128     @param {Number} idx the content index
129     @returns {Boolean} YES, NO, or SC.MIXED_STATE
130   */
131   contentIndexOutlineLevel: function(view, content, idx) {
132     return -1;
133   },
134   
135   /**
136     Returns a constant indicating the disclosure state of the item.  Must be
137     one of SC.BRANCH_OPEN, SC.BRANCH_CLOSED, SC.LEAF_NODE.  If you return one
138     of the BRANCH options then the item may be rendered with a disclosure 
139     triangle open or closed.  If you return SC.LEAF_NODe then the item will 
140     be rendered as a leaf node.  
141 
142     Default returns SC.LEAF_NODE.
143     
144     @param {SC.CollectionView} view the collection view
145     @param {SC.Array} content the content object
146     @param {Number} idx the content index
147     @returns {Boolean} YES, NO, or SC.MIXED_STATE
148   */
149   contentIndexDisclosureState: function(view, content, idx) {
150     return SC.LEAF_NODE;    
151   },
152   
153   /**
154     Called to expand a content index item if it is currently in a closed 
155     disclosure state.  The default implementation does nothing.
156     
157     @param {SC.CollectionView} view the collection view
158     @param {SC.Array} content the content object
159     @param {Number} idx the content index
160     @returns {void}
161   */
162   contentIndexExpand: function(view, content, idx) {
163     console.log('contentIndexExpand(%@, %@, %@)'.fmt(view,content,idx));
164   },
165   
166   /**
167     Called to collapse a content index item if it is currently in an open 
168     disclosure state.  The default implementation does nothing.  
169     
170     @param {SC.CollectionView} view the collection view
171     @param {SC.Array} content the content object
172     @param {Number} idx the content index
173     @returns {void}
174   */
175   contentIndexCollapse: function(view, content, idx) {
176     console.log('contentIndexCollapse(%@, %@, %@)'.fmt(view,content,idx));
177   }
178     
179 };
