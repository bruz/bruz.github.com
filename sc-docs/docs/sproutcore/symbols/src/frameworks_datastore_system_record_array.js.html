 10 /**
 11   @class
 12 
 13   A RecordArray wraps an array of storeKeys and, optionally, a Query object.
 14   When you access the items of a RecordArray it will automatically convert the
 15   storeKeys into actual SC.Record objects that the rest of your application
 16   can work with.
 17   
 18   Normally you do not create RecordArray's yourself.  Instead, a RecordArray
 19   is returned when you call SC.Store.findAll(), already properly configured.
 20   You can usually just work with the RecordArray instance just like another
 21   array.
 22   
 23   The information below about RecordArray internals is only intended for those
 24   who need to override this class for some reason to do some special.
 25   
 26   h2. Internal Notes
 27   
 28   Normally the RecordArray behavior is very simple.  Any array-like operations
 29   will be translated into similar calls onto the underlying array of 
 30   storeKeys.  The underlying array can be a real array or it may be a 
 31   SparseArray, which is how you implement incremental loading.
 32   
 33   If the RecordArray is created with an SC.Query objects as well (and it 
 34   almost always will have a Query object), then the RecordArray will also 
 35   consult the query for various delegate operations such as determining if 
 36   the record array should update automatically whenever records in the store
 37   changes.
 38   
 39   It will also ask the Query to refresh the storeKeys whenever records change
 40   in the store.
 41   
 42   @extends SC.Object
 43   @extends SC.Enumerable
 44   @extends SC.Array
 45   @since SproutCore 1.0
 46 */
 47

103   /** @private
104     Returned length is a pass-through to the storeKeys array.
105   */
106   length: function() {
107     this.flush(); // cleanup pending changes
108     var storeKeys = this.get('storeKeys');
109     return storeKeys ? storeKeys.get('length') : 0;
110   }.property('storeKeys').cacheable(),

112   _scra_records: null,
113   
114   /** @private
115     Looks up the store key in the store keys array and materializes a
116     records.
117     
118     @param {Number} idx index of the object
119     @return {SC.Record} materialized record
120   */
121   objectAt: function(idx) {
122 
123     this.flush(); // cleanup pending if needed
124 
125     var recs      = this._scra_records, 
126         storeKeys = this.get('storeKeys'),
127         store     = this.get('store'),
128         storeKey, ret ;
129     
130     if (!storeKeys || !store) return undefined; // nothing to do
131     if (recs && (ret=recs[idx])) return ret ; // cached
132     
133     // not in cache, materialize
134     if (!recs) this._scra_records = recs = [] ; // create cache
135     storeKey = storeKeys.objectAt(idx);
136     
137     if (storeKey) {
138       // if record is not loaded already, then ask the data source to 
139       // retrieve it
140       if (store.peekStatus(storeKey) === SC.Record.EMPTY) {
141         store.retrieveRecord(null, null, storeKey);
142       }
143       recs[idx] = ret = store.materializeRecord(storeKey);
144     }
145     return ret ;
146   },
147 
148   /** @private - optimized forEach loop. */
149   forEach: function(callback, target) {
150     this.flush();
151     
152     var recs      = this._scra_records, 
153         storeKeys = this.get('storeKeys'),
154         store     = this.get('store'), 
155         len       = storeKeys ? storeKeys.get('length') : 0,
156         idx, storeKey, rec;
157         
158     if (!storeKeys || !store) return this; // nothing to do    
159     if (!recs) recs = this._scra_records = [] ;
160     if (!target) target = this;
161     
162     for(idx=0;idx<len;idx++) {
163       rec = recs[idx];
164       if (!rec) {
165         rec = recs[idx] = store.materializeRecord(storeKeys.objectAt(idx));
166       }
167       callback.call(target, rec, idx, this);
168     }
169     
170     return this;
171   },

173   /** @private
174     Pass through to the underlying array.  The passed in objects must be
175     records, which can be converted to storeKeys.
176     
177     @param {Number} idx start index
178     @param {Number} amt end index
179     @param {SC.RecordArray} recs to replace with records
180     @return {SC.RecordArray} 'this' after replace
181   */
182   replace: function(idx, amt, recs) {
183 
184     this.flush(); // cleanup pending if needed
185     
186     var storeKeys = this.get('storeKeys'), 
187         len       = recs ? (recs.get ? recs.get('length') : recs.length) : 0,
188         i, keys;
189         
190     if (!storeKeys) throw "storeKeys required";
191 
192     var query = this.get('query');
193     if (query && !query.get('isEditable')) throw SC.RecordArray.NOT_EDITABLE;
194     
195     // you can't modify an array whose store keys are autogenerated from a 
196     // query.
197     
198     // map to store keys
199     keys = [] ;
200     for(i=0;i<len;i++) keys[i] = recs.objectAt(i).get('storeKey');
201     
202     // pass along - if allowed, this should trigger the content observer 
203     storeKeys.replace(idx, amt, keys);
204     return this; 
205   },

345   /** @private
346     Called whenever the store initiates a refresh of the query.  Sets the 
347     status of the record array to the appropriate status.
348     
349     @param {SC.Query} query
350     @returns {SC.RecordArray} receiver
351   */
352   storeWillFetchQuery: function(query) {
353     var status = this.get('status'),
354         K      = SC.Record;
355     if ((status === K.EMPTY) || (status === K.ERROR)) status = K.BUSY_LOADING;
356     if (status & K.READY) status = K.BUSY_REFRESH;
357     this.setIfChanged('status', status);
358     return this ;
359   },

364     @param {SC.Query} query
365     @returns {SC.RecordArray} receiver
366   */
367   storeDidFetchQuery: function(query) {
368     this.setIfChanged('status', SC.Record.READY_CLEAN);
369     return this ;
370   },

372   /** @private
373     Called whenever the store has cancelled a refresh.  Sets the 
374     status of the record array to the appropriate status.
375     
376     @param {SC.Query} query
377     @returns {SC.RecordArray} receiver
378   */
379   storeDidCancelQuery: function(query) {
380     var status = this.get('status'),
381         K      = SC.Record;
382     if (status === K.BUSY_LOADING) status = K.EMPTY;

400   /** @private
401     Called by the store whenever it changes the state of certain store keys.
402     If the receiver cares about these changes, it will mark itself as dirty.
403     The next time you try to access the record array it will update any 
404     pending changes.
405     
406     @param {SC.Array} storeKeys the effected store keys
407     @param {SC.Set} recordTypes the record types for the storeKeys.
408     @returns {SC.RecordArray} receiver
409   */
410   storeDidChangeStoreKeys: function(storeKeys, recordTypes) {
411     var query =  this.get('query');
412     // fast path exits
413     if (query.get('location') !== SC.Query.LOCAL) return this;
414     if (!query.containsRecordTypes(recordTypes)) return this;   
415     
416     // ok - we're interested.  mark as dirty and save storeKeys.
417     var changed = this._scq_changedStoreKeys;
418     if (!changed) changed = this._scq_changedStoreKeys = SC.IndexSet.create();
419     changed.addEach(storeKeys);
420     
421     this.set('needsFlush', YES);
422     this.enumerableContentDidChange();
423 
424     return this;
425   },
426   
427   /**
428     Applies the query to any pending changed store keys, updating the record
429     array contents as necessary.  This method is called automatically anytime
430     you access the RecordArray to make sure it is up to date, but you can 
431     call it yourself as well if you need to force the record array to fully
432     update immediately.
433     
434     Currently this method only has an effect if the query location is 
435     SC.Query.LOCAL.  You can call this method on any RecordArray however,
436     without an error.
437     
438     @param {Boolean} _flush to force it - use reload() to trigger it
439     @returns {SC.RecordArray} receiver
440   */
441   flush: function(_flush) {
442     // Are we already inside a flush?  If so, then don't do it again, to avoid
443     // never-ending recursive flush calls.  Instead, we'll simply mark
444     // ourselves as needing a flush again when we're done.
445     if (this._insideFlush) {
446       this.set('needsFlush', YES);
447       return this;
448     }
449     
450     if (!this.get('needsFlush') && !_flush) return this; // nothing to do
451     this.set('needsFlush', NO); // avoid running again.
452     
453     // fast exit
454     var query = this.get('query'),
455         store = this.get('store'); 
456     if (!store || !query || query.get('location') !== SC.Query.LOCAL) {
457       return this;
458     }
459     
460     this._insideFlush = YES;
461     
462     // OK, actually generate some results
463     var storeKeys = this.get('storeKeys'),
464         changed   = this._scq_changedStoreKeys,
465         didChange = NO,
466         K         = SC.Record,
467         rec, status, recordType, sourceKeys, scope, included;
468 
469     // if we have storeKeys already, just look at the changed keys
470     var oldStoreKeys = storeKeys;
471     if (storeKeys && !_flush) {
472       if (changed) {
473         changed.forEach(function(storeKey) {
474           // get record - do not include EMPTY or DESTROYED records
475           status = store.peekStatus(storeKey);
476           if (!(status & K.EMPTY) && !((status & K.DESTROYED) || (status === K.BUSY_DESTROYING))) {
477             rec = store.materializeRecord(storeKey);
478             included = !!(rec && query.contains(rec));
479           } else included = NO ;
480           
481           // if storeKey should be in set but isn't -- add it.
482           if (included) {
483             if (storeKeys.indexOf(storeKey)<0) {
484               if (!didChange) storeKeys = storeKeys.copy(); 
485               storeKeys.pushObject(storeKey); 
486             }
487           // if storeKey should NOT be in set but IS -- remove it
488           } else {
489             if (storeKeys.indexOf(storeKey)>=0) {
490               if (!didChange) storeKeys = storeKeys.copy();
491               storeKeys.removeObject(storeKey);
492             } // if (storeKeys.indexOf)
493           } // if (included)
494         }, this);
495         // make sure resort happens
496         didChange = YES ;
497       } // if (changed)
498     
499     // if no storeKeys, then we have to go through all of the storeKeys 
500     // and decide if they belong or not.  ick.
501     } else {
502       
503       // collect the base set of keys.  if query has a parent scope, use that
504       if (scope = query.get('scope')) {
505         sourceKeys = scope.flush().get('storeKeys');
506 
507       // otherwise, lookup all storeKeys for the named recordType...
508       } else if (recordType = query.get('expandedRecordTypes')) {
509         sourceKeys = SC.IndexSet.create();
510         recordType.forEach(function(cur) { 
511           sourceKeys.addEach(store.storeKeysFor(recordType));
512         });
513       }
514 
515       // loop through storeKeys to determine if it belongs in this query or 
516       // not.
517       storeKeys = [];
518       sourceKeys.forEach(function(storeKey) {
519         status = store.peekStatus(storeKey);
520         if (!(status & K.EMPTY) && !((status & K.DESTROYED) || (status === K.BUSY_DESTROYING))) {
521           rec = store.materializeRecord(storeKey);
522           if (rec && query.contains(rec)) storeKeys.push(storeKey);
523         }
524       });
525       
526       didChange = YES ;
527     }
528 
529     // clear set of changed store keys
530     if (changed) changed.clear();
531     
532     // only resort and update if we did change
533     if (didChange) {
534       
535       // storeKeys must be a new instance because orderStoreKeys() works on it
536       if (storeKeys && (storeKeys===oldStoreKeys)) {
537         storeKeys = storeKeys.copy();
538       }
539       
540       storeKeys = SC.Query.orderStoreKeys(storeKeys, query, store);
541       if (SC.compare(oldStoreKeys, storeKeys) !== 0){
542         this.set('storeKeys', SC.clone(storeKeys)); // replace content
543       }
544     }
545 
546     this._insideFlush = NO;
547     return this;
548   },

600   /** @private 
601     Invoked whenever the storeKeys array changes.  Observes changes.
602   */
603   _storeKeysDidChange: function() {
604     var storeKeys = this.get('storeKeys');
605     
606     var prev = this._prevStoreKeys, 
607         f    = this._storeKeysContentDidChange,
608         fs   = this._storeKeysStateDidChange;
609     
610     if (storeKeys === prev) return; // nothing to do
611     
612     if (prev) prev.removeObserver('[]', this, f);
613     this._prevStoreKeys = storeKeys;
614     if (storeKeys) storeKeys.addObserver('[]', this, f);
615     
616     var rev = (storeKeys) ? storeKeys.propertyRevision : -1 ;
617     this._storeKeysContentDidChange(storeKeys, '[]', storeKeys, rev);
618     
619   }.observes('storeKeys'),
620   
621   /** @private
622     Invoked whenever the content of the storeKeys array changes.  This will
623     dump any cached record lookup and then notify that the enumerable content
624     has changed.
625   */
626   _storeKeysContentDidChange: function(target, key, value, rev) {
627     if (this._scra_records) this._scra_records.length=0 ; // clear cache
628     
629     this.beginPropertyChanges()
630       .notifyPropertyChange('length')
631       .enumerableContentDidChange()
632     .endPropertyChanges();
633   },
634   
635   /** @private */
636   init: function() {
637     sc_super();
638     this._storeKeysDidChange();
639   }
