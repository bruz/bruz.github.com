  8 /**
  9   @class
 10 
 11   Displays a progress bar.  You can display both a defined and an 
 12   indeterminate progressbar.  The progress bar itself is designed to be styled
 13   using CSS classes with the following structure:
 14   
 15   <div class="sc-progress-view"><div class="inner"></div></div>
 16   
 17   The outer can form the boundary of the bar while the inner will be adjusted 
 18   to fit the percentage of the progress.
 19   
 20   Creating a ProgressView accepts a number of properties, for example:
 21   {
 22     value: 50, 
 23     minimum: 0, 
 24     maximum: 100,
 25     isIndeterminate: NO,
 26     isEnabled: YES
 27   }
 28   
 29   Default isEnabled value is YES.
 30 
 31   @extends SC.View
 32   @extends SC.Control
 33   @since SproutCore 1.0
 34 */
 35 SC.ProgressView = SC.View.extend(SC.Control, {
 36   
 37   // ........................................
 38   // PROPERTIES
 39   //
 40 
 41   /**
 42     Bind this to the current value of the progress bar.  Note that by default 
 43     an empty value will disable the progress bar and a multiple value will make 
 44     it indeterminate.
 45   */
 46   value: 0.50,
 47   valueBindingDefault: SC.Binding.single().notEmpty(),
 48   
 49   /**
 50     The minimum value of the progress.
 51   */ 
 52   minimum: 0,
 53   minimumBindingDefault: SC.Binding.single().notEmpty(),
 54 
 55   /**
 56     Optionally specify the key used to extract the minimum progress value 
 57     from the content object.  If this is set to null then the minimum value
 58     will not be derived from the content object.
 59     
 60     @property {String}
 61   */
 62   contentMinimumKey: null,
 63   
 64   /**
 65     The maximum value of the progress bar.
 66   */
 67   maximum: 1.0,
 68   maximumBindingDefault: SC.Binding.single().notEmpty(),
 69 
 70   /**
 71     The value of the progress inner offset range. Should be the same as width 
 72     of image. Default it to 24
 73 
 74     @type Integer
 75   */
 76   offsetRange: 24,
 77 
 78   /**
 79     Optionally specify the key used to extract the maximum progress value 
 80     from the content object.  If this is set to null then the maximum value
 81     will not be derived from the content object.
 82     
 83     @property {String}
 84   */
 85   contentMaximumKey: null,
 86 
 87   /** 
 88     Set to true if the item in progress is indeterminate.  This may be 
 89     overridden by the actual value.
 90     @returns {Boolean} 
 91   */
 92   isIndeterminate: NO,
 93   isIndeterminateBindingDefault: SC.Binding.bool(),
 94 
 95   /**
 96     Set to YES when the process is currently running.  This will cause the 
 97     progress bar to animate, especially if it is indeterminate.  
 98   */
 99   isRunning: NO,
100   isRunningBindingDefault: SC.Binding.bool(),
101 
102   /** 
103     Set to the matrix used for background image position for animation.
104     [1st image y-location, offset, total number of images]
105     @property {Array}
106   */
107   animatedBackgroundMatrix: [],
108   
109   /**
110     Optionally specify the key used to extract the isIndeterminate value 
111     from the content object.  If this is set to null then the isIndeterminate 
112     value will not be derived from the content object.
113     
114     @property {String}
115   */
116   contentIsIndeterminateKey: null,
117 
118   // ........................................
119   // STRUCTURE
120   //
121 
122   classNames: 'sc-progress-view',
123   
124   // ........................................
125   // INTERNAL SUPPORT
126   //
127 
128   _backgroundOffset: 0,
129   _currentBackground: 1,
130   _nextBackground: 1,
131   
132   // start animating at the end of the init() method.  note that we call this
133   // here because we want this to make sure this function is called anytime 
134   // the progress view is instantiated.
135   init: function() {
136     sc_super();
137     this.animateProgressBar(); // start animating...  
138   },
139   
140   // start/stop running animation based on isRunning state.
141   animateProgressBar: function() {
142     if (this.get('isRunning') && this.get('isVisibleInWindow')) {
143       this._animateProgressBar(500); // wait to start to avoid probs
144     }
145   }.observes('isRunning', 'isVisibleInWindow'),
146 
147   _animateProgressBar: function(delay) {  
148     if (delay===0) delay = 1000/30;
149     if (this.get('isRunning') && this.get('isVisibleInWindow')) {
150       this.displayDidChange();
151       this.invokeLater(this._animateProgressBar, delay, 600);
152     }
153   },
154   
155   displayProperties: 'value minimum maximum isIndeterminate'.w(),
156   
157   render: function(context, firstTime) {
158     var inner, animatedBackground, value, cssString, backPosition,
159         isIndeterminate = this.get('isIndeterminate'),
160         isRunning = this.get('isRunning'),
161         isEnabled = this.get('isEnabled'),
162         offsetRange = this.get('offsetRange'),
163         offset = (isIndeterminate && isRunning) ? 
164                 (Math.floor(Date.now()/75)%offsetRange-offsetRange) : 0;
165   
166     // compute value for setting the width of the inner progress
167     if (!isEnabled) {
168       value = "0%" ;
169     } else if (isIndeterminate) {
170       value = "120%";
171     } else {
172       value = (this.get("_percentageNumeric") * 100) + "%";
173     }
174 
175     var classNames = {
176       'sc-indeterminate': isIndeterminate,
177       'sc-empty': (value <= 0),
178       'sc-complete': (value >= 100)
179     };
180     
181     if(firstTime) {
182       var classString = this._createClassNameString(classNames);
183       context.push('<div class="sc-inner ', classString, '" style="width: ', 
184                     value, ';left: ', offset, 'px;">',
185                     '<div class="sc-inner-head">','</div>',
186                     '<div class="sc-inner-tail"></div></div>',
187                     '<div class="sc-outer-head"></div>',
188                     '<div class="sc-outer-tail"></div>');
189     }
190     else {
191       context.setClass(classNames);
192       inner = this.$('.sc-inner');
193       animatedBackground = this.get('animatedBackgroundMatrix');
194       cssString = "width: "+value+"; ";
195       cssString = cssString + "left: "+offset+"px; ";
196       if (animatedBackground.length === 3 ) {
197         inner.css('backgroundPosition', '0px -'+ 
198                 (animatedBackground[0] + 
199                 animatedBackground[1]*this._currentBackground)+'px');
200         if(this._currentBackground===animatedBackground[2]-1
201            || this._currentBackground===0){
202           this._nextBackground *= -1;
203         }
204         this._currentBackground += this._nextBackground;
205         
206         cssString = cssString + "backgroundPosition: "+backPosition+"px; ";
207         //Instead of using css() set attr for faster perf.
208         inner.attr('style', cssString);
209       }else{
210         inner.attr('style', cssString);
211       }
212     }
213     
214   },
215   
216   contentPropertyDidChange: function(target, key) {
217     var content = this.get('content');
218     this.beginPropertyChanges()
219       .updatePropertyFromContent('value', key, 'contentValueKey', content)
220       .updatePropertyFromContent('minimum', key, 'contentMinimumKey', content)
221       .updatePropertyFromContent('maximum', key, 'contentMaximumKey', content)
222       .updatePropertyFromContent('isIndeterminate', key, 'contentIsIndeterminateKey', content)
223     .endPropertyChanges();
224   },
225   
226   _percentageNumeric: function(){
227     var minimum = this.get('minimum') || 0.0,
228         maximum = this.get('maximum') || 1.0,
229         value = this.get('value') || 0.0;
230     value = (value - minimum) / (maximum - minimum);
231     if (value > 1.0) value = 1.0;
232 
233     if(isNaN(value)) value = 0.0;
234     // cannot be smaller then minimum
235     if(value<minimum) value = 0.0;
236     // cannot be larger then maximum
237     if(value>maximum) value = 1.0;
238     return value;
239   }.property('value').cacheable(),
240   
241   _createClassNameString: function(classNames) {
242     var classNameArray = [], key;
243     for(key in classNames) {
244       if(!classNames.hasOwnProperty(key)) continue;
245       if(classNames[key]) classNameArray.push(key);
246     }
247     return classNameArray.join(" ");
248   }
249   
250 }) ;
