 80   /**
 81     @private
 82     Calculates the maximum offset given content and container sizes, and the
 83     alignment.
 84   */
 85   maximumScrollOffset: function(contentSize, containerSize, align) {
 86     // if our content size is larger than or the same size as the container, it's quite
 87     // simple to calculate the answer. Otherwise, we need to do some fancy-pants
 88     // alignment logic (read: simple math)
 89     if (contentSize >= containerSize) return contentSize - containerSize;
 90     
 91     // alignment, yeah
 92     if (align === SC.ALIGN_LEFT || align === SC.ALIGN_TOP) {
 93       // if we left-align something, and it is smaller than the view, does that not mean
 94       // that it's maximum (and minimum) offset is 0, because it should be positioned at 0?
 95       return 0;
 96     } else if (align === SC.ALIGN_MIDDLE || align === SC.ALIGN_CENTER) {
 97       // middle align means the difference divided by two, because we want equal parts on each side.
 98       return 0 - Math.round((containerSize - contentSize) / 2);
 99     } else {
100       // right align means the entire difference, because we want all that space on the left
101       return 0 - (containerSize - contentSize);
102     }
103   },

105   /**
106     @private
107     Calculates the minimum offset given content and container sizes, and the
108     alignment.
109   */
110   minimumScrollOffset: function(contentSize, containerSize, align) {
111     // if the content is larger than the container, we have no need to change the minimum
112     // away from the natural 0 position.
113     if (contentSize > containerSize) return 0;
114     
115     // alignment, yeah
116     if (align === SC.ALIGN_LEFT || align === SC.ALIGN_TOP) {
117       // if we left-align something, and it is smaller than the view, does that not mean
118       // that it's maximum (and minimum) offset is 0, because it should be positioned at 0?
119       return 0;
120     } else if (align === SC.ALIGN_MIDDLE || align === SC.ALIGN_CENTER) {
121       // middle align means the difference divided by two, because we want equal parts on each side.
122       return 0 - Math.round((containerSize - contentSize) / 2);
123     } else {
124       // right align means the entire difference, because we want all that space on the left
125       return 0 - (containerSize - contentSize);
126     }
127   },

131     size and the size of the scroll view.  If horizontal scrolling is 
132     disabled, this will always return 0.
133     
134     @property {Number}

137     var view = this.get('contentView'),
138         contentWidth = 0, calculatedWidth = 0;
139         
140     if (view && view.get('frame')) contentWidth = view.get('frame').width;
141     if (view) calculatedWidth = view.calculatedWidth || 0;

144     // to avoid looking at the incorrect value calculated by frame.
145     if(view && view.calculatedWidth && view.calculatedWidth!==0){
146       contentWidth = view.calculatedWidth; 
147     }
148     contentWidth *= this._scale;
149     
150     var containerWidth = this.get('containerView').get('frame').width ;

152     // we still must go through minimumScrollOffset even if we can't scroll
153     // because we need to adjust for alignment. So, just make sure it won't allow scrolling.

460   /**
461     Scrolls the receiver to the specified x,y coordinate.  This should be the
462     offset into the contentView you want to appear at the top-left corner of
463     the scroll view.
464     
465     This method will contrain the actual scroll based on whether the view
466     can scroll in the named direction and the maximum distance it can
467     scroll.
468     
469     If you only want to scroll in one direction, pass null for the other 
470     direction.  You can also optionally pass a Hash for the first parameter 
471     with x and y coordinates.
472     
473     @param x {Number} the x scroll location
474     @param y {Number} the y scroll location
475     @returns {SC.ScrollView} receiver
476   */
477   scrollTo: function(x,y) {
478     // normalize params
479     if (y===undefined && SC.typeOf(x) === SC.T_HASH) {
480       y = x.y; x = x.x;
481     }
482     
483     if (!SC.none(x)) {
484       x = Math.max(this.get('minimumHorizontalScrollOffset'),Math.min(this.get('maximumHorizontalScrollOffset'), x)) ;
485       this.set('horizontalScrollOffset', x) ;
486     }
487     
488     if (!SC.none(y)) {
489       y = Math.max(this.get('minimumVerticalScrollOffset'),Math.min(this.get('maximumVerticalScrollOffset'), y)) ;
490       this.set('verticalScrollOffset', y) ;
491     }
492     
493     return this ;
494   },
495   
496   /**
497     Scrolls the receiver in the horizontal and vertical directions by the 
498     amount specified, if allowed.  The actual scroll amount will be 
499     constrained by the current scroll view settings.
500     
501     If you only want to scroll in one direction, pass null or 0 for the other 
502     direction.  You can also optionally pass a Hash for the first parameter 
503     with x and y coordinates.
504     
505     @param x {Number} change in the x direction (or hash)
506     @param y {Number} change in the y direction
507     @returns {SC.ScrollView} receiver
508   */
509   scrollBy: function(x , y) {
510     // normalize params
511     if (y===undefined && SC.typeOf(x) === SC.T_HASH) {
512       y = x.y; x = x.x;
513     }
514     
515     // if null, undefined, or 0, pass null; otherwise just add current offset
516     x = (x) ? this.get('horizontalScrollOffset')+x : null ;
517     y = (y) ? this.get('verticalScrollOffset')+y : null ;
518     return this.scrollTo(x,y) ;
519   },

521   /**
522     Scroll the view to make the view's frame visible.  For this to make sense,
523     the view should be a subview of the contentView.  Otherwise the results
524     will be undefined.
525     
526     @param {SC.View} view view to scroll or null to scroll receiver visible
527     @returns {Boolean} YES if scroll position was changed
528   */
529   scrollToVisible: function(view) {
530     
531     // if no view is passed, do default
532     if (arguments.length === 0) return sc_super(); 
533     
534     var contentView = this.get('contentView') ;
535     if (!contentView) return NO; // nothing to do if no contentView.
536 
537     // get the frame for the view - should work even for views with static 
538     // layout, assuming it has been added to the screen.
539     var vf = view.get('frame');
540     if (!vf) return NO; // nothing to do
541     
542     // convert view's frame to an offset from the contentView origin.  This
543     // will become the new scroll offset after some adjustment.
544     vf = contentView.convertFrameFromView(vf, view.get('parentView')) ;
545     
546     return this.scrollToRect(vf);
547   },
548   
549   /**
550     Scroll to the supplied rectangle.
551     @param {rect} Rectangle to scroll to.
552     @returns {Boolean} YES if scroll position was changed.
553   */
554   scrollToRect: function(rect) {
555     // find current visible frame.
556     var vo = SC.cloneRect(this.get('containerView').get('frame')) ;
557     
558     vo.x = this.get('horizontalScrollOffset') ;
559     vo.y = this.get('verticalScrollOffset') ;
560 
561     var origX = vo.x, origY = vo.y;
562     
563     // if top edge is not visible, shift origin
564     vo.y -= Math.max(0, SC.minY(vo) - SC.minY(rect)) ;
565     vo.x -= Math.max(0, SC.minX(vo) - SC.minX(rect)) ;
566     
567     // if bottom edge is not visible, shift origin
568     vo.y += Math.max(0, SC.maxY(rect) - SC.maxY(vo)) ;
569     vo.x += Math.max(0, SC.maxX(rect) - SC.maxX(vo)) ;
570     
571     // scroll to that origin.
572     if ((origX !== vo.x) || (origY !== vo.y)) {
573       this.scrollTo(vo.x, vo.y);
574       return YES ;
575     } else return NO;
576   },

683   /**
684     Adjusts the layout for the various internal views.  This method is called
685     once when the scroll view is first configured and then anytime a scroller
686     is shown or hidden.  You can call this method yourself as well to retile.
687     
688     You may also want to override this method to handle layout for any
689     additional controls you have added to the view.
690   */
691   tile: function() {
692     
693     // for now just hide scroll bars on touch platforms
694     var isTouch = SC.platform.touch;
695     
696     // get horizontal scroller/determine if we should have a scroller
697     var hscroll = this.get('hasHorizontalScroller') ? this.get('horizontalScrollerView') : null ;
698     var hasHorizontal = hscroll && !isTouch && this.get('isHorizontalScrollerVisible');

1053   _adjustForEdgeResistance: function(offset, minOffset, maxOffset, resistanceCoefficient, asymptote) {
1054     var distanceFromEdge;
1055     
1056     // find distance from edge
1057     if (offset < minOffset) distanceFromEdge = offset - minOffset;
1058     else if (offset > maxOffset) distanceFromEdge = maxOffset - offset;
1059     else return offset;
1060     
1061     // manipulate logarithmically
1062     distanceFromEdge = Math.pow(resistanceCoefficient, Math.abs(distanceFromEdge)) * asymptote;
1063     
1064     // adjust mathematically
1065     if (offset < minOffset) distanceFromEdge = distanceFromEdge - asymptote;
1066     else distanceFromEdge = -distanceFromEdge + asymptote;
1067     
1068     // generate final value
1069     return Math.min(Math.max(minOffset, offset), maxOffset) + distanceFromEdge;

1243   /**
1244     @private
1245     Does bounce calculations, adjusting velocity.
1246     
1247     Bouncing is fun. Functions that handle it should have fun names,
1248     don'tcha think?
1249     
1250     P.S.: should this be named "bouncityBounce" instead?
1251   */
1252   bouncyBounce: function(velocity, value, minValue, maxValue, de, ac, additionalAcceleration) {
1253     // we have 4 possible paths. On a higher level, we have two leaf paths that can be applied
1254     // for either of two super-paths.
1255     //
1256     // The first path is if we are decelerating past an edge: in this case, this function must
1257     // must enhance that deceleration. In this case, our math boils down to taking the amount
1258     // by which we are past the edge, multiplying it by our deceleration factor, and reducing
1259     // velocity by that amount.
1260     //
1261     // The second path is if we are not decelerating, but are still past the edge. In this case,
1262     // we must start acceleration back _to_ the edge. The math here takes the distance we are from
1263     // the edge, multiplies by the acceleration factor, and then performs two additional things:
1264     // First, it speeds up the acceleration artificially  with additionalAcceleration; this will
1265     // make the stop feel more sudden, as it will still have this additional acceleration when it reaches
1266     // the edge. Second, it ensures the result does not go past the final value, so we don't end up
1267     // bouncing back and forth all crazy-like.
1268     if (value < minValue) {
1269       if (velocity < 0) velocity = velocity + ((minValue - value) * de);
1270       else {
1271         velocity = Math.min((minValue-value) * ac + additionalAcceleration, minValue - value - 0.01);
1272       }
1273     } else if (value > maxValue) {
1274       if (velocity > 0) velocity = velocity - ((value - maxValue) * de);
1275       else {
1276         velocity = -Math.min((value - maxValue) * ac + additionalAcceleration, value - maxValue - 0.01);
1277       }
1278     }
1279     return velocity;
1280   },

1284     // However, note maxOffsetX/Y takes into account the scale;
1285     // also, newX/Y adds in the current deceleration velocity (the deceleration velocity will
1286     // be changed later in this function).
1287     var touch = this.touch,
1288         scale = this._scale,
1289         minOffsetX = this.minimumScrollOffset(touch.contentSize.width * this._scale, touch.containerSize.width, this.get("horizontalAlign")),
1290         minOffsetY = this.minimumScrollOffset(touch.contentSize.height * this._scale, touch.containerSize.height, this.get("verticalAlign")),
1291         maxOffsetX = this.maximumScrollOffset(touch.contentSize.width * this._scale, touch.containerSize.width, this.get("horizontalAlign")),
1292         maxOffsetY = this.maximumScrollOffset(touch.contentSize.height * this._scale, touch.containerSize.height, this.get("verticalAlign")),
1293         
1294         now = Date.now(),
1295         t = Math.max(now - touch.lastEventTime, 1),
1296         
1297         newX = this._scroll_horizontalScrollOffset + touch.decelerationVelocity.x * (t/10),
1298         newY = this._scroll_verticalScrollOffset + touch.decelerationVelocity.y * (t/10);
1299     
1300     var de = touch.decelerationFromEdge, ac = touch.accelerationToEdge;
1301     
1302     // determine if position was okay before adjusting scale (which we do, in
1303     // a lovely, animated way, for the scaled out/in too far bounce-back).
1304     // if the position was okay, then we are going to make sure that we keep the
1305     // position okay when adjusting the scale.
1306     //
1307     // Position OKness, here, referring to if the position is valid (within
1308     // minimum and maximum scroll offsets)
1309     var validXPosition = newX >= minOffsetX && newX <= maxOffsetX;
1310     var validYPosition = newY >= minOffsetY && newY <= maxOffsetY;
1311     
1312     // We are going to change scale in a moment, but the position should stay the
1313     // same, if possible (unless it would be more jarring, as described above, in
1314     // the case of starting with a valid position and ending with an invalid one).
1315     //
1316     // Because we are changing the scale, we need to make the position scale-neutral.
1317     // we'll make it non-scale-neutral after applying scale.
1318     //
1319     // Question: might it be better to save the center position instead, so scaling
1320     // bounces back around the center of the screen?
1321     newX /= this._scale;
1322     newY /= this._scale;
1323     
1324     // scale velocity (amount to change) starts out at 0 each time, because 
1325     // it is calculated by how far out of bounds it is, rather than by the
1326     // previous such velocity.
1327     var sv = 0;
1328     
1329     // do said calculation; we'll use the same bouncyBounce method used for everything
1330     // else, but our adjustor that gives a minimum amount to change by and (which, as we'll
1331     // discuss, is to make the stop feel slightly more like a stop), we'll leave at 0 
1332     // (scale doesn't really need it as much; if you disagree, at least come up with 
1333     // numbers more appropriate for scale than the ones for X/Y)
1334     sv = this.bouncyBounce(sv, scale, touch.minimumScale, touch.maximumScale, de, ac, 0);
1335     
1336     // add the amount to scale. This is linear, rather than multiplicative. If you think
1337     // it should be multiplicative (or however you say that), come up with a new formula.
1338     this._scale = scale = scale + sv;
1339     
1340     // now we can convert newX/Y back to scale-specific coordinates...
1341     newX *= this._scale;
1342     newY *= this._scale;
1343     
1344     // It looks very weird if the content started in-bounds, but the scale animation
1345     // made it not be in bounds; it causes the position to animate snapping back, and,
1346     // well, it looks very weird. It is more proper to just make sure it stays in a valid
1347     // position. So, we'll determine the new maximum/minimum offsets, and then, if it was
1348     // originally a valid position, we'll adjust the new position to a valid position as well.
1349     
1350     
1351     // determine new max offset
1352     minOffsetX = this.minimumScrollOffset(touch.contentSize.width * this._scale, touch.containerSize.width, this.get("horizontalAlign"));
1353     minOffsetY = this.minimumScrollOffset(touch.contentSize.height * this._scale, touch.containerSize.height, this.get("verticalAlign"));
1354     maxOffsetX = this.maximumScrollOffset(touch.contentSize.width * this._scale, touch.containerSize.width, this.get("horizontalAlign"));
1355     maxOffsetY = this.maximumScrollOffset(touch.contentSize.height * this._scale, touch.containerSize.height, this.get("verticalAlign"));
1356     
1357     // see if scaling messed up the X position (but ignore if 'tweren't right to begin with).
1358     if (validXPosition && (newX < minOffsetX || newX > maxOffsetX)) {
1359       // Correct the position
1360       newX = Math.max(minOffsetX, Math.min(newX, maxOffsetX));
1361     }
1362     
1363     // now the y
1364     if (validYPosition && (newY < minOffsetY || newY > maxOffsetY)) {
1365       // again, correct it...
1366       newY = Math.max(minOffsetY, Math.min(newY, maxOffsetY));
1367     }
1368     
1369     
1370     // now that we are done modifying the position, we may update the actual scroll
1371     this._scroll_horizontalScrollOffset = newX;
1372     this._scroll_verticalScrollOffset = newY;
1373     
1374     this._applyCSSTransforms(touch.layer); // <- Does what it sounds like.
1375 
1376     SC.RunLoop.begin();
1377     this._touchScrollDidChange();
1378     SC.RunLoop.end();
1379     
1380     // Now we have to adjust the velocities. The velocities are simple x and y numbers that
1381     // get added to the scroll X/Y positions each frame.
1382     // The default decay rate is .950 per frame. To achieve some semblance of accuracy, we
1383     // make it to the power of the elapsed number of frames. This is not fully accurate,
1384     // as this is applying the elapsed time between this frame and the previous time to
1385     // modify the velocity for the next frame. My mind goes blank when I try to figure out
1386     // a way to fix this (given that we don't want to change the velocity on the first frame),
1387     // and as it seems to work great as-is, I'm just leaving it.
1388     var decay = touch.decelerationRate;
1389     touch.decelerationVelocity.y *= Math.pow(decay, (t / 10));
1390     touch.decelerationVelocity.x *= Math.pow(decay, (t / 10));
1391     
1392     // We have a bouncyBounce method that adjusts the velocity for bounce. That is, if it is
1393     // out of range and still going, it will slow it down. This step is decelerationFromEdge.
1394     // If it is not moving (or has come to a stop from decelerating), but is still out of range, 
1395     // it will start it moving back into range (accelerationToEdge)
1396     // we supply de and ac as these properties.
1397     // The .3 artificially increases the acceleration by .3; this is actually to make the final
1398     // stop a bit more abrupt.
1399     touch.decelerationVelocity.x = this.bouncyBounce(touch.decelerationVelocity.x, newX, minOffsetX, maxOffsetX, de, ac, 0.3);
1400     touch.decelerationVelocity.y = this.bouncyBounce(touch.decelerationVelocity.y, newY, minOffsetY, maxOffsetY, de, ac, 0.3);
1401  
1402     // if we ain't got no velocity... then we must be finished, as there is no where else to go.
1403     // to determine our velocity, we take the absolue value, and use that; if it is less than .01, we
1404     // must be done. Note that we check scale's most recent velocity, calculated above using bouncyBounce,
1405     // as well.
1406     var absXVelocity = Math.abs(touch.decelerationVelocity.x);
1407     var absYVelocity = Math.abs(touch.decelerationVelocity.y);
1408     if (absYVelocity < 0.01 && absXVelocity < 0.01 && Math.abs(sv) < 0.01) {
1409       // we can reset the timeout, as it will no longer be required, and we don't want to re-cancel it later.
1410       touch.timeout = null;
1411       this.touch = null;
1412       
1413       // we aren't in a run loop right now (see below, where we trigger the timer)
1414       // so, we must start one.
1415       SC.RunLoop.begin();
1416       
1417       // trigger scroll end
1418       this._touchScrollDidEnd();
1419       
1420       // set the scale, vertical, and horizontal offsets to what they technically already are,
1421       // but don't know they are yet. This will finally update things like, say, the clipping frame.
1422       this.beginPropertyChanges();
1423       this.set("scale", this._scale);
1424       this.set("verticalScrollOffset", this._scroll_verticalScrollOffset);
1425       this.set("horizontalScrollOffset", this._scroll_horizontalScrollOffset);
1426       this.endPropertyChanges();
1427       
1428       // and now we're done, so just end the run loop and return.
1429       SC.RunLoop.end();
1430       return;
1431     }
1432     
1433     // We now set up the next round. We are doing this as raw as we possibly can, not touching the
1434     // run loop at all. This speeds up performance drastically--keep in mind, we're on comparatively
1435     // slow devices, here. So, we'll just make a closure, saving "this" into "self" and calling
1436     // 10ms later (or however long it takes). Note also that we save both the last event time
1437     // (so we may calculate elapsed time) and the timeout we are creating, so we may cancel it in future.
1438     var self = this;
1439     touch.lastEventTime = Date.now();
1440     this.touch.timeout = setTimeout(function(){
1441       self.decelerateAnimation();
1442     }, 10);
1443   },
