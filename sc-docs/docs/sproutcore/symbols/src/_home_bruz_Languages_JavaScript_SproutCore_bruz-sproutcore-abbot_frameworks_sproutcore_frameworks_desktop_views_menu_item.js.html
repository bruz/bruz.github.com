 78 
 79     @property {SC.MenuView}

 81   subMenu: function() {
 82     var content = this.get('content'), menuItems, parentMenu;
 83 
 84     if (!content) return null;
 85 
 86     parentMenu = this.get('parentMenu');
 87     menuItems = content.get(parentMenu.itemSubMenuKey );
 88     if (menuItems) {
 89       if (SC.kindOf(menuItems, SC.MenuPane)) {
 90         menuItems.set('isModal', NO);
 91         menuItems.set('isSubMenu', YES);
 92         menuItems.set('parentMenu', parentMenu);
 93         return menuItems;
 94       } else {
 95         return SC.MenuPane.create({
 96           layout: { width: 200 },
 97           items: menuItems,
 98           isModal: NO,
 99           isSubMenu: YES,
100           parentMenu: parentMenu,
101           controlSize: parentMenu.get('controlSize')
102         });
103       }
104     }
105 
106     return null;
107   }.property('content').cacheable(),

126   /**
127     Fills the passed html-array with strings that can be joined to form the
128     innerHTML of the receiver element.  Also populates an array of classNames
129     to set on the outer element.
130 
131     @param {SC.RenderContext} context
132     @param {Boolean} firstTime
133     @returns {void}
134   */
135   render: function(context, firstTime) {
136     var content = this.get('content'),
137         key, val,
138         menu = this.get('parentMenu'),
139         itemWidth = this.get('itemWidth') || menu.layout.width,
140         itemHeight = this.get('itemHeight') || SC.DEFAULT_MENU_ITEM_HEIGHT ;
141     this.set('itemWidth',itemWidth);
142     this.set('itemHeight',itemHeight);
143 
144     context = context.begin('a').addClass('menu-item');
145 
146     if (content.get(menu.itemSeparatorKey)) {
147       context.push('<span class="separator"></span>');
148       context.addClass('disabled');
149     } else {
150       val = content.get(menu.itemIconKey);
151       if (val) {
152         this.renderImage(context, val);
153         context.addClass('has-icon');
154       }
155 
156       val = this.get('title');
157       if (SC.typeOf(val) !== SC.T_STRING) val = val.toString();
158       this.renderLabel(context, val);
159 
160       if (this.getContentProperty('itemCheckboxKey')) {
161         context.push('<div class="checkbox"></div>');
162       }
163 
164       if (this.get('hasSubMenu')) {
165         this.renderBranch(context);
166       }
167 
168       val = this.getContentProperty('itemShortCutKey');
169       if (val) {
170         this.renderShortcut(context, val);
171       }
172     }
173 
174     context = context.end();
175   },

177   /**
178    Generates the image used to represent the image icon. override this to
179    return your own custom HTML
180 
181    @param {SC.RenderContext} context the render context
182    @param {String} the source path of the image
183    @returns {void}
184   */
185   renderImage: function(context, image) {
186     // get a class name and url to include if relevant
187 
188     var url, className ;
189     if (image && SC.ImageView.valueIsUrl(image)) {
190       url = image ;
191       className = '' ;
192     } else {
193       className = image ;
194       url = SC.BLANK_IMAGE_URL;
195     }
196     // generate the img element...
197     context.begin('img').addClass('image').addClass(className).attr('src', url).end() ;
198   },

279   mouseUp: function(evt) {
280     // SproutCore's event system will deliver the mouseUp event to the view
281     // that got the mouseDown event, but for menus we want to track the mouse,
282     // so we'll do our own dispatching.
283     var targetMenuItem;
284 
285     targetMenuItem = this.getPath('parentMenu.rootMenu.targetMenuItem');
286 
287     if (targetMenuItem) targetMenuItem.performAction();
288     return YES ;
289   },

291   /** @private
292     Called on mouse down to send the action to the target.
293 
294     This method will start flashing the menu item to indicate to the user that
295     their selection has been received, unless disableMenuFlash has been set to
296     YES on the menu item.
297 
298     @returns {Boolean}
299   */
300   performAction: function() {
301     // Disabled menu items and menu items with submenus should not have
302     // actions.
303     if (!this.get('isEnabled')||this.get('hasSubMenu')) return NO;
304 
305     var disableFlash = this.getContentProperty('itemDisableMenuFlashKey'),
306         menu;
307 
308     if (disableFlash) {
309       // Menu flashing has been disabled for this menu item, so perform
310       // the action immediately.
311       this.sendAction();
312     } else {
313       // Flash the highlight of the menu item to indicate selection,
314       // then actually send the action once its done.
315       this._flashCounter = 0;
316 
317       // Set a flag on the root menu to indicate that we are in a
318       // flashing state. In the flashing state, no other menu items
319       // should become selected.
320       menu = this.getPath('parentMenu.rootMenu');
321       menu._isFlashing = YES;
322       this.invokeLater(this.flashHighlight, 25);
323       this.invokeLater(this.sendAction, 150);
324     }
325 
326     return YES;
327   },
328 
329   /**
330     Actually sends the action of the menu item to the target.
331     @private
332   */
333   sendAction: function() {
334     var action = this.getContentProperty('itemActionKey'),
335         target = this.getContentProperty('itemTargetKey'),
336         rootMenu = this.getPath('parentMenu.rootMenu'),
337         responder;
338 
339     // Close the menu
340     this.getPath('parentMenu.rootMenu').remove();
341     // We're no longer flashing
342     rootMenu._isFlashing = NO;
343 
344     action = (action === undefined) ? rootMenu.get('action') : action;
345     target = (target === undefined) ? rootMenu.get('target') : target;
346 
347     // Notify the root menu pane that the selection has changed
348     rootMenu.set('selectedItem', this.get('content'));
349 
350     // Legacy support for actions that are functions
351     if (SC.typeOf(action) === SC.T_FUNCTION) {
352       action.apply(target, [rootMenu]);
353       //@if(debug)
354       SC.Logger.warn('Support for menu item action functions has been deprecated. Please use target and action.');
355       //@endif
356     } else {
357       responder = this.getPath('pane.rootResponder') || SC.RootResponder.responder;
358 
359       if (responder) {
360         // Send the action down the responder chain
361         responder.sendAction(action, target, this);
362       }
363     }
364

367   /**
368     Toggles the focus class name on the menu item layer to quickly flash the
369     highlight. This indicates to the user that a selection has been made.
370 
371     This is initially called by performAction(). flashHighlight then keeps
372     track of how many flashes have occurred, and calls itself until a maximum
373     has been reached.
374 
375     @private
376   */
377   flashHighlight: function() {
378     var flashCounter = this._flashCounter, layer = this.$();
379     if (flashCounter % 2 === 0) {
380       layer.addClass('focus');
381     } else {
382       layer.removeClass('focus');
383     }
384 
385     if (flashCounter <= 2) {
386       this.invokeLater(this.flashHighlight, 50);
387       this._flashCounter++;
388     }
389   },

396   /** @private */
397   mouseEntered: function(evt) {
398     var menu = this.get('parentMenu'),
399         rootMenu = menu.get('rootMenu');
400 
401     // Ignore mouse entering if we're in the middle of
402     // a menu flash.
403     if (rootMenu._isFlashing) return;
404 
405     menu.set('mouseHasEntered', YES);
406     this.set('mouseHasEntered', YES);
407     menu.set('currentMenuItem', this);
408 
409     // Become first responder to show highlight
410     if (this.get('isEnabled')) {
411       this.becomeFirstResponder();
412     }
413 
414     if(this.get('hasSubMenu')) {
415       this._subMenuTimer = this.invokeLater(this.showSubMenu,100) ;
416     }
417 	  return YES ;
418   },
419 
420   /** @private
421     Set the focus based on whether the current menu item is selected or not.
422 
423     @returns Boolean
424   */
425   mouseExited: function(evt) {
426     var parentMenu, timer;
427 
428     // If we have a submenu, we need to give the user's mouse time to get
429     // to the new menu before we remove highlight.
430     if (this.get('hasSubMenu')) {
431       // If they are exiting the view before we opened the submenu,
432       // make sure we don't open it once they've left.
433       timer = this._subMenuTimer;
434       if (timer) {
435         timer.invalidate();
436       } else {
437         this.invokeLater(this.checkMouseLocation, 100);
438       }
439     } else {
440       parentMenu = this.get('parentMenu');
441 
442       if (parentMenu.get('currentMenuItem') === this) {
443         parentMenu.set('currentMenuItem', null);
444       }
445     }
446 
447     return YES ;
448   },

467   checkMouseLocation: function() {
468     var subMenu = this.get('subMenu'), parentMenu = this.get('parentMenu'),
469         currentMenuItem, previousMenuItem;
470 
471     if (!subMenu.get('mouseHasEntered')) {
472       currentMenuItem = parentMenu.get('currentMenuItem');
473       if (currentMenuItem === this || currentMenuItem === null) {
474         previousMenuItem = parentMenu.get('previousMenuItem');
475 
476         if (previousMenuItem) {
477           previousMenuItem.resignFirstResponder();
478         }
479         this.resignFirstResponder();
480         subMenu.remove();
481       }
482     }

596   /**
597     @private
598 
599     Add an observer to ensure that we invalidate our cached properties
600     whenever the content object’s associated property changes.
601   */
602   contentDidChange: function() {
603     var content    = this.get('content'),
604         oldContent = this._content;
605 
606     if (content === oldContent) return ;
607 
608     var f = this.contentPropertyDidChange;
609     // remove an observer from the old content if necessary
610     if (oldContent  &&  oldContent.removeObserver) oldContent.removeObserver('*', this, f) ;
611 
612     // add observer to new content if necessary.
613     this._content = content ;
614     if (content  &&  content.addObserver) content.addObserver('*', this, f) ;
615 
616     // notify that value did change.
617     this.contentPropertyDidChange(content, '*') ;
618   }.observes('content'),
619 
620 
621   /**
622     @private
623 
624     Invalidate our cached property whenever the content object’s associated
625     property changes.
626   */
627   contentPropertyDidChange: function(target, key) {
628     // If the key that changed in the content is one of the fields for which
629     // we (potentially) cache a value, update our cache.
630     var menu = this.get('parentMenu') ;
631     if (!menu) return ;
632 
633     var mapping           = SC.MenuItemView._contentPropertyToMenuItemPropertyMapping,
634         contentProperties = SC.keys(mapping),
635         i, len, contentProperty, menuItemProperty ;
636 
637 
638     // Are we invalidating all keys?
639     if (key === '*') {
640       for (i = 0, len = contentProperties.length;  i < len;  ++i) {
641         contentProperty  = contentProperties[i] ;
642         menuItemProperty = mapping[contentProperty] ;
643         this.notifyPropertyChange(menuItemProperty) ;
644       }
645     }
646     else {
647       for (i = 0, len = contentProperties.length;  i < len;  ++i) {
648         contentProperty  = contentProperties[i] ;
649         if (menu.get(contentProperty) === key) {
650           menuItemProperty = mapping[contentProperty] ;
651           this.notifyPropertyChange(menuItemProperty) ;
652 
653           // Note:  We won't break here in case the menu is set up to map
654           //        multiple properties to the same content key.
655         }
656       }
657     }
658   }

667 /** @private
668   A mapping of the "content property key" keys to the properties we use to
669   wrap them.  This hash is used in 'contentPropertyDidChange' to ensure that
670   when the content changes a property that is locally cached inside the menu
671   item, the cache is properly invalidated.
672 
673   Implementor note:  If you add such a cached property, you must add it to
674                      this mapping.
675 */
676 SC.MenuItemView._contentPropertyToMenuItemPropertyMapping = {
677   itemTitleKey:     'title',
678   itemIsEnabledKey: 'isEnabled',
679   itemSeparatorKey: 'isSeparator',
680   itemSubMenuKey:   'subMenu'
