  5 require('views/list') ;
  6 
  7 /** @class
  8 
  9   A grid view renders a collection of items in a grid of rows and columns.
 10 
 11   @extends SC.CollectionView
 12   @author    Charles Jolley  
 13   @version 1.0
 14 */
 15 SC.GridView = SC.ListView.extend(
 16 /** @scope SC.GridView.prototype */ {
 17     classNames: ['sc-grid-view'],
 18   
 19   layout: { left:0, right:0, top:0, bottom:0 },
 20 
 21   /** 
 22     The common row height for grid items.
 23     
 24     The value should be an integer expressed in pixels.
 25   */
 26   rowHeight: 48,
 27   
 28   /**
 29     The minimum column width for grid items.  Items will actually
 30     be laid out as needed to completely fill the space, but the minimum
 31     width of each item will be this value.
 32   */
 33   columnWidth: 64,
 34 
 35   /**
 36     The default example item view will render text-based items.
 37     
 38     You can override this as you wish.
 39   */
 40   exampleView: SC.LabelView,
 41   
 42   insertionOrientation: SC.HORIZONTAL_ORIENTATION,
 43   
 44   /** @private */
 45   itemsPerRow: function() {
 46     var f = this.get('frame'),
 47         columnWidth = this.get('columnWidth') || 0 ;
 48 
 49     return (columnWidth <= 0) ? 1 : Math.floor(f.width / columnWidth) ;
 50   }.property('clippingFrame', 'columnWidth').cacheable(),
 51   
 52   /** @private
 53     Find the contentIndexes to display in the passed rect. Note that we 
 54     ignore the width of the rect passed since we need to have a single
 55     contiguous range.
 56   */
 57   contentIndexesInRect: function(rect) {
 58     var rowHeight = this.get('rowHeight') || 48 ,
 59         itemsPerRow = this.get('itemsPerRow'),
 60         min = Math.floor(SC.minY(rect) / rowHeight) * itemsPerRow,
 61         max = Math.ceil(SC.maxY(rect) / rowHeight) * itemsPerRow ;
 62     return SC.IndexSet.create(min, max-min);
 63   },
 64   
 65   /** @private */
 66   layoutForContentIndex: function(contentIndex) {
 67     var rowHeight = this.get('rowHeight') || 48,
 68         frameWidth = this.get('clippingFrame').width,
 69         itemsPerRow = this.get('itemsPerRow'),
 70         columnWidth = Math.floor(frameWidth/itemsPerRow),
 71         row = Math.floor(contentIndex / itemsPerRow),
 72         col = contentIndex - (itemsPerRow*row) ;
 73     return { 
 74       left: col * columnWidth,
 75       top: row * rowHeight,
 76       height: rowHeight,
 77       width: columnWidth
 78     };
 79   },
 80   
 81   /** @private
 82     Overrides default CollectionView method to compute the minimim height
 83     of the list view.
 84   */
 85   computeLayout: function() {
 86     var content = this.get('content'),
 87         count = (content) ? content.get('length') : 0,
 88         rowHeight = this.get('rowHeight') || 48,
 89         itemsPerRow = this.get('itemsPerRow'),
 90         rows = Math.ceil(count / itemsPerRow) ;
 91   
 92     // use this cached layout hash to avoid allocing memory...
 93     var ret = this._cachedLayoutHash ;
 94     if (!ret) ret = this._cachedLayoutHash = {};
 95     
 96     // set minHeight
 97     ret.minHeight = rows * rowHeight ;
 98     this.calculatedHeight = ret.minHeight;
 99     return ret; 
100   },
101   
102   insertionPointClass: SC.View.extend({
103     classNames: ['grid-insertion-point'],
104     
105     render: function(context, firstTime) {
106       if (firstTime) context.push('<span class="anchor"></span>') ;
107     }
108 
109   }),
110   
111   showInsertionPoint: function(itemView, dropOperation) {
112     if (!itemView) return ;
113     
114     // if drop on, then just add a class...
115     if (dropOperation === SC.DROP_ON) {
116       if (itemView !== this._dropOnInsertionPoint) {
117         this.hideInsertionPoint() ;
118         //itemView.addClassName('drop-target') ;
119         this._dropOnInsertionPoint = itemView ;
120       }
121       
122     } else {
123       
124       if (this._dropOnInsertionPoint) {
125         //this._dropOnInsertionPoint.removeClassName('drop-target') ;
126         this._dropOnInsertionPoint = null ;
127       }
128     
129       if (!this._insertionPointView) {
130         this._insertionPointView = this.insertionPointClass.create() ;
131       }
132     
133       var insertionPoint = this._insertionPointView ;
134       var itemViewFrame = itemView.get('frame') ;
135       var f = { height: itemViewFrame.height - 6, 
136             x: itemViewFrame.x, 
137             y: itemViewFrame.y + 6, 
138             width: 0 
139           };
140 
141       if (!SC.rectsEqual(insertionPoint.get('frame'), f)) {
142         insertionPoint.set('frame', f) ;
143       }
144 
145       if (insertionPoint.parentNode !== itemView.parentNode) {
146         itemView.parentNode.appendChild(insertionPoint) ;
147       }
148     }
149     
150   },
151     
152   hideInsertionPoint: function() {
153     var insertionPoint = this._insertionPointView ;
154     if (insertionPoint) insertionPoint.removeFromParent() ;
155 
156     if (this._dropOnInsertionPoint) {
157       //this._dropOnInsertionPoint.removeClassName('drop-target') ;
158       this._dropOnInsertionPoint = null ;
159     }
160   },
161   
162   // // We can do this much faster programatically using the rowHeight
163   insertionIndexForLocation: function(loc, dropOperation) {  
164     var f = this.get('frame'),
165         sf = this.get('clippingFrame'),
166         itemsPerRow = this.get('itemsPerRow'),
167         columnWidth = Math.floor(f.width / itemsPerRow),
168         row = Math.floor((loc.y - f.y - sf.y) / this.get('rowHeight')) ;
169 
170     var retOp = SC.DROP_BEFORE,
171         offset = (loc.x - f.x - sf.x),
172         col = Math.floor(offset / columnWidth),
173         percentage = (offset / columnWidth) - col ;
174     
175     // if the dropOperation is SC.DROP_ON and we are in the center 60%
176     // then return the current item.
177     if (dropOperation === SC.DROP_ON) {
178       if (percentage > 0.80) col++ ;
179       if ((percentage >= 0.20) && (percentage <= 0.80)) {
180         retOp = SC.DROP_ON;
181       }
182     } else {
183       if (percentage > 0.45) col++ ;
184     }
185     
186     // convert to index
187     var ret= (row*itemsPerRow) + col ;
188     return [ret, retOp] ;
189   },
190 
191   /** @private
192     If the size of the clipping frame changes, all of the item views
193     on screen are potentially in the wrong position.  Update all of their
194     layouts if different.
195   */
196   _gv_clippingFrameDidChange: function() {
197     var nowShowing = this.get('nowShowing'), itemView, idx, len;
198     this.notifyPropertyChange('itemsPerRow');
199 
200     len = nowShowing.get('length');
201 
202     for (idx=0; idx < len; idx++) {
203       itemView = this.itemViewForContentIndex(idx);
204       itemView.adjust(this.layoutForContentIndex(idx));
205     }
206   }.observes('clippingFrame')
207 }) ;
