  8 // Constants
  9 SC.WELL_CONTAINER_PADDING=15;
 10 
 11 /** @class
 12 
 13   A WellView is a ContainerView with a border. It's useful when you want to
 14   group a set of views. It allows you to easily switch its contents too.
 15   
 16   It has a default contentLayout that will replace the layout of the contentView.
 17   
 18   @extends SC.ContainerView
 19   @since SproutCore 1.0
 20   @test in progress
 21 */
 22 SC.WellView = SC.ContainerView.extend(
 23 /** @scope SC.WellView.prototype */ {
 24   
 25   classNames: 'sc-well-view',
 26   
 27   /**
 28     Layout for the content of the container view.
 29     @property {Object}
 30   */
 31   contentLayout: {
 32     top:SC.WELL_CONTAINER_PADDING, bottom:SC.WELL_CONTAINER_PADDING,
 33     left:SC.WELL_CONTAINER_PADDING, right:SC.WELL_CONTAINER_PADDING},
 34   
 35   
 36   /**
 37      Overrides createChildViews and replaces the layout of the contentView
 38      with the one in contentLayout.
 39    */
 40   
 41   createChildViews: function() {
 42     // if contentView is defined, then create the content
 43     var view = this.get('contentView') ;
 44     if (view) {
 45       view = this.contentView = this.createChildView(view) ;
 46       view.set('layout', this.contentLayout);
 47       this.childViews = [view] ;
 48     } 
 49   },
 50   
 51   /**
 52      The render method for the WellView simply add the html necessary for
 53      the border.
 54      
 55    */
 56   
 57   render: function(context, firstTime) {
 58     if(firstTime){
 59      context.push("<div class='top-left-edge'></div>",
 60        "<div class='top-edge'></div>",
 61        "<div class='top-right-edge'></div>",
 62        "<div class='right-edge'></div>",
 63        "<div class='bottom-right-edge'></div>",
 64        "<div class='bottom-edge'></div>",
 65        "<div class='bottom-left-edge'></div>",
 66        "<div class='left-edge'></div>",
 67        "<div class='content-background'></div>");
 68      }    
 69      sc_super();
 70   },
 71   
 72   /**
 73      Invoked whenever the content property changes.  This method will simply
 74      call replaceContent and set the contentLayout in the new contentView.
 75      
 76      Override replaceContent to change how the view is
 77      swapped out.
 78    */
 79   contentViewDidChange: function() {
 80     var view = this.get('contentView');
 81     view.set('layout', this.contentLayout);
 82     this.replaceContent(view);
 83   }.observes('contentView')
 84   
 85 }) ;
