 10   A response represents a single response from a server request.  An instance
 11   of this class is returned whenever you call SC.Request.send().

 18 SC.Response = SC.Object.extend(
 19 /** @scope SC.Response.prototype */ {
 20   
 21   /**
 22     Becomes true if there was a failure.  Makes this into an error object.
 23     
 24     @property {Boolean}
 25   */
 26   isError: NO,
 27   
 28   /**
 29     Always the current response
 30     
 31     @property {SC.Response}
 32   */
 33   errorValue: function() {
 34     return this;
 35   }.property().cacheable(),
 36   
 37   /**
 38     The error object generated when this becomes an error
 39     
 40     @property {SC.Error}
 41   */
 42   errorObject: null,
 43   
 44   /** 
 45     Request used to generate this response.  This is a copy of the original
 46     request object as you may have modified the original request object since
 47     then.
 48    
 49     To retrieve the original request object use originalRequest.
 50     
 51     @property {SC.Request}
 52   */
 53   request: null,
 54   
 55   /**
 56     The request object that originated this request series.  Mostly this is
 57     useful if you are looking for a reference to the original request.  To
 58     inspect actual properties you should use request instead.
 59     
 60     @property {SC.Request}
 61   */
 62   originalRequest: function() {
 63     var ret = this.get('request');
 64     while (ret.get('source')) ret = ret.get('source');

 90     @property {Boolean}

 95 
 96   /**
 97     If set, then will attempt to automatically parse response as XML
 98     regarldess of headers.
 99     
100     @property {Boolean}
101   */
102   isXML: function() {
103     return this.getPath('request.isXML') || NO ;
104   }.property('request').cacheable(),

106   /**

108     
109     @property {Hash}

113   }.property('request').cacheable(),
114   
115   /**
116     The response status code.
117     
118     @property {Number}

120   status: -100, // READY
121

123     Headers from the response.  Computed on-demand
124     
125     @property {Hash}
126   */
127   headers: null,
128   
129   /**
130     The raw body retrieved from the transport.  This may be further parsed
131     by the body property, which is probably what you want.
132     
133     @property {String}
134   */
135   encodedBody: null,
136   
137   /**
138     Response body. If isJSON was set, will be parsed automatically.
139     
140     @response {Hash|String|SC.Error} the response body or the parsed JSON.
141       Returns a SC.Error instance if there is a JSON parsing error.
142   */
143   body: function() {
144     // TODO: support XML
145     var ret = this.get('encodedBody');
146     if (ret && this.get('isJSON')) {
147       try {
148         ret = SC.json.decode(ret);
149       } catch(e) {
150         return SC.Error.create({
151           message: e.name + ': ' + e.message,
152           label: 'Response',
153           errorValue: this });
154       }
155     }
156     return ret;
157   }.property('encodedBody').cacheable(),

159   /**

161     @deprecated
162   
163     Alias for body.  Provides compatibility with older code.
164     
165     @property {Hash|String}
166   */
167   response: function() {
168     return this.get('body');
169   }.property('body').cacheable(),
170   
171   /**
172     Set to YES if response is cancelled
173   */
174   isCancelled: NO,
175   
176   /**
177     Set to YES if the request timed out.  Set to NO if the request has
178     completed before the timeout value.  Set to null if the timeout timer is
179     still ticking.
180   */
181   timedOut: null,
182   
183   /**
184     The timer tracking the timeout
185   */
186   timeoutTimer: null,

192   /**
193     Called by the request manager when its time to actually run.  This will
194     invoke any callbacks on the source request then invoke transport() to 
195     begin the actual request.
196   */
197   fire: function() {
198     var req = this.get('request'),
199         source = req ? req.get('source') : null;
200     
201     
202     // first give the source a chance to fixup the request and response
203     // then freeze req so no more changes can happen.
204     if (source && source.willSend) source.willSend(req, this);
205     req.freeze();
206 
207     // if the source did not cancel the request, then invoke the transport
208     // to actually trigger the request.  This might receive a response 
209     // immediately if it is synchronous.
210     if (!this.get('isCancelled')) this.invokeTransport();
211 
212 
213     // If the request specified a timeout value, then set a timer for it now.
214     var timeout = req.get('timeout');
215     if (timeout) {
216       var timer = SC.Timer.schedule({
217         target:   this, 
218         action:   'timeoutReached', 
219         interval: timeout,
220         repeats:  NO
221       });
222       this.set('timeoutTimer', timer);
223     }
224 
225 
226     // if the transport did not cancel the request for some reason, let the
227     // source know that the request was sent
228     if (!this.get('isCancelled') && source && source.didSend) {
229       source.didSend(req, this);
230     }
231   },
232 
233   invokeTransport: function() {
234     this.receive(function(proceed) { this.set('status', 200); }, this);
235   },
236   
237   /**
238     Invoked by the transport when it receives a response.  The passed-in
239     callback will be invoked to actually process the response.  If cancelled
240     we will pass NO.  You should clean up instead.
241     
242     Invokes callbacks on the source request also.
243     
244     @param {Function} callback the function to receive
245     @param {Object} context context to execute the callback in
246     @returns {SC.Response} receiver
247   */
248   receive: function(callback, context) {
249     // If we timed out, we should ignore this response.
250     if (!this.get('timedOut')) {
251       // If we had a timeout timer scheduled, invalidate it now.
252       var timer = this.get('timeoutTimer');
253       if (timer) timer.invalidate();
254       this.set('timedOut', NO);
255     
256       var req = this.get('request');
257       var source = req ? req.get('source') : null;
258     
259       SC.RunLoop.begin();
260     
261       // invoke the source, giving a chance to fixup the reponse or (more 
262       // likely) cancel the request.
263       if (source && source.willReceive) source.willReceive(req, this);
264     
265       // invoke the callback.  note if the response was cancelled or not
266       callback.call(context, !this.get('isCancelled'));
267     
268       // if we weren't cancelled, then give the source first crack at handling
269       // the response.  if the source doesn't want listeners to be notified,
270       // it will cancel the response.
271       if (!this.get('isCancelled') && source && source.didReceive) {
272         source.didReceive(req, this);
273       }
274 
275       // notify listeners if we weren't cancelled.
276       if (!this.get('isCancelled')) this.notify();
277 
278       SC.RunLoop.end();
279     }
280     
281     // no matter what, remove from inflight queue
282     SC.Request.manager.transportDidClose(this) ;
283     return this;
284   },
285   
286   /**
287     Default method just closes the connection.  It will also mark the request
288     as cancelled, which will not call any listeners.
289   */
290   cancel: function() {
291     if (!this.get('isCancelled')) {
292       this.set('isCancelled', YES) ;
293       this.cancelTransport() ;
294       SC.Request.manager.transportDidClose(this) ;
295     }
296   },
297   
298   /**
299     Default method just closes the connection.
300   */
301   timeoutReached: function() {
302     // If we already received a response yet the timer still fired for some
303     // reason, do nothing.
304     if (this.get('timedOut') === null) {
305       this.set('timedOut', YES);
306       this.cancelTransport();
307       SC.Request.manager.transportDidClose(this);
308       
309       // Set our value to an error.
310       var error = SC.$error("HTTP Request timed out", "Request", 408) ;
311       error.set("errorValue", this) ;
312       this.set('isError', YES);
313       this.set('errorObject', error);
314       
315       // Invoke the didTimeout callback.
316       var req = this.get('request');
317       var source = req ? req.get('source') : null;
318       if (!this.get('isCancelled') && source && source.didTimeout) {
319         source.didTimeout(req, this);
320       }
321     }
322   },
323   
324   /**
325     Override with concrete implementation to actually cancel the transport.
326   */
327   cancelTransport: function() {},
328   
329   
330   /** @private
331     Will notify each listener.
332   */
333   _notifyListener: function(listeners, status) {
334     var info = listeners[status], params, target, action;
335     if (!info) return NO ;
336     
337     params = (info.params || []).copy();
338     params.unshift(this);
339     
340     target = info.target;
341     action = info.action;
342     if (SC.typeOf(action) === SC.T_STRING) action = target[action];
343     
344     return action.apply(target, params);
345   },
346   
347   /**
348     Notifies any saved target/action.  Call whenever you cancel, or end.
349     
350     @returns {SC.Response} receiver
351   */
352   notify: function() {
353     var listeners = this.get('listeners'), 
354         status    = this.get('status'),
355         baseStat  = Math.floor(status / 100) * 100,
356         handled   = NO ;
357         
358     if (!listeners) return this ; // nothing to do
359     
360     handled = this._notifyListener(listeners, status);
361     if (!handled) handled = this._notifyListener(listeners, baseStat);
362     if (!handled) handled = this._notifyListener(listeners, 0);
363     
364     return this ;
365   },
366   
367   /**
368     String representation of the response object
369   */
370   toString: function() {
371     var ret = sc_super();
372     return "%@<%@ %@, status=%@".fmt(ret, this.get('type'), this.get('address'), this.get('status'));
373   }

377 /**
378   Concrete implementation of SC.Response that implements support for using 
379   XHR requests.
380   
381   @extends SC.Response
382   @since SproutCore 1.0
383 */
384 SC.XHRResponse = SC.Response.extend({
385 
386   /**
387     Implement transport-specific support for fetching all headers
388   */
389   headers: function() {
390     var xhr = this.get('rawRequest'),
391         str = xhr ? xhr.getAllResponseHeaders() : null,
392         ret = {};
393         
394     if (!str) return ret;
395     
396     str.split("\n").forEach(function(header) {
397       var idx = header.indexOf(':'),
398           key, value;
399       if (idx>=0) {
400         key = header.slice(0,idx);
401         value = header.slice(idx+1).trim();
402         ret[key] = value ;
403       }
404     }, this);
405     
406     return ret ;
407   }.property('status').cacheable(),
408   
409   // returns a header value if found...
410   header: function(key) {
411     var xhr = this.get('rawRequest');
412     return xhr ? xhr.getResponseHeader(key) : null;    
413   },
414   
415   /**
416     Implement transport-specific support for fetching tasks
417   */
418   encodedBody: function() {
419     var xhr = this.get('rawRequest'), ret ;
420     if (!xhr) ret = null;
421     else if (this.get('isXML')) ret = xhr.responseXML;
422     else ret = xhr.responseText;
423     return ret ;
424   }.property('status').cacheable(),
425   
426 
427   cancelTransport: function() {
428     var rawRequest = this.get('rawRequest');
429     if (rawRequest) rawRequest.abort();
430     this.set('rawRequest', null);
431   },
432   
433   invokeTransport: function() {
434     
435     var rawRequest, transport, handleReadyStateChange, async, headers;
436     
437     // Get an XHR object
438     function tryThese() {
439       for (var i=0; i < arguments.length; i++) {
440         try {
441           var item = arguments[i]() ;
442           return item ;
443         } catch (e) {}
444       }
445       return NO;
446     }
447     
448     rawRequest = tryThese(
449       function() { return new XMLHttpRequest(); },
450       function() { return new ActiveXObject('Msxml2.XMLHTTP'); },
451       function() { return new ActiveXObject('Microsoft.XMLHTTP'); }
452     );
453     
454     // save it 
455     this.set('rawRequest', rawRequest);
456     
457     // configure async callback - differs per browser...
458     async = !!this.getPath('request.isAsynchronous') ;
459     if (async) {
460       if (!SC.browser.msie) {
461         SC.Event.add(rawRequest, 'readystatechange', this, 
462                      this.finishRequest, rawRequest) ;
463       } else {
464         transport=this;
465         handleReadyStateChange = function() {
466           if (!transport) return null ;
467           var ret = transport.finishRequest();
468           if (ret) transport = null ; // cleanup memory
469           return ret ;
470         };
471         rawRequest.onreadystatechange = handleReadyStateChange;
472       }
473     }
474     
475     // initiate request.  
476     rawRequest.open(this.get('type'), this.get('address'), async ) ;
477     
478     // headers need to be set *after* the open call.
479     headers = this.getPath('request.headers') ;
480     for (var headerKey in headers) {
481       rawRequest.setRequestHeader(headerKey, headers[headerKey]) ;
482     }
483 
484     // now send the actual request body - for sync requests browser will
485     // block here
486     rawRequest.send(this.getPath('request.encodedBody')) ;
487     if (!async) this.finishRequest() ; // not async
488     
489     return rawRequest ;
490   },
491   
492   /**  @private
493   
494     Called by the XHR when it responds with some final results.
495     
496     @param {XMLHttpRequest} rawRequest the actual request
497     @returns {SC.XHRRequestTransport} receiver
498   */
499   finishRequest: function(evt) {
500     var rawRequest = this.get('rawRequest'),
501         readyState = rawRequest.readyState,
502         error, status, msg;
503 
504     if (readyState === 4) {
505       this.receive(function(proceed) {
506 
507         if (!proceed) return ; // skip receiving...
508       
509         // collect the status and decide if we're in an error state or not
510         status = -1 ;
511         try {
512           status = rawRequest.status || 0;
513         } catch (e) {}
514 
515         // if there was an error - setup error and save it
516         if ((status < 200) || (status >= 300)) {
517           
518           try {
519             msg = rawRequest.statusText || '';
520           } catch(e2) {
521             msg = '';
522           }
523           
524           error = SC.$error(msg || "HTTP Request failed", "Request", status) ;
525           error.set("errorValue", this) ;
526           this.set('isError', YES);
527           this.set('errorObject', error);
528         }
529 
530         // set the status - this will trigger changes on relatedp properties
531         this.set('status', status);
532       
533       }, this);
534       // avoid memory leak in MSIE: clean up
535       rawRequest.onreadystatechange = function() {} ;
536       return YES;
537     }
538     return NO; 
539   }
