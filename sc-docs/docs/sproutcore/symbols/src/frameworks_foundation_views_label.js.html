 11 SC.ALIGN_LEFT = 'left';
 12 SC.ALIGN_RIGHT = 'right';
 13 SC.ALIGN_CENTER = 'center';
 14 
 15 SC.REGULAR_WEIGHT = 'normal';
 16 SC.BOLD_WEIGHT = 'bold';

 18 /**
 19   @class
 20   
 21   Displays a static string of text.
 22   
 23   You use a label view anytime you need to display a static string of text 
 24   or to display text that may need to be edited using only an inline control.
 25   
 26   @extends SC.View
 27   @extends SC.Control
 28   @extends SC.InlineEditorDelegate
 29   @since SproutCore 1.0
 30 */
 31 SC.LabelView = SC.View.extend(SC.Control,
 32 /** @scope SC.LabelView.prototype */ {
 33 
 34   classNames: ['sc-label-view'],
 35 
 36   /**
 37     Specify the font weight for this.  You may pass SC.REGULAR_WEIGHT, or SC.BOLD_WEIGHT.
 38   */
 39   fontWeight: SC.REGULAR_WEIGHT,
 40   
 41   /**
 42     If true, value will be escaped to avoid scripting attacks.
 43     
 44     This is a default value that can be overridden by the
 45     settings on the owner view.
 46   */
 47   escapeHTML: true,
 48   escapeHTMLBindingDefault: SC.Binding.oneWay().bool(),
 49 
 50   /**
 51     If true, then the value will be localized.
 52     
 53     This is a default default that can be overidden by the
 54     settings in the owner view.
 55   */
 56   localize: false,
 57   localizeBindingDefault: SC.Binding.oneWay().bool(),
 58   
 59   /**
 60     Set this to a validator or to a function and the value
 61     will be passed through it before being set.
 62     
 63     This is a default default that can be overidden by the
 64     settings in the owner view.
 65   */
 66   formatter: null,
 67 
 68   /** 
 69     The value of the label.
 70     
 71     You may also set the value using a content object and a contentValueKey.
 72     
 73     @field {String}
 74   */
 75   value: '',
 76   
 77   /**
 78     The hint to display if no value is set.  Should be used only if isEditable
 79     is set to YES.
 80   */
 81   hint: null,
 82 
 83   /**
 84     The exampleInlineTextFieldView property is by default a 
 85     SC.InlineTextFieldView but it can be set to a customized inline text field
 86     view.
 87   
 88     @property
 89     @type {SC.View}
 90     @default {SC.InlineTextFieldView}
 91   */
 92   exampleInlineTextFieldView: SC.InlineTextFieldView,
 93   
 94   /**
 95     An optional icon to display to the left of the label.  Set this value
 96     to either a CSS class name (for spriting) or an image URL.
 97   */
 98   icon: null,
 99   
100   /**
101     Set the alignment of the label view.
102   */
103   textAlign: SC.ALIGN_LEFT,
104   
105   /**
106     If you want the inline editor to be multiline set this property to YES.
107   */
108   isInlineEditorMultiline: NO,
109   
110   /**
111     [RO] The value that will actually be displayed.
112     
113     This property is dynamically computed by applying localization, 
114     string conversion and other normalization utilities.
115     
116     @field
117   */
118   displayValue: function() {
119     var value, formatter;
120     
121     value = this.get('value') ;
122     
123     // 1. apply the formatter
124     formatter = this.getDelegateProperty('formatter', this.displayDelegate) ;
125     if (formatter) {
126       var formattedValue = (SC.typeOf(formatter) === SC.T_FUNCTION) ? 
127           formatter(value, this) : formatter.fieldValueForObject(value, this) ;
128       if (!SC.none(formattedValue)) value = formattedValue ;
129     }
130     
131     // 2. If the returned value is an array, convert items to strings and 
132     // join with commas.
133     if (SC.typeOf(value) === SC.T_ARRAY) {
134       var ary = [];
135       for(var idx=0, idxLen = value.get('length'); idx< idxLen;idx++) {
136         var x = value.objectAt(idx) ;
137         if (!SC.none(x) && x.toString) x = x.toString() ;
138         ary.push(x) ;
139       }
140       value = ary.join(',') ;
141     }
142     
143     // 3. If value is not a string, convert to string. (handles 0)
144     if (!SC.none(value) && value.toString) value = value.toString() ;
145     
146     // 4. Localize
147     if (value && this.getDelegateProperty('localize', this.displayDelegate)) value = value.loc() ;
148 
149     // 5. escapeHTML if needed
150     if (this.get('escapeHTML')) value = SC.RenderContext.escapeHTML(value);
151     
152     return value ;
153   }.property('value', 'localize', 'formatter', 'escapeHTML').cacheable(),
154   
155   
156   /**
157     [RO] The hint value that will actually be displayed.
158     
159     This property is dynamically computed by applying localization 
160     and other normalization utilities.
161     
162   */
163   hintValue: function() {
164     var hintVal = this.get('hint');
165     if (this.get('escapeHTML')) hintVal = SC.RenderContext.escapeHTML(hintVal);
166     return hintVal ;
167   }.property('hint', 'escapeHTML').cacheable(),
168   
169   /**
170     Enables editing using the inline editor.
171   */
172   isEditable: NO,
173   isEditableBindingDefault: SC.Binding.bool(),
174 
175   /**
176     YES if currently editing label view.
177   */
178   isEditing: NO,
179   
180   /**
181     Validator to use during inline editing.
182     
183     If you have set isEditing to YES, then any validator you set on this
184     property will be used when the label view is put into edit mode.
185     
186     @type {SC.Validator}
187   */
188   validator: null,
189 
190   /**
191     Event dispatcher callback.
192     If isEditable is set to true, opens the inline text editor view.
193 
194     @param {DOMMouseEvent} evt DOM event
195     
196   */
197   doubleClick: function( evt ) { return this.beginEditing(); },
198   
199   
200   /**
201     Opens the inline text editor (closing it if it was already open for 
202     another view).
203     
204     @return {Boolean} YES if did begin editing
205   */
206   beginEditing: function() {
207     if (this.get('isEditing')) return YES ;
208     if (!this.get('isEditable')) return NO ;
209 
210     var el = this.$(),
211         value = this.get('value') || '',
212         f = SC.viewportOffset(el[0]),
213         frameTemp = this.convertFrameFromView(this.get('frame'), null) ;
214     f.width=frameTemp.width;
215     f.height=frameTemp.height;
216     
217     SC.InlineTextFieldView.beginEditing({
218       frame: f,
219       delegate: this,
220       exampleElement: el,
221       value: value, 
222       multiline: this.get('isInlineEditorMultiline'), 
223       isCollection: NO,
224       validator: this.get('validator'),
225       exampleInlineTextFieldView: this.get('exampleInlineTextFieldView')
226     });
227   },
228   
229   /**
230     Cancels the current inline editor and then exits editor. 
231     
232     @return {Boolean} NO if the editor could not exit.
233   */
234   discardEditing: function() {
235     if (!this.get('isEditing')) return YES ;
236     return SC.InlineTextFieldView.discardEditing() ;
237   },
238   
239   /**
240     Commits current inline editor and then exits editor.
241     
242     @return {Boolean} NO if the editor could not exit
243   */
244   commitEditing: function() {
245     if (!this.get('isEditing')) return YES ;
246     return SC.InlineTextFieldView.commitEditing() ;
247   },
248 
249   /** @private
250     Set editing to true so edits will no longer be allowed.
251   */
252   inlineEditorWillBeginEditing: function(inlineEditor) {
253     this.set('isEditing', YES);
254   },
255 
256   /** @private 
257     Hide the label view while the inline editor covers it.
258   */
259   inlineEditorDidBeginEditing: function(inlineEditor) {
260     var layer = this.$();
261     this._oldOpacity = layer.css('opacity') ;
262     layer.css('opacity', 0.0);
263   },
264   
265   /** @private
266     Delegate method defaults to the isEditable property
267   */
268   inlineEditorShouldBeginEditing: function(){
269     return this.get('isEditable');
270   },
271   
272   /** @private
273     Could check with a validator someday...
274   */
275   inlineEditorShouldEndEditing: function(inlineEditor, finalValue) {
276     return YES ;
277   },
278 
279   /** @private
280     Update the field value and make it visible again.
281   */
282   inlineEditorDidEndEditing: function(inlineEditor, finalValue) {
283     this.setIfChanged('value', finalValue) ;
284     this.$().css('opacity', this._oldOpacity);
285     this._oldOpacity = null ;
286     this.set('isEditing', NO) ;
287   },
288 
289   displayProperties: 'displayValue textAlign fontWeight icon'.w(),
290   
291   _TEMPORARY_CLASS_HASH: {},
292   
293   render: function(context, firstTime) {
294     var value = this.get('displayValue'),
295         icon = this.get('icon'),
296         hint = this.get('hintValue'),
297         classes, stylesHash, text,
298         iconChanged = false, textChanged = false;
299     
300     if (icon) {
301       var url = (icon.indexOf('/')>=0) ? icon : SC.BLANK_IMAGE_URL,
302           className = (url === icon) ? '' : icon ;
303       icon = '<img src="'+url+'" alt="" class="icon '+className+'" />';
304       if(icon!==this._iconCache) {
305         this._iconCache=icon;
306         iconChanged = true;
307       }
308     }
309     
310     if (hint && (!value || value === '')) {
311       text = '<span class="sc-hint">'+hint+'</span>';
312     }else{
313       text = value;
314     }
315     if(text!==this._textCache) {
316       this._textCache=text;
317       textChanged = true;
318     }
319         
320     if(firstTime || textChanged || iconChanged){
321       context.push(icon, text);
322     }
323     
324     // and setup alignment and font-weight on styles
325     stylesHash = { 
326       'text-align': this.get('textAlign'), 
327       'font-weight': this.get('fontWeight')
328     };
329            
330     // if we are editing, set the opacity to 0
331     if (this.get('isEditing')) stylesHash['opacity']=0;
332     context.addStyle(stylesHash);
333     
334     classes = this._TEMPORARY_CLASS_HASH;
335     classes.icon = !!this.get('icon');
336     context.setClass(classes);
337   }
338   
339 });
