  8 /**
  9   @class
 10   
 11   Implements some enhancements to the built-in Number object that makes it
 12   easier to handle rounding and display of numbers.
 13   
 14   @since SproutCore 1.0
 15   @author Colin Campbell
 16 */
 17 SC.Math = SC.Object.create({
 18   
 19   /**
 20     Checks to see if the number is near the supplied parameter to a certain lambda.
 21     
 22     @param {Number} n1 First number in comparison.
 23     @param {Number} n2 Number to compare against the first.
 24     @param {Number} lambda The closeness sufficient for a positive result. Default 0.00001
 25     @returns {Boolean}
 26   */
 27   near: function(n1, n2, lambda) {
 28     if (!lambda) lambda = 0.00001;
 29     return Math.abs(n1 - n2) <= lambda;
 30   },
 31   
 32   /**
 33     Rounds a number to a given decimal place. If a negative decimalPlace
 34     parameter is provided, the number will be rounded outward (ie. providing
 35     -3 will round to the thousands).
 36     
 37     Function is insufficient for high negative values of decimalPlace parameter.
 38     For example, (123456.789).round(-5) should evaluate to 100000 but instead
 39     evaluates to 99999.999... 
 40     
 41     @param {Number} n The number to round
 42     @param {Integer} decimalPlace
 43     @returns {Number}
 44   */
 45   round: function(n, decimalPlace) {
 46     if (!decimalPlace) decimalPlace = 0;
 47     var factor = Math.pow(10, decimalPlace);
 48     if (decimalPlace < 0) {
 49        // stop rounding errors from hurting the factor...
 50       var s = factor.toString();
 51       factor = s.substring(0, s.indexOf("1")+1);
 52     }
 53     n = n.valueOf();
 54     return Math.round(n * factor) / factor;
 55   }
 56   
 57 }) ;
