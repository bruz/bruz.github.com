  7 SC.BORDER_BEZEL  = 'sc-bezel-border';
  8 SC.BORDER_BLACK  = 'sc-black-border';
  9 SC.BORDER_GRAY   = 'sc-gray-border';
 10 SC.BORDER_TOP    = 'sc-top-border';
 11 SC.BORDER_BOTTOM = 'sc-bottom-border';
 12 SC.BORDER_NONE   = null ;
 13 
 14 /**
 15   @namespace
 16 
 17   The Border mixin can be applied to any view to give it a visual border.
 18   In addition to specifying the mixin itself, you should specify the border
 19   style with the borderStyle property on your view.
 20 
 21   Border style can be any predefined CSS class name or a border color.
 22 
 23   If you specify a CSS class name, it must end in "-border". Additionally,
 24   you should set the borderTop, borderRight, borderBottom, and borderLeft
 25   properties so SproutCore can accurately account for the size of your view.
 26 
 27   SproutCore pre-defines several useful border styles including:
 28 
 29 - SC.BORDER_BEZEL  - displays an inlaid bezel
 30 - SC.BORDER_BLACK  - displays a black border
 31 - SC.BORDER_GRAY   - displays a gray border
 32 - SC.BORDER_TOP    - displays a border on the top only
 33 - SC.BORDER_BOTTOM - displays a border on the bottom only
 34 - SC.BORDER_NONE   - disables the border
 35 
 36   @since SproutCore 1.0
 37 */
 38 SC.Border = {
 39   /**
 40     The thickness of the top border.
 41 
 42     @property {Number}
 43     @commonTask Border Dimensions
 44   */
 45   borderTop: 0,
 46 
 47   /**
 48     The thickness of the right border.
 49 
 50     @property {Number}
 51     @commonTask Border Dimensions
 52   */
 53   borderRight: 0,
 54 
 55   /**
 56     The thickness of the bottom border.
 57 
 58     @property {Number}
 59     @commonTask Border Dimensions
 60   */
 61   borderBottom: 0,
 62 
 63   /**
 64     The thickness of the left border.
 65 
 66     @property {Number}
 67     @commonTask Border Dimensions
 68   */
 69   borderLeft: 0,
 70 
 71   /**
 72     The style of the border. You may specify a color string (like 'red' or
 73     '#fff'), a CSS class name, or one of:
 74 - SC.BORDER_BEZEL
 75 - SC.BORDER_BLACK
 76 - SC.BORDER_GRAY
 77 - SC.BORDER_TOP
 78 - SC.BORDER_BOTTOM
 79 - SC.BORDER_NONE
 80 
 81     If you specify a CSS class name, it must end in "-border".
 82   */
 83   borderStyle: SC.BORDER_GRAY,
 84 
 85   /**
 86     Walk like a duck
 87 
 88     @private
 89   */
 90   hasBorder: YES,
 91 
 92   /**
 93     Make sure we re-render if the borderStyle property changes.
 94     @private
 95   */
 96   displayProperties: ['borderStyle'],
 97 
 98   /** @private */
 99   _BORDER_REGEXP: (/-border$/),
100 
101   /** @private */
102   initMixin: function() {
103     this._sc_border_borderStyleDidChange();
104   },
105 
106   /** @private */
107   renderMixin: function(context, firstTime) {
108     var style = this.get('borderStyle');
109     if (style) {
110       if (this._BORDER_REGEXP.exec(style)) {
111         context.addClass(style);
112       } else context.addStyle('border', '1px '+style+' solid');
113     }
114   },
115 
116   /** @private */
117   _sc_border_borderStyleDidChange: function() {
118     var borderStyle = this.get('borderStyle'),
119         borderSize = SC.Border.dimensions[borderStyle];
120 
121     if (borderSize) {
122       this.borderTop    = borderSize;
123       this.borderRight  = borderSize;
124       this.borderBottom = borderSize;
125       this.borderLeft   = borderSize;
126     }
127   }
128 };
129 
130 SC.mixin(SC.Border, {
131   dimensions: {
132     'sc-bezel-border': 1,
133     'sc-black-border': 1,
134     'sc-gray-border': 1,
135     'sc-top-border': 1,
136     'sc-bottom-border': 1
137   }
138 });
