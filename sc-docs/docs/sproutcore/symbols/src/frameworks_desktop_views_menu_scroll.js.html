 10 /** @class
 11 
 12   Implements a complete scroller view for menus.  This class implements the
 13   arrows displayed in a menu to scroll.
 14   
 15   The main difference with SC.ScrollerView is that there is only vertical 
 16   scrollers. Value Syncing between SC.MenuScrollView and SC.MenuScrollerView
 17   is done using valueBinding.
 18   
 19   @extends SC.ScrollerView
 20   @since SproutCore 1.0

 23 SC.MenuScrollerView = SC.ScrollerView.extend({
 24   classNames: ['sc-menu-scroller-view'],
 25   
 26   // ..........................................................
 27   // PROPERTIES
 28   // 
 29   
 30   /**
 31    Used to set the scrolling direction of the scroller.
 32   */
 33   scrollDown: NO,
 34   
 35   /** 
 36     The scroller offset value.  This value will adjust between the minimum
 37     and maximum values that you set. Default is 0.
 38     
 39     @property
 40   */
 41   value: function(key, val) {
 42     if (val !== undefined) {
 43       // Don't enforce the maximum now, because the scroll view could change
 44       // height and we want our content to stay put when it does.
 45       this._value = val ;
 46     } else {
 47       var value = this._value || 0 ; // default value is at top/left
 48       return Math.min(value, this.get('maximum')) ;
 49     }
 50   }.property('maximum').cacheable(),
 51   
 52   /**
 53     The maximum offset value for the scroller.  This will be used to calculate
 54     the internal height/width of the scroller itself. It is not necessarily
 55     the same as the height of a scroll view's content view.
 56     
 57     When set less than the height of the scroller, the scroller is disabled.
 58     
 59     @property {Number}
 60   */
 61   maximum: 0,
 62   
 63   /**
 64     YES if enable scrollbar, NO to disable it.  Scrollbars will automatically 
 65     disable if the maximum scroll width does not exceed their capacity.
 66     
 67     @property
 68   */
 69   isEnabled: YES,
 70   
 71   /**
 72     Determine the layout direction.  Determines whether the scrollbar should 
 73     appear horizontal or vertical.  This must be set when the view is created.
 74     Changing this once the view has been created will have no effect.
 75     
 76     @property
 77   */
 78   layoutDirection: SC.LAYOUT_VERTICAL,
 79   
 80   /** 
 81      Amount to scroll one vertical line.
 82      Defaults to 20px.
 83   */
 84   verticalLineScroll: 20,
 85 
 86   /**
 87     This function overrides the default function in SC.Scroller as 
 88     menus only have vertical scrolling.
 89     
 90     @property {String}
 91   */
 92   ownerScrollValueKey: function() {
 93     return 'verticalScrollOffset' ;  
 94   }.property('layoutDirection').cacheable(),
 95 
 96   // ..........................................................
 97   // INTERNAL SUPPORT
 98   //
 99 
100   init: function() {
101     // Set the scrollerThickness based on controlSize
102     switch (this.get('controlSize')) {
103       case SC.TINY_CONTROL_SIZE:
104         this.set('scrollerThickness', SC.MenuScrollerView.TINY_SCROLLER_THICKNESS);
105         break;
106       case SC.SMALL_CONTROL_SIZE:
107         this.set('scrollerThickness', SC.MenuScrollerView.SMALL_SCROLLER_THICKNESS);
108         break;
109       case SC.REGULAR_CONTROL_SIZE:
110         this.set('scrollerThickness', SC.MenuScrollerView.REGULAR_SCROLLER_THICKNESS);
111         break;
112       case SC.LARGE_CONTROL_SIZE:
113         this.set('scrollerThickness', SC.MenuScrollerView.LARGE_SCROLLER_THICKNESS);
114         break;
115       case SC.HUGE_CONTROL_SIZE:
116         this.set('scrollerThickness', SC.MenuScrollerView.HUGE_SCROLLER_THICKNESS);
117         break;
118     }
119 
120     return sc_super();
121   },
122   
123   render: function(context, firstTime) {
124     context.addClass('sc-vertical') ;
125     context.addClass(this.get('controlSize'));
126     if (firstTime) {
127       var direction = this.get('scrollDown') ? 'arrowDown' : 'arrowUp' ;
128       context.push('<span class="scrollArrow '+direction+'"> </span>') ;
129     } 
130     context.setClass('disabled', !this.get('isEnabled')) ;
131   },

151   mouseEntered: function(evt) {
152     this.set('isMouseOver', YES);
153     this._invokeScrollOnMouseOver();
154   },
155   
156   mouseExited: function(evt) {
157     this.set('isMouseOver', NO);

160   /** @private */
161   
162   /**
163     This function overrides the default function in SC.Scroller. 
164     SC.MenuScroller and SC.MenuScroll use valueBinding so this function is
165     not neccesary.
166   */
167   _sc_scroller_valueDidChange: function() {
168     
169   }.observes('value'),
170   
171 
172   // after 50msec, fire event again
173   _sc_scroller_armScrollTimer: function() {
174     if (!this._sc_scrollTimer) {
175       SC.RunLoop.begin() ;
176       var method = this._sc_scroller_scrollDidChange ;
177       this._sc_scrollTimer = this.invokeLater(method, 50) ;
178       SC.RunLoop.end() ;
179     }
180   },
181   
182   _sc_scroller_scrollDidChange: function() {
183     var now = Date.now(), 
184         last = this._sc_lastScroll, 
185         layer = this.get('layer'), 
186         scroll = 0 ;
187     
188     if (last && (now-last)<50) return this._sc_scroller_armScrollTimer() ;
189     this._sc_scrollTimer = null ;
190     this._sc_lastScroll = now ;
191     
192     SC.RunLoop.begin();
193     
194     if (!this.get('isEnabled')) return ; // nothing to do.
195     
196     this._sc_scrollValue = scroll = layer.scrollTop ;
197     this.set('value', scroll) ; // will now enforce minimum and maximum
198     
199     SC.RunLoop.end();
200   },
201   
202   
203   /**
204     Scroll the menu if it is is an up or down arrow. This is called by
205     the function that simulates mouseOver.
206   */
207   _scrollMenu: function(){
208     var val = this.get('value'), newval;
209     if(this.get('scrollDown')) {
210       newval = val+this.verticalLineScroll;
211       if(newval<=this.get('maximum')){
212         this.set('value', newval);
213       }
214     }
215     else {
216       newval = val-this.verticalLineScroll;
217       if(newval>=0){
218         this.set('value', newval);
219       }else if(val<=this.verticalLineScroll && val>0){
220         this.set('value', 0);
221       }
222     }
223     return YES;
224   },
225   
226   /**
227     We use this function to simulate mouseOver. It checks for the flag 
228     isMouseOver which is turned on when mouseEntered is called and turned off
229     when mouseExited is called. 
230   */
231   _invokeScrollOnMouseOver: function(){
232     this._scrollMenu();
233     if(this.get('isMouseOver')){
234       this.invokeLater(this._invokeScrollOnMouseOver, 100);
235     }
236   }

250 /** @class
251 
252   Implements a scroll view for menus.  This class extends SC.ScrollView for 
253   menus. 
254   
255   The main difference with SC.ScrollView is that there is only vertical 
256   scrolling. Value Syncing between SC.MenuScrollView and SC.MenuScrollerView
257   is done using valueBinding.
258   
259   @extends SC.ScrollView
260   @since SproutCore 1.0
261 */
262 SC.MenuScrollView = SC.ScrollView.extend({
263 
264   classNames: ['sc-menu-scroll-view'],
265   
266   // ..........................................................
267   // PROPERTIES
268   // 
269   
270   
271   /**
272     The maximum horizontal scroll offset allowed given the current contentView 
273     size and the size of the scroll view.  If horizontal scrolling is 
274     disabled, this will always return 0.
275     
276     @property {Number}
277   */
278   maximumHorizontalScrollOffset: 0,
279     
280        
281   // ..........................................................
282   // SCROLLERS
283   // 
284   
285   /** 
286     YES if the view should maintain a horizontal scroller.   This property 
287     must be set when the view is created.
288     
289     @property {Boolean}
290   */
291   hasHorizontalScroller: NO,
292   
293   /**
294     The horizontal scroller view class. This will be replaced with a view 
295     instance when the ScrollView is created unless hasHorizontalScroller is 
296     NO.
297     
298     @property {SC.View}
299   */
300   horizontalScrollerView: SC.MenuScrollerView,
301   
302   /**
303     YES if the horizontal scroller should be visible.  You can change this 
304     property value anytime to show or hide the horizontal scroller.  If you 
305     do not want to use a horizontal scroller at all, you should instead set 
306     hasHorizontalScroller to NO to avoid creating a scroller view in the 
307     first place.
308     
309     @property {Boolean}
310   */
311   isHorizontalScrollerVisible: NO,
312 
313   /**
314     Returns YES if the view both has a horizontal scroller, the scroller is
315     visible.
316     
317     @property {Boolean}
318   */
319   canScrollHorizontal: NO,
320    
321   /**
322     If YES, the horizontal scroller will autohide if the contentView is
323     smaller than the visible area.  You must set hasHorizontalScroller to YES 
324     for this property to have any effect.  
325   */
326   autohidesHorizontalScroller: NO,
327   
328   /** 
329     YES if the view shuld maintain a vertical scroller.   This property must 
330     be set when the view is created.
331     
332     @property {Boolean}
333   */
334   hasVerticalScroller: YES,
335   
336   /**
337     The vertical scroller view class. This will be replaced with a view 
338     instance when the ScrollView is created unless hasVerticalScroller is NO.
339     
340     @property {SC.View}
341   */
342   verticalScrollerView: SC.MenuScrollerView,
343   verticalScrollerView2: SC.MenuScrollerView,
344   
345   /**
346     YES if the vertical scroller should be visible.  For SC.MenuScroll the
347     vertical scroller is always there we just hide the arrows to scroll.
348     
349     @property {Boolean}
350   */
351   isVerticalScrollerVisible: YES,
352 
353   
354   canScrollVertical: YES,
355 
356   /**
357     If YES, the vertical scroller will autohide if the contentView is
358     smaller than the visible area.  You must set hasVerticalScroller to YES 
359     for this property to have any effect.  
360   */
361   autohidesVerticalScroller: YES,
362   
363   /**
364     Use this property to set the 'bottom' offset of your vertical scroller, 
365     to make room for a thumb view or other accessory view. Default is 0.
366     
367     @property {Number}
368   */
369   verticalScrollerBottom: 0,
370   
371   
372   // ..........................................................
373   // CUSTOM VIEWS
374   // 
375 
376   /**
377     Control Size for Menu content: change verticalLineScroll
378   */
379   controlSize: SC.REGULAR_CONTROL_SIZE,
380   
381   /**
382     The container view that will contain your main content view.  You can 
383     replace this property with your own custom subclass if you prefer.
384     
385     @type {SC.ContainerView}
386   */
387   containerView: SC.ContainerView,
388   
389   // ..........................................................
390   // METHODS
391   // 
392 
393   
394   /**
395     Adjusts the layout for the various internal views.  This method is called
396     once when the scroll view is first configured and then anytime a scroller
397     is shown or hidden.  You can call this method yourself as well to retile.
398     
399     You may also want to override this method to handle layout for any
400     additional controls you have added to the view.
401   */
402   tile: function() {
403     // get vertical scroller/determine if we should have a scroller
404     var hasScroller, vscroll, vscroll2, hasVertical, clip, clipLayout, viewportHeight;
405     hasScroller = this.get('hasVerticalScroller');
406     vscroll = hasScroller ? this.get('verticalScrollerView') : null ;
407     vscroll2 = hasScroller ? this.get('verticalScrollerView2') : null ;
408     hasVertical = vscroll && this.get('isVerticalScrollerVisible') ;
409     
410     // get the containerView
411     clip = this.get('containerView') ;
412     clipLayout = { left: 0, top: 0 } ;
413     
414     if (hasVertical) {
415       viewportHeight =0;
416       var scrollerThickness = vscroll.get('scrollerThickness') || vscroll2.get('scrollerThickness');
417       var view   = this.get('contentView'), view2, 
418             f      = (view) ? view.get('frame') : null, 
419             height = (f) ? f.height : 0,
420             elem = this.containerView.$()[0],
421             verticalOffset = this.get('verticalScrollOffset'),
422             topArrowInvisible = { height: 0, top: 0, right: 0, left: 0 },
423             topArrowVisible = { height: scrollerThickness, top: 0, right: 0, left: 0 },
424             bottomArrowVisible = { height: scrollerThickness, bottom: 0, right: 0, left: 0 },
425             bottomArrowInvisible = { height: 0, bottom: 0, right: 0, left: 0 };
426       
427       if(elem) viewportHeight = elem.offsetHeight;
428       
429       if(verticalOffset===0){
430         clipLayout.top = 0 ;
431         clipLayout.bottom = scrollerThickness;
432         vscroll.set('layout', topArrowInvisible) ;
433         vscroll2.set('layout', bottomArrowVisible) ;
434       }else if(verticalOffset>=(height-viewportHeight-scrollerThickness)){
435         clipLayout.top = scrollerThickness ;
436         clipLayout.bottom = 0 ;
437         vscroll.set('layout', topArrowVisible) ;
438         vscroll2.set('layout', bottomArrowInvisible) ;
439       }else{
440         clipLayout.top = scrollerThickness ;
441         clipLayout.bottom = scrollerThickness ;
442         vscroll.set('layout', topArrowVisible) ;
443         vscroll2.set('layout', bottomArrowVisible) ;
444       }
445     } 
446     if (vscroll){
447      vscroll.set('isVisible', hasVertical) ;
448      vscroll2.set('isVisible', hasVertical) ;
449     }
450     clip.set('layout', clipLayout) ;
451   },
452   
453   /** @private
454     Called whenever a scroller visibility changes.  Calls the tile() method.
455   */
456   scrollerVisibilityDidChange: function() {
457     this.tile();
458   }.observes('isVerticalScrollerVisible', 'isHorizontalScrollerVisible', 'verticalScrollOffset'),
459

465   /** @private
466     Instantiate scrollers & container views as needed.  Replace their classes
467     in the regular properties.
468   */
469   createChildViews: function() {
470     var childViews = [], view, view2, controlSize = this.get('controlSize') ;
471     
472     // create the containerView.  We must always have a container view. 
473     // also, setup the contentView as the child of the containerView...
474     if (SC.none(view = this.containerView)) view = SC.ContainerView;
475     
476     childViews.push(this.containerView = this.createChildView(view, {
477       contentView: this.contentView
478     }));
479     
480     // and replace our own contentView...
481     this.contentView = this.containerView.get('contentView');
482     
483     // create a vertical scroller 
484     if ((view=this.verticalScrollerView) && (view2=this.verticalScrollerView2)) {
485       if (this.get('hasVerticalScroller')) {
486         view = this.verticalScrollerView = this.createChildView(view, {
487           layout: {top: 0, left: 0, right: 0},
488           controlSize: controlSize,
489           valueBinding: '*owner.verticalScrollOffset'
490         }) ;
491         childViews.push(view);
492         view2 = this.verticalScrollerView2 = this.createChildView(view2, {
493           scrollDown: YES,
494           layout: {bottom: 0, left: 0, right: 0 },
495           controlSize: controlSize,
496           valueBinding: '*owner.verticalScrollOffset'
497         }) ;
498         childViews.push(view2);
499       } else {
500         this.verticalScrollerView = null ;
501         this.verticalScrollerView2 = null ;
502       }
503     }
504     
505     // set childViews array.
506     this.childViews = childViews ;
507     
508     this.contentViewFrameDidChange() ; // setup initial display...
509     this.tile() ; // set up initial tiling
510   },
511   
512   init: function() {
513     sc_super();
514     
515     // start observing initial content view.  The content view's frame has
516     // already been setup in prepareDisplay so we don't need to call 
517     // viewFrameDidChange...
518     this._scroll_contentView = this.get('contentView') ;
519     var contentView = this._scroll_contentView ;
520 
521     if (contentView) {
522       contentView.addObserver('frame', this, this.contentViewFrameDidChange) ;
523     }
524 
525     if (this.get('isVisibleInWindow')) this._scsv_registerAutoscroll() ;
526   },
527   
528   /** @private Registers/deregisters view with SC.Drag for autoscrolling */
529   _scsv_registerAutoscroll: function() {
530     if (this.get('isVisibleInWindow')) SC.Drag.addScrollableView(this);
531     else SC.Drag.removeScrollableView(this);
532   }.observes('isVisibleInWindow'),
533   
534   /** @private
535     Invoked whenever the contentView's frame changes.  This will update the 
536     scroller maxmimum and optionally update the scroller visibility if the
537     size of the contentView changes.  We don't care about the origin since
538     that is tracked separately from the offset values.
539   */
540   contentViewFrameDidChange: function() {
541     var view   = this.get('contentView'), view2, 
542         f      = (view) ? view.get('frame') : null,
543         width  = (f) ? f.width : 0,
544         height = (f) ? f.height : 0,
545         dim    = this.get('frame'),
546         viewportHeight, elem ;
547         
548     // cache out scroll settings...
549     //if ((width === this._scroll_contentWidth) && (height === this._scroll_contentHeight)) return ;
550     this._scroll_contentWidth = width;
551     this._scroll_contentHeight = height ;
552     
553     if (this.get('hasVerticalScroller') && (view = this.get('verticalScrollerView')) && (view2 = this.get('verticalScrollerView2'))) {
554       height -= 1 ; // accurately account for our layout
555       // decide if it should be visible or not
556       if (this.get('autohidesVerticalScroller')) {
557         this.set('isVerticalScrollerVisible', height > dim.height);
558       }
559       height -= this.get('verticalScrollerBottom') ;
560       viewportHeight = 0;
561       elem = this.containerView.$()[0];
562       if(elem) viewportHeight = elem.offsetHeight;
563       height = height - viewportHeight;
564       view.setIfChanged('maximum', height) ;
565       view2.setIfChanged('maximum', height) ;
566     }
567   },
568   
569   /** @private
570     Whenever the horizontal scroll offset changes, update the scrollers and 
571     edit the location of the contentView.
572   */
573   _scroll_horizontalScrollOffsetDidChange: function() {},
574    
575   /** @private
576     Whenever the vertical scroll offset changes, update the scrollers and 
577     edit the location of the contentView.
578   */
579   _scroll_verticalScrollOffsetDidChange: function() {
580     var offset = this.get('verticalScrollOffset') ;
581     
582     // update the offset for the contentView...
583     var contentView = this.get('contentView');
584     if (contentView) contentView.adjust('top', 0-offset) ;
585     
586   }.observes('verticalScrollOffset')
