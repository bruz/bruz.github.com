  8 /**
  9   @class
 10   
 11   The UserDefaults object provides an easy way to store user preferences in
 12   your application on the local machine.  You use this by providing built-in
 13   defaults using the SC.userDefaults.defaults() method.  You can also
 14   implement the UserDefaultsDelegate interface to be notified whenever a
 15   default is required.  
 16   
 17   You should also set the userDomain property on the defaults on page load.
 18   This will allow the UserDefaults application to store/fetch keys from 
 19   localStorage for the correct user.
 20   
 21   You can also set an appDomain property if you want.  This will be 
 22   automatically prepended to key names with no slashes in them.
 23   
 24   SC.userDefaults.getPath("global:contactInfo.userName");
 25   
 26   @extends SC.Object
 27   @since SproutCore 1.0
 28 */
 29 SC.UserDefaults = SC.Object.extend(/** @scope SC.UserDefaults.prototype */ {
 30   
 31   ready: NO,
 32   
 33   /** 
 34     the default domain for the user.  This will be used to store keys in
 35     local storage.  If you do not set this property, the wrong values may be
 36     returned.
 37   */
 38   userDomain: null,
 39   
 40   /**
 41     The default app domain for the user.  Any keys that do not include a 
 42     slash will be prefixed with this app domain key when getting/setting.
 43   */
 44   appDomain: null,
 45   
 46   /** @private
 47     Defaults.  These will be used if not defined on localStorage.
 48   */
 49   _defaults: null,
 50   
 51   _safari3DB: null,
 52   
 53   /**
 54     Invoke this method to set the builtin defaults.  This will cause all
 55     properties to change.
 56   */
 57   defaults: function(newDefaults) {
 58     this._defaults = newDefaults ;
 59     this.allPropertiesDidChange();
 60   },
 61   
 62   /**
 63     Attempts to read a user default from local storage.  If not found on 
 64     localStorage, use the the local defaults, if defined.  If the key passed
 65     does not include a slash, then add the appDomain or use "app/".
 66     
 67     @param {String} keyName
 68     @returns {Object} read value
 69   */
 70   readDefault: function(keyName) {
 71     var ret= undefined, userKeyName, localStorage, key, del, storageSafari3;
 72     // namespace keyname
 73     keyName = this._normalizeKeyName(keyName);
 74     userKeyName = this._userKeyName(keyName);
 75 
 76     // look into recently written values
 77     if (this._written) ret = this._written[userKeyName];
 78     
 79     // attempt to read from localStorage
 80     
 81     if(SC.browser.msie=="7.0"){
 82       localStorage=document.body;
 83       try{
 84         localStorage.load("SC.UserDefaults");
 85       }catch(e){
 86         console.err("Couldn't load userDefaults in IE7: "+e.description);
 87       }
 88     }else if(this.HTML5DB_noLocalStorage){
 89       storageSafari3 = this._safari3DB;
 90     }else{
 91       localStorage = window.localStorage ;
 92       if (!localStorage && window.globalStorage) {
 93         localStorage = window.globalStorage[window.location.hostname];
 94       }
 95     }
 96     if (localStorage || storageSafari3) {
 97       key=["SC.UserDefaults",userKeyName].join('-at-');
 98       if(SC.browser.msie=="7.0"){
 99         ret=localStorage.getAttribute(key.replace(/\W/gi, ''));        
100       }else if(storageSafari3){
101         ret = this.dataHash[key];
102         
103       }else{
104         ret = localStorage[key];
105       }
106       if (!SC.none(ret)) {
107         try {
108           ret = SC.json.decode(ret);
109         } 
110         catch(ex) {
111           ret = undefined;
112         }
113       } else ret = undefined;
114     }
115     
116     // if not found in localStorage, try to notify delegate
117     del =this.delegate ;
118     if (del && del.userDefaultsNeedsDefault) {
119       ret = del.userDefaultsNeedsDefault(this, keyName, userKeyName);
120     }
121     
122     // if not found in localStorage or delegate, try to find in defaults
123     if ((ret===undefined) && this._defaults) {
124       ret = this._defaults[userKeyName] || this._defaults[keyName];
125     }
126     
127     return ret ;
128   },
129   
130   /**
131     Attempts to write the user default to local storage or at least saves them
132     for now.  Also notifies that the value has changed.
133     
134     @param {String} keyName
135     @param {Object} value
136     @returns {SC.UserDefault} receiver
137   */
138   writeDefault: function(keyName, value) {
139     var userKeyName, written, localStorage, key, del, storageSafari3;
140     
141     keyName = this._normalizeKeyName(keyName);
142     userKeyName = this._userKeyName(keyName);
143     
144     // save to local hash
145     written = this._written ;
146     if (!written) written = this._written = {};
147     written[userKeyName] = value ;
148     
149     // save to local storage
150     
151     if(SC.browser.msie=="7.0"){
152       localStorage=document.body;
153     }else if(this.HTML5DB_noLocalStorage){
154       storageSafari3 = this._safari3DB;
155     }else{
156        localStorage = window.localStorage ;
157        if (!localStorage && window.globalStorage) {
158          localStorage = window.globalStorage[window.location.hostname];
159        }
160     }
161     key=["SC.UserDefaults",userKeyName].join('-at-');
162     if (localStorage || storageSafari3) {
163       var encodedValue = SC.json.encode(value);
164       if(SC.browser.msie=="7.0"){
165         localStorage.setAttribute(key.replace(/\W/gi, ''), encodedValue);
166         localStorage.save("SC.UserDefaults");
167       }else if(storageSafari3){
168         var obj = this;
169         storageSafari3.transaction(
170           function (t) {
171             t.executeSql("delete from SCLocalStorage where key = ?", [key], 
172               function (){
173                 t.executeSql("insert into SCLocalStorage(key, value)"+
174                             " VALUES ('"+key+"', '"+encodedValue+"');", 
175                             [], obj._nullDataHandler, obj.killTransaction
176                 );
177               }
178             );
179           }
180         );
181         this.dataHash[key] = encodedValue;
182       }else{
183         try{
184           localStorage[key] = encodedValue;
185         }catch(e){
186           console.error("Failed using localStorage. "+e);
187         }
188       }
189     }
190     
191     // also notify delegate
192     del = this.delegate;
193     if (del && del.userDefaultsDidChange) {
194       del.userDefaultsDidChange(this, keyName, value, userKeyName);
195     }
196     
197     return this ;
198   },
199   
200   /**
201     Removed the passed keyName from the written hash and local storage.
202     
203     @param {String} keyName
204     @returns {SC.UserDefaults} receiver
205   */
206   resetDefault: function(keyName) {  
207     var fullKeyName, userKeyName, written, localStorage, key, storageSafari3;
208     fullKeyName = this._normalizeKeyName(keyName);
209     userKeyName = this._userKeyName(fullKeyName);
210     
211     this.propertyWillChange(keyName);
212     this.propertyWillChange(fullKeyName);
213     
214     written = this._written;
215     if (written) delete written[userKeyName];
216     
217     if(SC.browser.msie=="7.0"){
218        localStorage=document.body;
219     }else if(this.HTML5DB_noLocalStorage){
220          storageSafari3 = this._safari3DB;
221     }else{
222        localStorage = window.localStorage ;
223        if (!localStorage && window.globalStorage) {
224          localStorage = window.globalStorage[window.location.hostname];
225        }
226     }
227 
228     key=["SC.UserDefaults",userKeyName].join('-at-');
229 
230     if (localStorage) {
231       if(SC.browser.msie=="7.0"){
232         localStorage.setAttribute(key.replace(/\W/gi, ''), null);
233         localStorage.save("SC.UserDefaults");
234       } else if(storageSafari3){
235         var obj = this;
236         storageSafari3.transaction(
237           function (t) {
238             t.executeSql("delete from SCLocalStorage where key = ?", [key], null);
239           }
240         );
241         delete this.dataHash[key];
242       }else{
243         delete localStorage[key];
244       }
245     }
246     
247 
248     this.propertyDidChange(keyName);
249     this.propertyDidChange(fullKeyName);
250     return this ;
251   },
252   
253   /**
254     Is called whenever you .get() or .set() values on this object
255     
256     @param {Object} key
257     @param {Object} value
258     @returns {Object}
259   */
260   unknownProperty: function(key, value) {
261     if (value === undefined) {
262       return this.readDefault(key) ;
263     } else {
264       this.writeDefault(key, value);
265       return value ;
266     }
267   },
268   
269   /**
270     Normalize the passed key name.  Used by all accessors to automatically 
271     insert an appName if needed.
272   */
273   _normalizeKeyName: function(keyName) {
274     if (keyName.indexOf(':')<0) {
275       var domain = this.get('appDomain') || 'app';
276       keyName = [domain, keyName].join(':');
277     } 
278     return keyName;
279   },
280   
281   /** 
282     Builds a user key name from the passed key name
283   */
284   _userKeyName: function(keyName) {
285     var user = this.get('userDomain') || '(anonymous)' ;
286     return [user,keyName].join('-at-');
287   },
288   
289   _domainDidChange: function() {
290     var didChange = NO;
291     if (this.get("userDomain") !== this._scud_userDomain) {
292       this._scud_userDomain = this.get('userDomain');
293       didChange = YES;
294     }
295     
296     if (this.get('appDomain') !== this._scud_appDomain) {
297       this._scud_appDomain = this.get('appDomain');
298       didChange = YES;
299     }
300     
301     if (didChange) this.allPropertiesDidChange();
302   }.observes('userDomain', 'appDomain'),
303   
304   init: function() {
305     sc_super();
306     if(SC.userDefaults && SC.userDefaults.get('dataHash')){
307       var dh = SC.userDefaults.get('dataHash');
308       if (dh) this.dataHash=SC.userDefaults.get('dataHash')
309     }
310     this._scud_userDomain = this.get('userDomain');
311     this._scud_appDomain  = this.get('appDomain');
312     if(SC.browser.msie=="7.0"){
313       //Add user behavior userData. This works in all versions of IE.
314       //Adding to the body as is the only element never removed.
315       document.body.addBehavior('#default#userData');
316     }
317     this.HTML5DB_noLocalStorage = ((parseInt(SC.browser.safari, 0)>523) && (parseInt(SC.browser.safari, 0)<528));
318     if(this.HTML5DB_noLocalStorage){
319       var myDB;
320       try {
321         if (!window.openDatabase) {
322           console.error("Trying to load a database with safari version 3.1 "+
323                   "to get SC.UserDefaults to work. You are either in a"+
324                   " previous version or there is a problem with your browser.");
325           return;
326         } else {
327           var shortName = 'scdb',
328               version = '1.0',
329               displayName = 'SproutCore database',
330               maxSize = 65536; // in bytes,
331           myDB = openDatabase(shortName, version, displayName, maxSize);
332     
333           // You should have a database instance in myDB.
334     
335         }
336       } catch(e) {
337         console.error("Trying to load a database with safari version 3.1 "+
338                 "to get SC.UserDefaults to work. You are either in a"+
339                 " previous version or there is a problem with your browser.");
340         return;
341       }
342     
343       if(myDB){
344         var obj = this;
345         myDB.transaction(
346           function (transaction) {
347             transaction.executeSql('CREATE TABLE IF NOT EXISTS SCLocalStorage'+
348               '(key TEXT NOT NULL PRIMARY KEY, value TEXT NOT NULL);', 
349               [], obj._nullDataHandler, obj.killTransaction);
350           }
351         );
352         myDB.transaction(
353           function (transaction) {
354             
355             transaction.parent = obj;
356             transaction.executeSql('SELECT * from SCLocalStorage;', 
357                 [], function(transaction, results){
358                   var hash={}, row;
359                   for(var i=0, iLen=results.rows.length; i<iLen; i++){
360                     row=results.rows.item(i);
361                     hash[row['key']]=row['value'];
362                   }
363                   transaction.parent.dataHash = hash;
364                   SC.run(function() { SC.userDefaults.set('ready', YES); });
365                 }, obj.killTransaction);
366           }
367         );
368         this._safari3DB=myDB;
369       }
370     }else{
371       this.set('ready', YES);
372     }
373   },
374 
375 
376   //Private methods to use if user defaults uses the database in safari 3
377   _killTransaction: function(transaction, error){
378     return true; // fatal transaction error
379   },
380 
381   _nullDataHandler: function(transaction, results){},
382         
383   readyCallback: function(ob, func){
384     this.func = func;
385     this.ob = ob;
386   },
387   
388   readyChanged: function(){
389     if(this.ready===YES){
390       var f = this.func;
391       if(f) f.apply(this.ob);
392     }
393   }.observes('ready')  
394 });
395 
396 /** global user defaults. */
397 SC.userDefaults = SC.UserDefaults.create();
