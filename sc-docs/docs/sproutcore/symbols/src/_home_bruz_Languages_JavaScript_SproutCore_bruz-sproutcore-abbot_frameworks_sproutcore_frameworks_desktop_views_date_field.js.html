  9 /** @class
 10 
 11   A Date field add behaviour to the Text Field to support date management, 
 12   for example, disabling deletion, and special behaviour to tabs commands.
 13   
 14   This field view is tighly inregrated with SC.DateTime
 15   
 16   By default the Date Field View show Date only, but if you need to show the Time do:
 17   
 18   {{{
 19     dateAndTime: Shared.DateFieldView.design({
 20       showTime: YES,
 21       valueBinding: '...'
 22     }),
 23   }}}
 24   
 25   and if you only need to show time: 
 26   
 27   {{{
 28     timeOnly: Shared.DateFieldView.design({
 29       showTime: YES,
 30       showDate: NO,
 31       valueBinding: '...'
 32     })
 33   }}}
 34   
 35   The default format are:
 36   - formatTime: '%I:%M %p',
 37   - formatDate: '%d/%m/%Y',
 38   - formatDateTime: '%d/%m/%Y %I:%M %p',
 39   
 40   Example usage with special format:
 41 
 42   specialDate: Shared.DateFieldView.design({
 43     formatDate: '%d %b of %Y',
 44     valueBinding: '...'
 45   }),
 46   
 47   You can override these format as you like, but has some limitations,
 48   actually only support these KEY from SC.DateTime:
 49   
 50   %a %b %d %H %I %j %m %M %p %S %U %W %y %Y
 51   
 52   These are keys that has FIXED length, so we can control the selection and tabing.
 53   
 54   @extends SC.TextFieldView
 55   @since SproutCore 1.0
 56   @author Juan Pablo Goldfinger
 57 */
 58 SC.DateFieldView = SC.TextFieldView.extend(
 59 /** @scope SC.DateFieldView.prototype */
 60 {
 61 
 62   value: null,
 63   
 64   // Default Behaviour
 65   showDate: YES,
 66   showTime: NO,
 67   
 68   // Default formats to choose depending of the behaviour.
 69   formatTime: '%I:%M %p',
 70   formatDate: '%d/%m/%Y',
 71   formatDateTime: '%d/%m/%Y %I:%M %p',
 72   
 73   // DateTime constants (with fixed width, like numbers or abbs with fixed length)
 74   // original: '%a %A %b %B %c %d %h %H %i %I %j %m %M %p %S %U %W %x %X %y %Y %Z %%'.w(),
 75   // NOTE: I think that %a and %b areb't useful because is more adecuato to represente day
 76   // with 1..31 without zeros at start, but causes the lenght not to be fixed)
 77   _dtConstants: '%a %b %d %H %I %j %m %M %p %S %U %W %y %Y'.w(),
 78   // Width constants for each representation %@.
 79   _wtConstants: [3,3,2,2,2,3,2,2,2,2,2,2,2,4],
 80   
 81   // PRIVATE ATTRIBUTES
 82   activeSelection: 0,

100   /**
101     The current format to apply for Validator and to show.
102     
103     @property
104     @type {String}
105   */
106   format: function() {
107     var st = this.get('showTime');
108     var sd = this.get('showDate');
109     if (st === YES && sd === YES) return this.get('formatDateTime');
110     if (st === YES) return this.get('formatTime');
111     return this.get('formatDate');
112   }.property('showTime', 'showDate').cacheable(),
113   
114   /**
115     The current validator to format the Date to the input field and viceversa.
116     
117     @property
118     @type {SC.Validator.DateTime}
119   */
120   validator: function() {
121     return SC.Validator.DateTime.extend({ format: this.get('format') });
122   }.property('format').cacheable(),
123   
124   /**
125     Array of Key/TextSelection found for the current format.
126     
127     @property
128     @type {SC.Array}
129   */  
130   tabsSelections: function() {
131     var arr = [];
132     var ft = this.get('format');
133     var _dt = this.get('_dtConstants');
134     var _wt = this.get('_wtConstants');
135     
136     // Parse the string format to retrieve and build 
137     // a TextSelection array ordered to support tabs behaviour
138     if (SC.empty(ft)) {
139       throw 'The format string is empty, and must be a valid string.';
140     }
141     
142     var pPos, key, keyPos, startAt = 0, nPos = 0, oPos = 0;
143     while(startAt < ft.length && ft.indexOf('%', startAt) !== -1) {
144       pPos = ft.indexOf('%', startAt);
145       key = ft.substring(pPos, pPos + 2);
146       startAt = pPos + 2;
147       
148       keyPos = _dt.indexOf(key);
149       if (keyPos === -1) {
150         throw "SC.DateFieldView: The format's key '%@' is not supported.".fmt(key); 
151       }
152       nPos = nPos + pPos - oPos;
153       arr.push(SC.Object.create({
154         key: key,
155         textSelection: SC.TextSelection.create({ start: nPos, end: nPos + _wt[keyPos] })
156       }));
157       nPos = nPos + _wt[keyPos];
158       oPos = startAt;   
159     }
160     pPos = key = keyPos = null;
161     
162     return arr;
163   }.property('format').cacheable(),
164   
165   /** @private
166     If the activeSelection changes or the value changes, update the "TextSelection" to show accordingly.
167    
168   */
169   updateTextSelecitonObserver: function() {
170     var as = this.get('activeSelection');
171     var ts = this.get('tabsSelections');
172     if (this.get('isEditing')) {
173       this.selection(null, ts[as].get('textSelection'));
174     }
175   }.observes('activeSelection', 'value'),
176   
177   /** @private
178     Updates the value according the key.
179    
180   */
181   updateValue: function(key, upOrDown) {
182     // 0 is DOWN - 1 is UP
183     var newValue = (upOrDown === 0) ? -1 : 1;
184     var value = this.get('value'), hour;
185     switch(key) {
186       case '%a': case '%d': case '%j': this.set('value', value.advance({ day: newValue })); break;
187       case '%b': case '%m': this.set('value', value.advance({ month: newValue })); break;
188       case '%H': case '%I': this.set('value', value.advance({ hour: newValue })); break;
189       case '%M': this.set('value', value.advance({ minute: newValue })); break;
190       case '%p': {
191         hour = value.get('hour') >= 12 ? -12 : 12;
192         this.set('value', value.advance({ hour: hour }));
193         break;
194       }
195       case '%S': this.set('value', value.advance({ second: newValue })); break;
196       case '%U': this.set('value', value.advance({ week1: newValue })); break;
197       case '%W': this.set('value', value.advance({ week0: newValue })); break;
198       case '%y': case '%Y': this.set('value', value.advance({ year: newValue })); break;
199     } 
200   },

223   keyDown: function(evt) {
224     if (this.interpretKeyEvents(evt)) {
225       evt.stop();
226       return YES;
227     }
228     return sc_super();
229   },
230   
231   ctrl_a: function() {
232     return YES;
233   },
234 
235   moveUp: function(evt) {
236     var as = this.get('activeSelection');
237     var ts = this.get('tabsSelections');
238     this.updateValue(ts[as].get('key'), 1);
239     return YES;
240   },
241 
242   moveDown: function(evt) {
243     var as = this.get('activeSelection');
244     var ts = this.get('tabsSelections');
245     this.updateValue(ts[as].get('key'), 0);
246     return YES;
247   },
248 
249   insertText: function(evt) {
250     return YES;
251   },
252 
253   moveRight: function(evt) {
254     var ts = this.get('tabsSelections');
255     var ns = this.get('activeSelection') + 1;
256     if (ns === ts.length) {
257       ns = 0;
258     }
259     this.set('activeSelection', ns);
260     return YES;
261   },
262     
263   moveLeft: function(evt) {
264     var ts = this.get('tabsSelections');
265     var ns = this.get('activeSelection') - 1;
266     if (ns === -1) {
267       ns = ts.length - 1;
268     }
269     this.set('activeSelection', ns);
270     return YES;
271   },
272 
273   insertTab: function(evt) {
274     var ts = this.get('tabsSelections');
275     var ns = this.get('activeSelection') + 1;
276     if (ns < ts.length) { 
277       this.set('activeSelection', ns);
278       return YES;   
279     }
280     return NO;
281   },
282 
283   insertBacktab: function(evt) {
284     var ns = this.get('activeSelection') - 1;
285     if (ns !== -1) { 
286       this.set('activeSelection', ns);
287       return YES;   
288     }
289     return NO;
290   },
291 
292   mouseUp: function(evt) {
293     var ret = sc_super();
294     var cs = this.get('selection');
295     if (SC.none(cs)) {
296       this.set('activeSelection', 0);
297     } else {
298       var caret = cs.get('start');
299       var ts = this.get('tabsSelections');
300       var _tsLen = ts.length, cts;
301       for(var i=0; i<_tsLen; i++) {
302         cts = ts[i].get('textSelection');
303         if (caret >= cts.get('start') && caret <= cts.get('end')) {
304           this.set('activeSelection', i);
305         }
306       }
307     }
308     return ret;
309   },
310 
311   deleteBackward: function(evt) {
312     return YES;
313   },
314 
315   deleteForward: function(evt) {
316     return YES;
317   }
