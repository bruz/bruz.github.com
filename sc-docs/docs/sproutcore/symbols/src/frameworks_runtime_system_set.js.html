 13 // IMPORTANT NOTE:  This file actually defines two classes: 
 14 // SC.Set is a fully observable set class documented below. 
 15 // SC._CoreSet is just like SC.Set but is not observable.  This is required
 16 // because SC.Observable is built on using sets and requires sets without 
 17 // observability.
 18 //
 19 // We use pointer swizzling below to swap around the actual definitions so 
 20 // that the documentation will turn out right.  (The docs should only 
 21 // define SC.Set - not SC._CoreSet)
 22 
 23 /**
 24   @class 
 25 
 26   An unordered collection of objects.
 27 
 28   A Set works a bit like an array except that its items are not ordered.  
 29   You can create a set to efficiently test for membership for an object. You 
 30   can also iterate through a set just like an array, even accessing objects
 31   by index, however there is no gaurantee as to their order.
 32 
 33   Note that SC.Set is a primitive object, like an array.  It does implement
 34   limited key-value observing support but it does not extend from SC.Object
 35   so you should not subclass it.
 36 
 37   h1. Creating a Set
 38 
 39   You can create a set like you would most objects using SC.Set.create().  
 40   Most new sets you create will be empty, but you can also initialize the set 
 41   with some content by passing an array or other enumerable of objects to the 
 42   constructor.
 43 
 44   Finally, you can pass in an existing set and the set will be copied.  You
 45   can also create a copy of a set by calling SC.Set#clone().
 46 
 47   {{{
 48     // creates a new empty set
 49     var foundNames = SC.Set.create();
 50 
 51     // creates a set with four names in it.
 52     var names = SC.Set.create(["Charles", "Peter", "Chris", "Erich"]) ;
 53 
 54     // creates a copy of the names set.
 55     var namesCopy = SC.Set.create(names);
 56 
 57     // same as above.
 58     var anotherNamesCopy = names.clone();
 59   }}}
 60 
 61   h1. Adding/Removing Objects
 62 
 63   You generally add or removed objects from a set using add() or remove().
 64   You can add any type of object including primitives such as numbers,
 65   strings, and booleans.
 66 
 67   Note that objects can only exist one time in a set.  If you call add() on
 68   a set with the same object multiple times, the object will only be added 
 69   once.  Likewise, calling remove() with the same object multiple times will
 70   remove the object the first time and have no effect on future calls until 
 71   you add the object to the set again.
 72 
 73   Note that you cannot add/remove null or undefined to a set.  Any attempt to
 74   do so will be ignored.  
 75 
 76   In addition to add/remove you can also call push()/pop().  Push behaves just
 77   like add() but pop(), unlike remove() will pick an arbitrary object, remove
 78   it and return it.  This is a good way to use a set as a job queue when you
 79   don't care which order the jobs are executed in.
 80 
 81   h1. Testing for an Object
 82 
 83   To test for an object's presence in a set you simply call SC.Set#contains().
 84   This method tests for the object's hash, which is generally the same as the
 85   object's _guid but if you implement the hash() method on the object, it will
 86   use the return value from that method instead.
 87 
 88   @extends SC.Enumerable 
 89   @extends SC.Observable
 90   @extends SC.Copyable
 91   @extends SC.Freezable
 92 
 93   @since SproutCore 1.0
 94 */
 95 SC.Set = SC.mixin({}, 
 96   SC.Enumerable, 
 97   SC.Observable, 
 98   SC.Freezable, 
 99 /** @scope SC.Set.prototype */ {
100 
101   /** 
102     Creates a new set, with the optional array of items included in the 
103     return set.
104 
105     @param {SC.Enumerable} items items to add
106     @return {SC.Set}
107   */
108   create: function(items) {
109     var ret, idx, pool = SC.Set._pool, isObservable = this.isObservable;
110     if (!isObservable && items===undefined && pool.length>0) ret = pool.pop();
111     else {
112       ret = SC.beget(this);
113       if (isObservable) ret.initObservable();
114       
115       if (items && items.isEnumerable && items.get('length')>0) {
116 
117         ret.isObservable = NO; // suspend change notifications
118         
119         // arrays and sets get special treatment to make them a bit faster
120         if (items.isSCArray) {
121           idx = items.get ? items.get('length') : items.length;
122           while(--idx>=0) ret.add(items.objectAt(idx));
123         
124         } else if (items.isSet) {
125           idx = items.length;
126           while(--idx>=0) ret.add(items[idx]);
127           
128         // otherwise use standard SC.Enumerable API
129         } else items.forEach(function(i) { ret.add(i); }, this);
130         
131         ret.isObservable = isObservable;
132       }
133     }
134     return ret ;
135   },
136   
137   /**
138     Walk like a duck
139     
140     @property {Boolean}
141   */
142   isSet: YES,
143   
144   /**
145     This property will change as the number of objects in the set changes.
146 
147     @property {Number}
148   */
149   length: 0,
150 
151   /**
152     Returns the first object in the set or null if the set is empty
153     
154     @property {Object}
155   */
156   firstObject: function() {
157     return (this.length>0) ? this[0] : undefined ;
158   }.property(),
159   
160   /**
161     Clears the set 
162     
163     @returns {SC.Set}
164   */
165   clear: function() { 
166     if (this.isFrozen) throw SC.FROZEN_ERROR;
167     this.length = 0;
168     return this ;
169   },
170 
171   /**
172     Call this method to test for membership.
173     
174     @returns {Boolean}
175   */
176   contains: function(obj) {
177 
178     // because of the way a set is "reset", the guid for an object may 
179     // still be stored as a key, but points to an index that is beyond the
180     // length.  Therefore the found idx must both be defined and less than
181     // the current length.
182     if (obj === null) return NO ;
183     var idx = this[SC.hashFor(obj)] ;
184     return (!SC.none(idx) && (idx < this.length) && (this[idx]===obj)) ;
185   },
186   
187   /**
188     Returns YES if the passed object is also a set that contains the same 
189     objects as the receiver.
190   
191     @param {SC.Set} obj the other object
192     @returns {Boolean}
193   */
194   isEqual: function(obj) {
195     // fail fast
196     if (!obj || !obj.isSet || (obj.get('length') !== this.get('length'))) {
197       return NO ;
198     }
199     
200     var loc = this.get('length');
201     while(--loc>=0) {
202       if (!obj.contains(this[loc])) return NO ;
203     }
204     
205     return YES;
206   },
207 
208   /**
209     Call this method to add an object. performs a basic add.
210 
211     If the object is already in the set it will not be added again.
212 
213     @param obj {Object} the object to add
214     @returns {SC.Set} receiver
215   */
216   add: function(obj) {
217     if (this.isFrozen) throw SC.FROZEN_ERROR;
218     
219     // cannot add null to a set.
220     if (obj===null || obj===undefined) return this; 
221 
222     // Implementation note:  SC.hashFor() is inlined because sets are
223     // fundamental in SproutCore, and the inlined code is ~ 25% faster than
224     // calling SC.hashFor() in IE8.
225     var hashFunc,
226         guid = (obj && (hashFunc = obj.hash) && (typeof hashFunc === SC.T_FUNCTION)) ? hashFunc.call(obj) : SC.guidFor(obj),
227         idx  = this[guid],
228         len  = this.length;
229     if ((idx===null || idx===undefined) || (idx >= len) || (this[idx]!==obj)) {
230       this[len] = obj;
231       this[guid] = len;
232       this.length = len+1;
233     }
234     
235     if (this.isObservable) this.enumerableContentDidChange();
236     
237     return this ;
238   },
239 
240   /**
241     Add all the items in the passed array or enumerable
242 
243     @returns {SC.Set} receiver
244   */
245   addEach: function(objects) {
246     if (this.isFrozen) throw SC.FROZEN_ERROR;
247     if (!objects || !objects.isEnumerable) {
248       throw "%@.addEach must pass enumerable".fmt(this);
249     }
250 
251     var idx, isObservable = this.isObservable ;
252     
253     if (isObservable) this.beginPropertyChanges();
254     if (objects.isSCArray) {
255       idx = objects.get('length');
256       while(--idx >= 0) this.add(objects.objectAt(idx)) ;
257     } else if (objects.isSet) {
258       idx = objects.length;
259       while(--idx>=0) this.add(objects[idx]);
260       
261     } else objects.forEach(function(i) { this.add(i); }, this);
262     if (isObservable) this.endPropertyChanges();
263     
264     return this ;
265   },  
266 
267   /**
268     Removes the object from the set if it is found.
269 
270     If the object is not in the set, nothing will be changed.
271 
272     @param obj {Object} the object to remove
273     @returns {SC.Set} receiver
274   */  
275   remove: function(obj) {
276     if (this.isFrozen) throw SC.FROZEN_ERROR;
277 
278     // Implementation note:  SC.none() and SC.hashFor() are inlined because
279     // sets are fundamental in SproutCore, and the inlined code is ~ 25%
280     // faster than calling them "normally" in IE8.
281     if (obj === null || obj === undefined) return this ;
282 
283     var hashFunc,
284         guid = (obj && (hashFunc = obj.hash) && (typeof hashFunc === SC.T_FUNCTION)) ? hashFunc.call(obj) : SC.guidFor(obj),
285         idx  = this[guid],
286         len  = this.length;
287 
288     // not in set.
289     // (SC.none is inlined for the reasons given above)
290     if ((idx === null || idx === undefined) || (idx >= len) || (this[idx] !== obj)) return this; 
291 
292     // clear the guid key
293     delete this[guid];
294 
295     // to clear the index, we will swap the object stored in the last index.
296     // if this is the last object, just reduce the length.
297     if (idx < (len-1)) {
298       obj = this[idx] = this[len-1];
299       guid = (obj && (hashFunc = obj.hash) && (typeof hashFunc === SC.T_FUNCTION)) ? hashFunc.call(obj) : SC.guidFor(obj);
300       this[guid] = idx;
301     }
302 
303     // reduce the length
304     this.length = len-1;
305     if (this.isObservable) this.enumerableContentDidChange();
306     return this ;
307   },
308 
309   /**
310     Removes an arbitrary object from the set and returns it.
311 
312     @returns {Object} an object from the set or null
313   */
314   pop: function() {
315     if (this.isFrozen) throw SC.FROZEN_ERROR;
316     var obj = (this.length > 0) ? this[this.length-1] : null ;
317     if (obj) this.remove(obj) ;
318     return obj ;
319   },
320 
321   /**
322     Removes all the items in the passed array.
323 
324     @returns {SC.Set} receiver
325   */
326   removeEach: function(objects) {
327     if (this.isFrozen) throw SC.FROZEN_ERROR;
328     if (!objects || !objects.isEnumerable) {
329       throw "%@.addEach must pass enumerable".fmt(this);
330     }
331 
332     var idx, isObservable = this.isObservable ;
333     
334     if (isObservable) this.beginPropertyChanges();
335     if (objects.isSCArray) {
336       idx = objects.get('length');
337       while(--idx >= 0) this.remove(objects.objectAt(idx)) ;
338     } else if (objects.isSet) {
339       idx = objects.length;
340       while(--idx>=0) this.remove(objects[idx]);
341     } else objects.forEach(function(i) { this.remove(i); }, this);
342     if (isObservable) this.endPropertyChanges();
343     
344     return this ;
345   },  
346 
347   /**
348    Clones the set into a new set.  
349 
350     @returns {SC.Set} new copy
351   */
352   copy: function() {
353     return this.constructor.create(this);    
354   },
355 
356   /**
357     Return a set to the pool for reallocation.
358 
359     @returns {SC.Set} receiver
360   */
361   destroy: function() {
362     this.isFrozen = NO ; // unfreeze to return to pool
363     if (!this.isObservable) SC.Set._pool.push(this.clear());
364     return this;
365   },
366   
367   // .......................................

370 
371   /** @private - optimized */
372   forEach: function(iterator, target) {
373     var len = this.length;
374     if (!target) target = this ;
375     for(var idx=0;idx<len;idx++) iterator.call(target, this[idx], idx, this);
376     return this ;
377   },
378 
379   /** @private */
380   toString: function() {
381     var len = this.length, idx, ary = [];
382     for(idx=0;idx<len;idx++) ary[idx] = this[idx];
383     return "SC.Set<%@>".fmt(ary.join(',')) ;
384   },
385   
386   // the pool used for non-observable sets
387   _pool: [],
388 
389   /** @private */
390   isObservable: YES
391 
392 }) ;
