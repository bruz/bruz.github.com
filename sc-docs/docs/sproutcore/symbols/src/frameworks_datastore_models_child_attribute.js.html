  7 sc_require('models/record');
  8 sc_require('models/record_attribute');

 10 /** @class
 11   
 12   ChildAttribute is a subclass of RecordAttribute and handles to-one 
 13   relationship for child record
 14   
 15   When setting ( .set() ) the value of a toMany attribute, make sure
 16   to pass in an array of SC.Record objects.
 17   
 18   There are many ways you can configure a ManyAttribute:
 19   
 20   {{{
 21     contacts: SC.ChildAttribute.attr('SC.Child');
 22   }}}
 23   
 24   @extends SC.RecordAttribute
 25   @since SproutCore 1.0
 26 */
 27 SC.ChildAttribute = SC.RecordAttribute.extend(
 28   /** @scope SC.ChildAttribute.prototype */ {
 29     
 30   isChildRecordTransform: YES,
 31       
 32   // ..........................................................
 33   // LOW-LEVEL METHODS
 34   //
 35   
 36   /**  @private - adapted for to many relationship */
 37   toType: function(parentRecord, key, hash) {
 38     var ret   = null,
 39         cacheKey  = SC.keyFor('__kid__', SC.guidFor(this)),
 40         recordType  = this.get('typeClass');
 41     
 42     if (parentRecord[cacheKey]) return parentRecord[cacheKey];
 43     
 44     if (!parentRecord) {
 45       throw 'SC.Child: Error during transform: Unable to retrieve parent record.';
 46     }
 47 
 48     // If no hash, return null.
 49     if (hash){
 50       // Get the record type.
 51       var childNS = parentRecord.get('childRecordNamespace');
 52       if (hash.type && !SC.none(childNS)) {
 53         recordType = childNS[hash.type];
 54       }
 55 
 56       if (!recordType || SC.typeOf(recordType) !== SC.T_CLASS) {
 57         throw 'SC.Child: Error during transform: Invalid record type.';
 58       }
 59       // Create an instance of the record by registering it with the parent and return.
 60       ret = parentRecord[cacheKey] = parentRecord.registerChildRecord(recordType, hash);
 61     }
 62     return ret;
 63   },
 64   
 65   // Default fromType is just returning itself
 66   fromType: function(record, key, value){
 67     return value;
 68   },
 69     
 70   /**
 71     The core handler.  Called from the property.
 72     
 73     @param {SC.Record} record the record instance
 74     @param {String} key the key used to access this attribute on the record
 75     @param {Object} value the property value if called as a setter
 76     @returns {Object} property value
 77   */
 78   call: function(record, key, value) {
 79     var attrKey = this.get('key') || key,
 80         cacheKey = SC.keyFor('__kid__', SC.guidFor(this)), 
 81         nvalue;
 82     if (value !== undefined) {
 83       // careful: don't overwrite value here.  we want the return value to 
 84       // cache.
 85       this.orphan(record);
 86       nvalue = this.fromType(record, key, value) ; // convert to attribute.
 87       record[cacheKey] = null;
 88       record.writeAttribute(attrKey, nvalue);
 89       value = this.toType(record, key, value); // need to convert to the child record for caching
 90     } else {
 91       value = record.readAttribute(attrKey);
 92       if (SC.none(value) && (value = this.get('defaultValue'))) {
 93         if (typeof value === SC.T_FUNCTION) {
 94           value = this.defaultValue(record, key, this);
 95           // write default value so it doesn't have to be executed again
 96           if(record.attributes()) record.writeAttribute(attrKey, value, true);
 97         }
 98       } else value = this.toType(record, key, value);
 99     }
100 
101     return value ;
102   },
103   
104   orphan: function(parentRecord){
105     var cacheKey = SC.keyFor('__kid__', SC.guidFor(this)),
106         store, storeKey, attrs, key, param, cRef;
107     cRef = parentRecord ? parentRecord[cacheKey] : null;
108     if (cRef) {
109       attrs = cRef.get('readOnlyAttributes');
110       for(key in attrs) {
111         param = cRef[key];
112         // Orphan all the child record and child records in a tree to clean up the store
113         if(param && param.isChildRecordTransform) param.orphan(parentRecord);
114       }
115       store = cRef.get('store');
116       if(store) storeKey = cRef.storeKey;
117       if(storeKey) store.unloadRecord(undefined, undefined, storeKey);
118     }
119   }
120 });
