 13 /**
 14   @class
 15 
 16   A dynamically filled array.  A SparseArray makes it easy for you to create 
 17   very large arrays of data but then to defer actually populating that array
 18   until it is actually needed.  This is often much faster than generating an
 19   array up front and paying the cost to load your data then.
 20   
 21   Although technically all arrays in JavaScript are "sparse" (in the sense 
 22   that you can read and write properties at arbitrary indexes), this array
 23   keeps track of which elements in the array have been populated already 
 24   and which ones have not.  If you try to get a value at an index that has 
 25   not yet been populated, the SparseArray will notify a delegate object first,
 26   giving the delegate a chance to populate the component.
 27   
 28   Most of the time, you will use a SparseArray to incrementally load data 
 29   from the server.  For example, if you have a contact list with 3,000
 30   contacts in it, you may create a SparseArray with a length of 3,000 and set
 31   that as the content for a ListView.  As the ListView tries to display the
 32   visible contacts, it will request them from the SparseArray, which will in
 33   turn notify your delegate, giving you a chance to load the contact data from
 34   the server.
 35   
 36   @extends SC.Enumerable
 37   @extends SC.Array
 38   @extends SC.Observable
 39   @extends SC.DelegateSupport
 40   @since SproutCore 1.0
 41 */
 42 
 43 SC.SparseArray = SC.Object.extend(SC.Observable, SC.Enumerable, SC.Array, 
 44   SC.DelegateSupport, /** @scope SC.SparseArray.prototype */ {  
 45 
 46   // ..........................................................
 47   // LENGTH SUPPORT
 48   // 
 49 
 50   _requestingLength: 0,  
 51   _requestingIndex: 0,
 52    
 53   /**
 54     The length of the sparse array.  The delegate for the array should set 
 55     this length.
 56     
 57     @property {Number}
 58   */
 59   length: function() {
 60     var del = this.delegate ;
 61     if (del && SC.none(this._length) && del.sparseArrayDidRequestLength) {
 62       this._requestingLength++ ;
 63       del.sparseArrayDidRequestLength(this);
 64       this._requestingLength-- ;
 65     }
 66     return this._length || 0 ;
 67   }.property().cacheable(),
 68 
 69   /**
 70     Call this method from a delegate to provide a length for the sparse array.
 71     If you pass null for this property, it will essentially "reset" the array
 72     causing your delegate to be called again the next time another object 
 73     requests the array length.
 74   
 75     @param {Number} length the length or null
 76     @returns {SC.SparseArray} receiver
 77   */
 78   provideLength: function(length) {
 79     if (SC.none(length)) this._sa_content = null ;
 80     if (length !== this._length) {
 81       this._length = length ;
 82       if (this._requestingLength <= 0) this.enumerableContentDidChange() ;
 83     }
 84     return this ;
 85   },
 86 
 87   // ..........................................................
 88   // READING CONTENT 
 89   // 
 90 
 91   /** 
 92     The minimum range of elements that should be requested from the delegate.
 93     If this value is set to larger than 1, then the sparse array will always
 94     fit a requested index into a range of this size and request it.
 95     
 96     @property {Number}
 97   */
 98   rangeWindowSize: 1,
 99   
100   /*
101     This array contains all the start_indexes of ranges requested. This is to 
102     avoid calling sparseArrayDidRequestRange to often. Indexes are removed and 
103     added as range requests are completed.
104   */
105   requestedRangeIndex: [],
106   
107   /** 
108     Returns the object at the specified index.  If the value for the index
109     is currently undefined, invokes the didRequestIndex() method to notify
110     the delegate.
111     
112     @param  {Number} idx the index to get
113     @return {Object} the object
114   */
115   objectAt: function(idx) {
116     var content = this._sa_content, ret ;
117     if (!content) content = this._sa_content = [] ;
118     if ((ret = content[idx]) === undefined) {
119       this.requestIndex(idx);
120       ret = content[idx]; // just in case the delegate provided immediately
121     }
122     return ret ;
123   },
124 
125   /**
126     Returns the set of indexes that are currently defined on the sparse array.
127     If you pass an optional index set, the search will be limited to only 
128     those indexes.  Otherwise this method will return an index set containing
129     all of the defined indexes.  Currently this can be quite expensive if 
130     you have a lot of indexes defined.
131     
132     @param {SC.IndexSet} indexes optional from indexes
133     @returns {SC.IndexSet} defined indexes
134   */
135   definedIndexes: function(indexes) {
136     var ret = SC.IndexSet.create(),
137         content = this._sa_content,
138         idx, len;
139         
140     if (!content) return ret.freeze(); // nothing to do
141     
142     if (indexes) {
143       indexes.forEach(function(idx) { 
144         if (content[idx] !== undefined) ret.add(idx);
145       });
146     } else {      
147       len = content.length;
148       for(idx=0;idx<len;idx++) {
149         if (content[idx] !== undefined) ret.add(idx);
150       }
151     }
152     
153     return ret.freeze();
154   },
155   
156   _TMP_RANGE: {},
157   
158   /**
159     Called by objectAt() whenever you request an index that has not yet been
160     loaded.  This will possibly expand the index into a range and then invoke
161     an appropriate method on the delegate to request the data.
162     
163     It will check if the range has been already requested.
164     
165     @param {Number} idx the index to retrieve
166     @returns {SC.SparseArray} receiver
167   */
168   requestIndex: function(idx) {
169     var del = this.delegate;
170     if (!del) return this; // nothing to do
171     
172     // adjust window
173     var len = this.get('rangeWindowSize'), start = idx;
174     if (len > 1) start = start - Math.floor(start % len);
175     if (len < 1) len = 1 ;
176     
177     // invoke appropriate callback
178     this._requestingIndex++;
179     if (del.sparseArrayDidRequestRange) {
180       var range = this._TMP_RANGE;
181       if(this.wasRangeRequested(start)===-1){
182         range.start = start;
183         range.length = len;
184         del.sparseArrayDidRequestRange(this, range);
185         this.requestedRangeIndex.push(start);
186       }
187     } else if (del.sparseArrayDidRequestIndex) {
188       while(--len >= 0) del.sparseArrayDidRequestIndex(this, start + len);
189     }
190     this._requestingIndex--;
191 
192     return this ;
193   },
194   
195   /*
196     This method is called by requestIndex to check if the range has already 
197     been requested. We assume that rangeWindowSize is not changed often.
198     
199      @param {Number} startIndex
200      @return {Number} index in requestRangeIndex
201   */
202   wasRangeRequested: function(rangeStart) {
203     var i, ilen;
204     for(i=0, ilen=this.requestedRangeIndex.length; i<ilen; i++){
205       if(this.requestedRangeIndex[i]===rangeStart) return i;
206     }
207     return -1;
208   },
209   
210   /*
211     This method has to be called after a request for a range has completed.
212     To remove the index from the sparseArray to allow future updates on the 
213     range.
214     
215      @param {Number} startIndex
216      @return {Number} index in requestRangeIndex
217   */
218   rangeRequestCompleted: function(start) { 
219     var i = this.wasRangeRequested(start);
220     if(i>=0) { 
221       this.requestedRangeIndex.removeAt(i,1);
222       return YES;
223     }
224     return NO;
225   },
226   
227   /**
228     This method sets the content for the specified to the objects in the 
229     passed array.  If you change the way SparseArray implements its internal
230     tracking of objects, you should override this method along with 
231     objectAt().
232     
233     @param {Range} range the range to apply to
234     @param {Array} array the array of objects to insert
235     @returns {SC.SparseArray} reciever
236   */
237   provideObjectsInRange: function(range, array) {
238     var content = this._sa_content ;
239     if (!content) content = this._sa_content = [] ;
240     var start = range.start, len = range.length;
241     while(--len >= 0) content[start+len] = array[len];
242     if (this._requestingIndex <= 0) this.enumerableContentDidChange() ;
243     return this ;
244   },
245 
246   _TMP_PROVIDE_ARRAY: [],
247   _TMP_PROVIDE_RANGE: { length: 1 },
248   
249   /**
250     Convenience method to provide a single object at a specified index.  Under
251     the covers this calls provideObjectsInRange() so you can override only 
252     that method and this one will still work.
253     
254     @param {Number} index the index to insert
255     @param {Object} the object to insert
256     @return {SC.SparseArray} receiver
257   */
258   provideObjectAtIndex: function(index, object) {
259     var array = this._TMP_PROVIDE_ARRAY, range = this._TMP_PROVIDE_RANGE;
260     array[0] = object;
261     range.start = index;
262     return this.provideObjectsInRange(range, array);
263   },
264 
265   /**
266     Invalidates the array content in the specified range.  This is not the 
267     same as editing an array.  Rather it will cause the array to reload the
268     content from the delegate again when it is requested.
269     
270     @param {Range} the range
271     @returns {SC.SparseArray} receiver
272   */
273   objectsDidChangeInRange: function(range) {
274 
275     // delete cached content
276     var content = this._sa_content ;
277     if (content) {
278       // if range covers entire length of cached content, just reset array
279       if (range.start === 0 && SC.maxRange(range)>=content.length) {
280         this._sa_content = null ;
281         
282       // otherwise, step through the changed parts and delete them.
283       } else {
284         var start = range.start, loc = Math.min(start + range.length, content.length);
285         while (--loc>=start) content[loc] = undefined;
286       }
287     }    
288     this.enumerableContentDidChange(range) ; // notify
289     return this ;
290   },
291   
292   /**
293     Optimized version of indexOf().  Asks the delegate to provide the index 
294     of the specified object.  If the delegate does not implement this method
295     then it will search the internal array directly.
296     
297     @param {Object} obj the object to search for
298     @returns {Number} the discovered index or -1 if not found
299   */
300   indexOf: function(obj) {
301     var del = this.delegate ;
302     if (del && del.sparseArrayDidRequestIndexOf) {
303       return del.sparseArrayDidRequestIndexOf(this, obj);
304     } else {
305       var content = this._sa_content ;
306       if (!content) content = this._sa_content = [] ;
307       return content.indexOf(obj) ;
308     }
309   },  
310   
311   // ..........................................................
312   // EDITING
313   // 
314 
315   /**
316     Array primitive edits the objects at the specified index unless the 
317     delegate rejects the change.
318     
319     @param {Number} idx the index to begin to replace
320     @param {Number} amt the number of items to replace
321     @param {Array} objects the new objects to set instead
322     @returns {SC.SparseArray} receiver
323   */
324   replace: function(idx, amt, objects) {
325     objects = objects || [] ;
326 
327     // if we have a delegate, get permission to make the replacement.
328     var del = this.delegate ;
329     if (del) {
330       if (!del.sparseArrayShouldReplace || 
331           !del.sparseArrayShouldReplace(this, idx, amt, objects)) {
332             return this;
333       }
334     }
335 
336     // go ahead and apply to local content.
337     var content = this._sa_content ;
338     if (!content) content = this._sa_content = [] ;
339     content.replace(idx, amt, objects) ;
340     
341     // update length
342     var len = objects ? (objects.get ? objects.get('length') : objects.length) : 0;
343     var delta = len - amt ;
344     if (!SC.none(this._length)) {
345       this.propertyWillChange('length');
346       this._length += delta;
347       this.propertyDidChange('length');
348     }
349 
350     this.enumerableContentDidChange(idx, amt, delta) ;
351     return this ;
352   },
353 
354   /** 
355     Resets the SparseArray, causing it to reload its content from the 
356     delegate again.
357     
358     @returns {SC.SparseArray} receiver
359   */
360   reset: function() {
361     this._sa_content = null ;
362     this._length = null ;
363     this.enumerableContentDidChange() ;
364     this.invokeDelegateMethod(this.delegate, 'sparseArrayDidReset', this);
365     return this ;
366   }
367       
368 }) ;
