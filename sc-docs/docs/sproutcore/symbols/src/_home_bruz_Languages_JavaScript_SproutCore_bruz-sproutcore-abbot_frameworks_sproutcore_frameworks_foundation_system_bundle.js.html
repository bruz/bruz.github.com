 15     @property
 16     @default NO
 17     @type {Boolean}

 19     If YES, log bundle loading.

 23   /**
 24     Returns YES is bundleName is loaded; NO if bundleName is not loaded or
 25     no information is available.
 26     
 27     @param bundleName {String}
 28     @returns {Boolean}
 29   */
 30   bundleIsLoaded: function(bundleName) {
 31     var bundleInfo = SC.BUNDLE_INFO[bundleName] ;
 32     return bundleInfo ? !!bundleInfo.loaded : NO ;
 33   },
 34   
 35   /**
 36     @private
 37     
 38     Execute callback function.
 39   */
 40   _scb_bundleDidLoad: function(bundleName, target, method, args) {
 41     var m = method, t = target ;
 42 
 43     if(SC.typeOf(target) === SC.T_STRING) {
 44       t = SC.objectForPropertyPath(target);
 45     }
 46 
 47     if(SC.typeOf(method) === SC.T_STRING) {
 48       m = SC.objectForPropertyPath(method, t);
 49     }
 50     
 51     if(!m) {
 52 
 53       if(SC.LAZY_INSTANTIATION[bundleName]) {
 54         var lazyInfo = SC.LAZY_INSTANTIATION[bundleName];
 55 
 56       if (SC.logBundleLoading) console.log("SC.loadBundle(): Bundle '%@' is marked for lazy instantiation, instantiating it now…".fmt(bundleName));            
 57         
 58         for(var i=0, iLen = lazyInfo.length; i<iLen; i++) {
 59           try { 
 60             lazyInfo[i]();
 61           }catch(e) {
 62             console.log("SC.loadBundle(): Failted to lazily instatiate entry for  '%@'".fmt(bundleName));  
 63           }
 64         }
 65         delete SC.LAZY_INSTANTIATION[bundleName];
 66 
 67         if(SC.typeOf(target) === SC.T_STRING) {
 68           t = SC.objectForPropertyPath(target);
 69         }
 70         if(SC.typeOf(method) === SC.T_STRING) {
 71           m = SC.objectForPropertyPath(method, t);
 72         }
 73 
 74         if(!method) {
 75           throw "SC.loadBundle(): could not find callback for lazily instantiated bundle '%@'".fmt(bundleName);
 76 
 77         }
 78       } else {
 79         throw "SC.loadBundle(): could not find callback for '%@'".fmt(bundleName);
 80       }
 81     }
 82 
 83     if(!args) {
 84       args = [];
 85     }
 86 
 87     args.push(bundleName);
 88     
 89     var needsRunLoop = !!SC.RunLoop.currentRunLoop;
 90     if (needsRunLoop) SC.RunLoop.begin() ;
 91     m.apply(t, args) ;
 92     if (needsRunLoop) SC.RunLoop.end() 
 93   },
 94   
 95   tryToLoadBundle: function(bundleName, target, method, args) {
 96     var m, t;
 97     
 98     // First see if it is already defined.
 99     if(SC.typeOf(target) === SC.T_STRING) {
100       t = SC.objectForPropertyPath(target);
101     }
102     if(SC.typeOf(method) === SC.T_STRING) {
103       m = SC.objectForPropertyPath(method, t);
104     }
105 
106     // If the method exists, try to call it. It could have been loaded 
107     // through other means but the SC.BUNDLE_INFO entry doesn't exist.
108     if(m || SC.LAZY_INSTANTIATION[bundleName]) {
109       if(SC.logBundleLoading) console.log("SC.loadBundle(): Bundle '%@' found through other means, will attempt to load…".fmt(bundleName));
110       SC.BUNDLE_INFO[bundleName] = {loaded: YES};
111       return SC.BUNDLE_INFO[bundleName]; 
112     }
113     return NO;
114   },
115     
116   /**
117     Dynamically load bundleName if not already loaded. Call the target and 
118     method with any given arguments.
119     
120     @param bundleName {String}
121     @param target {Function} 
122     @param method {Function}
123   */
124   loadBundle: function(bundleName, target, method) {
125     var idx, len;
126     if(method === undefined && SC.typeOf(target) === SC.T_FUNCTION) {
127       method = target;
128       target = null;
129     }
130 
131     var bundleInfo = SC.BUNDLE_INFO[bundleName], callbacks, targets,
132         args       = SC.A(arguments).slice(3),
133         log        = SC.logBundleLoading;
134 
135     if (log) {
136       console.log("SC.loadBundle(): Attempting to load '%@'".fmt(bundleName));
137     }
138     
139     if (!bundleInfo) {
140       if (log) console.log("SC.loadBundle(): Attemping to load %@ without SC.BUNDLE_INFO entry… could be loaded through other means.".fmt(bundleName));
141       bundleInfo = this.tryToLoadBundle(bundleName, target, method, args);
142     }
143     
144 
145     if (!bundleInfo) {        
146       throw "SC.loadBundle(): could not find bundle '%@'".fmt(bundleName) ;
147     } else if (bundleInfo.loaded) {
148 
149       if (log) console.log("SC.loadBundle(): Bundle '%@' already loaded, skipping.".fmt(bundleName));
150 
151       if(method) {
152         // call callback immediately if we're already loaded and SC.isReady
153         if (SC.isReady) {
154           SC._scb_bundleDidLoad(bundleName, target, method, args);
155         } else {
156           // queue callback for when SC is ready
157           SC.ready(SC, function() {
158             SC._scb_bundleDidLoad(bundleName, target, method, args);        
159           });
160         }
161       }
162     } else {
163 
164       if (log) console.log("SC.loadBundle(): Bundle '%@' is not loaded, loading now.".fmt(bundleName));
165 
166       // queue callback for later
167       callbacks = bundleInfo.callbacks || [] ;
168 
169       if (method) {
170         callbacks.push(function() {
171           SC._scb_bundleDidLoad(bundleName, target, method, args);        
172         });
173         bundleInfo.callbacks = callbacks ;
174       }
175 
176       if (!bundleInfo.loading) {
177         // load bundle's dependencies first
178         var requires = bundleInfo.requires || [] ;
179         var dependenciesMet = YES ;
180         for (idx=0, len=requires.length; idx<len; ++idx) {
181           var targetName = requires[idx] ;
182           var targetInfo = SC.BUNDLE_INFO[targetName] ;
183           if (!targetInfo) {
184             throw "SC.loadBundle(): could not find required bundle '%@' for bundle '%@'".fmt(targetName, bundleName) ;
185           } else {
186             if (targetInfo.loading) {
187               dependenciesMet = NO ;
188               break ;
189             } else if (targetInfo.loaded) {
190               continue ;
191             } else {
192               dependenciesMet = NO ;
193               
194               // register ourself as a dependent bundle (used by 
195               // SC.bundleDidLoad()...)
196               var dependents = targetInfo.dependents;
197               if(!dependents) targetInfo.dependents = dependents = [];
198 
199               dependents.push(bundleName) ;
200 
201               if (log) console.log("SC.loadBundle(): '%@' depends on '%@', loading dependency…".fmt(bundleName, targetName));
202               
203               // recursively load targetName so it's own dependencies are
204               // loaded first.
205               SC.loadBundle(targetName) ;
206               break ;
207             }
208           }
209         }
210         
211         if (dependenciesMet) {
212           // add <script> and <link> tags to DOM for bundle's resources
213           var styles, scripts, url, el, head, body;
214           head = document.getElementsByTagName('head')[0] ;
215           if (!head) head = document.documentElement ; // fix for Opera
216           styles = bundleInfo.styles || [] ;
217           for (idx=0, len=styles.length; idx<len; ++idx) {
218             url = styles[idx] ;
219             if (url.length > 0) {
220               el = document.createElement('link') ;
221               el.setAttribute('href', url) ;
222               el.setAttribute('rel', "stylesheet") ;
223               el.setAttribute('type', "text/css") ;
224               head.appendChild(el) ;
225             }
226           }
227 
228           // Push the URLs on the the queue and then start the loading.
229           var jsBundleLoadQueue = this._jsBundleLoadQueue;
230           if(!jsBundleLoadQueue) this._jsBundleLoadQueue = jsBundleLoadQueue = {};
231           jsBundleLoadQueue[bundleName] = [];
232           var q = jsBundleLoadQueue[bundleName] ;
233           scripts = bundleInfo.scripts || [] ;
234           
235           for (idx=0, len=scripts.length; idx<len; ++idx) {
236             url = scripts[idx] ;
237             if (url.length > 0) {
238               q.push(url);
239             }
240           }
241           
242           // and remember that we're loading
243           bundleInfo.loading = YES ;
244           
245           // Start the load process.
246           this.scriptDidLoad(bundleName);
247         }
248       }
249     }
250   },
251 
252   /**
253     Load the next script in the queue now that the caller of this function
254     is complete.
255     
256     @param {String} bundleName The name of the bundle.
257   */
258   scriptDidLoad: function(bundleName) {
259     var jsBundleLoadQueue = this._jsBundleLoadQueue;
260     if(jsBundleLoadQueue) {
261       var q = jsBundleLoadQueue[bundleName];
262       if(q) {
263         var url = q.shift();
264         
265         if (SC.logBundleLoading) console.log("SC.scriptDidLoad(): Loading next file in '%@' -> '%@'".fmt(bundleName, url));
266 
267         var el = document.createElement('script') ;
268         el.setAttribute('type', "text/javascript") ;
269         el.setAttribute('src', url) ;
270         document.body.appendChild(el) ;
271       }
272     }
273   },
274   
275   /** @private
276     Called by bundle_loaded.js immediately after a framework/bundle is loaded.
277     Any pending callbacks are called (if SC.isReady), and any dependent 
278     bundles which were waiting for this bundle to load are notified so they 
279     can continue loading.
280     
281     @param bundleName {String} the name of the bundle that just loaded
282   */
283   bundleDidLoad: function(bundleName) {
284     var bundleInfo = SC.BUNDLE_INFO[bundleName], 
285         log        = SC.logBundleLoading,
286         callbacks, targets ;
287     if (!bundleInfo) {
288       bundleInfo = SC.BUNDLE_INFO[bundleName] = { loaded: YES} ;
289       return;
290     }
291     if (bundleInfo.loaded && log) {
292       console.log("SC.bundleDidLoad() called more than once for bundle '%@'. Skipping.".fmt(bundleName));
293       return ;
294     }
295     
296     // remember that we're loaded
297     delete bundleInfo.loading ;
298     bundleInfo.loaded = YES ;
299     
300     // call our callbacks (if SC.isReady), otherwise queue them for later
301     if (SC.isReady) {
302       SC._invokeCallbacksForBundle(bundleName) ;
303     } else {
304       SC.ready(SC, function() {
305         SC._invokeCallbacksForBundle(bundleName) ;
306       });
307     }
308     
309     // for each dependent bundle, try and load them again...
310     var dependents = bundleInfo.dependents || [] ;
311     for (var idx=0, len=dependents.length; idx<len; ++idx) {
312       if (log) console.log("SC.loadBundle(): Bundle '%@' has completed loading, loading '%@' that depended on it.".fmt(bundleName, dependents[idx]));
313       SC.loadBundle(dependents[idx]) ;
314     }
315   },
316   
317   /** @private Invoke queued callbacks for bundleName. */
318   _invokeCallbacksForBundle: function(bundleName) {
319     var bundleInfo = SC.BUNDLE_INFO[bundleName], callbacks ;
320     if (!bundleInfo) return ; // shouldn't happen, but recover anyway
321     
322     if (SC.logBundleLoading) console.log("SC.loadBundle(): Bundle '%@' has completed loading, invoking callbacks.".fmt(bundleName));
323 
324     callbacks = bundleInfo.callbacks || [] ;
325     
326     SC.RunLoop.begin() ;
327     for (var idx=0, len=callbacks.length; idx<len; ++idx) {
328       callbacks[idx]() ;
329     }
330     SC.RunLoop.end() ;
331   }
