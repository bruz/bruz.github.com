 51 /** @class
 52 
 53   A Scanner reads a string and interprets the characters into numbers. You
 54   assign the scanner's string on initialization and the scanner progresses
 55   through the characters of that string from beginning to end as you request
 56   items.
 57   
 58   Scanners are used by DateTime to convert strings into DateTime objects.
 59   
 60   @extends SC.Object
 61   @since SproutCore 1.0
 62   @author Martin Ottenwaelter
 63 */
 64 SC.Scanner = SC.Object.extend(
 65 /** @scope SC.Scanner.prototype */ {
 66   
 67   /**
 68     The string to scan. You usually pass it to the create method:
 69     
 70     {{{
 71       SC.Scanner.create({string: 'May, 8th'});
 72     }}}
 73     
 74     @property
 75     @type {String}
 76   */
 77   string: null,
 78   
 79   /**
 80     The current scan location. It is incremented by the scanner as the
 81     characters are processed.
 82     The default is 0: the beginning of the string.
 83     
 84     @property
 85     @type {integer}
 86   */
 87   scanLocation: 0,
 88   
 89   /**
 90     Reads some characters from the string, and increments the scan location
 91     accordingly. 
 92     
 93     @param {integer} len the amount of characters to read
 94     @throws {SC.SCANNER_OUT_OF_BOUNDS_ERROR} if asked to read too many characters
 95     @returns {String} the characters
 96   */
 97   scan: function(len) {
 98     if (this.scanLocation + len > this.length) throw SC.SCANNER_OUT_OF_BOUNDS_ERROR;
 99     var str = this.string.substr(this.scanLocation, len);
100     this.scanLocation += len;
101     return str;
102   },
103   
104   /**
105     Reads some characters from the string and interprets it as an integer.
106     
107     @param {integer} len the amount of characters to read
108     @throws {SC.SCANNER_INT_ERROR} if asked to read non numeric characters
109     @returns {integer} the scanned integer
110   */
111   scanInt: function(len) {
112     var str = this.scan(len);
113     var re = new RegExp("\\d{"+len+"}");
114     if (!str.match(re)) throw SC.SCANNER_INT_ERROR;
115     return parseInt(str, 10);
116   },
117   
118   /**
119     Attempts to skip a given string.
120     
121     @param {String} str the string to skip
122     @throws {SC.SCANNER_SKIP_ERROR} if the given string could not be scanned
123     @returns {Boolean} YES if the given string was successfully scanned
124   */
125   skipString: function(str) {
126     if (this.scan(str.length) !== str) throw SC.SCANNER_SKIP_ERROR;
127     return YES;
128   },
129   
130   /**
131     Attempts to scan any string in a given array.
132     
133     @param {Array} ary the array of strings to scan
134     @throws {SC.SCANNER_SCAN_ARRAY_ERROR} if no string of the given array is found
135     @returns {integer} the index of the scanned string of the given array
136   */
137   scanArray: function(ary) {
138     for (var i = 0, len = ary.length; i < len; i++) {
139       if (this.scan(ary[i].length) === ary[i]) {
140         return i;
141       }
142       this.scanLocation -= ary[i].length;
143     }
144     throw SC.SCANNER_SCAN_ARRAY_ERROR;
145   }
146   
147 });

150 /** @class
151 
152   A class representation of a date and time. It's basically a wrapper around
153   the Date javascript object, KVO friendly and with common date/time
154   manipulation methods.
155 
156   This object differs from the standard JS Date object, however, in that it
157   supports time zones other than UTC and that local to the machine on which
158   it is running.  Any time zone can be specified when creating an SC.DateTime
159   object, e.g
160 
161     // Creates a DateTime representing 5am in Washington, DC and 10am in London
162     var d = SC.DateTime.create({ hour: 5, timezone: 300 }); // -5 hours from UTC
163     var e = SC.DateTime.create({ hour: 10, timezone: 0 }); // same time, specified in UTC
164     
165     and it is true that d.isEqual(e).
166 
167   The time zone specified upon creation is permanent, and any calls to get() on that
168   instance will return values expressed in that time zone.  So,
169   
170     d.get('hour') returns 5.
171     e.get('hour') returns 10.
172     
173     but
174     
175     d.get('milliseconds') === e.get('milliseconds') is true, since they are technically the same position in time.
176   
177   @extends SC.Object
178   @extends SC.Freezable
179   @extends SC.Copyable
180   @author Martin Ottenwaelter
181   @author Jonathan Lewis
182   @author Josh Holt
183   @since SproutCore 1.0
184 */
185 SC.DateTime = SC.Object.extend(SC.Freezable, SC.Copyable,
186   /** @scope SC.DateTime.prototype */ {
187   
188   /** @private
189     Internal representation of a date: the number of milliseconds
190     since January, 1st 1970 00:00:00.0 UTC.
191     
192     @property
193     @type {Integer}
194   */
195   _ms: 0,
196   
197   /** @read-only
198     The offset, in minutes, between UTC and the object's timezone.
199     All calls to get() will use this time zone to translate date/time
200     values into the zone specified here.
201     
202     @property
203     @type {Integer}
204   */
205   timezone: 0,
206   
207   /**
208     A DateTime instance is frozen by default for better performance.
209     
210     @property
211     @type {Boolean}
212   */
213   isFrozen: YES,
214   
215   /**
216     Returns a new DateTime object where one or more of the elements have been
217     changed according to the options parameter. The time options (hour,
218     minute, sec, usec) reset cascadingly, so if only the hour is passed, then
219     minute, sec, and usec is set to 0. If the hour and minute is passed, then
220     sec and usec is set to 0.
221     
222     (Parts copied from the Ruby On Rails documentation)
223     
224     If a time zone is passed in the options hash, all dates and times are assumed
225     to be local to it, and the returned DateTime instance has that time zone.  If
226     none is passed, it defaults to SC.DateTime.timezone.
227 
228     Note that passing only a time zone does not affect the actual milliseconds since
229     Jan 1, 1970, only the time zone in which it is expressed when displayed.
230     
231     @see SC.DateTime#create for the list of options you can pass
232     @returns {DateTime} copy of receiver
233   */
234   adjust: function(options, resetCascadingly) {
235     var timezone;
236     
237     options = options ? SC.clone(options) : {};
238     timezone = (options.timezone !== undefined) ? options.timezone : (this.timezone !== undefined) ? this.timezone : 0;
239     
240     return this.constructor._adjust(options, this._ms, timezone, resetCascadingly)._createFromCurrentState();
241   },
242   
243   /**
244     Returns a new DateTime object advanced according the the given parameters.
245     Don't use floating point values, it might give unpredicatble results.
246     
247     @see SC.DateTime#create for the list of options you can pass
248     @param {Hash} options the amount of date/time to advance the receiver
249     @returns {DateTime} copy of the receiver
250   */
251   advance: function(options) {
252     return this.constructor._advance(options, this._ms, this.timezone)._createFromCurrentState();
253   },
254   
255   /**
256     Generic getter.
257     
258     The properties you can get are:
259       - 'year'
260       - 'month' (January is 1, contrary to JavaScript Dates for which January is 0)
261       - 'day'
262       - 'dayOfWeek' (Sunday is 0)
263       - 'hour'
264       - 'minute'
265       - 'second'
266       - 'millisecond'
267       - 'milliseconds', the number of milliseconds since
268         January, 1st 1970 00:00:00.0 UTC
269       - 'isLeapYear', a boolean value indicating whether the receiver's year
270         is a leap year
271       - 'daysInMonth', the number of days of the receiver's current month
272       - 'dayOfYear', January 1st is 1, December 31th is 365 for a common year
273       - 'week' or 'week1', the week number of the current year, starting with
274         the first Sunday as the first day of the first week (00..53)
275       - 'week0', the week number of the current year, starting with
276         the first Monday as the first day of the first week (00..53)
277       - 'lastMonday', 'lastTuesday', etc., 'nextMonday', 'nextTuesday', etc.,
278         the date of the last or next weekday in comparison to the receiver.
279     
280     @param {String} key the property name to get
281     @return the value asked for
282   */
283   unknownProperty: function(key) {
284     return this.constructor._get(key, this._ms, this.timezone);
285   },
286   
287   /**
288     Formats the receiver according to the given format string. Should behave
289     like the C strftime function.
290   
291     The format parameter can contain the following characters:
292       - %a - The abbreviated weekday name (``Sun'')
293       - %A - The full weekday name (``Sunday'')
294       - %b - The abbreviated month name (``Jan'')
295       - %B - The full month name (``January'')
296       - %c - The preferred local date and time representation
297       - %d - Day of the month (01..31)
298       - %h - Hour of the day, 24-hour clock (0..23)
299       - %H - Hour of the day, 24-hour clock (00..23)
300       - %i - Hour of the day, 12-hour clock (1..12)
301       - %I - Hour of the day, 12-hour clock (01..12)
302       - %j - Day of the year (001..366)
303       - %m - Month of the year (01..12)
304       - %M - Minute of the hour (00..59)
305       - %p - Meridian indicator (``AM'' or ``PM'')
306       - %S - Second of the minute (00..60)
307       - %U - Week number of the current year,
308           starting with the first Sunday as the first
309           day of the first week (00..53)
310       - %W - Week number of the current year,
311           starting with the first Monday as the first 
312           day of the first week (00..53)
313       - %w - Day of the week (Sunday is 0, 0..6)
314       - %x - Preferred representation for the date alone, no time
315       - %X - Preferred representation for the time alone, no date
316       - %y - Year without a century (00..99)
317       - %Y - Year with century
318       - %Z - Time zone (ISO 8601 formatted)
319       - %% - Literal ``%'' character
320     
321     @param {String} format the format string
322     @return {String} the formatted string
323   */
324   toFormattedString: function(fmt) {
325     return this.constructor._toFormattedString(fmt, this._ms, this.timezone);
326   },
327   
328   /**
329     Formats the receiver according ISO 8601 standard. It is equivalent to
330     calling toFormattedString with the '%Y-%m-%dT%H:%M:%S%Z' format string.
331     
332     @return {String} the formatted string
333   */
334   toISO8601: function(){
335     return this.constructor._toFormattedString(SC.DATETIME_ISO8601, this._ms, this.timezone);
336   },
337   
338   /** @private
339     Creates a string representation of the receiver.
340     (Debuggers call all the time the toString method. Because of the way
341     DateTime is designed, calling SC.DateTime._toFormattedString would
342     have a nasty side effect. We shouldn't therefore call any of SC.DateTime's
343     methods from toString)
344     
345     @returns {String}
346   */
347   toString: function() {
348     return "UTC: " +
349            new Date(this._ms).toUTCString() +
350            ", timezone: " +
351            this.timezone;
352   },
353   
354   /**
355     Returns YES if the passed DateTime is equal to the receiver, ie: if their
356     number of milliseconds since January, 1st 1970 00:00:00.0 UTC are equal.
357     This is the preferred method for testing equality.
358   
359     @see SC.DateTime#compare
360     @param {SC.DateTime} aDateTime the DateTime to compare to
361     @returns {Boolean}
362   */
363   isEqual: function(aDateTime) {
364     return SC.DateTime.compare(this, aDateTime) === 0;
365   },
366   
367   /**
368     Returns a copy of the receiver. Because of the way DateTime is designed,
369     it just returns the receiver.
370     
371     @returns {DateTime}
372   */
373   copy: function() {
374     return this;
375   },
376   
377   /**
378     Returns a copy of the receiver with the timezone set to the passed
379     timezone. The returned value is equal to the receiver (ie SC.Compare
380     returns 0), it is just the timezone representation that changes.
381     
382     If you don't pass any argument, the target timezone is assumed to be 0,
383     ie UTC.
384 
385     Note that this method does not change the underlying position in time,
386     but only the time zone in which it is displayed.  In other words, the underlying
387     number of milliseconds since Jan 1, 1970 does not change.
388     
389     @return {DateTime}
390   */
391   toTimezone: function(timezone) {
392     if (timezone === undefined) timezone = 0;
393     return this.advance({ timezone: timezone - this.timezone });
394   }
395   
396 });

528   /** @private
529     Both args are optional, but will only overwrite _date and _tz if defined.
530     This method does not affect the DateTime instance's actual time, but simply

532     (this._date is just a resource optimization)

534     This is mainly used as a way to store a recursion starting state during

536

538     'timezone' is the current time zone we want to be working in internally.

541     are wanted for later restoration.

557     'timezone' will be used, or the last this._tz.

560     var tz = (timezone !== undefined) ? timezone : this._tz; // use the last-known time zone if necessary

562     return this._setCalcState(ms, tz); // now call the one we really wanted

565   /** @private
566     @see SC.DateTime#unknownProperty
567   */
568   _get: function(key, start, timezone) {
569     var ms, tz, doy, m, y, firstDayOfWeek, dayOfWeek, dayOfYear, prefix, suffix;
570     var currentWeekday, targetWeekday;
571     var d = this._date;
572     var originalTime, v = null;
573 
574     // Set up an absolute date/time using the given milliseconds since Jan 1, 1970.
575     // Only do it if we're given a time value, though, otherwise we want to use the
576     // last one we had because this _get() method is recursive.
577     //
578     // Note that because these private time calc methods are recursive, and because all DateTime instances
579     // share an internal this._date and this._tz state for doing calculations, methods
580     // that modify this._date or this._tz should restore the last state before exiting
581     // to avoid obscure calculation bugs.  So we save the original state here, and restore
582     // it before returning at the end.
583     originalTime = this._setCalcState(start, timezone); // save so we can restore it to how it was before we got here
584 
585     // Check this first because it is an absolute value -- no tweaks necessary when calling for milliseconds
586     if (key === 'milliseconds') {
587       v = d.getTime();
588     }
589     else if (key === 'timezone') {
590       v = this._tz;
591     }
592     
593     // 'nextWeekday' or 'lastWeekday'.
594     // We want to do this calculation in local time, before shifting UTC below.
595     if (v === null) {
596       prefix = key.slice(0, 4);
597       suffix = key.slice(4);
598       if (prefix === 'last' || prefix === 'next') {
599         currentWeekday = this._get('dayOfWeek');
600         targetWeekday = this._englishDayNames.indexOf(suffix);    
601         if (targetWeekday >= 0) {
602           var delta = targetWeekday - currentWeekday;
603           if (prefix === 'last' && delta >= 0) delta -= 7;
604           if (prefix === 'next' && delta <  0) delta += 7;
605           this._advance({ day: delta }, start, timezone);
606           v = this._createFromCurrentState();
607         }
608       }
609     }
610     
611     if (v === null) {
612       // need to adjust for alternate display time zone.
613       // Before calculating, we need to get everything into a common time zone to
614       // negate the effects of local machine time (so we can use all the 'getUTC...() methods on Date).
615       if (timezone !== undefined) {
616         this._setCalcState(d.getTime() - (timezone * 60000), 0); // make this instance's time zone the new UTC temporarily
617       }
618     
619       // simple keys
620       switch (key) {
621         case 'year':
622           v = d.getUTCFullYear(); //TODO: investigate why some libraries do getFullYear().toString() or getFullYear()+""
623           break;
624         case 'month':
625           v = d.getUTCMonth()+1; // January is 0 in JavaScript
626           break;
627         case 'day':
628           v = d.getUTCDate();
629           break;
630         case 'dayOfWeek':
631           v = d.getUTCDay();
632           break;
633         case 'hour':
634           v = d.getUTCHours();
635           break;
636         case 'minute':
637           v = d.getUTCMinutes();
638           break;
639         case 'second':
640           v = d.getUTCSeconds();
641           break;
642         case 'millisecond':
643           v = d.getUTCMilliseconds();
644           break;
645       }
646       
647       // isLeapYear
648       if ((v === null) && (key === 'isLeapYear')) {
649         y = this._get('year');
650         v = (y%4 === 0 && y%100 !== 0) || y%400 === 0;
651       }
652 
653       // daysInMonth
654       if ((v === null) && (key === 'daysInMonth')) {
655         switch (this._get('month')) {
656           case 4:
657           case 6:
658           case 9:
659           case 11:
660             v = 30;
661             break;
662           case 2:
663             v = this._get('isLeapYear') ? 29 : 28;
664             break;
665           default:
666             v = 31;
667             break;
668         }
669       }
670     
671       // dayOfYear
672       if ((v === null) && (key === 'dayOfYear')) {
673         ms = d.getTime(); // save time
674         doy = this._get('day');
675         this._setCalcStateFromHash({ day: 1 });
676         for (m = this._get('month') - 1; m > 0; m--) {
677           this._setCalcStateFromHash({ month: m });
678           doy += this._get('daysInMonth');
679         }
680         d.setTime(ms); // restore time
681         v = doy;
682       }
683 
684       // week, week0 or week1
685       if ((v === null) && (key.slice(0, 4) === 'week')) {
686         // firstDayOfWeek should be 0 (Sunday) or 1 (Monday)
687         firstDayOfWeek = key.length === 4 ? 1 : parseInt(key.slice('4'), 10);
688         dayOfWeek = this._get('dayOfWeek');
689         dayOfYear = this._get('dayOfYear') - 1;
690         if (firstDayOfWeek === 0) {
691           v = parseInt((dayOfYear - dayOfWeek + 7) / 7, 10);
692         }
693         else {
694           v = parseInt((dayOfYear - (dayOfWeek - 1 + 7) % 7 + 7) / 7, 10);
695         }
696       }
697     }
698 
699     // restore the internal calculation state in case someone else was in the
700     // middle of a calculation (we might be recursing).
701     this._setCalcState(originalTime.milliseconds, originalTime.timezone);
702 
703     return v;
704   },

706   /**
707     Sets the internal calculation state to something specified.
708   */
709   _adjust: function(options, start, timezone, resetCascadingly) {
710     var opts = options ? SC.clone(options) : {};

734   /* @private
735     Converts a standard date/time options hash to an integer representing that position
736     in time relative to Jan 1, 1970
737   */
738   _toMilliseconds: function(options, start, timezone, resetCascadingly) {
739     var opts = options ? SC.clone(options) : {};
740     var d = this._date;
741     var previousMilliseconds = d.getTime(); // rather than create a new Date object, we'll reuse the instance we have for calculations, then restore it
742     var ms, tz;
743 
744     // Initialize our internal for-calculations Date object to our current date/time.
745     // Note that this object was created in the local machine time zone, so when we set
746     // its params later, it will be assuming these values to be in the same time zone as it is.
747     // It's ok for start to be null, in which case we'll just keep whatever we had in 'd' before.
748     if (!SC.none(start)) {
749       d.setTime(start); // using milliseconds here specifies an absolute location in time, regardless of time zone, so that's nice
750     }
751     
752     // We have to get all time expressions, both in 'options' (assume to be in time zone 'timezone')
753     // and in 'd', to the same time zone before we can any calculations correctly.  So because the Date object provides
754     // a suite of UTC getters and setters, we'll temporarily redefine 'timezone' as our new
755     // 'UTC', so we don't have to worry about local machine time.  We do this by subtracting
756     // milliseconds for the time zone offset.  Then we'll do all our calculations, then convert
757     // it back to real UTC.
758     
759     // (Zero time zone is considered a valid value.)
760     tz = (timezone !== undefined) ? timezone : (this.timezone !== undefined) ? this.timezone : 0;
761     d.setTime(d.getTime() - (tz * 60000)); // redefine 'UTC' to establish a new local absolute so we can use all the 'getUTC...()' Date methods
762     
763     // the time options (hour, minute, sec, millisecond)
764     // reset cascadingly (see documentation)
765     if (resetCascadingly === undefined || resetCascadingly === YES) {
766       if ( !SC.none(opts.hour) && SC.none(opts.minute)) {
767         opts.minute = 0;
768       }
769       if (!(SC.none(opts.hour) && SC.none(opts.minute))
770           && SC.none(opts.second)) {
771         opts.second = 0;
772       }
773       if (!(SC.none(opts.hour) && SC.none(opts.minute) && SC.none(opts.second))
774           && SC.none(opts.millisecond)) {
775         opts.millisecond = 0;
776       }
777     }
778     
779     // Get the current values for any not provided in the options hash.
780     // Since everything is in 'UTC' now, use the UTC accessors.  We do this because,
781     // according to javascript Date spec, you have to set year, month, and day together
782     // if you're setting any one of them.  So we'll use the provided Date.UTC() method
783     // to get milliseconds, and we need to get any missing values first...
784     if (SC.none(opts.year))        opts.year = d.getUTCFullYear();
785     if (SC.none(opts.month))       opts.month = d.getUTCMonth() + 1; // January is 0 in JavaScript
786     if (SC.none(opts.day))         opts.day = d.getUTCDate();
787     if (SC.none(opts.hour))        opts.hour = d.getUTCHours();
788     if (SC.none(opts.minute))      opts.minute = d.getUTCMinutes();
789     if (SC.none(opts.second))      opts.second = d.getUTCSeconds();
790     if (SC.none(opts.millisecond)) opts.millisecond = d.getUTCMilliseconds();
791 
792     // Ask the JS Date to calculate milliseconds for us (still in redefined UTC).  It
793     // is best to set them all together because, for example, a day value means different things
794     // to the JS Date object depending on which month or year it is.  It can now handle that stuff
795     // internally as it's made to do.
796     ms = Date.UTC(opts.year, opts.month - 1, opts.day, opts.hour, opts.minute, opts.second, opts.millisecond);
797 
798     // Now that we've done all our calculations in a common time zone, add back the offset
799     // to move back to real UTC.
800     d.setTime(ms + (tz * 60000));
801     ms = d.getTime(); // now get the corrected milliseconds value
802     
803     // Restore what was there previously before leaving in case someone called this method
804     // in the middle of another calculation.
805     d.setTime(previousMilliseconds);
806 
807     return ms;
808   },

810   /**
811     Returns a new DateTime object advanced according the the given parameters.
812     The parameters can be:
813     - none, to create a DateTime instance initialized to the current
814       date and time in the local timezone,
815     - a integer, the number of milliseconds since
816       January, 1st 1970 00:00:00.0 UTC
817     - a options hash that can contain any of the following properties: year,
818       month, day, hour, minute, second, millisecond, timezone
819       
820     Note that if you attempt to create a SC.DateTime instance that has already
821     been created, then, for performance reasons, a cached value may be
822     returned.
823     
824     The timezone option is the offset, in minutes, between UTC and local time.
825     If you don't pass a timezone option, the date object is created in the
826     local timezone. If you want to create a UTC+2 (CEST) date, for example,
827     then you should pass a timezone of -120.
828     
829     @param options one of the three kind of parameters descibed above
830     @returns {DateTime} the DateTime instance that corresponds to the
831       passed parameters, possibly fetched from cache
832   */
833   create: function() {
834     var arg = arguments.length === 0 ? {} : arguments[0];
835     var timezone;
836     
837     // if simply milliseconds since Jan 1, 1970 are given, just use those
838     if (SC.typeOf(arg) === SC.T_NUMBER) {
839       arg = { milliseconds: arg };
840     }
841 
842     // Default to local machine time zone if none is given
843     timezone = (arg.timezone !== undefined) ? arg.timezone : this.timezone;
844     if (timezone === undefined) timezone = 0;
845 
846     // Desired case: create with milliseconds if we have them.
847     // If we don't, convert what we have to milliseconds and recurse.
848     if (!SC.none(arg.milliseconds)) {
849 
850       // quick implementation of a FIFO set for the cache
851       var key = 'nu' + arg.milliseconds + timezone, cache = this._dt_cache;
852       var ret = cache[key];
853       if (!ret) {
854         var previousKey, idx = this._dt_cache_index, C = this;
855         ret = cache[key] = new C([{ _ms: arg.milliseconds, timezone: timezone }]);
856         idx = this._dt_cache_index = (idx + 1) % this._DT_CACHE_MAX_LENGTH;
857         previousKey = cache[idx];
858         if (previousKey !== undefined && cache[previousKey]) delete cache[previousKey];
859         cache[idx] = key;
860       }
861       return ret;
862     }
863     // otherwise, convert what we have to milliseconds and try again
864     else {
865       var now = new Date();
866 
867       return this.create({ // recursive call with new arguments
868         milliseconds: this._toMilliseconds(arg, now.getTime(), timezone, arg.resetCascadingly),
869         timezone: timezone
870       });
871     }
872   },

886   /**
887     Returns a DateTime object created from a given string parsed with a given
888     format. Returns null if the parsing fails.
889 
890     @see SC.DateTime#toFormattedString for a description of the format parameter
891     @param {String} str the string to parse
892     @param {String} fmt the format to parse the string with
893     @returns {DateTime} the DateTime corresponding to the string parameter
894   */
895   parse: function(str, fmt) {
896     var re = /(?:\%([aAbBcdHIjmMpSUWwxXyYZ\%])|(.))/g;
897     var d, parts, opts = {}, check = {}, scanner = SC.Scanner.create({string: str});
898     try {
899       while ((parts = re.exec(fmt)) !== null) {
900         switch(parts[1]) {
901           case 'a': check.dayOfWeek = scanner.scanArray(this.abbreviatedDayNames); break;
902           case 'A': check.dayOfWeek = scanner.scanArray(this.dayNames); break;
903           case 'b': opts.month = scanner.scanArray(this.abbreviatedMonthNames) + 1; break;
904           case 'B': opts.month = scanner.scanArray(this.monthNames) + 1; break;
905           case 'c': throw "%c is not implemented";
906           case 'd': opts.day = scanner.scanInt(2); break;
907           case 'H': opts.hour = scanner.scanInt(2); break;
908           case 'I': opts.hour = scanner.scanInt(2); break;
909           case 'j': throw "%j is not implemented";
910           case 'm': opts.month = scanner.scanInt(2); break;
911           case 'M': opts.minute = scanner.scanInt(2); break;
912           case 'p': opts.meridian = scanner.scanArray(['AM', 'PM']); break;
913           case 'S': opts.second = scanner.scanInt(2); break;
914           case 'U': throw "%U is not implemented";
915           case 'W': throw "%W is not implemented";
916           case 'w': throw "%w is not implemented";
917           case 'x': throw "%x is not implemented";
918           case 'X': throw "%X is not implemented";
919           case 'y': opts.year = scanner.scanInt(2); opts.year += (opts.year > 70 ? 1900 : 2000); break;
920           case 'Y': opts.year = scanner.scanInt(4); break;
921           case 'Z':
922             var modifier = scanner.scan(1);
923             if (modifier === 'Z') {
924               opts.timezone = 0;
925             } else if (modifier === '+' || modifier === '-' ) {
926               var h = scanner.scanInt(2);
927               if (scanner.scan(1) !== ':') scanner.scan(-1);
928               var m = scanner.scanInt(2);
929               opts.timezone = (modifier === '+' ? -1 : 1) * (h*60 + m);
930             }
931             break;
932           case '%': scanner.skipString('%'); break;
933           default:  scanner.skipString(parts[0]); break;
934         }
935       }
936     } catch (e) {
937       console.log('SC.DateTime.createFromString ' + e.toString());
938       return null;
939     }
940     
941     if (!SC.none(opts.meridian) && !SC.none(opts.hour)) {
942       if (opts.meridian === 1) opts.hour = (opts.hour + 12) % 24;
943       delete opts.meridian;
944     }
945     
946     d = SC.DateTime.create(opts);
947     
948     if (!SC.none(check.dayOfWeek) && d.get('dayOfWeek') !== check.dayOfWeek) {
949       return null;
950     }
951     
952     return d;
953   },

970   /** @private
971     @see SC.DateTime#_toFormattedString
972   */
973   __toFormattedString: function(part, start, timezone) {
974     var hour, offset;
975 
976     // Note: all calls to _get() here should include only one
977     // argument, since _get() is built for recursion and behaves differently
978     // if arguments 2 and 3 are included.
979     //
980     // This method is simply a helper for this._toFormattedString() (one underscore);
981     // this is only called from there, and _toFormattedString() has already
982     // set up the appropriate internal date/time/timezone state for it.
983     
984     switch(part[1]) {
985       case 'a': return this.abbreviatedDayNames[this._get('dayOfWeek')];
986       case 'A': return this.dayNames[this._get('dayOfWeek')];
987       case 'b': return this.abbreviatedMonthNames[this._get('month')-1];
988       case 'B': return this.monthNames[this._get('month')-1];
989       case 'c': return this._date.toString();
990       case 'd': return this._pad(this._get('day'));
991       case 'D': return this._get('day');
992       case 'h': return this._get('hour');
993       case 'H': return this._pad(this._get('hour'));
994       case 'i':
995         hour = this._get('hour');
996         return (hour === 12 || hour === 0) ? 12 : (hour + 12) % 12;
997       case 'I': 
998         hour = this._get('hour');
999         return this._pad((hour === 12 || hour === 0) ? 12 : (hour + 12) % 12);
1000       case 'j': return this._pad(this._get('dayOfYear'), 3);
1001       case 'm': return this._pad(this._get('month'));
1002       case 'M': return this._pad(this._get('minute'));
1003       case 'p': return this._get('hour') > 11 ? 'PM' : 'AM';
1004       case 'S': return this._pad(this._get('second'));
1005       case 'u': return this._pad(this._get('utc')); //utc
1006       case 'U': return this._pad(this._get('week0'));
1007       case 'W': return this._pad(this._get('week1'));
1008       case 'w': return this._get('dayOfWeek');
1009       case 'x': return this._date.toDateString();
1010       case 'X': return this._date.toTimeString();
1011       case 'y': return this._pad(this._get('year') % 100);
1012       case 'Y': return this._get('year');
1013       case 'Z':
1014         offset = -1 * timezone;
1015         return (offset >= 0 ? '+' : '-')
1016                + this._pad(parseInt(Math.abs(offset)/60, 10))
1017                + ':'
1018                + this._pad(Math.abs(offset)%60);
1019       case '%': return '%';
1020     }
1021   },
