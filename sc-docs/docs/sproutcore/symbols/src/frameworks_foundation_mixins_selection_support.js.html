  7 /**
  8   @namespace
  9   
 10   Implements common selection management properties for controllers.
 11   
 12   Selection can be managed by any controller in your applications.  This
 13   mixin provides some common management features you might want such as
 14   disabling selection, or restricting empty or multiple selections.
 15   
 16   To use this mixin, simply add it to any controller you want to manage 
 17   selection and call updateSelectionAfterContentChange()
 18   whenever your source content changes.  You can also override the properties
 19   defined below to configure how the selection management will treat your 
 20   content.
 21   
 22   This mixin assumes the arrangedObjects property will return an SC.Array of 
 23   content you want the selection to reflect.
 24   
 25   Add this mixin to any controller you want to manage selection.  It is 
 26   already applied to the CollectionController and ArrayController.
 27   
 28   @since SproutCore 1.0
 29 */
 30 SC.SelectionSupport = {
 31 
 32   // ..........................................................
 33   // PROPERTIES
 34   // 
 35   /**
 36     Walk like a duck.
 37     
 38     @property {Boolean}
 39   */
 40   hasSelectionSupport: YES,
 41 
 42   /**
 43     If YES, selection is allowed. Default is YES.
 44     
 45     @property {Boolean}
 46   */
 47   allowsSelection: YES,
 48 
 49   /**
 50     If YES, multiple selection is allowed. Default is YES.
 51     
 52     @property {Boolean}
 53   */
 54   allowsMultipleSelection: YES,
 55 
 56   /**
 57     If YES, allow empty selection Default is YES.
 58     
 59     @property {Boolean}
 60   */
 61   allowsEmptySelection: YES,
 62 
 63   /**
 64     Override to return the first selectable object.  For example, if you 
 65     have groups or want to otherwise limit the kinds of objects that can be
 66     selected.
 67     
 68     the default imeplementation returns firstObject property.
 69     
 70     @returns {Object} first selectable object
 71   */
 72   firstSelectableObject: function() {
 73     return this.get('firstObject');
 74   }.property(),
 75 
 76   /**
 77     This is the current selection.  You can make this selection and another
 78     controller's selection work in concert by binding them together. You
 79     generally have a master selection that relays changes TO all the others.
 80     
 81     @property {SC.SelectionSet}
 82   */
 83   selection: function(key, value) {
 84 
 85     var old = this._scsel_selection,
 86     oldlen = old ? old.get('length') : 0,
 87     content,
 88     empty,
 89     len;
 90 
 91     // whenever we have to recompute selection, reapply all the conditions to
 92     // the selection.  This ensures that changing the conditions immediately
 93     // updates the selection.
 94     // 
 95     // Note also if we don't allowSelection, we don't clear the old selection;
 96     // we just don't allow it to be changed.
 97     if ((value === undefined) || !this.get('allowsSelection')) value = old;
 98 
 99     len = (value && value.isEnumerable) ? value.get('length') : 0;
100 
101     // if we don't allow multiple selection
102     if ((len > 1) && !this.get('allowsMultipleSelection')) {
103 
104       if (oldlen > 1) {
105         value = SC.SelectionSet.create().addObject(old.get('firstObject')).freeze();
106         len = 1;
107       } else {
108         value = old;
109         len = oldlen;
110       }
111     }
112 
113     // if we don't allow empty selection, block that also.  select first 
114     // selectable item if necessary.
115     if ((len === 0) && !this.get('allowsEmptySelection')) {
116       if (oldlen === 0) {
117         value = this.get('firstSelectableObject');
118         if (value) value = SC.SelectionSet.create().addObject(value).freeze();
119         else value = SC.SelectionSet.EMPTY;
120         len = value.get('length');
121 
122       } else {
123         value = old;
124         len = oldlen;
125       }
126     }
127 
128     // if value is empty or is not enumerable, then use empty set
129     if (len === 0) value = SC.SelectionSet.EMPTY;
130 
131     // always use a frozen copy...
132     value = value.frozenCopy();
133     this._scsel_selection = value;
134 
135     return value;
136 
137   }.property('arrangedObjects', 'allowsEmptySelection', 'allowsMultipleSelection', 'allowsSelection').cacheable(),
138 
139   /**
140     YES if the receiver currently has a non-zero selection.
141     
142     @property {Boolean}
143   */
144   hasSelection: function() {
145     var sel = this.get('selection');
146     return !! sel && (sel.get('length') > 0);
147   }.property('selection').cacheable(),
148 
149   // ..........................................................
150   // METHODS
151   // 
152   /**
153     Selects the passed objects in your content.  If you set "extend" to YES,
154     then this will attempt to extend your selection as well.
155   
156     @param {SC.Enumerable} objects objects to select
157     @param {Boolean} extend optionally set to YES to extend selection
158     @returns {Object} receiver
159   */
160   selectObjects: function(objects, extend) {
161 
162     // handle passing an empty array
163     if (!objects || objects.get('length') === 0) {
164       if (!extend) this.set('selection', SC.SelectionSet.EMPTY);
165       return this;
166     }
167 
168     var sel = this.get('selection');
169     if (extend && sel) sel = sel.copy();
170     else sel = SC.SelectionSet.create();
171 
172     sel.addObjects(objects).freeze();
173     this.set('selection', sel);
174     return this;
175   },
176 
177   /**
178     Selects a single passed object in your content.  If you set "extend" to 
179     YES then this will attempt to extend your selection as well.
180     
181     @param {Object} object object to select
182     @param {Boolean} extend optionally set to YES to extend selection
183     @returns {Object} receiver
184   */
185   selectObject: function(object, extend) {
186     if (object === null) {
187       if (!extend) this.set('selection', null);
188       return this;
189 
190     } else return this.selectObjects([object], extend);
191   },
192 
193   /**
194     Deselects the passed objects in your content.
195     
196     @param {SC.Enumerable} objects objects to select
197     @returns {Object} receiver
198   */
199   deselectObjects: function(objects) {
200 
201     if (!objects || objects.get('length') === 0) return this; // nothing to do
202     var sel = this.get('selection');
203     if (!sel || sel.get('length') === 0) return this; // nothing to do
204     // find index for each and remove it
205     sel = sel.copy().removeObjects(objects).freeze();
206     this.set('selection', sel.freeze());
207     return this;
208   },
209 
210   /**
211     Deselects the passed object in your content.
212     
213     @param {SC.Object} object single object to select
214     @returns {Object} receiver
215   */
216   deselectObject: function(object) {
217     if (!object) return this; // nothing to do
218     else return this.deselectObjects([object]);
219   },
220 
221   /**
222     Call this method whenever your source content changes to ensure the 
223     selection always remains up-to-date and valid.
224     
225     @returns {Object}
226   */
227   updateSelectionAfterContentChange: function() {
228     var arrangedObjects = this.get('arrangedObjects');
229     var selectionSet = this.get('selection');
230     var allowsEmptySelection = this.get('allowsEmptySelection');
231     var indexSet; // Selection index set for arranged objects
232 
233     // If we don't have any selection, there's nothing to update
234     if (!selectionSet) return this;
235     // Remove any selection set objects that are no longer in the content
236     indexSet = selectionSet.indexSetForSource(arrangedObjects);
237     if ((indexSet && (indexSet.get('length') !== selectionSet.get('length'))) || (!indexSet && (selectionSet.get('length') > 0))) { // then the selection content has changed
238       selectionSet = selectionSet.copy().constrain(arrangedObjects).freeze();
239       this.set('selection', selectionSet);
240     }
241     
242     // Reselect an object if required (if content length > 0)
243     if ((selectionSet.get('length') === 0) && (arrangedObjects.get('length') > 0) && !allowsEmptySelection) {
244       this.selectObject(this.get('firstSelectableObject'), NO);
245     }
246 
247     return this;
248   }
249 
250 };
