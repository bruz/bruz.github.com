  8 require("tasks/task");
  9 
 10 // default callback
 11 SC.didPreloadBundle = function() {};
 12 
 13 /**
 14   @private
 15   A task that preloads a bundle, supplying a target and action to be called
 16   on bundle load completion.
 17 */
 18 SC.PreloadBundleTask = SC.Task.extend({
 19   /**
 20     The identifier of the bundle to load.
 21   */
 22   bundle: null,
 23   
 24   /**
 25     The target to supply to SC.loadBundle.
 26   */
 27   target: "SC",
 28   
 29   /**
 30     The action to supply to SC.loadBundle.
 31   */
 32   action: "preloaded",
 33   
 34   run: function(queue) {
 35     var bundle;
 36     if (bundle = this.get("bundle")) {
 37       var st = Date.now();
 38       SC.loadBundle(this.get("bundle"), this.get("target"), this.get("action"));
 39     }
 40   }
 41 });
