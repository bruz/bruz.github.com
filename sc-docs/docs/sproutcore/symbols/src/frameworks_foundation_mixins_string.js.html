109   SproutCore implements a variety of enhancements to the built-in String

119   /**
120     Localizes the string.  This will look up the reciever string as a key 
121     in the current Strings hash.  If the key matches, the loc'd value will be
122     used.  The resulting string will also be passed through fmt() to insert
123     any variables.
124     
125     @param args {Object...} optional arguments to interpolate also
126     @returns {String} the localized and formatted string.
127   */
128   loc: function() {
129     // NB: This could be implemented as a wrapper to locWithDefault() but
130     // it would add some overhead to deal with the arguments and adds stack
131     // frames, so we are keeping the implementation separate.
132     if(!SC.Locale.currentLocale) SC.Locale.createCurrentLocale();
133     var str = SC.Locale.currentLocale.locWithDefault(this);
134     if (SC.typeOf(str) !== SC.T_STRING) str = this;
135     return str.fmt.apply(str,arguments) ;
136   },
137 
138   /**
139     Works just like loc() except that it will return the passed default 
140     string if a matching key is not found.
141     
142     @param {String} def the default to return
143     @param {Object...} args optional formatting arguments
144     @returns {String} localized and formatted string
145   */
146   locWithDefault: function(def) {
147     if(!SC.Locale.currentLocale) SC.Locale.createCurrentLocale();
148     var str = SC.Locale.currentLocale.locWithDefault(this, def);
149     if (SC.typeOf(str) !== SC.T_STRING) str = this;
150     var args = SC.$A(arguments); args.shift(); // remove def param
151     return str.fmt.apply(str,args) ;
152   },

154   /** 
155     Capitalizes a string.
156 
157     h2. Examples
158     
159     | *Input String* | *Output String* |
160     | my favorite items | My favorite items |
161     | css-class-name | Css-class-name |
162     | action_name | Action_name |
163     | innerHTML | InnerHTML |
164 
165     @return {String} capitalized string
166   */
167   capitalize: function() {
168     return this.charAt(0).toUpperCase() + this.slice(1) ;

171   /**
172     Capitalizes every word in a string.  Unlike titleize, spaces or dashes 
173     will remain in-tact.
174     
175     h2. Examples
176     
177     | *Input String* | *Output String* |
178     | my favorite items | My Favorite Items |

192   /**
193     Converts a string to a title.  This will decamelize the string, convert
194     separators to spaces and capitalize every word.
195 
196     h2. Examples
197     
198     | *Input String* | *Output String* |
199     | my favorite items | My Favorite Items |
200     | css-class-name | Css Class Name |
201     | action_name | Action Name |
202     | innerHTML | Inner HTML |
203 
204     @return {String} titleized string.
205   */
206   titleize: function() {
207     var ret = this.replace(SC.STRING_DECAMELIZE_REGEXP,'$1_$2'); // decamelize
208     return ret.replace(SC.STRING_TITLEIZE_REGEXP, 
209       function(str,separater,character) { 
210         return (character) ? (' ' + character.toUpperCase()) : ' ';
211       }).capitalize() ;
212   },

214   /**
215     Camelizes a string.  This will take any words separated by spaces, dashes
216     or underscores and convert them into camelCase.
217     
218     h2. Examples
219     
220     | *Input String* | *Output String* |
221     | my favorite items | myFavoriteItems |
222     | css-class-name | cssClassName |
223     | action_name | actionName |
224     | innerHTML | innerHTML |
225 
226     @returns {String} camelized string
227   */
228   camelize: function() {
229     var ret = this.replace(SC.STRING_TITLEIZE_REGEXP, 
230       function(str,separater,character) { 
231         return (character) ? character.toUpperCase() : '' ;
232       }) ;
233     var first = ret.charAt(0), lower = first.toLowerCase() ;
234     return (first !== lower) ? (lower + ret.slice(1)) : ret ;
235   },

277   /**
278     Converts a camelized string or a string with spaces or underscores into
279     a string with components separated by dashes.
280     
281     h2. Examples
282     
283     | *Input String* | *Output String* |
284     | my favorite items | my-favorite-items |
285     | css-class-name | css-class-name |
286     | action_name | action-name |
287     | innerHTML | inner-html |
288 
289     @returns {String} the dasherized string.
290   */
291   dasherize: function() {
292     // Do we have the item in our cache?
293     var cache = SC.STRING_DASHERIZE_CACHE,
294         ret   = cache[this];
295 
296     if (ret) {
297       return ret;
298     }
299     else {
300       ret = this.decamelize().replace(SC.STRING_DASHERIZE_REGEXP,'-') ;
301 
302       // Add the item to our cache.
303       cache[this] = ret;
304     }
305 
306     return ret;
307   },

461   /**
462     Converts a word into its plural form. 
463     
464     @returns {String} the plural form of the string
465   */
466   pluralize: function() {
467 			var idx, len,
468 			 		compare = this.split(/\s/).pop(), //check only the last word of a string
469 					restOfString = this.replace(compare,''),
470 					isCapitalized = compare.charAt(0).match(/[A-Z]/) ? true : false;

473       for (idx=0, len=SC.INFLECTION_CONSTANTS.UNCOUNTABLE.length; idx < len; idx++) {

475           if (compare == uncountable) {	
476               return this.toString();
477           }
478       }
479       for (idx=0, len=SC.INFLECTION_CONSTANTS.IRREGULAR.length; idx < len; idx++) {
480           var singular = SC.INFLECTION_CONSTANTS.IRREGULAR[idx][0],
481 							plural   = SC.INFLECTION_CONSTANTS.IRREGULAR[idx][1];
482           if ((compare == singular) || (compare == plural)) {
483 							if(isCapitalized) plural = plural.capitalize();
484               return restOfString + plural;
485           }
486       }
487       for (idx=0, len=SC.INFLECTION_CONSTANTS.PLURAL.length; idx < len; idx++) {
488           var regex          = SC.INFLECTION_CONSTANTS.PLURAL[idx][0],
489 							replace_string = SC.INFLECTION_CONSTANTS.PLURAL[idx][1];
490           if (regex.test(compare)) {
491               return this.replace(regex, replace_string);
492           }
493       }
494   },
495 
496   /**
497     Converts a word into its singular form. 
498     
499     @returns {String} the singular form of the string
500   */
501   singularize: function() {
502 			var idx, len,
503 					compare = this.split(/\s/).pop(), //check only the last word of a string								
504 					restOfString = this.replace(compare,''),	
505 					isCapitalized = compare.charAt(0).match(/[A-Z]/) ? true : false;
506 
507 			compare = compare.toLowerCase();
508       for (idx=0, len=SC.INFLECTION_CONSTANTS.UNCOUNTABLE.length; idx < len; idx++) {
509           var uncountable = SC.INFLECTION_CONSTANTS.UNCOUNTABLE[idx];
510           if (compare == uncountable) {
511               return this.toString();
512           }
513       }
514       for (idx=0, len=SC.INFLECTION_CONSTANTS.IRREGULAR.length; idx < len; idx++) {
515           var singular = SC.INFLECTION_CONSTANTS.IRREGULAR[idx][0],
516 							plural   = SC.INFLECTION_CONSTANTS.IRREGULAR[idx][1];
517           if ((compare == singular) || (compare == plural)) {
518 							if(isCapitalized) singular = singular.capitalize();
519               return restOfString + singular;
520           }
521       }
522       for (idx=0, len=SC.INFLECTION_CONSTANTS.SINGULAR.length; idx < len; idx++) {
523           var regex          = SC.INFLECTION_CONSTANTS.SINGULAR[idx][0],
524 							replace_string = SC.INFLECTION_CONSTANTS.SINGULAR[idx][1];
525           if (regex.test(compare)) {
526               return this.replace(regex, replace_string);
527           }
528       }
529   }
