 16 /**
 17   @namespace
 18   
 19   This module implements Observer-friendly Array-like behavior.  This mixin is 
 20   picked up by the Array class as well as other controllers, etc. that want to  
 21   appear to be arrays.
 22   
 23   Unlike SC.Enumerable, this mixin defines methods specifically for 
 24   collections that provide index-ordered access to their contents.  When you
 25   are designing code that needs to accept any kind of Array-like object, you
 26   should use these methods instead of Array primitives because these will 
 27   properly notify observers of changes to the array. 
 28   
 29   Although these methods are efficient, they do add a layer of indirection to
 30   your application so it is a good idea to use them only when you need the 
 31   flexibility of using both true JavaScript arrays and "virtual" arrays such
 32   as controllers and collections.
 33   
 34   You can use the methods defined in this module to access and modify array 
 35   contents in a KVO-friendly way.  You can also be notified whenever the 
 36   membership if an array changes by changing the syntax of the property to
 37   .observes('*myProperty.[]') .
 38   
 39   To support SC.Array in your own class, you must override two
 40   primitives to use it: replace() and objectAt().  
 41   
 42   Note that the SC.Array mixin also incorporates the SC.Enumerable mixin.  All
 43   SC.Array-like objects are also enumerable.
 44   
 45   @extends SC.Enumerable
 46   @since SproutCore 0.9.0
 47 */

129   /**
130     Remove an object at the specified index using the replace() primitive 
131     method.  You can pass either a single index, a start and a length or an
132     index set.
133     
134     If you pass a single index or a start and length that is beyond the 
135     length this method will throw an SC.OUT_OF_RANGE_EXCEPTION
136     
137     @param {Number|SC.IndexSet} start index, start of range, or index set
138     @param {Number} length length of passing range
139     @returns {Object} receiver
140   */
141   removeAt: function(start, length) {
142     
143     var delta = 0, // used to shift range
144         empty = [];
145     
146     if (typeof start === SC.T_NUMBER) {
147       
148       if ((start < 0) || (start >= this.get('length'))) {
149         throw SC.OUT_OF_RANGE_EXCEPTION;
150       }
151       
152       // fast case
153       if (length === undefined) {
154         this.replace(start,1,empty);
155         return this ;
156       } else {
157         start = SC.IndexSet.create(start, length);
158       }
159     }
160     
161     this.beginPropertyChanges();
162     start.forEachRange(function(start, length) {
163       start -= delta ;
164       delta += length ;
165       this.replace(start, length, empty); // remove!
166     }, this);
167     this.endPropertyChanges();
168     
169     return this ;
170   },

172   /**
173     Search the array of this object, removing any occurrences of it.
174     @param {object} obj object to remove

177     var loc = this.get('length') || 0;
178     while(--loc >= 0) {
179       var curObject = this.objectAt(loc) ;
180       if (curObject == obj) this.removeAt(loc) ;
181     }
182     return this ;

271   /**  
272     Compares each item in the array.  Returns true if they are equal.
273   */
274   isEqual: function(ary) {
275     if (!ary) return false ;
276     if (ary == this) return true;
277     
278     var loc = ary.get('length') ;
279     if (loc != this.get('length')) return false ;
280 
281     while(--loc >= 0) {
282       if (!SC.isEqual(ary.objectAt(loc), this.objectAt(loc))) return false ;
283     }
284     return true ;
285   },

351   /**
352     Creates a new range observer on the receiver.  The target/method callback
353     you provide will be invoked anytime any property on the objects in the 
354     specified range changes.  It will also be invoked if the objects in the
355     range itself changes also.
356     
357     The callback for a range observer should have the signature:
358     
359     {{{
360       function rangePropertyDidChange(array, objects, key, indexes, conext)
361     }}}
362     
363     If the passed key is '[]' it means that the object itself changed.
364     
365     The return value from this method is an opaque reference to the 
366     range observer object.  You can use this reference to destroy the 
367     range observer when you are done with it or to update its range.
368     
369     @param {SC.IndexSet} indexes indexes to observe
370     @param {Object} target object to invoke on change
371     @param {String|Function} method the method to invoke
372     @param {Object} context optional context
373     @returns {SC.RangeObserver} range observer
374   */
375   addRangeObserver: function(indexes, target, method, context) {
376     var rangeob = this._array_rangeObservers;
377     if (!rangeob) rangeob = this._array_rangeObservers = SC.CoreSet.create() ;
378 
379     // The first time a range observer is added, cache the current length so
380     // we can properly notify observers the first time through
381     if (this._array_oldLength===undefined) {
382       this._array_oldLength = this.get('length') ;
383     }
384     
385     var C = this.rangeObserverClass ;
386     var isDeep = NO; //disable this feature for now
387     var ret = C.create(this, indexes, target, method, context, isDeep) ;
388     rangeob.add(ret);
389     
390     // first time a range observer is added, begin observing the [] property
391     if (!this._array_isNotifyingRangeObservers) {
392       this._array_isNotifyingRangeObservers = YES ;
393       this.addObserver('[]', this, this._array_notifyRangeObservers);
394     }
395     
396     return ret ;
397   },

402     addRangeObserver().

405     pass in.

415   /**
416     Removes a range observer from the receiver.  The range observer must
417     already be active on the array.
418     
419     The return value should replace the old range observer object.  It will
420     usually be null.
421     
422     @param {SC.RangeObserver} rangeObserver the range observer
423     @returns {SC.RangeObserver} updated range observer or null
424   */
425   removeRangeObserver: function(rangeObserver) {
426     var ret = rangeObserver.destroy(this);
427     var rangeob = this._array_rangeObservers;
428     if (rangeob) rangeob.remove(rangeObserver) ; // clear
429     return ret ;
430   },

432   /**
433     Updates observers with content change.  To support range observers, 
434     you must pass three change parameters to this method.  Otherwise this
435     method will assume the entire range has changed.
436     
437     This also assumes you have already updated the length property.
438     @param {Number} start the starting index of the change
439     @param {Number} amt the final range of objects changed
440     @param {Number} delta if you added or removed objects, the delta change
441     @returns {SC.Array} receiver
442   */
443   enumerableContentDidChange: function(start, amt, delta) {
444     var rangeob = this._array_rangeObservers, 
445         oldlen  = this._array_oldLength,
446         newlen, length, changes ;
447 
448     this.beginPropertyChanges();    
449     this.notifyPropertyChange('length'); // flush caches
450 
451     // schedule info for range observers
452     if (rangeob && rangeob.length>0) {
453 
454       // if no oldLength has been cached, just assume 0
455       if (oldlen === undefined) oldlen = 0;    
456       this._array_oldLength = newlen = this.get('length');
457       
458       // normalize input parameters
459       // if delta was not passed, assume it is the different between the 
460       // new and old length.
461       if (start === undefined) start = 0;
462       if (delta === undefined) delta = newlen - oldlen ;
463       if (delta !== 0 || amt === undefined) {
464         length = newlen - start ;
465         if (delta<0) length -= delta; // cover removed range as well
466       } else {
467         length = amt ;
468       }
469       
470       changes = this._array_rangeChanges;
471       if (!changes) changes = this._array_rangeChanges = SC.IndexSet.create();
472       changes.add(start, length);
473     }
474     
475     this.notifyPropertyChange('[]') ;
476     this.endPropertyChanges();
477     
478     return this ;
479   },
480   
481   /**  @private
482     Observer fires whenever the '[]' property changes.  If there are 
483     range observers, will notify observers of change.
484   */
485   _array_notifyRangeObservers: function() {
486     var rangeob = this._array_rangeObservers,
487         changes = this._array_rangeChanges,
488         len     = rangeob ? rangeob.length : 0, 
489         idx, cur;
490         
491     if (len > 0 && changes && changes.length > 0) {
492       for(idx=0;idx<len;idx++) rangeob[idx].rangeDidChange(changes);
493       changes.clear(); // reset for later notifications
494     }
495   }

505 /**
506   Returns a new array that is a slice of the receiver.  This implementation
507   uses the observable array methods to retrieve the objects for the new 
508   slice.
509   
510   @param beginIndex {Integer} (Optional) index to begin slicing from.     
511   @param endIndex {Integer} (Optional) index to end the slice at.
512   @returns {Array} New array with specified slice
513 */
514 SC.Array.slice = function(beginIndex, endIndex) {
515   var ret = []; 
516   var length = this.get('length') ;
517   if (SC.none(beginIndex)) beginIndex = 0 ;
518   if (SC.none(endIndex) || (endIndex > length)) endIndex = length ;
519   while(beginIndex < endIndex) ret[ret.length] = this.objectAt(beginIndex++) ;
520   return ret ;
521 }  ;

523 /**
524   Returns the index for a particular object in the index.
525   
526   @param {Object} object the item to search for
527   @param {NUmber} startAt optional starting location to search, default 0
528   @returns {Number} index of -1 if not found
529 */
530 SC.Array.indexOf = function(object, startAt) {
531   var idx, len = this.get('length');
532   
533   if (startAt === undefined) startAt = 0;
534   else startAt = (startAt < 0) ? Math.ceil(startAt) : Math.floor(startAt);
535   if (startAt < 0) startAt += len;
536   
537   for(idx=startAt;idx<len;idx++) {
538     if (this.objectAt(idx) === object) return idx ;
539   }
540   return -1;
541 };
542 
543 // Some browsers do not support indexOf natively.  Patch if needed
544 if (!Array.prototype.indexOf) Array.prototype.indexOf = SC.Array.indexOf;

546 /**
547   Returns the last index for a particular object in the index.
548   
549   @param {Object} object the item to search for
550   @param {NUmber} startAt optional starting location to search, default 0
551   @returns {Number} index of -1 if not found
552 */
553 SC.Array.lastIndexOf = function(object, startAt) {
554   var idx, len = this.get('length');
555   
556   if (startAt === undefined) startAt = len-1;
557   else startAt = (startAt < 0) ? Math.ceil(startAt) : Math.floor(startAt);
558   if (startAt < 0) startAt += len;
559   
560   for(idx=startAt;idx>=0;idx--) {
561     if (this.objectAt(idx) === object) return idx ;
562   }
563   return -1;
564 };

568   Array.prototype.lastIndexOf = SC.Array.lastIndexOf;

572 // ARRAY SUPPORT
573 //
574 // Implement the same enhancements on Array.  We use specialized methods
575 // because working with arrays are so common.
576 (function() {
577   SC.mixin(Array.prototype, {
578     
579     // primitive for array support.
580     replace: function(idx, amt, objects) {
581       if (this.isFrozen) throw SC.FROZEN_ERROR ;
582       if (!objects || objects.length === 0) {
583         this.splice(idx, amt) ;
584       } else {
585         var args = [idx, amt].concat(objects) ;
586         this.splice.apply(this,args) ;
587       }
588       
589       // if we replaced exactly the same number of items, then pass only the
590       // replaced range.  Otherwise, pass the full remaining array length 
591       // since everything has shifted
592       var len = objects ? (objects.get ? objects.get('length') : objects.length) : 0;
593       this.enumerableContentDidChange(idx, amt, len - amt) ;
594       return this ;
595     },
596     
597     // If you ask for an unknown property, then try to collect the value
598     // from member items.
599     unknownProperty: function(key, value) {
600       var ret = this.reducedProperty(key, value) ;
601       if ((value !== undefined) && ret === undefined) {
602         ret = this[key] = value;
603       }
604       return ret ;
605     }
606     
607   });
608     
609   // If browser did not implement indexOf natively, then override with
610   // specialized version
611   var indexOf = Array.prototype.indexOf;
612   if (!indexOf || (indexOf === SC.Array.indexOf)) {
613     Array.prototype.indexOf = function(object, startAt) {
614       var idx, len = this.length;
615       
616       if (startAt === undefined) startAt = 0;
617       else startAt = (startAt < 0) ? Math.ceil(startAt) : Math.floor(startAt);
618       if (startAt < 0) startAt += len;
619       
620       for(idx=startAt;idx<len;idx++) {
621         if (this[idx] === object) return idx ;
622       }
623       return -1;
624     } ; 
625   }
626   
627   var lastIndexOf = Array.prototype.lastIndexOf ;
628   if (!lastIndexOf || (lastIndexOf === SC.Array.lastIndexOf)) {
629     Array.prototype.lastIndexOf = function(object, startAt) {
630       var idx, len = this.length;
631       
632       if (startAt === undefined) startAt = len-1;
633       else startAt = (startAt < 0) ? Math.ceil(startAt) : Math.floor(startAt);
634       if (startAt < 0) startAt += len;
635       
636       for(idx=startAt;idx>=0;idx--) {
637         if (this[idx] === object) return idx ;
638       }
639       return -1;
640     };
641   }
