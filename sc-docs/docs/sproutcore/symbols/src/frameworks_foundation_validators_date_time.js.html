  8 /**
  9   This validates a SC.DateTime, used in SC.DateFieldView.
 10   
 11   @class
 12   @extends SC.Validator
 13   @author Juan Pablo Goldfinger
 14   @version 1.0
 15 */
 16 SC.Validator.DateTime = SC.Validator.extend({
 17 
 18   /**
 19     The standard format you want the validator to convert dates to.
 20   */
 21   format: '%d/%m/%Y',
 22 
 23   /**
 24     if we have a number, then convert to a date object.
 25   */
 26   fieldValueForObject: function(object, form, field) {
 27     if (SC.kindOf(object, SC.DateTime)) {
 28       object = object.toFormattedString(this.get('format'));
 29     } else {
 30       object = null;
 31     }
 32     return object;
 33   },
 34 
 35   /**
 36     Try to pase value as a date. convert into a number, or return null if
 37     it could not be parsed.
 38   */
 39   objectForFieldValue: function(value, form, field) {
 40     if (value) {
 41       value = SC.DateTime.parse(value, this.get('format'));
 42     }
 43     return value;
 44   }
 45 
 46 });
