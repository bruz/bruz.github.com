  9 /** @class
 10 
 11   SC.PopupButtonView displays a pop-up menu when clicked, from which the user
 12   can select an item.
 13 
 14   To use, create the SC.PopupButtonView as you would a standard SC.ButtonView,
 15   then set the menu property to an instance of SC.MenuPane. For example:
 16 
 17 {{{
 18 SC.PopupButtonView.design({
 19   layout: { width: 200, height: 18 },
 20   menuBinding: 'MyApp.menuController.menuPane'
 21 });
 22 }}}
 23 
 24   You would then have your MyApp.menuController return an instance of the menu
 25   to display.
 26 
 27   @extends SC.ButtonView
 28   @author Santosh Shanbhogue
 29   @author Tom Dale
 30   @copyright 2008-2010, Sprout Systems, Inc. and contributors.
 31   @version 1.0
 32 */
 33 SC.PopupButtonView = SC.ButtonView.extend(
 34 /** @scope SC.PopupButtonView.prototype */ {
 35   classNames: ['sc-popup-button'],
 36 
 37   // ..........................................................
 38   // PROPERTIES
 39   //
 40 
 41   /**
 42     The prefer matrix to use when displaying the menu.
 43 
 44     @property
 45   */
 46   preferMatrix: null,
 47 
 48   /**
 49     The SC.MenuPane that should be displayed when the button is clicked.
 50 
 51     @type {SC.MenuPane}
 52     @default null
 53   */
 54   menu: null,
 55   
 56   /**
 57     If YES and the menu is a class, this will cause a task that will instantiate the menu
 58     to be added to SC.backgroundTaskQueue.
 59   */
 60   shouldLoadInBackground: NO,
 61 
 62   // ..........................................................
 63   // INTERNAL SUPPORT
 64   //
 65   
 66   /**
 67     @private
 68     If necessary, adds the loading of the menu to the background task queue.
 69   */
 70   init: function() {
 71     sc_super();
 72     this._setupMenu();
 73     if (this.get('shouldLoadInBackground')) {
 74       SC.backgroundTaskQueue.push(SC.PopupButtonMenuLoader.create({ popupButton: this }));
 75     }
 76   },
 77   
 78   /**
 79     @private
 80     Sets up binding on the menu, removing any old ones if necessary.
 81   */
 82   _setupMenu: function() {
 83     var menu = this.get('instantiatedMenu');
 84     
 85     // clear existing bindings
 86     if (this.isActiveBinding) this.isActiveBinding.disconnect();
 87     this.isActiveBinding = null;
 88     
 89     // if there is a menu
 90     if (menu && !menu.isClass) {
 91       this.isActiveBinding = this.bind('isActive', menu, 'isVisibleInWindow');
 92     }
 93   },
 94   
 95   /**
 96     Setup the bindings for menu...
 97   */
 98   _popup_menuDidChange: function() {
 99     this._setupMenu();
100   }.observes('menu'),
101 
102   /** @private
103     isActive is NO, but when the menu is instantiated, it is bound to the menu's isVisibleInWindow property.
104   */
105   isActive: NO,
106   
107   /**
108     @private
109     Instantiates the menu if it is not already instantiated.
110   */
111   _instantiateMenu: function() {
112     // get menu
113     var menu = this.get('menu');
114     
115     // if it is already instantiated or does not exist, we cannot do anything
116     if (!menu.isClass || !menu) return;
117     
118     // create
119     this.menu = menu.create();
120     
121     // setup
122     this._setupMenu();
123   },
124   
125   acceptsFirstResponder: YES,
126   
127   /**
128     The guaranteed-instantiated menu.
129   */
130   instantiatedMenu: function() {
131     // get the menu
132     var menu = this.get('menu');
133     
134     // if it is a class, we need to instantiate it
135     if (menu && menu.isClass) {
136       // do so
137       this._instantiateMenu();
138       
139       // get the new version of the local
140       menu = this.get('menu');
141     }
142     
143     // return
144     return menu;
145   }.property('menu').cacheable(),
146 
147   /** @private
148     Displays the menu.
149 
150     @param {SC.Event} evt
151   */
152   action: function(evt)
153   {
154     var menu = this.get('instantiatedMenu') ;
155 
156     if (!menu) {
157       //@ if (debug)
158       SC.Logger.warn("SC.PopupButton - Unable to show menu because the menu property is set to %@.".fmt(menu));
159       //@ endif
160       return NO ;
161     }
162 
163     menu.popup(this, this.get('preferMatrix')) ;
164     return YES;
165   },
166 
167   /** @private
168     On mouse down, we set the state of the button, save some state for further
169     processing, then call the button's action method.
170 
171     @param {SC.Event} evt
172     @returns {Boolean}
173   */
174   mouseDown: function(evt) {
175     // If disabled, handle mouse down but ignore it.
176     if (!this.get('isEnabled')) return YES ;
177 
178     this._isMouseDown = YES;
179 
180     this._action() ;
181 
182     // Store the current timestamp. We register the timestamp at the end of
183     // the runloop so that the menu has been rendered, in case that operation
184     // takes more than a few hundred milliseconds.
185 
186     // One mouseUp, we'll use this value to determine how long the mouse was
187     // pressed.
188     this.invokeLast(this._recordMouseDownTimestamp);
189 
190     this.becomeFirstResponder();
191 
192     return YES ;
193   },
194 
195   /** @private
196     Records the current timestamp. This is invoked at the end of the runloop
197     by mouseDown. We use this value to determine the delay between mouseDown
198     and mouseUp.
199   */
200   _recordMouseDownTimestamp: function() {
201     this._menuRenderedTimestamp = new Date().getTime();
202   },
203 
204   /** @private
205     Because we responded YES to the mouseDown event, we have responsibility
206     for handling the corresponding mouseUp event.
207 
208     However, the user may click on this button, then drag the mouse down to a
209     menu item, and release the mouse over the menu item. We therefore need to
210     delegate any mouseUp events to the menu's menu item, if one is selected.
211 
212     We also need to differentiate between a single click and a click and hold.
213     If the user clicks and holds, we want to close the menu when they release.
214     Otherwise, we should wait until they click on the menu's modal pane before
215     removing our active state.
216 
217     @param {SC.Event} evt
218     @returns {Boolean}
219   */
220   mouseUp: function(evt) {
221     var timestamp = new Date().getTime(),
222         previousTimestamp = this._menuRenderedTimestamp,
223         menu = this.get('instantiatedMenu'),
224         touch = SC.platform.touch,
225         targetMenuItem;
226 
227     if (menu) {
228       // Get the menu item the user is currently hovering their mouse over
229       targetMenuItem = menu.getPath('rootMenu.targetMenuItem');
230 
231       if (targetMenuItem) {
232         // Have the menu item perform its action.
233         // If the menu returns NO, it had no action to
234         // perform, so we should close the menu immediately.
235         if (!targetMenuItem.performAction()) menu.remove();
236       } else {
237         // If the user waits more than certain amount of time between
238         // mouseDown and mouseUp, we can assume that they are clicking and
239         // dragging to the menu item, and we should close the menu if they
240         //mouseup anywhere not inside the menu.
241         if (!touch && (timestamp - previousTimestamp > SC.ButtonView.CLICK_AND_HOLD_DELAY)) {
242           menu.remove();
243         }
244       }
245     }
246 
247     // Reset state.
248     this._isMouseDown = NO;
249     sc_super();
250     return YES;
251   },
252 
253   /** @private
254     Overrides ButtonView's mouseExited method to remove the behavior where the
255     active state is removed on mouse exit. We want the button to remain active
256     as long as the menu is visible.
257 
258     @param {SC.Event} evt
259     @returns {Boolean}
260   */
261   mouseExited: function(evt) {
262     return YES;
263   },
264 
265   /** @private
266     Overrides performKeyEquivalent method to pass any keyboard shortcuts to
267     the menu.
268 
269     @param {String} charCode string corresponding to shortcut pressed (e.g.,
270     alt_shift_z)
271     @param {SC.Event} evt
272   */
273   performKeyEquivalent: function( charCode, evt )
274   {
275     if (!this.get('isEnabled')) return NO ;
276     var menu = this.get('instantiatedMenu') ;
277     return (!!menu && menu.performKeyEquivalent(charCode, evt)) ;
278   },
279 
280   /** @private */
281   acceptsFirstResponder: function() {
282     if(!SC.SAFARI_FOCUS_BEHAVIOR) return this.get('isEnabled');
283     else return NO;
284   }.property('isEnabled')
285 
286 });

288 /**
289   @private
290   Handles lazy instantiation of popup button menu.
291 */
292 SC.PopupButtonMenuLoader = SC.Task.extend({
293   popupButton: null,
294   run: function() {
295     if (this.popupButton) this.popupButton._instantiateMenu();
296   }
