 14 /**
 15   @class
 16   
 17   The image cache can be used to control the order of loading images into the
 18    browser cache.
 19   
 20   Images queues are necessary because browsers impose strict limits on the 
 21   number of concurrent connections that can be open at any one time to any one 
 22   host. By controlling the order and timing of your loads using this image 
 23   queue, you can improve the percieved performance of your application by 
 24   ensuring the images you need most load first.
 25   
 26   Note that if you use the SC.ImageView class, it will use this image cache 
 27   for you automatically.
 28   
 29   h1. Loading Images
 30   
 31   When you need to display an image, simply call the loadImage() method with 
 32   the URL of the image, along with a target/method callback. The signature of 
 33   your callback should be:
 34   
 35   {{{
 36     imageDidLoad: function(imageUrl, imageOrError) {
 37       //...
 38     }
 39   }}}
 40 
 41   The "imageOrError" parameter will contain either an image object or an error 
 42   object if the image could not be loaded for some reason.  If you receive an 
 43   error object, it will be one of SC.IMAGE_ABORTED_ERROR or 
 44   SC.IMAGE_FAILED_ERROR.
 45   
 46   You can also optionally specify that the image should be loaded in the 
 47   background.  Background images are loaded with a lower priority than 
 48   foreground images.
 49   
 50   h1. Aborting Image Loads
 51   
 52   If you request an image load but then no longer require the image for some 
 53   reason, you should notify the imageCache by calling the releaseImage() 
 54   method.  Pass the URL, target and method that you included in your original 
 55   loadImage() request.  
 56   
 57   If you have requested an image before, you should always call releaseImage() 
 58   when you are finished with it, even if the image has already loaded.  This 
 59   will allow the imageCache to properly manage its own internal resources.
 60   
 61   This method may remove the image from the queue of images that need or load 
 62   or it may abort an image load in progress to make room for other images.  If 
 63   the image is already loaded, this method will have no effect.
 64   
 65   h1. Reloading an Image
 66   
 67   If you have already loaded an image, the imageCache will avoid loading the 
 68   image again.  However, if you need to force the imageCache to reload the 
 69   image for some reason, you can do so by calling reloadImage(), passing the 
 70   URL. 
 71   
 72   This will cause the image cache to attempt to load the image again the next 
 73   time you call loadImage on it.
 74   
 75   @extends SC.Object
 76   @since SproutCore 1.0
 77 */
 78 SC.imageCache = SC.Object.create(/** @scope SC.imageCache.prototype */ {
 79 
 80   /**
 81     The maximum number of images that can load from a single hostname at any
 82     one time.  For most browsers 4 is a reasonable number, though you may 
 83     tweak this on a browser-by-browser basis.
 84   */
 85   loadLimit: 4,
 86   
 87   /**
 88     The number of currently active requests on the cache. 
 89   */
 90   activeRequests: 0,
 91   
 92   /**
 93     Loads an image from the server, calling your target/method when complete.
 94     
 95     You should always pass at least a URL and optionally a target/method.  If 
 96     you do not pass the target/method, the image will be loaded in background 
 97     priority.  Usually, however, you will want to pass a callback to be 
 98     notified when the image has loaded.  Your callback should have a signature 
 99     like:
100 
101     {{{
102       imageDidLoad: function(imageUrl, imageOrError) { .. }
103     }}}
104     
105     If you do pass a target/method you can optionally also choose to load the 
106     image either in the foreground or in the background.  The image cache 
107     prioritizes foreground images over background images.  This does not impact 
108     how many images load at one time.
109     
110     @param {String} url
111     @param {Object} target
112     @param {String|Function} method
113     @param {Boolean} isBackgroundFlag
114     @returns {SC.imageCache} receiver
115   */
116   loadImage: function(url, target, method, isBackgroundFlag) {
117     // normalize params
118     var type = SC.typeOf(target);
119     if (SC.none(method) && SC.typeOf(target)===SC.T_FUNCTION) {
120       target = null; method = target ;
121     }
122     if (SC.typeOf(method) === SC.T_STRING) {
123       method = target[method];      
124     }
125     // if no callback is passed, assume background image.  otherwise, assume
126     // foreground image.
127     if (SC.none(isBackgroundFlag)) {
128       isBackgroundFlag = SC.none(target) && SC.none(method);
129     }
130     
131     // get image entry in cache.  If entry is loaded, just invoke callback
132     // and quit.
133     var entry = this._imageEntryFor(url) ;
134     if (entry.status === this.IMAGE_LOADED) {
135       if (method) method.call(target || entry.image, entry.url, entry.image);
136       
137     // otherwise, add to list of callbacks and queue image.
138     } else {
139       if (target || method) this._addCallback(entry, target, method);
140       entry.retainCount++; // increment retain count, regardless of callback
141       this._scheduleImageEntry(entry, isBackgroundFlag);
142     }
143   },
144   
145   /**
146     Invoke this method when you are finished with an image URL.  If you 
147     passed a target/method, you should also pass it here to remove it from
148     the list of callbacks.
149     
150     @param {String} url
151     @param {Object} target
152     @param {String|Function} method
153     @returns {SC.imageCache} receiver
154   */
155   releaseImage: function(url, target, method) {
156     
157     // get entry.  if there is no entry, just return as there is nothing to 
158     // do.
159     var entry = this._imageEntryFor(url, NO) ;
160     if (!entry) return this ;
161     
162     // there is an entry, decrement the retain count.  If <=0, delete!
163     if (--entry.retainCount <= 0) {
164       this._deleteEntry(entry); 
165     
166     // if >0, just remove target/method if passed
167     } else if (target || method) {
168       // normalize
169       var type = SC.typeOf(target);
170       if (SC.none(method) && SC.typeOf(target)===SC.T_FUNCTION) {
171         target = null; method = target ;
172       }
173       if (SC.typeOf(method) === SC.T_STRING) {
174         method = target[method];      
175       }
176 
177       // and remove
178       this._removeCallback(entry, target, method) ;
179     }
180   },
181 
182   /** 
183     Forces the image to reload the next time you try to load it.
184   */
185   reloadImage: function(url) {
186     var entry = this._imageEntryFor(url, NO); 
187     if (entry && entry.status===this.IMAGE_LOADED) {
188       entry.status = this.IMAGE_WAITING;
189     }
190   },
191   
192   /**
193     Initiates a load of the next image in the image queue.  Normally you will
194     not need to call this method yourself as it will be initiated 
195     automatically when the queue becomes active.
196   */
197   loadNextImage: function() {
198     var entry = null, queue;
199 
200     // only run if we don't have too many active request...
201     if (this.get('activeRequests')>=this.get('loadLimit')) return; 
202     
203     // first look in foreground queue
204     queue = this._foregroundQueue ;
205     while(queue.length>0 && !entry) entry = queue.shift();
206     
207     // then look in background queue
208     if (!entry) {
209       queue = this._backgroundQueue ;
210       while(queue.length>0 && !entry) entry = queue.shift();
211     }
212     this.set('isLoading', !!entry); // update isLoading...
213     
214     // if we have an entry, then initiate an image load with the proper 
215     // callbacks.
216     if (entry) {
217       // var img = (entry.image = new Image()) ;
218       var img = entry.image ;
219       img.onabort = this._imageDidAbort ;
220       img.onerror = this._imageDidError ;
221       img.onload = this._imageDidLoad ;
222       img.src = entry.url ;
223 
224       // add to loading queue.
225       this._loading.push(entry) ;
226     
227       // increment active requests and start next request until queue is empty
228       // or until load limit is reached.
229       this.incrementProperty('activeRequests');
230       this.loadNextImage();
231     } 
232   },
233   
234   // ..........................................................
235   // SUPPORT METHODS
236   // 
237 
238   /** @private Find or create an entry for the URL. */
239   _imageEntryFor: function(url, createIfNeeded) {
240     if (createIfNeeded === undefined) createIfNeeded = YES;
241     var entry = this._images[url] ;
242     if (!entry && createIfNeeded) {
243       var img = new Image() ;
244       entry = this._images[url] = { 
245         url: url, status: this.IMAGE_WAITING, callbacks: [], retainCount: 0, image: img
246       };
247       img.entry = entry ; // provide a link back to the image
248     }
249     return entry ;
250   },
251   
252   /** @private deletes an entry from the image queue, descheduling also */
253   _deleteEntry: function(entry) {
254     this._unscheduleImageEntry(entry) ;
255     delete this._images[entry.url];    
256   },
257   
258   /** @private 
259     Add a callback to the image entry.  First search the callbacks to make
260     sure this is only added once.
261   */
262   _addCallback: function(entry, target, method) {
263     var callbacks = entry.callbacks;
264 
265     // try to find in existing array
266     var handler = callbacks.find(function(x) {
267       return x[0]===target && x[1]===method;
268     }, this);
269     
270     // not found, add...
271     if (!handler) callbacks.push([target, method]);
272     callbacks = null; // avoid memory leaks
273     return this ;
274   },
275   
276   /** @private
277     Removes a callback from the image entry.  Removing a callback just nulls
278     out that position in the array.  It will be skipped when executing.
279   */
280   _removeCallback: function(entry, target, method) {
281     var callbacks = entry.callbacks ;
282     callbacks.forEach(function(x, idx) {
283       if (x[0]===target && x[1]===method) callbacks[idx] = null;
284     }, this);
285     callbacks = null; // avoid memory leaks
286     return this ;
287   },
288   
289   /** @private 
290     Adds an entry to the foreground or background queue to load.  If the 
291     loader is not already running, start it as well.  If the entry is in the
292     queue, but it is in the background queue, possibly move it to the
293     foreground queue.
294   */
295   _scheduleImageEntry: function(entry, isBackgroundFlag) {
296 
297     var background = this._backgroundQueue ;
298     var foreground = this._foregroundQueue ;
299     
300     // if entry is loaded, nothing to do...
301     if (entry.status === this.IMAGE_LOADED) return this;
302 
303     // if image is already in background queue, but now needs to be
304     // foreground, simply remove from background queue....
305     if ((entry.status===this.IMAGE_QUEUED) && !isBackgroundFlag && entry.isBackground) {
306       background[background.indexOf(entry)] = null ;
307       entry.status = this.IMAGE_WAITING ;
308     }
309     
310     // if image is not in queue already, add to queue.
311     if (entry.status!==this.IMAGE_QUEUED) {
312       var queue = (isBackgroundFlag) ? background : foreground ;
313       queue.push(entry);
314       entry.status = this.IMAGE_QUEUED ;
315       entry.isBackground = isBackgroundFlag ;
316     }
317     
318     // if the image loader is not already running, start it...
319     if (!this.isLoading) this.invokeLater(this.loadNextImage, 100);
320     this.set('isLoading', YES);
321     
322     return this ; // done!
323   },
324   
325   /** @private
326     Removes an entry from the foreground or background queue.  
327   */
328   _unscheduleImageEntry: function(entry) {
329     // if entry is not queued, do nothing
330     if (entry.status !== this.IMAGE_QUEUED) return this ;
331     
332     var queue = entry.isBackground ? this._backgroundQueue : this._foregroundQueue ;
333     queue[queue.indexOf(entry)] = null; 
334     
335     // if entry is loading, abort it also.  Call local abort method in-case
336     // browser decides not to follow up.
337     if (this._loading.indexOf(entry) >= 0) {
338       queue.image.abort();
339       this.imageStatusDidChange(entry, this.ABORTED);
340     }
341     
342     return this ;
343   },
344   
345   /** @private invoked by Image().  Note that this is the image instance */
346   _imageDidAbort: function() {
347     SC.RunLoop.begin();
348     SC.imageCache.imageStatusDidChange(this.entry, SC.imageCache.ABORTED);
349     SC.RunLoop.end();
350   },
351   
352   _imageDidError: function() {
353     SC.RunLoop.begin();
354     SC.imageCache.imageStatusDidChange(this.entry, SC.imageCache.ERROR);
355     SC.RunLoop.end();
356   },
357   
358   _imageDidLoad: function() {
359     SC.RunLoop.begin();
360     SC.imageCache.imageStatusDidChange(this.entry, SC.imageCache.LOADED);
361     SC.RunLoop.end();
362   },
363 
364   /** @private called whenever the image loading status changes.  Notifies
365     items in the queue and then cleans up the entry.
366   */
367   imageStatusDidChange: function(entry, status) {
368     if (!entry) return; // nothing to do...
369     
370     var url = entry.url ;
371     
372     // notify handlers.
373     var value ;
374     switch(status) {
375       case this.LOADED:
376         value = entry.image;
377         break;
378       case this.ABORTED:
379         value = SC.IMAGE_ABORTED_ERROR;
380         break;
381       case this.ERROR:
382         value = SC.IMAGE_FAILED_ERROR ;
383         break;
384       default:
385         value = SC.IMAGE_FAILED_ERROR ;
386         break;
387     }
388     entry.callbacks.forEach(function(x){
389       var target = x[0], method = x[1];
390       method.call(target, url, value);
391     },this);
392     
393     // now clear callbacks so they aren't called again.
394     entry.callbacks = [];
395     
396     // finally, if the image loaded OK, then set the status.  Otherwise
397     // set it to waiting so that further attempts will load again
398     entry.status = (status === this.LOADED) ? this.IMAGE_LOADED : this.IMAGE_WAITING ;
399     
400     // now cleanup image...
401     var image = entry.image ;
402     if (image) {
403       image.onload = image.onerror = image.onabort = null ; // no more notices
404       if (status !== this.LOADED) entry.image = null;
405     }
406 
407     // remove from loading queue and periodically compact
408     this._loading[this._loading.indexOf(entry)]=null;
409     if (this._loading.length > this.loadLimit*2) {
410       this._loading = this._loading.compact();
411     }
412     
413     this.decrementProperty('activeRequests');
414     this.loadNextImage() ;
415   },
416   
417   init: function() {
418     sc_super();
419     this._images = {};
420     this._loading = [] ;
421     this._foregroundQueue = [];
422     this._backgroundQueue = [];
423   },
424   
425   IMAGE_LOADED: "loaded",
426   IMAGE_QUEUED: "queued",
427   IMAGE_WAITING: "waiting",
428   
429   ABORTED: 'aborted',
430   ERROR: 'error',
431   LOADED: 'loaded'
432 });
