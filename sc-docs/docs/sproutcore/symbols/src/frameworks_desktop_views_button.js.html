 10 /** @class
 11 
 12   Implements a push-button-style button.  This class is used to implement 
 13   both standard push buttons and tab-style controls.  See also SC.CheckboxView
 14   and SC.RadioView which are implemented as field views, but can also be 
 15   treated as buttons.
 16   
 17   By default, a button uses the SC.Control mixin which will apply CSS 
 18   classnames when the state of the button changes:
 19     - active     when button is active
 20     - sel        when button is toggled to a selected state
 21   
 22   @extends SC.View
 23   @extends SC.Control
 24   @extends SC.Button
 25   @since SproutCore 1.0  
 26 */

153   /**
154     Called when the user presses a shortcut key, such as return or cancel,
155     associated with this button.
156 
157     Highlights the button to show that it is being triggered, then, after a
158     delay, performs the button's action.
159 
160     Does nothing if the button is disabled.
161 
162     @param {Event} evt
163     @returns {Boolean} success/failure of the request
164   */
165   triggerAction: function(evt) {
166     // If this button is disabled, we have nothing to do
167     if (!this.get('isEnabled')) return NO;
168 
169     // Set active state of the button so it appears highlighted
170     this.set('isActive', YES);
171 
172     // Invoke the actual action method after a small delay to give the user a
173     // chance to see the highlight. This is especially important if the button
174     // closes a pane, for example.
175     this.invokeLater('_triggerActionAfterDelay', 200, evt);
176     return YES;
177   },

179   /** @private
180     Called by triggerAction after a delay; this method actually
181     performs the action and restores the button's state.
182 
183     @param {Event} evt
184   */
185   _triggerActionAfterDelay: function(evt) {
186     this._action(evt, YES);
187     this.didTriggerAction();
188     this.set('isActive', NO);
189   },

241   render: function(context, firstTime) {
242     // add href attr if tagName is anchor...
243     var href, toolTip, classes, theme;
244     if (this.get('tagName') === 'a') {
245       href = this.get('href');
246       if (!href || (href.length === 0)) href = "javascript:;";
247       context.attr('href', href);
248     } else {
249       context.attr('role','button');
250     }
251 
252     // If there is a toolTip set, grab it and localize if necessary.
253     toolTip = this.get('toolTip') ;
254     if (SC.typeOf(toolTip) === SC.T_STRING) {
255       if (this.get('localize')) toolTip = toolTip.loc() ;
256       context.attr('title', toolTip) ;
257       context.attr('alt', toolTip) ;
258     }
259     
260     // add some standard attributes & classes.
261     classes = this._TEMPORARY_CLASS_HASH;
262     classes.def = this.get('isDefault');
263     classes.cancel = this.get('isCancel');
264     classes.icon = !!this.get('icon');
265     context.attr('role', 'button').setClass(classes);
266     theme = this.get('theme');
267     if (theme) context.addClass(theme);
268     
269     // render inner html 
270     this[this.get('renderStyle')](context, firstTime);
271   },

292     }

294       this.renderTitle(context, firstTime) ;
295     }

312   /** @private
313     Whenever the isDefault or isCancel property changes, update the display and change the keyEquivalent.
314   */  
315   _isDefaultOrCancelDidChange: function() {
316     var isDef = !!this.get('isDefault'),
317         isCancel = !isDef && this.get('isCancel') ;
318     
319     if(this.didChangeFor('defaultCancelChanged','isDefault','isCancel')) {
320       this.displayDidChange() ; // make sure to update the UI
321       if (isDef) {
322         this.set('keyEquivalent', 'return'); // change the key equivalent
323       } else if (isCancel) {
324         this.setIfChanged('keyEquivalent', 'escape') ;
325       } else {
326         //restore the default key equivalent
327         this.set("keyEquivalent",this._defaultKeyEquivalent);
328       }
329     }
330

335   /** @private 
336     On mouse down, set active only if enabled.
337   */    
338   mouseDown: function(evt) {
339     var buttonBehavior = this.get('buttonBehavior');
340 
341     if (!this.get('isEnabled')) return YES ; // handled event, but do nothing
342     this.set('isActive', YES);
343     this._isMouseDown = YES;
344 
345     if (buttonBehavior === SC.HOLD_BEHAVIOR) {
346       this._action(evt);
347     } else if (!this._isFocused && (buttonBehavior!==SC.PUSH_BEHAVIOR)) {
348       this._isFocused = YES ;
349       this.becomeFirstResponder();
350       if (this.get('isVisibleInWindow')) {
351         this.$()[0].focus();
352       }
353     }
354 
355     return YES ;
356   },
357 
358   /** @private
359     Remove the active class on mouseOut if mouse is down.
360   */  
361   mouseExited: function(evt) {
362     if (this._isMouseDown) {
363       this.set('isActive', NO);
364     }
365     return YES;
366   },
367 
368   /** @private
369     If mouse was down and we renter the button area, set the active state again.
370   */  
371   mouseEntered: function(evt) {
372     if (this._isMouseDown) {
373       this.set('isActive', YES);
374     }
375     return YES;
376   },
377 
378   /** @private
379     ON mouse up, trigger the action only if we are enabled and the mouse was released inside of the view.
380   */  
381   mouseUp: function(evt) {
382     if (this._isMouseDown) this.set('isActive', NO); // track independently in case isEnabled has changed
383     this._isMouseDown = false;
384 
385     if (this.get('buttonBehavior') !== SC.HOLD_BEHAVIOR) {
386       var inside = this.$().within(evt.target) ;
387       if (inside && this.get('isEnabled')) this._action(evt) ;
388     }
389 
390     return YES ;
391   },
392   
393   touchStart: function(touch){
394     var buttonBehavior = this.get('buttonBehavior');
395 
396     if (!this.get('isEnabled')) return YES ; // handled event, but do nothing
397     this.set('isActive', YES);
398 
399     if (buttonBehavior === SC.HOLD_BEHAVIOR) {
400       this._action(touch);
401     } else if (!this._isFocused && (buttonBehavior!==SC.PUSH_BEHAVIOR)) {
402       this._isFocused = YES ;
403       this.becomeFirstResponder();
404       if (this.get('isVisibleInWindow')) {
405         this.$()[0].focus();
406       }
407     }
408 
409     // don't want to do whatever default is...
410     touch.preventDefault();
411 
412     return YES;
413   },
414   
415   touchesDragged: function(evt, touches) {
416     if (!this.touchIsInBoundary(evt)) {
417       if (!this._touch_exited) this.set('isActive', NO);
418       this._touch_exited = YES;
419     } else {
420       if (this._touch_exited) this.set('isActive', YES);
421       this._touch_exited = NO;
422     }
423     
424     evt.preventDefault();
425     return YES;
426   },
427   
428   touchEnd: function(touch){
429     this._touch_exited = NO;
430     this.set('isActive', NO); // track independently in case isEnabled has changed
431 
432     if (this.get('buttonBehavior') !== SC.HOLD_BEHAVIOR) {
433       if (this.touchIsInBoundary(touch)) this._action();
434     }
435     
436     touch.preventDefault();
437     return YES ;
438   },

441   /** @private */
442   keyDown: function(evt) {
443     // handle tab key
444     if (evt.which === 9) {
445       var view = evt.shiftKey ? this.get('previousValidKeyView') : this.get('nextValidKeyView');
446       if(view) view.becomeFirstResponder();
447       else evt.allowDefault();
448       return YES ; // handled
449     }    
450     if (evt.which === 13) {
451       this.triggerAction(evt);
452       return YES ; // handled
453     }
454     return NO; 
455   },

457   /** @private  Perform an action based on the behavior of the button.
458   
459    - toggle behavior: switch to on/off state
460    - on behavior: turn on.
461    - off behavior: turn off.
462    - otherwise: invoke target/action
463   */
464   _action: function(evt, skipHoldRepeat) {
465     switch(this.get('buttonBehavior')) {
466       
467     // When toggling, try to invert like values. i.e. 1 => 0, etc.
468     case SC.TOGGLE_BEHAVIOR:
469       var sel = this.get('isSelected') ;
470       if (sel) {
471         this.set('value', this.get('toggleOffValue')) ;
472       } else {
473         this.set('value', this.get('toggleOnValue')) ;
474       }
475       break ;
476       
477     // set value to on.  change 0 => 1.
478     case SC.TOGGLE_ON_BEHAVIOR:
479       this.set('value', this.get('toggleOnValue')) ;
480       break ;
481       
482     // set the value to false. change 1 => 0
483     case SC.TOGGLE_OFF_BEHAVIOR:
484       this.set('value', this.get('toggleOffValue')) ;
485       break ;
486 
487     case SC.HOLD_BEHAVIOR:
488       this._runHoldAction(evt, skipHoldRepeat);
489       break ;
490 
491     // otherwise, just trigger an action if there is one.
492     default:
493       //if (this.action) this.action(evt);
494       this._runAction(evt);
495     }
496   },
497 
498   /** @private */
499   _runAction: function(evt) {
500     var action = this.get('action'),
501         target = this.get('target') || null,
502         rootResponder = this.getPath('pane.rootResponder');
503 
504     if (action) {
505       if (this._hasLegacyActionHandler()) {
506         // old school... V
507         this._triggerLegacyActionHandler(evt);
508       } else {
509         if (rootResponder) {
510           // newer action method + optional target syntax...
511           rootResponder.sendAction(action, target, this, this.get('pane'));
512         }
513       }
514     }
515   },
516 
517   /** @private */
518   _runHoldAction: function(evt, skipRepeat) {
519     if (this.get('isActive')) {
520       this._runAction();
521 
522       if (!skipRepeat) {
523         // This run loop appears to only be necessary for testing
524         SC.RunLoop.begin();
525         this.invokeLater('_runHoldAction', this.get('holdInterval'), evt);
526         SC.RunLoop.end();
527       }
528     }
529   },
530   
531   /** @private */
532   _hasLegacyActionHandler: function()
533   {
534     var action = this.get('action');
535     if (action && (SC.typeOf(action) === SC.T_FUNCTION)) return true;
536     if (action && (SC.typeOf(action) === SC.T_STRING) && (action.indexOf('.') != -1)) return true;
537     return false;
538   },
539 
540   /** @private */
541   _triggerLegacyActionHandler: function( evt )
542   {
543     if (!this._hasLegacyActionHandler()) return false;
544     
545     var action = this.get('action');
546     if (SC.typeOf(action) === SC.T_FUNCTION) this.action(evt);
547     if (SC.typeOf(action) === SC.T_STRING) {
548       eval("this.action = function(e) { return "+ action +"(this, e); };");
549       this.action(evt);
550     }
551   },
552   
553   /** tied to the isEnabled state */
554   acceptsFirstResponder: function() {
555     if(!SC.SAFARI_FOCUS_BEHAVIOR) return this.get('isEnabled');
556     else return NO;
557   }.property('isEnabled'),
558   
559   willBecomeKeyResponderFrom: function(keyView) {
560     // focus the text field.
561     if (!this._isFocused) {
562       this._isFocused = YES ;
563       this.becomeFirstResponder();
564       if (this.get('isVisibleInWindow')) {
565         var elem=this.$()[0];
566         if (elem) elem.focus();
567       }
568     }
569   },
570   
571   willLoseKeyResponderTo: function(responder) {
572     if (this._isFocused) this._isFocused = NO ;
573   },
574   
575   didAppendToDocument: function() {
576     if(parseInt(SC.browser.msie, 0)===7 && this.get('useStaticLayout')){
577       var layout = this.get('layout'),
578           elem = this.$(), w=0;
579       if(elem && elem[0] && (w=elem[0].clientWidth) && w!==0 && this._labelMinWidthIE7===0){
580         var label = this.$('.sc-button-label'),
581             paddingRight = parseInt(label.css('paddingRight'),0),
582             paddingLeft = parseInt(label.css('paddingLeft'),0),
583             marginRight = parseInt(label.css('marginRight'),0),
584             marginLeft = parseInt(label.css('marginLeft'),0);
585         if(marginRight=='auto') console.log(marginRight+","+marginLeft+","+paddingRight+","+paddingLeft);
586         if(!paddingRight && isNaN(paddingRight)) paddingRight = 0;
587         if(!paddingLeft && isNaN(paddingLeft)) paddingLeft = 0;
588         if(!marginRight && isNaN(marginRight)) marginRight = 0;
589         if(!marginLeft && isNaN(marginLeft)) marginLeft = 0;
590         
591         this._labelMinWidthIE7 = w-(paddingRight + paddingLeft)-(marginRight + marginLeft);
592         label.css('minWidth', this._labelMinWidthIE7+'px');
593       }else{
594         this.invokeLater(this.didAppendToDocument, 1);
595       }
596     }
597   }
