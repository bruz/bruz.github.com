 12 SC.mixin({
 13   _isReadyBound: NO,
 14   
 15   /** @private configures the ready event handler if needed */
 16   _bindReady: function() {
 17     if (this._isReadyBound) return;
 18     this._isReadyBound = YES ;
 19 
 20     // Mozilla, Opera (see further below for it) and webkit nightlies 
 21     // currently support this event.  Use the handy event callback
 22     if ( document.addEventListener && !SC.browser.opera) {
 23       document.addEventListener( "DOMContentLoaded", SC._didBecomeReady, NO );
 24     }
 25 
 26     // If IE is used and is not in a frame
 27     // Continually check to see if the document is ready
 28     if (SC.browser.msie && (window === top)) {
 29       (function() {
 30         if (SC.isReady) return;
 31         try {
 32           // If IE is used, use the trick by Diego Perini
 33           // http://javascript.nwbox.com/IEContentLoaded/
 34           document.documentElement.doScroll("left");
 35         } catch( error ) {
 36           setTimeout( arguments.callee, 0 );
 37           return;
 38         }
 39         // and execute any waiting functions
 40         SC._didBecomeReady();
 41       })();
 42     }
 43 
 44     if ( SC.browser.opera ) {
 45       document.addEventListener( "DOMContentLoaded", function () {
 46         if (SC.isReady) return;
 47         for (var i = 0; i < document.styleSheets.length; i++) {
 48           if (document.styleSheets[i].disabled) {
 49             setTimeout( arguments.callee, 0 );
 50             return;
 51           }
 52         }
 53         // and execute any waiting functions
 54         SC._didBecomeReady();
 55       }, NO);
 56     }
 57 
 58     if (SC.browser.safari && SC.browser.safari < 530.0 ) {
 59       console.error("ready() is not yet supported on Safari 3.1 and earlier");
 60       // TODO: implement ready() in < Safari 4 
 61       // var numStyles;
 62       // (function(){
 63       //   if (SC.isReady) return;
 64       //   if ( document.readyState != "loaded" && document.readyState != "complete" ) {
 65       //     setTimeout( arguments.callee, 0 );
 66       //     return;
 67       //   }
 68       //   if ( numStyles === undefined ) numStyles = 0 ;
 69       //     //numStyles = SC.$("style, link[rel=stylesheet]").length;
 70       //   if ( document.styleSheets.length != numStyles ) {
 71       //     setTimeout( arguments.callee, 0 );
 72       //     return;
 73       //   }
 74       //   // and execute any waiting functions
 75       //   SC._didBecomeReady();
 76       // })();
 77     }
 78 
 79     // A fallback to window.onload, that will always work
 80     SC.Event.add( window, "load", SC._didBecomeReady);
 81   },
 82 
 83   /** @private handlers scheduled to execute on ready. */
 84   _readyQueue: [],
 85   
 86   _afterReadyQueue: [],
 87 
 88   isReady: NO,
 89   
 90   /** @private invoked when the document becomes ready. */
 91   _didBecomeReady: function() {
 92     // Only call once
 93     if (SC.isReady) return ;
 94     if (typeof SC.mapDisplayNames === SC.T_FUNCTION) SC.mapDisplayNames();
 95     if (typeof SC.addInvokeOnceLastDebuggingInfo === SC.T_FUNCTION) SC.addInvokeOnceLastDebuggingInfo();
 96      
 97     // setup locale
 98     SC.Locale.createCurrentLocale();
 99     
100     // if there is a body tag on the document, set the language
101     if (document && document.getElementsByTagName) {
102       var body = document.getElementsByTagName('body')[0];
103       if (body) {
104         var className = body.className ;
105         var language = SC.Locale.currentLanguage.toLowerCase() ;
106         body.className = (className && className.length>0) ? [className, language].join(' ') : language ;
107       }
108     }
109 
110     SC.Benchmark.start('ready') ;
111     
112     // Begin runloop
113     SC.RunLoop.begin();
114     
115     var handler, ary, idx, len ;
116 
117     // correctly handle queueing new SC.ready() calls
118     do {
119       ary = SC._readyQueue ;
120       SC._readyQueue = [] ; // reset
121       for (idx=0, len=ary.length; idx<len; idx++) {
122         handler = ary[idx] ;
123         var target = handler[0] || document ;
124         var method = handler[1] ;
125         if (method) method.call(target) ;
126       }
127     } while (SC._readyQueue.length > 0) ;
128 
129     // okay, now we're ready (any SC.ready() calls will now be called immediately)
130     SC.isReady = YES ;
131     
132     // clear the queue
133     SC._readyQueue = null ;
134     
135     // trigger any bound ready events
136     SC.Event.trigger(document, "ready", null, NO);
137     
138     // Remove any loading div
139     if (SC.removeLoading) SC.$('#loading').remove();
140     
141     // Now execute main, if defined and SC.UserDefaults is ready
142     if(SC.userDefaults.get('ready')){
143       if ((SC.mode === SC.APP_MODE) && (typeof main != "undefined") && (main instanceof Function) && !SC.suppressMain) main();
144     } 
145     else {
146       SC.userDefaults.readyCallback(window, main);
147     }
148     
149     // end run loop.  This is probably where a lot of bindings will trigger
150     SC.RunLoop.end() ; 
151     
152     SC.Benchmark.end('ready') ;
153     SC.Benchmark.log();
154   },
155   
156   /** 
157     Add the passed target and method to the queue of methods to invoke when
158     the document is ready.  These methods will be called after the document
159     has loaded and parsed, but before the main() function is called.
160     
161     Methods are called in the order they are added.
162   
163     If you add a ready handler when the main document is already ready, then
164     your handler will be called immediately.
165     
166     @param target {Object} optional target object
167     @param method {Function} method name or function to execute
168     @returns {SC}
169   */
170   ready: function(target, method) {
171     var queue = this._readyQueue;
172     
173     // normalize
174     if (method === undefined) {
175       method = target; target = null ;
176     } else if (SC.typeOf(method) === SC.T_STRING) {
177       method = target[method] ;
178     }
179     
180     if (!method) return this; // nothing to do.
181     
182     // if isReady, execute now.
183     if (this.isReady) return method.call(target || document) ;
184     
185     // otherwise, add to queue.
186     queue.push([target, method]) ;
187     return this ; 
188   }
189   
190 }) ;
191 
192 SC._bindReady() ;
193 SC.removeLoading = YES;
