 10 /** @class
 11 
 12   A cascading data source will actually forward requests onto an array of 
 13   additional data sources, stopping when one of the data sources returns YES,
 14   indicating that it handled the request.  
 15   
 16   You can use a cascading data source to tie together multiple data sources,
 17   treating them as a single namespace.
 18   
 19   h2. Configuring a Cascade Data Source
 20   
 21   You will usually define your cascading data source in your main method after
 22   all the classes you have are loaded.
 23   
 24   {{{
 25     MyApp.dataSource = SC.CascadeDataSource.create({
 26       dataSources: "prefs youtube photos".w(),
 27       
 28       prefs:   MyApp.PrefsDataSource.create({ root: "/prefs" }),
 29       youtube: YouTube.YouTubeDataSource.create({ apiKey: "123456" }),
 30       photos:  MyApp.PhotosDataSource.create({ root: "photos" })
 31       
 32     });
 33     
 34     MyApp.store.set('dataSource', MyApp.dataSource);
 35   }}}
 36   
 37   Note that the order you define your dataSources property will determine the
 38   order in which requests will cascade from the store.
 39   
 40   Alternatively, you can use a more jQuery-like API for defining your data
 41   sources:
 42   
 43   {{{
 44     MyApp.dataSource = SC.CascadeDataSource.create()
 45       .from(MyApp.PrefsDataSource.create({ root: "/prefs" }))
 46       .from(YouTube.YouTubeDataSource.create({ apiKey: "123456" }))
 47       .from(MyApp.PhotosDataSource.create({ root: "photos" }));
 48 
 49     MyApp.store.set('dataSource', MyApp.dataSource);
 50   }}}
 51 
 52   In this case, the order you call from() will determine the order the request
 53   will cascade.
 54   
 55   @extends SC.DataSource
 56   @since SproutCore 1.0
 57 */
 58 SC.CascadeDataSource = SC.DataSource.extend( 
 59   /** @scope SC.CascadeDataSource.prototype */ {
 60 
 61   /**
 62     The data sources used by the cascade, in the order that they are to be 
 63     followed.  Usually when you define the cascade, you will define this
 64     array.
 65     
 66     @property {Array}
 67   */
 68   dataSources: null,
 69 
 70   /**
 71     Add a data source to the list of sources to use when cascading.  Used to
 72     build the data source cascade effect.
 73 
 74     @param {SC.DataSource} dataSource a data source instance to add.
 75     @returns {SC.CascadeDataSource} receiver
 76   */
 77   from: function(dataSource) {
 78     var dataSources = this.get('dataSources');
 79     if (!dataSources) this.set('dataSources', dataSources = []);
 80     dataSources.push(dataSource);
 81     return this ;
 82   },
 83     
 84   // ..........................................................
 85   // SC.STORE ENTRY POINTS
 86   // 
 87   
 88   /** @private - just cascades */
 89   fetch: function(store, query) {
 90     var sources = this.get('dataSources'), 
 91         len     = sources ? sources.length : 0,
 92         ret     = NO,
 93         cur, source, idx;
 94     
 95     for(idx=0; (ret !== YES) && idx<len; idx++) {
 96       source = sources.objectAt(idx);
 97       cur = source.fetch ? source.fetch.call(source, store, query) : NO;
 98       ret = this._handleResponse(ret, cur);
 99     }
100     
101     return ret ;
102   },
103   
104   
105   /** @private - just cascades */
106   retrieveRecords: function(store, storeKeys) {
107     var sources = this.get('dataSources'), 
108         len     = sources ? sources.length : 0,
109         ret     = NO,
110         cur, source, idx;
111     
112     for(idx=0; (ret !== YES) && idx<len; idx++) {
113       source = sources.objectAt(idx);
114       cur = source.retrieveRecords.call(source, store, storeKeys);
115       ret = this._handleResponse(ret, cur);
116     }
117     
118     return ret ;
119   },
120 
121   /** @private - just cascades */
122   commitRecords: function(store, createStoreKeys, updateStoreKeys, destroyStoreKeys) {
123     var sources = this.get('dataSources'), 
124         len     = sources ? sources.length : 0,
125         ret     = NO,
126         cur, source, idx;
127     
128     for(idx=0; (ret !== YES) && idx<len; idx++) {
129       source = sources.objectAt(idx);
130       cur = source.commitRecords.call(source, store, createStoreKeys, updateStoreKeys, destroyStoreKeys);
131       ret = this._handleResponse(ret, cur);
132     }
133     
134     return ret ;
135   },
136 
137   /** @private - just cascades */
138   cancel: function(store, storeKeys) {
139     var sources = this.get('dataSources'), 
140         len     = sources ? sources.length : 0,
141         ret     = NO,
142         cur, source, idx;
143     
144     for(idx=0; (ret !== YES) && idx<len; idx++) {
145       source = sources.objectAt(idx);
146       cur = source.cancel.call(source, store, storeKeys);
147       ret = this._handleResponse(ret, cur);
148     }
149     
150     return ret ;
151   },
152   
153   // ..........................................................
154   // INTERNAL SUPPORT
155   // 
156   
157   /** @private */
158   init: function() {
159     sc_super();
160     
161     // if a dataSources array is defined, look for any strings and lookup 
162     // the same on the data source.  Replace.
163     var sources = this.get('dataSources'),
164         idx     = sources ? sources.get('length') : 0,
165         source;
166     while(--idx>=0) {
167       source = sources[idx];
168       if (SC.typeOf(source) === SC.T_STRING) sources[idx] = this.get(source);
169     }
170     
171   },
172 
173   /** @private - Determine the proper return value. */
174   _handleResponse: function(current, response) {
175     if (response === YES) return YES ;
176     else if (current === NO) return (response === NO) ? NO : SC.MIXED_STATE ;
177     else return SC.MIXED_STATE ;
178   }
179     
180 });
