 10 /**
 11   Handles parsing and validating of numbers.
 12   
 13   @extends SC.Validator
 14   @author Charles Jolley
 15   @version 1.0
 16   @class
 17 */
 18 SC.Validator.Number = SC.Validator.extend(
 19 /** @scope SC.Validator.Number.prototype */ {
 20 
 21   /**
 22     Number of decimal places to show.  
 23     
 24     If 0, then numbers will be treated as integers.  Otherwise, numbers will
 25     show with a fixed number of decimals.
 26   */
 27   places: 0,
 28   
 29   fieldValueForObject: function(object, form, field) {
 30     switch(SC.typeOf(object)) {
 31       case SC.T_NUMBER:
 32         object = object.toFixed(this.get('places')) ;
 33         break ;
 34       case SC.T_NULL:
 35       case SC.T_UNDEFINED:
 36         object = '';
 37         break ;
 38     }
 39     return object ;
 40   },
 41 
 42   objectForFieldValue: function(value, form, field) {
 43     // strip out commas
 44     var result;
 45     value = value.replace(/,/g,'');
 46     switch(SC.typeOf(value)) {
 47       case SC.T_STRING:
 48         if (value.length === 0) {
 49           value = null ;
 50         } else if (this.get('places') > 0) {
 51           value = parseFloat(value) ;
 52         } else {
 53           if(value.length==1 && value.match(/-/)) value = null;
 54           else {
 55             result = parseInt(value,0) ;
 56             if(isNaN(result)){
 57               value = SC.uniJapaneseConvert(value);
 58               value = parseInt(value,0) ;
 59               if(isNaN(value)) value='';
 60             }else value = result;
 61           }
 62         }
 63         break ;
 64       case SC.T_NULL:
 65       case SC.T_UNDEFINED:
 66         value = null ;
 67         break ;
 68     }
 69     return value ;
 70   },
 71   
 72   validate: function(form, field) { 
 73     var value = field.get('fieldValue') ;
 74     return (value === '') || !(isNaN(value) || isNaN(parseFloat(value))) ; 
 75   },
 76   
 77   validateError: function(form, field) {
 78     var label = field.get('errorLabel') || 'Field' ;
 79     return SC.$error("Invalid.Number(%@)".loc(label), label) ;
 80   },
 81   
 82   /** 
 83     Allow only numbers, dashes, period, and commas
 84   */
 85   validateKeyDown: function(form, field, charStr) {
 86     var text = field.$input().val();
 87     if (!text) text='';
 88     text+=charStr;
 89     if(this.get('places')===0){
 90       if(charStr.length===0) return true;
 91       else return text.match(/^[\-{0,1}]?[0-9,\0]*/)[0]===text;
 92     }else {
 93       if(charStr.length===0) return true;
 94       else return text.match(/^[\-{0,1}]?[0-9,\0]*\.?[0-9\0]+/)===text;
 95     }
 96   }
 97     
 98 }) ;
