 10 /**
 11   @class
 12   
 13   Implements support for Ajax requests using XHR, JSON-P and other prototcols.
 14   
 15   SC.Request is much like an inverted version of the request/response objects
 16   you receive when implementing HTTP servers.  
 17   
 18   To send a request, you just need to create your request object, configure
 19   your options, and call send() to initiate the request.
 20   
 21   @extends SC.Object
 22   @extends SC.Copyable
 23   @extends SC.Freezable
 24   @since SproutCore 1.0
 25 */

209   COPY_KEYS: 'isAsynchronous isJSON isXML address type timeout body responseClass willSend didSend willReceive didReceive'.w(),
210   
211   /**
212     Returns a copy of the current request.  This will only copy certain
213     properties so if you want to add additional properties to the copy you
214     will need to override copy() in a subclass.
215     
216     @returns {SC.Request} new request
217   */
218   copy: function() {
219     var ret = {},
220         keys = this.COPY_KEYS,
221         loc  = keys.length, 
222         key, listeners, headers;
223         
224     while(--loc>=0) {
225       key = keys[loc];
226       if (this.hasOwnProperty(key)) ret[key] = this.get(key);
227     }
228     
229     if (this.hasOwnProperty('listeners')) {
230       ret.listeners = SC.copy(this.get('listeners'));
231     }
232     
233     if (this.hasOwnProperty('_headers')) {
234       ret._headers = SC.copy(this._headers);
235     }
236     
237     ret.source = this.get('source') || this ;
238     
239     return this.constructor.create(ret);
240   },
241   
242   /**
243     To set headers on the request object.  Pass either a single key/value 
244     pair or a hash of key/value pairs.  If you pass only a header name, this
245     will return the current value of the header.
246     
247     @param {String|Hash} key
248     @param {String} value
249     @returns {SC.Request|Object} receiver
250   */
251   header: function(key, value) {
252     var headers;
253     
254     if (SC.typeOf(key) === SC.T_STRING) {
255       headers = this._headers ;
256       if (arguments.length===1) {
257         return headers ? headers[key] : null;
258       } else {
259         this.propertyWillChange('headers');
260         if (!headers) headers = this._headers = {};
261         headers[key] = value;
262         this.propertyDidChange('headers');
263         return this;
264       }
265     
266     // handle parsing hash of parameters
267     } else if (value === undefined) {
268       headers = key;
269       this.beginPropertyChanges();
270       for(key in headers) {
271         if (!headers.hasOwnProperty(key)) continue ;
272         this.header(key, headers[key]);
273       }
274       this.endPropertyChanges();
275       return this;
276     }
277 
278     return this ;
279   },

282     Converts the current request to be asynchronous.

316   /** 
317     Called just before a request is enqueued.  This will encode the body 
318     into JSON if it is not already encoded.
319   */
320   _prep: function() {
321     var hasContentType = !!this.header('Content-Type');
322     if (this.get('isJSON') && !hasContentType) {
323       this.header('Content-Type', 'application/json');
324     } else if (this.get('isXML') && !hasContentType) {
325       this.header('Content-Type', 'text/xml');
326     }
327     return this ;
328   },
329   
330   /**
331     Will fire the actual request.  If you have set the request to use JSON 
332     mode then you can pass any object that can be converted to JSON as the 
333     body.  Otherwise you should pass a string body.
334     
335     @param {String|Object} body (optional)
336     @returns {SC.Response} new response object
337   */  
338   send: function(body) {
339     // Sanity-check:  Be sure a timeout value was not specified if the request
340     // is synchronous (because it wouldn't work).
341     var timeout = this.get('timeout');
342     if (timeout) {
343       if (!this.get('isAsynchronous')) throw "Timeout values cannot be used with synchronous requests";
344     }
345     else if (timeout === 0) {
346       throw "The timeout value must either not be specified or must be greater than 0";
347     }
348     
349     if (body) this.set('body', body);
350     return SC.Request.manager.sendRequest(this.copy()._prep());
351   },
352 
353   /**
354     Resends the current request.  This is more efficient than calling send()
355     for requests that have already been used in a send.  Otherwise acts just
356     like send().  Does not take a body argument.
357     
358     @returns {SC.Response} new response object
359   */
360   resend: function() {
361     var req = this.get('source') ? this : this.copy()._prep();
362     return SC.Request.manager.sendRequest(req);
363   },
364   
365   /**
366     Configures a callback to execute when a request completes.  You must pass
367     at least a target and action/method to this and optionally a status code.
368     You may also pass additional parameters which will be passed along to your
369     callback.
370     
371     h2. Scoping With Status Codes
372     
373     If you pass a status code as the first option to this method, then your 
374     notification callback will only be called if the response status matches
375     the code.  For example, if you pass 201 (or SC.Request.CREATED) then 
376     your method will only be called if the response status from the server
377     is 201.
378     
379     You can also pass "generic" status codes such as 200, 300, or 400, which
380     will be invoked anytime the status code is the range if a more specific 
381     notifier was not registered first and returned YES.  
382     
383     Finally, passing a status code of 0 or no status at all will cause your
384     method to be executed no matter what the resulting status is unless a 
385     more specific notifier was registered and returned YES.
386     
387     h2. Callback Format
388     
389     Your notification callback should expect to receive the Response object
390     as the first parameter plus any additional parameters that you pass.  
391     
392     @param {Number} status
393     @param {Object} target
394     @param {String|function} action
395     @param {Hash} params
396     @returns {SC.Request} receiver
397   */
398   notify: function(status, target, action, params) {
399     
400     // normalize status
401     var hasStatus = YES ;
402     if (SC.typeOf(status) !== SC.T_NUMBER) {
403       params = SC.A(arguments).slice(2);
404       action = target;
405       target = status;
406       status = 0 ;
407       hasStatus = NO ;
408     } else params = SC.A(arguments).slice(3);
409     
410     var listeners = this.get('listeners');
411     if (!listeners) this.set('listeners', listeners = {});
412     listeners[status] = { target: target, action: action, params: params };
413 
414     return this;
415   }

471 /**
472   The request manager coordinates all of the active XHR requests.  It will
473   only allow a certain number of requests to be active at a time; queuing 
474   any others.  This allows you more precise control over which requests load
475   in which order.
476 */
477 SC.Request.manager = SC.Object.create( SC.DelegateSupport, {
478 
479   /**
480     Maximum number of concurrent requests allowed.  6 for all browsers.
481     
482     @property {Number}
483   */
484   maxRequests: 6,
485 
486   /**
487     Current requests that are inflight.
488     
489     @property {Array}
490   */
491   inflight: [],
492   
493   /**
494     Requests that are pending and have not been started yet.
495   
496     @property {Array}
497   */
498   pending: [],
499 
500   // ..........................................................
501   // METHODS
502   // 
503   
504   /**
505     Invoked by the send() method on a request.  This will create a new low-
506     level transport object and queue it if needed.
507     
508     @param {SC.Request} request the request to send
509     @returns {SC.Object} response object
510   */
511   sendRequest: function(request) {
512     if (!request) return null ;
513     
514     // create low-level transport.  copy all critical data for request over
515     // so that if the request has been reconfigured the transport will still
516     // work.
517     var response = request.get('responseClass').create({ request: request });
518 
519     // add to pending queue
520     this.get('pending').pushObject(response);
521     this.fireRequestIfNeeded();
522     
523     return response ;
524   },
525 
526   /** 
527     Cancels a specific request.  If the request is pending it will simply
528     be removed.  Otherwise it will actually be cancelled.
529     
530     @param {Object} response a response object
531     @returns {Boolean} YES if cancelled
532   */
533   cancel: function(response) {
534 
535     var pending = this.get('pending'),
536         inflight = this.get('inflight'),
537         idx ;
538 
539     if (pending.indexOf(response) >= 0) {
540       this.propertyWillChange('pending');
541       pending.removeObject(response);
542       this.propertyDidChange('pending');
543       return YES;
544       
545     } else if (inflight.indexOf(response) >= 0) {
546       
547       response.cancel();
548       
549       inflight.removeObject(response);
550       this.fireRequestIfNeeded();
551       return YES;
552 
553     } else return NO ;
554   },  
555 
556   /**
557     Cancels all inflight and pending requests.  
558     
559     @returns {Boolean} YES if any items were cancelled.
560   */
561   cancelAll: function() {
562     if (this.get('pending').length || this.get('inflight').length) {
563       this.set('pending', []);
564       this.get('inflight').forEach(function(r) { r.cancel(); });
565       this.set('inflight', []);
566       return YES;
567       
568     } else return NO ;
569   },
570   
571   /**
572     Checks the inflight queue.  If there is an open slot, this will move a 
573     request from pending to inflight.
574     
575     @returns {Object} receiver
576   */
577   fireRequestIfNeeded: function() {
578     var pending = this.get('pending'), 
579         inflight = this.get('inflight'),
580         max = this.get('maxRequests'),
581         next ;
582         
583     if ((pending.length>0) && (inflight.length<max)) {
584       next = pending.shiftObject();
585       inflight.pushObject(next);
586       next.fire();
587     }
588   },
589 
590   /**
591     Called by a response/transport object when finishes running.  Removes 
592     the transport from the queue and kicks off the next one.
593   */
594   transportDidClose: function(response) {
595     this.get('inflight').removeObject(response);
596     this.fireRequestIfNeeded();
597   }
598   
599 });
