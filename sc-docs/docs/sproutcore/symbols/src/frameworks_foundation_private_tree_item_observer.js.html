 11 /** 
 12   @ignore
 13   @class
 14   
 15   A TreeNode is an internal class that will manage a single item in a tree
 16   when trying to display the item in a hierarchy. 
 17   
 18   When displaying a tree of objects, a tree item object will be nested to 
 19   cover every object that might have child views.
 20   
 21   TreeNode stores an array which contains either a number pointing to the 
 22   next place in the array there is a child item or it contains a child item.
 23   
 24   @extends SC.Object
 25   @extends SC.Array
 26   @extends SC.CollectionContent
 27   @since SproutCore 1.0
 28 */

 94   /**
 95     IndexSet of children with branches.  This will ask the delegate to name 
 96     these indexes.  The default implementation will iterate over the children
 97     of the item but a more optimized version could avoid touching each item.
 98     
 99     @property
100     @type SC.IndexSet
101   */
102   branchIndexes: function() {
103     var item = this.get('item'), 
104         len, pitem, idx, children, ret;
105     
106     // no item - no branches
107     if (!item) return SC.IndexSet.EMPTY;
108     
109     // if item is treeItemContent then ask it directly
110     else if (item.isTreeItemContent) {
111       pitem  = this.get('parentItem');
112       idx    = this.get('index') ;
113       return item.treeItemBranchIndexes(pitem, idx);
114       
115     // otherwise, loop over children and determine disclosure state for each
116     } else {
117       children = this.get('children');
118       if (!children) return null; // no children - no branches
119       ret = SC.IndexSet.create();
120       len = children.get('length');
121       pitem = item ; // save parent
122       
123       for(idx=0;idx<len;idx++) {
124         if (!(item = children.objectAt(idx))) continue ;
125         if (!this._computeChildren(item, pitem, idx)) continue; // no chil'en
126         if (this._computeDisclosureState(item, pitem, idx) !== SC.LEAF_NODE) {
127           ret.add(idx);
128         }
129       }
130 
131       return ret.get('length')>0 ? ret : null;
132     }

152   /**
153     Get the object at the specified index.  This will talk the tree info
154     to determine the proper place.  The offset should be relative to the 
155     start of this tree item.  Calls recursively down the tree.
156     
157     This should only be called with an index you know is in the range of item
158     or its children based on looking at the length.
159   */
160   objectAt: function(index) {
161     var len   = this.get('length'),
162         item  = this.get('item'), 
163         cache = this._objectAtCache,
164         cur   = index,
165         loc   = 0,
166         indexes, children;
167      
168     if (index >= len) return undefined;
169     if (this.get('isHeaderVisible')) {
170       if (index === 0) return item;
171       else cur--;
172     }
173     item = null; 
174 
175     if (!cache) cache = this._objectAtCache = [];
176     if ((item = cache[index]) !== undefined) return item ;
177 
178     children = this.get('children');
179     if (!children) return undefined; // no children - nothing to get
180     
181     // loop through branch indexes, reducing the offset until it matches 
182     // something we might actually return.
183     if (indexes = this.get('branchIndexes')) {
184       indexes.forEach(function(i) {
185         if (item || (i > cur)) return ; // past end - nothing to do
186 
187         var observer = this.branchObserverAt(i), len;
188         if (!observer) return ; // nothing to do
189 
190         // if cur lands inside of this observer's length, use objectAt to get
191         // otherwise, just remove len from cur.
192         len = observer.get('length') ;
193         if (i+len > cur) {
194           item = observer.objectAt(cur-i);
195           cur  = -1;
196         } else cur -= len-1 ;
197         
198       },this);
199     }
200     
201     if (cur>=0) item = children.objectAt(cur); // get internal if needed
202     cache[index] = item ; // save in cache 
203     
204     return item ;
205   },

207   /**
208     Implements SC.Array.replace() primitive.  For this method to succeed, the
209     range you replace must lie entirely within the same parent item, otherwise
210     this will raise an exception.
211     
212     h3. The Operation Parameter
213     
214     Note that this replace method accepts an additional parameter "operation"
215     which is used when you try to insert an item on a boundary between 
216     branches whether it should be inserted at the end of the previous group
217     after the group.  If you don't pass operation, the default is 
218     SC.DROP_BEFORE, which is the expected behavior.
219     
220     Even if the operation is SC.DROP_AFTER, you should still pass the actual
221     index where you expect the item to be inserted.  For example, if you want
222     to insert AFTER the last index of an 3-item array, you would still call:
223     
224     {{{
225       observer.replace(3, 0, [object1 .. objectN], SC.DROP_AFTER)
226     }}}
227     
228     The operation is simply used to disambiguate whether the insertion is
229     intended to be AFTER the previous item or BEFORE the items you are
230     replacing.
231     
232     @param {Number} start the starting index
233     @param {Number} amt the number of items to replace
234     @param {SC.Array} objects array of objects to insert
235     @param {Number} operation either SC.DROP_BEFORE or SC.DROP_AFTER
236     @returns {SC.TreeItemObserver} receiver
237   */
238   replace: function(start, amt, objects, operation) {
239 
240     var cur      = start,
241         observer = null,
242         indexes, len, max;
243         
244     if (operation === undefined) operation = SC.DROP_BEFORE;
245     
246     // adjust the start location based on branches, possibly passing on to an
247     // observer.
248     if (this.get('isHeaderVisible')) cur--; // exclude my own header item 
249     if (cur < 0) throw "Tree Item cannot replace itself";
250 
251     // remove branch lengths.  If the adjusted start location lands inside of
252     // another branch, then just let that observer handle it.
253     if (indexes = this.get('branchIndexes')) {
254       indexes.forEach(function(i) {
255         if (observer || (i>=cur)) return ; // nothing to do
256         if (!(observer = this.branchObserverAt(i))) return; // nothing to do
257         len = observer.get('length');
258         
259         // if this branch range is before the start loc, just remove it and 
260         // go on.  If cur is somewhere inside of the range, then save to pass
261         // on.  Note use of operation to determine the abiguous end op.
262         if ((i+len === cur) && operation === SC.DROP_AFTER) cur -= i;
263         else if (i+len > cur) cur -= i; // put inside of nested range
264         else {
265           cur -= len-1; observer = null ;
266         }
267       }, this);      
268     }
269       
270     // if an observer was saved, pass on call.
271     if (observer) {
272       observer.replace(cur, amt, objects, operation);
273       return this;
274     }
275     
276     // no observer was saved, which means cur points to an index inside of
277     // our own range.  Now amt just needs to be adjusted to remove any
278     // visible branches as well.
279     max = cur + amt;
280     if (amt>1 && indexes) { // if amt is 1 no need...
281       indexes.forEachIn(cur, indexes.get('max')-cur, function(i) {
282         if (i > max) return; // nothing to do
283         if (!(observer = this.branchObserverAt(i))) return; // nothing to do
284         len = observer.get('length');
285         max -= len-1;
286       }, this);
287     }
288     
289     // get amt back out.  if amt is negative, it means that the range passed
290     // was not cleanly inside of this range.  raise an exception.
291     amt = max-cur; 
292     
293     // ok, now that we are adjusted, get the children and forward the replace
294     // call on.  if there are no children, bad news...
295     var children = this.get('children');
296     if (!children) throw "cannot replace() tree item with no children";
297 
298     if ((amt < 0) || (max>children.get('length'))) {
299       throw "replace() range must lie within a single tree item";
300     }
301 
302     children.replace(cur, amt, objects, operation);
303     
304     // don't call enumerableContentDidChange() here because, as an observer,
305     // we should be notified by the children array itself.
306     
307     return this;
308   },

310   /**
311     Called whenever the content for the passed observer has changed.  Default
312     version notifies the parent if it exists and updates the length.
313     
314     The start, amt and delta params should reflect changes to the children
315     array, not to the expanded range for the wrapper.
316   */
317   observerContentDidChange: function(start, amt, delta) {
318     
319     // clear caches
320     this.invalidateBranchObserversAt(start);
321     this._objectAtCache = this._outlineLevelCache = null;
322     this._disclosureStateCache = null;
323     this._contentGroupIndexes = NO;
324     this.notifyPropertyChange('branchIndexes');
325     
326     var oldlen = this.get('length'),
327         newlen = this._computeLength(),
328         parent = this.get('parentObserver'), set;
329     
330     // update length if needed
331     if (oldlen !== newlen) this.set('length', newlen);
332     
333     // if we have a parent, notify that parent that we have changed.
334     if (!this._notifyParent) return this; // nothing more to do
335     
336     if (parent) {
337       set = SC.IndexSet.create(this.get('index'));
338       parent._childrenRangeDidChange(parent.get('children'), null, '[]', set);
339       
340     // otherwise, note the enumerable content has changed.  note that we need
341     // to convert the passed change to reflect the computed range
342     } else {
343       if (oldlen === newlen) {
344         amt = this.expandChildIndex(start+amt);
345         start = this.expandChildIndex(start);
346         amt = amt - start ;
347         delta = 0 ;
348         
349       } else {
350         start = this.expandChildIndex(start);
351         amt   = newlen - start;
352         delta = newlen - oldlen ;
353       }
354 
355       this.enumerableContentDidChange(start, amt, delta);
356     }
357   },

359   /**
360     Accepts a child index and expands it to reflect any nested groups.
361   */
362   expandChildIndex: function(index) {
363     
364     var ret = index;
365     if (this.get('isHeaderVisible')) index++;
366 
367     // fast path
368     var branches = this.get('branchIndexes');
369     if (!branches || branches.get('length')===0) return ret;
370     
371     // we have branches, adjust for their length
372     branches.forEachIn(0, index, function(idx) {
373       ret += this.branchObserverAt(idx).get('length')-1;
374     }, this);
375     
376     return ret; // add 1 for item header
377   },

385   /**
386     Called by the collection view to return any group indexes.  The default 
387     implementation will compute the indexes one time based on the delegate 
388     treeItemIsGrouped
389   */
390   contentGroupIndexes: function(view, content) {
391     if (content !== this) return null; // only care about receiver
392 
393     var ret = this._contentGroupIndexes;
394     if (ret !== NO) return ret ;
395     
396     // if this is not the root item, never do grouping
397     if (this.get('parentObserver')) return null;
398     
399     var item = this.get('item'), group, indexes, len, cur, loc, children;
400     
401     if (item && item.isTreeItemContent) group = item.get('treeItemIsGrouped');
402     else group = !!this.delegate.get('treeItemIsGrouped');
403     
404     // if grouping is enabled, build an index set with all of our local 
405     // groups.
406     if (group) {
407       ret      = SC.IndexSet.create();
408       indexes  = this.get('branchIndexes');
409       children = this.get('children');
410       len      = children ? children.get('length') : 0;
411       cur = loc = 0;
412       
413       if (indexes) {
414         indexes.forEach(function(i) {
415           ret.add(cur, (i+1)-loc); // add loc -> i to set
416           cur += (i+1)-loc;
417           loc = i+1 ;
418           
419           var observer = this.branchObserverAt(i);
420           if (observer) cur += observer.get('length')-1;
421         }, this);
422       }
423 
424       if (loc<len) ret.add(cur, len-loc);
425     } else ret = null;
426     
427     this._contentGroupIndexes = ret ;
428     return ret;
429   },

436   /**
437     Returns the outline level for the specified index.
438   */
439   contentIndexOutlineLevel: function(view, content, index) {
440     if (content !== this) return -1; // only care about us
441     
442     var cache = this._outlineLevelCache;
443     if (cache && (cache[index] !== undefined)) return cache[index];
444     if (!cache) cache = this._outlineLevelCache = [];
445     
446     var len   = this.get('length'),
447         cur   = index,
448         loc   = 0,
449         ret   = null,
450         indexes, children, observer;
451     
452     if (index >= len) return -1;
453      
454     if (this.get('isHeaderVisible')) {
455       if (index === 0) return cache[0] = this.get('outlineLevel')-1;
456       else cur--;
457     }
458 
459     // loop through branch indexes, reducing the offset until it matches 
460     // something we might actually return.
461     if (indexes = this.get('branchIndexes')) {
462       indexes.forEach(function(i) {
463         if ((ret!==null) || (i > cur)) return ; // past end - nothing to do
464 
465         var observer = this.branchObserverAt(i), len;
466         if (!observer) return ; // nothing to do
467 
468         // if cur lands inside of this observer's length, use objectAt to get
469         // otherwise, just remove len from cur.
470         len = observer.get('length') ;
471         if (i+len > cur) {
472           ret  = observer.contentIndexOutlineLevel(view, observer, cur-i);
473           cur  = -1;
474         } else cur -= len-1 ;
475         
476       },this);
477     }
478     
479     if (cur>=0) ret = this.get('outlineLevel'); // get internal if needed
480     cache[index] = ret ; // save in cache 
481     return ret ;

507     // loop through branch indexes, reducing the offset until it matches

532   /**
533     Expands the specified content index.  This will search down until it finds
534     the branchObserver responsible for this item and then calls _collapse on
535     it.
536   */
537   contentIndexExpand: function(view, content, idx) {
538 
539     var indexes, cur = idx, children, item;
540     
541     if (content !== this) return; // only care about us
542     if (this.get('isHeaderVisible')) {
543       if (idx===0) {
544         this._expand(this.get('item'));
545         return;
546       } else cur--;
547     } 
548     
549     if (indexes = this.get('branchIndexes')) {
550       indexes.forEach(function(i) {
551         if (i >= cur) return; // past end - nothing to do
552         var observer = this.branchObserverAt(i), len;
553         if (!observer) return ; 
554         
555         len = observer.get('length');
556         if (i+len > cur) {
557           observer.contentIndexExpand(view, observer, cur-i);
558           cur = -1 ; //done
559         } else cur -= len-1;
560         
561       }, this);  
562     }
563     
564     // if we are still inside of the range then maybe pass on to a child item
565     if (cur>=0) {
566       children = this.get('children');  
567       item     = children ? children.objectAt(cur) : null;
568       if (item) this._expand(item, this.get('item'), cur);
569     }

572   /**
573     Called to collapse a content index item if it is currently in an open 
574     disclosure state.  The default implementation does nothing.  
575     
576     @param {SC.CollectionView} view the collection view
577     @param {SC.Array} content the content object
578     @param {Number} idx the content index
579     @returns {void}
580   */
581   contentIndexCollapse: function(view, content, idx) {
582 
583     var indexes, children, item, cur = idx;
584         
585     if (content !== this) return; // only care about us
586     if (this.get('isHeaderVisible')) {
587       if (idx===0) {
588         this._collapse(this.get('item'));
589         return;
590       } else cur--;
591     } 
592     
593     
594     if (indexes = this.get('branchIndexes')) {
595       indexes.forEach(function(i) {
596         if (i >= cur) return; // past end - nothing to do
597         var observer = this.branchObserverAt(i), len;
598         if (!observer) return ; 
599         
600         len = observer.get('length');
601         if (i+len > cur) {
602           observer.contentIndexCollapse(view, observer, cur-i);
603           cur = -1 ; //done
604         } else cur -= len-1;
605         
606       }, this);  
607     }
608 
609     // if we are still inside of the range then maybe pass on to a child item
610     if (cur>=0) {
611       children = this.get('children');  
612       item     = children ? children.objectAt(cur) : null;
613       if (item) this._collapse(item, this.get('item'), cur);
614     }

621   /**
622     Returns the branch item for the specified index.  If none exists yet, it
623     will be created.
624   */
625   branchObserverAt: function(index) {
626     var byIndex = this._branchObserversByIndex,
627         indexes = this._branchObserverIndexes,
628         ret, parent, pitem, item, children, guid, del ;
629         
630     if (!byIndex) byIndex = this._branchObserversByIndex = [];
631     if (!indexes) {
632       indexes = this._branchObserverIndexes = SC.IndexSet.create();
633     }
634 
635     if (ret = byIndex[index]) return ret ; // use cache
636 
637     // no observer for this content exists, create one
638     children = this.get('children');
639     item   = children ? children.objectAt(index) : null ;
640     if (!item) return null ; // can't create an observer for a null item
641     
642     byIndex[index] = ret = SC.TreeItemObserver.create({
643       item:     item,
644       delegate: this.get('delegate'),
645       parentObserver:   this,
646       index:  index,
647       outlineLevel: this.get('outlineLevel')+1
648     });
649 
650     indexes.add(index); // save for later invalidation
651     return ret ;
652   },

654   /**
655     Invalidates any branch observers on or after the specified index range.
656   */
657   invalidateBranchObserversAt: function(index) {
658     var byIndex = this._branchObserversByIndex,
659         indexes = this._branchObserverIndexes;
660 
661     if (!byIndex || byIndex.length<=index) return this ; // nothing to do
662     if (index < 0) index = 0 ;
663     
664     // destroy any observer on or after the range
665     indexes.forEachIn(index, indexes.get('max')-index, function(i) {
666       var observer = byIndex[i];
667       if (observer) observer.destroy();
668     }, this);
669     
670     byIndex.length = index; // truncate to dump extra indexes
671     
672     return this;
673   },

679   init: function() {
680     sc_super();
681     
682     // begin all properties on item if there is one.  This will allow us to
683     // track important property changes.
684     var item = this.get('item');
685     if (!item) throw "SC.TreeItemObserver.item cannot be null";

687     item.addObserver('*', this, this._itemPropertyDidChange);
688     this._itemPropertyDidChange(item, '*');
689     this._notifyParent = YES ; // avoid infinite loops
690   },

692   /**
693     Called just before a branch observer is removed.  Should stop any 
694     observering and invalidate any child observers.
695   */
696   destroy: function() {
697     this.invalidateBranchObserversAt(0);
698     this._objectAtCache = null ;
699     
700     // cleanup observing
701     var item = this.get('item');
702     if (item) item.removeObserver('*', this, this._itemPropertyDidChange);
703     
704     var children = this._children,
705         ro = this._childrenRangeObserver;
706     if (children && ro) children.removeRangeObserver(ro);
707     
708     sc_super();
709   },

711   /**
712     Called whenever a property changes on the item.  Determines if either the
713     children array or the disclosure state has changed and then notifies as 
714     necessary..
715   */
716   _itemPropertyDidChange: function(target, key) {
717     var children = this.get('children'),
718         state    = this.get('disclosureState'),
719         item     = this.get('item'),
720         next ;
721         
722     this.beginPropertyChanges();
723     
724     next = this._computeDisclosureState(item);
725     if (state !== next) this.set('disclosureState', next);
726     
727     next = this._computeChildren(item);
728     if (children !== next) this.set('children', next);
729     
730     this.endPropertyChanges();
731   },
732   
733   /**
734     Called whenever the children or disclosure state changes.  Begins or ends
735     observing on the children array so that changes can propogate outward.
736   */
737   _childrenDidChange: function() {
738     var state = this.get('disclosureState'),
739         cur   = state === SC.BRANCH_OPEN ? this.get('children') : null,
740         last  = this._children,
741         ro    = this._childrenRangeObserver;
742         
743     if (last === cur) return this; //nothing to do
744     if (ro) last.removeRangeObserver(ro);
745     if (cur) {
746       this._childrenRangeObserver = 
747           cur.addRangeObserver(null, this, this._childrenRangeDidChange);
748     } else this._childrenRangeObserver = null;
749     
750     this._children = cur ;
751     this._childrenRangeDidChange(cur, null, '[]', null);
752     
753   }.observes("children", "disclosureState"),
754 
755   /**
756     Called anytime the actual content of the children has changed.  If this 
757     changes the length property, then notifies the parent that the content
758     might have changed.
759   */
760   _childrenRangeDidChange: function(array, objects, key, indexes) {
761     var children = this.get('children'),
762         len = children ? children.get('length') : 0,
763         min = indexes ? indexes.get('min') : 0,
764         max = indexes ? indexes.get('max') : len,
765         old = this._childrenLen || 0;
766         
767     this._childrenLen = len; // save for future calls
768     this.observerContentDidChange(min, max-min, len-old);
769   },
770   
771   /**
772     Computes the current disclosure state of the item by asking the item or 
773     the delegate.  If no pitem or index is passed, the parentItem and idex 
774     will be used.
775   */
776   _computeDisclosureState: function(item, pitem, index) {
777     var key, del;
778 
779     // no item - assume leaf node
780     if (!item || !this._computeChildren(item)) return SC.LEAF_NODE;
781     
782     // item implement TreeItemContent - call directly
783     else if (item.isTreeItemContent) {
784       if (pitem === undefined) pitem = this.get('parentItem');
785       if (index === undefined) index = this.get('index');
786       return item.treeItemDisclosureState(pitem, index);
787       
788     // otherwise get treeItemDisclosureStateKey from delegate
789     } else {
790       key = this._treeItemIsExpandedKey ;
791       if (!key) {
792         del = this.get('delegate');
793         key = del ? del.get('treeItemIsExpandedKey') : 'treeItemIsExpanded';
794         this._treeItemIsExpandedKey = key ;
795       }
796       return item.get(key) ? SC.BRANCH_OPEN : SC.BRANCH_CLOSED;
797     }
798   },

800   /**
801     Collapse the item at the specified index.  This will either directly 
802     modify the property on the item or call the treeItemCollapse() method.
803   */
804   _collapse: function(item, pitem, index) {
805     var key, del;
806 
807     // no item - assume leaf node
808     if (!item || !this._computeChildren(item)) return this;
809     
810     // item implement TreeItemContent - call directly
811     else if (item.isTreeItemContent) {
812       if (pitem === undefined) pitem = this.get('parentItem');
813       if (index === undefined) index = this.get('index');
814       item.treeItemCollapse(pitem, index);
815       
816     // otherwise get treeItemDisclosureStateKey from delegate
817     } else {
818       key = this._treeItemIsExpandedKey ;
819       if (!key) {
820         del = this.get('delegate');
821         key = del ? del.get('treeItemIsExpandedKey') : 'treeItemIsExpanded';
822         this._treeItemIsExpandedKey = key ;
823       }
824       item.setIfChanged(key, NO);
825     }
826     
827     return this ;
828   },

860   /**
861     Computes the children for the passed item.
862   */
863   _computeChildren: function(item) {
864     var del, key;
865     
866     // no item - no children
867     if (!item) return null;
868     
869     // item implement TreeItemContent - call directly
870     else if (item.isTreeItemContent) return item.get('treeItemChildren');
871           
872     // otherwise get treeItemChildrenKey from delegate
873     else {
874       key = this._treeItemChildrenKey ;
875       if (!key) {
876         del = this.get('delegate');
877         key = del ? del.get('treeItemChildrenKey') : 'treeItemChildren';
878         this._treeItemChildrenKey = key ;
879       }
880       return item.get(key);
881     }

884   /**
885     Computes the length of the array by looking at children.
886   */
887   _computeLength: function() {
888     var ret = this.get('isHeaderVisible') ? 1 : 0,
889         state = this.get('disclosureState'),
890         children = this.get('children'),
891         indexes ;
892 
893     // if disclosure is open, add children count + length of branch observers.
894     if ((state === SC.BRANCH_OPEN) && children) {
895       ret += children.get('length');
896       if (indexes = this.get('branchIndexes')) {
897         indexes.forEach(function(idx) {
898           var observer = this.branchObserverAt(idx);
899           ret += observer.get('length')-1;
900         }, this);
901       }
902     } 
903     return ret ;
904   }
