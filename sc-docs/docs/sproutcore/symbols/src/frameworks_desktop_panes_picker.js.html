 11 /** 
 12   Popular customized picker position rules:
 13   default: initiated just below the anchor. 
 14            shift x, y to optimized picker visibility and make sure top-left corner is always visible.
 15   menu :   same as default rule +
 16            default(1,4,3) or custom offset below the anchor for default location to fine tunned visual alignment +
 17            enforce min left(7px)/right(8px) padding to the window
 18   fixed :  default(1,4,3) or custom offset below the anchor for default location to cope with specific anchor and skip fitPositionToScreen
 19   pointer :take default [0,1,2,3,2] or custom matrix to choose one of four perfect pointer positions.Ex:
 20            perfect right (0) > perfect left (1) > perfect top (2) > perfect bottom (3)
 21            fallback to perfect top (2)
 22   menu-pointer :take default [3,0,1,2,3] or custom matrix to choose one of four perfect pointer positions.Ex:
 23           perfect bottom (3) > perfect right (0) > perfect left (1) > perfect top (2)
 24           fallback to perfect bottom (3)
 25 */
 26 SC.PICKER_MENU = 'menu';
 27 SC.PICKER_FIXED = 'fixed';
 28 SC.PICKER_POINTER = 'pointer';
 29 SC.PICKER_MENU_POINTER = 'menu-pointer';
 30 /** 
 31   Pointer layout for perfect right/left/top/bottom
 32 */
 33 SC.POINTER_LAYOUT = ["perfectRight", "perfectLeft", "perfectTop", "perfectBottom"];

 35 /**
 36   @class
 37 
 38   Displays a non-modal, self anchor positioned picker pane.
 39 
 40   The default way to use the picker pane is to simply add it to your page like this:
 41   
 42   {{{
 43     SC.PickerPane.create({
 44       layout: { width: 400, height: 200 },
 45       contentView: SC.View.extend({
 46       })
 47     }).popup(anchor);
 48   }}}
 49   
 50   This will cause your picker pane to display.
 51   
 52   Picker pane is a simple way to provide non-modal messaging that won't 
 53   blocks the user's interaction with your application.  Picker panes are 
 54   useful for showing important detail informations with optimized position around anchor.
 55   They provide a better user experience than modal panel.
 56 
 57   Examples for applying popular customized picker position rules:
 58   
 59   1. default:   
 60   {{{
 61     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
 62     }).popup(anchor);
 63   }}}
 64 
 65   2. menu below the anchor with default offset matrix [1,4,3]:   
 66   {{{
 67     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
 68     }).popup(anchor, SC.PICKER_MENU);
 69   }}}
 70 
 71   3. menu on the right side of anchor with custom offset matrix [2,6,0]:   
 72   {{{
 73     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
 74     }).popup(anchor, SC.PICKER_MENU, [2,6,0]);
 75   }}}
 76 
 77   4. fixed below the anchor with default offset matrix [1,4,3]:   
 78   {{{
 79     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
 80     }).popup(anchor, SC.PICKER_FIXED);
 81   }}}
 82 
 83   5. fixed on the right side of anchor with custom offset matrix [-22,-17,0]:   
 84   {{{
 85     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
 86     }).popup(anchor, SC.PICKER_FIXED, [-22,-17,0]);
 87   }}}
 88 
 89   6. pointer with default position pref matrix [0,1,2,3,2]:   
 90   {{{
 91     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
 92     }).popup(anchor, SC.PICKER_POINTER);
 93   }}}
 94   perfect right (0) > perfect left (1) > perfect top (2) > perfect bottom (3)
 95   fallback to perfect top (2)
 96 
 97   7. pointer with custom position pref matrix [3,0,1,2,2]:   
 98   {{{
 99     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
100     }).popup(anchor, SC.PICKER_POINTER, [3,0,1,2,2]);
101   }}}
102 
103   perfect bottom (3) > perfect right (0) > perfect left (1) > perfect top (2)
104   fallback to perfect top (2)
105 
106   8. menu-pointer with default position pref matrix [3,0,1,2,3]:
107   {{{
108     SC.PickerPane.create({layout: { width: 400, height: 200 },contentView: SC.View.extend({})
109     }).popup(anchor, SC.PICKER_MENU_POINTER);
110   }}}
111   perfect bottom (3) > perfect right (0) > perfect left (1) > perfect top (2)
112   fallback to perfect bottom (3)
113   
114   @extends SC.PalettePane
115   @since SproutCore 1.0
116 */

174   popup: function(anchorViewOrElement, preferType, preferMatrix, pointerOffset) {
175     var anchor = anchorViewOrElement.isView ? anchorViewOrElement.get('layer') : anchorViewOrElement;
176     this.beginPropertyChanges();
177     this.set('anchorElement',anchor) ;
178     if (preferType) this.set('preferType',preferType) ;
179     if (preferMatrix) this.set('preferMatrix',preferMatrix) ;

181     this.endPropertyChanges();
182     this.positionPane();
183     this.append();

186   /** @private
187     The ideal position for a picker pane is just below the anchor that 
188     triggered it + offset of specific preferType. Find that ideal position, 
189     then call fitPositionToScreen to get final position. If anchor is missing, 
190     fallback to center.
191   */  
192   positionPane: function() {
193     var anchor       = this.get('anchorElement'),
194         preferType   = this.get('preferType'),
195         preferMatrix = this.get('preferMatrix'),
196         layout       = this.get('layout'),
197         origin;
198     
199     // usually an anchorElement will be passed.  The ideal position is just 
200     // below the anchor + default or custom offset according to preferType.
201     // If that is not possible, fitPositionToScreen will take care of that for 
202     // other alternative and fallback position.
203     if (anchor) {
204       anchor = this.computeAnchorRect(anchor);
205       if(anchor.x ===0 && anchor.y ===0) return ;
206       origin = SC.cloneRect(anchor);
207 
208       if (preferType) {
209         switch (preferType) {
210           case SC.PICKER_MENU:
211           case SC.PICKER_FIXED:
212             if(!preferMatrix || preferMatrix.length !== 3) {
213               // default below the anchor with fine tunned visual alignment 
214               // for Menu to appear just below the anchorElement.
215               this.set('preferMatrix', [1, 4, 3]) ;
216             }
217 
218             // fine tunned visual alignment from preferMatrix
219             origin.x += ((this.preferMatrix[2]===0) ? origin.width : 0) + this.preferMatrix[0] ;
220             origin.y += ((this.preferMatrix[2]===3) ? origin.height : 0) + this.preferMatrix[1];    
221             break;
222           default:
223             origin.y += origin.height ;
224             break;
225         }   
226       } else {
227         origin.y += origin.height ;
228       }
229       origin = this.fitPositionToScreen(origin, this.get('frame'), anchor) ;
230 
231       this.adjust({ width: origin.width, height: origin.height, left: origin.x, top: origin.y });
232     // if no anchor view has been set for some reason, just center.
233     } else {
234       this.adjust({ width: layout.width, height: layout.height, centerX: 0, centerY: 0 });
235     }
236     this.updateLayout();
237     return this ;
238   },

240   /** @private
241     This method will return ret (x, y, width, height) from a rectangular element
242     Notice: temp hack for calculating visiable anchor height by counting height 
243     up to window bottom only. We do have 'clippingFrame' supported from view.
244     But since our anchor can be element, we use this solution for now.
245   */  
246   computeAnchorRect: function(anchor) {
247     var bounding, ret, cq,
248         wsize = SC.RootResponder.responder.computeWindowSize();
249     // Some browsers natively implement getBoundingClientRect, so if it's
250     // available we'll use it for speed.
251     if (anchor.getBoundingClientRect) {
252       // Webkit and Firefox 3.5 will get everything they need by
253       // calling getBoundingClientRect()
254       bounding = anchor.getBoundingClientRect();
255       ret = {
256         x:      bounding.left,
257         y:      bounding.top,
258         width:  bounding.width,
259         height: bounding.height
260       };
261       // If width and height are undefined this means we are in IE or FF<3.5
262       // if we didnt get the frame dimensions the do the calculations
263       // based on an element
264       if(ret.width===undefined || ret.height===undefined){
265         cq = SC.$(anchor);
266         ret.width = cq.outerWidth();
267         ret.height = cq.outerHeight();
268       }
269     }
270     else {
271       // Only really old versions will have to go through this code path.
272       ret   = SC.viewportOffset(anchor); // get x & y
273       cq    = SC.$(anchor);
274       ret.width = cq.outerWidth();
275       ret.height = cq.outerHeight();
276     }
277     ret.height = (wsize.height-ret.y) < ret.height ? (wsize.height-ret.y) : ret.height;
278     if(!SC.browser.msie && window.scrollX>0 || window.scrollY>0){
279       ret.x+=window.scrollX;
280       ret.y+=window.scrollY;
281     }else if(SC.browser.msie && (document.documentElement.scrollTop>0 || document.documentElement.scrollLeft>0)){
282       ret.x+=document.documentElement.scrollLeft;
283       ret.y+=document.documentElement.scrollTop;
284     }
285     return ret ;
286   },

288   /** @private
289     This method will dispatch to the right re-position rule according to preferType
290   */  
291   fitPositionToScreen: function(preferredPosition, picker, anchor) {
292     // get window rect.
293     //if(this._prefPosX && this._prefPosY)
294     
295     var wsize = SC.RootResponder.responder.computeWindowSize() ;
296 
297     var wret = { x: 0, y: 0, width: wsize.width, height: wsize.height } ;
298     picker.x = preferredPosition.x ; picker.y = preferredPosition.y ;
299 
300     if(this.preferType) {
301       switch(this.preferType) {
302         case SC.PICKER_MENU:
303           // apply menu re-position rule
304           picker = this.fitPositionToScreenMenu(wret, picker, this.get('isSubMenu')) ;
305           break;
306         case SC.PICKER_MENU_POINTER:
307           this.setupPointer(anchor);
308           picker = this.fitPositionToScreenMenuPointer(wret, picker, anchor);
309           break;
310         case SC.PICKER_POINTER:
311           // apply pointer re-position rule
312           this.setupPointer(anchor);
313           picker = this.fitPositionToScreenPointer(wret, picker, anchor) ;
314           break;
315           
316         case SC.PICKER_FIXED:
317           // skip fitPositionToScreen
318           break;
319         default:
320           break;
321       }     
322     } else {
323       // apply default re-position rule
324       picker = this.fitPositionToScreenDefault(wret, picker, anchor) ;
325     }
326     this.displayDidChange();
327     return picker ;
328   },

330   /** @private
331     re-position rule migrated from old SC.OverlayPaneView. 
332     shift x, y to optimized picker visibility and make sure top-left corner is always visible.
333   */
334   fitPositionToScreenDefault: function(w, f, a) {
335     // make sure the right edge fits on the screen.  If not, anchor to 
336     // right edge of anchor or right edge of window, whichever is closer.
337     if (SC.maxX(f) > w.width) {
338       var mx = Math.max(SC.maxX(a), f.width) ;
339       f.x = Math.min(mx, w.width) - f.width ;
340     }
341 
342     // if the left edge is off of the screen, try to position at left edge
343     // of anchor.  If that pushes right edge off screen, shift back until 
344     // right is on screen or left = 0
345     if (SC.minX(f) < 0) {
346       f.x = SC.minX(Math.max(a,0)) ;
347       if (SC.maxX(f) > w.width) {
348         f.x = Math.max(0, w.width - f.width);
349       }
350     }
351 
352     // make sure bottom edge fits on screen.  If not, try to anchor to top
353     // of anchor or bottom edge of screen.
354     if (SC.maxY(f) > w.height) {
355       mx = Math.max((a.y - f.height), 0) ;
356       if (mx > w.height) {
357         f.y = Math.max(0, w.height - f.height) ;
358       } else f.y = mx ;
359     }
360 
361     // if Top edge is off screen, try to anchor to bottom of anchor. If that
362     // pushes off bottom edge, shift up until it is back on screen or top =0
363     if (SC.minY(f) < 0) {
364       mx = Math.min(SC.maxY(a), (w.height - a.height)) ;
365       f.y = Math.max(mx, 0) ;
366     }
367     return f ;    
368   },

370   /** @private
371     Reposition the pane in a way that is optimized for menus.
372 
373     Specifically, we want to ensure that the pane is at least 7 pixels from
374     the left side of the screen, and 20 pixels from the right side.
375 
376     If the menu is a submenu, we also want to reposition the pane to the left
377     of the parent menu if it would otherwise exceed the width of the viewport.
378   */
379   fitPositionToScreenMenu: function(windowFrame, paneFrame, subMenu) {
380 
381     // Set up init location for submenu
382     if (subMenu) {
383       paneFrame.x -= this.get('submenuOffsetX');
384       paneFrame.y -= Math.floor(this.get('menuHeightPadding')/2);
385     }
386 
387     // If the right edge of the pane is within 20 pixels of the right edge
388     // of the window, we need to reposition it.
389     if( (paneFrame.x + paneFrame.width) > (windowFrame.width-20) ) {
390       if (subMenu) {
391         // Submenus should be re-anchored to the left of the parent menu
392         paneFrame.x = paneFrame.x - (paneFrame.width*2);
393       } else {
394         // Otherwise, just position the pane 20 pixels from the right edge
395         paneFrame.x = windowFrame.width - paneFrame.width - 20;
396       }
397     }
398 
399     // Make sure we are at least 7 pixels from the left edge of the screen.
400     if( paneFrame.x < 7 ) paneFrame.x = 7;
401     
402     if (paneFrame.y < 7) {
403       paneFrame.height += paneFrame.y;
404       paneFrame.y = 7;
405     }
406 
407     // If the height of the menu is bigger than the window height, resize it.
408     if( paneFrame.height+paneFrame.y+35 >= windowFrame.height){
409       if (paneFrame.height+50 >= windowFrame.height) {
410         paneFrame.y = SC.MenuPane.VERTICAL_OFFSET;
411         paneFrame.height = windowFrame.height - (SC.MenuPane.VERTICAL_OFFSET*2);
412       } else {
413         paneFrame.y += (windowFrame.height - (paneFrame.height+paneFrame.y+35));
414       }
415     }
416 
417     return paneFrame ;
418   },

420   /** @private
421     Reposition the pane in a way that is optimized for menus that have a
422     point element.
423 
424     This simply calls fitPositionToScreenPointer, then ensures that the menu
425     does not exceed the height of the viewport.
426 
427     @returns {Rect}
428   */
429   fitPositionToScreenMenuPointer: function(w, f, a) {
430     f = this.fitPositionToScreenPointer(w,f,a);
431 
432     // If the height of the menu is bigger than the window height, resize it.
433     if( f.height+f.y+35 >= w.height){
434         f.height = w.height - f.y - (SC.MenuPane.VERTICAL_OFFSET*2) ;
435     }
436 
437     return f;
438   },

440   /** @private
441     re-position rule for triangle pointer picker.
442   */
443   fitPositionToScreenPointer: function(w, f, a) {
444     var offset = [this.pointerOffset[0], this.pointerOffset[1],
445                   this.pointerOffset[2], this.pointerOffset[3]];
446 
447     // initiate perfect positions matrix
448     // 4 perfect positions: right > left > top > bottom
449     // 2 coordinates: x, y
450     // top-left corner of 4 perfect positioned f  (4x2)
451     var prefP1    =[[a.x+a.width+offset[0],                   a.y+parseInt(a.height/2,0)-40],
452                     [a.x-f.width+offset[1],                   a.y+parseInt(a.height/2,0)-40],
453                     [a.x+parseInt((a.width/2)-(f.width/2),0), a.y-f.height+offset[2]],
454                     [a.x+parseInt((a.width/2)-(f.width/2),0), a.y+a.height+offset[3]]];
455     // bottom-right corner of 4 perfect positioned f  (4x2)
456     var prefP2    =[[a.x+a.width+f.width+offset[0],                   a.y+parseInt(a.height/2,0)+f.height-24],
457                     [a.x+offset[1],                                   a.y+parseInt(a.height/2,0)+f.height-24],
458                     [a.x+parseInt((a.width/2)-(f.width/2),0)+f.width, a.y+offset[2]],
459                     [a.x+parseInt((a.width/2)-(f.width/2),0)+f.width, a.y+a.height+f.height+offset[3]]];
460     // cutoff of 4 perfect positioned f: top, right, bottom, left  (4x4)
461     var cutoffPrefP =[[prefP1[0][1]>0 ? 0 : 0-prefP1[0][1], prefP2[0][0]<w.width ? 0 : prefP2[0][0]-w.width, prefP2[0][1]<w.height ? 0 : prefP2[0][1]-w.height, prefP1[0][0]>0 ? 0 : 0-prefP1[0][0]], 
462                       [prefP1[1][1]>0 ? 0 : 0-prefP1[1][1], prefP2[1][0]<w.width ? 0 : prefP2[1][0]-w.width, prefP2[1][1]<w.height ? 0 : prefP2[1][1]-w.height, prefP1[1][0]>0 ? 0 : 0-prefP1[1][0]],
463                       [prefP1[2][1]>0 ? 0 : 0-prefP1[2][1], prefP2[2][0]<w.width ? 0 : prefP2[2][0]-w.width, prefP2[2][1]<w.height ? 0 : prefP2[2][1]-w.height, prefP1[2][0]>0 ? 0 : 0-prefP1[2][0]],
464                       [prefP1[3][1]>0 ? 0 : 0-prefP1[3][1], prefP2[3][0]<w.width ? 0 : prefP2[3][0]-w.width, prefP2[3][1]<w.height ? 0 : prefP2[3][1]-w.height, prefP1[3][0]>0 ? 0 : 0-prefP1[3][0]]];
465 
466     var m = this.preferMatrix;
467 
468     // initiated with fallback position
469     // Will be used only if the following preferred alternative can not be found
470     if(m[4] === -1) {
471       //f.x = a.x>0 ? a.x+23 : 0; // another alternative align to left
472       f.x = a.x+parseInt(a.width/2,0);
473       f.y = a.y+parseInt(a.height/2,0)-parseInt(f.height/2,0);
474       this.set('pointerPos', SC.POINTER_LAYOUT[0]+' fallback');
475       this.set('pointerPosY', parseInt(f.height/2,0)-40);      
476     } else {
477       f.x = prefP1[m[4]][0];
478       f.y = prefP1[m[4]][1];
479       this.set('pointerPos', SC.POINTER_LAYOUT[m[4]]);
480       this.set('pointerPosY', 0);      
481     }
482     this.set('pointerPosX', 0);
483 
484     for(var i=0, cM, pointerLen=SC.POINTER_LAYOUT.length; i<pointerLen; i++) {
485       cM = m[i];
486       if (cutoffPrefP[cM][0]===0 && cutoffPrefP[cM][1]===0 && cutoffPrefP[cM][2]===0 && cutoffPrefP[cM][3]===0) {
487         // alternative i in preferMatrix by priority
488         if (m[4] !== cM) {
489           f.x = prefP1[cM][0] ;
490           f.y = prefP1[cM][1] ;
491           this.set('pointerPosY', 0);
492           this.set('pointerPos', SC.POINTER_LAYOUT[cM]);
493         }
494         i = SC.POINTER_LAYOUT.length;
495       } else if ((cM === 0 || cM === 1) && cutoffPrefP[cM][0]===0 && cutoffPrefP[cM][1]===0 && cutoffPrefP[cM][2] < f.height-91 && cutoffPrefP[cM][3]===0) {
496         if (m[4] !== cM) {
497           f.x = prefP1[cM][0] ;
498           this.set('pointerPos', SC.POINTER_LAYOUT[cM]);
499         }
500         f.y = prefP1[cM][1] - cutoffPrefP[cM][2];
501         this.set('pointerPosY', cutoffPrefP[cM][2]);
502         i = SC.POINTER_LAYOUT.length;
503       } else if ((cM === 0 || cM === 1) && cutoffPrefP[cM][0]===0 && cutoffPrefP[cM][1]===0 && cutoffPrefP[cM][2] <= f.height-51 && cutoffPrefP[cM][3]===0) {
504         if (m[4] !== cM) {
505           f.x = prefP1[cM][0] ;
506         }
507         f.y = prefP1[cM][1] - (f.height-51) ;
508         this.set('pointerPosY', (f.height-53));
509         this.set('pointerPos', SC.POINTER_LAYOUT[cM]+' extra-low');
510         i = SC.POINTER_LAYOUT.length;
511       } else if ((cM === 2 || cM === 3) && cutoffPrefP[cM][0]===0 && cutoffPrefP[cM][1]<= parseInt(f.width/2,0)-this.get('extraRightOffset') && cutoffPrefP[cM][2] ===0 && cutoffPrefP[cM][3]===0) {
512         if (m[4] !== cM) {
513           f.y = prefP1[cM][1] ;
514         }
515         f.x = prefP1[cM][0] - (parseInt(f.width/2,0)-this.get('extraRightOffset')) ;
516         this.set('pointerPos', SC.POINTER_LAYOUT[cM]+' extra-right');
517         i = SC.POINTER_LAYOUT.length;
518       } else if ((cM === 2 || cM === 3) && cutoffPrefP[cM][0]===0 && cutoffPrefP[cM][1]===0 && cutoffPrefP[cM][2] ===0 && cutoffPrefP[cM][3]<= parseInt(f.width/2,0)-this.get('extraRightOffset')) {
519         if (m[4] !== cM) {
520           f.y = prefP1[cM][1] ;
521         }
522         f.x = prefP1[cM][0] + (parseInt(f.width/2,0)-this.get('extraRightOffset')) ;
523         this.set('pointerPos', SC.POINTER_LAYOUT[cM]+' extra-left');
524         i = SC.POINTER_LAYOUT.length;
525       }
526     }
527     return f ;    
528   },

530   /** @private
531     This method will set up pointerOffset and preferMatrix according to type
532     and size if not provided explicitly.
533   */
534   setupPointer: function(a) {
535     var pointerOffset = this.pointerOffset,
536         K             = SC.PickerPane;
537     
538     // set up pointerOffset according to type and size if not provided explicitly
539     if (!pointerOffset || pointerOffset.length !== 4) {
540       if (this.get('preferType') == SC.PICKER_MENU_POINTER) {
541         switch (this.get('controlSize')) {
542           case SC.TINY_CONTROL_SIZE:
543             this.set('pointerOffset',    K.TINY_PICKER_MENU_POINTER_OFFSET) ;
544             this.set('extraRightOffset', K.TINY_PICKER_MENU_EXTRA_RIGHT_OFFSET) ;
545             break;
546           case SC.SMALL_CONTROL_SIZE:
547             this.set('pointerOffset',    K.SMALL_PICKER_MENU_POINTER_OFFSET) ;
548             this.set('extraRightOffset', K.SMALL_PICKER_MENU_EXTRA_RIGHT_OFFSET) ;
549             break;
550           case SC.REGULAR_CONTROL_SIZE:
551             this.set('pointerOffset',    K.REGULAR_PICKER_MENU_POINTER_OFFSET) ;
552             this.set('extraRightOffset', K.REGULAR_PICKER_MENU_EXTRA_RIGHT_OFFSET) ;
553             break;
554           case SC.LARGE_CONTROL_SIZE:
555             this.set('pointerOffset',    K.LARGE_PICKER_MENU_POINTER_OFFSET) ;
556             this.set('extraRightOffset', K.LARGE_PICKER_MENU_EXTRA_RIGHT_OFFSET) ;
557             break;
558           case SC.HUGE_CONTROL_SIZE:
559             this.set('pointerOffset',    K.HUGE_PICKER_MENU_POINTER_OFFSET) ;
560             this.set('extraRightOffset', K.HUGE_PICKER_MENU_EXTRA_RIGHT_OFFSET) ;
561             break;
562         }
563       }
564       else {
565         var overlapTuningX = (a.width < 16)  ? ((a.width < 4)  ? 9 : 6) : 0,
566             overlapTuningY = (a.height < 16) ? ((a.height < 4) ? 9 : 6) : 0,
567             offsetKey      = K.PICKER_POINTER_OFFSET;
568 
569         var offset = [offsetKey[0]+overlapTuningX,
570                       offsetKey[1]-overlapTuningX,
571                       offsetKey[2]-overlapTuningY,
572                       offsetKey[3]+overlapTuningY];
573         this.set('pointerOffset', offset) ;
574         this.set('extraRightOffset', K.PICKER_EXTRA_RIGHT_OFFSET) ;
575       }
576     }
577 
578     // set up preferMatrix according to type if not provided excplicitly:
579     // take default [0,1,2,3,2] for picker, [3,0,1,2,3] for menu picker if
580     // custom matrix not provided excplicitly
581     if(!this.preferMatrix || this.preferMatrix.length !== 5) {
582       // menu-picker default re-position rule :
583       // perfect bottom (3) > perfect right (0) > perfect left (1) > perfect top (2)
584       // fallback to perfect bottom (3)
585       // picker default re-position rule :
586       // perfect right (0) > perfect left (1) > perfect top (2) > perfect bottom (3)
587       // fallback to perfect top (2)
588       this.set('preferMatrix', this.get('preferType') == SC.PICKER_MENU_POINTER ? [3,0,1,2,3] : [0,1,2,3,2]) ;
589     }
590   },
591   
592   displayProperties: ["pointerPosY"],
593 
594   render: function(context, firstTime) {
595     var ret = sc_super();
596     if (context.needsContent) {
597       if (this.get('preferType') == SC.PICKER_POINTER || this.get('preferType') == SC.PICKER_MENU_POINTER) {
598         context.push('<div class="sc-pointer '+this.get('pointerPos')+'" style="margin-top: '+this.get('pointerPosY')+'px"></div>');
599         context.addClass(this.get('pointerPos'));
600       }
601     } else {
602       if (this.get('preferType') == SC.PICKER_POINTER || this.get('preferType') == SC.PICKER_MENU_POINTER) {
603         var el = this.$('.sc-pointer');
604         el.attr('class', "sc-pointer "+this.get('pointerPos'));
605         el.attr('style', "margin-top: "+this.get('pointerPosY')+"px");
606         context.addClass(this.get('pointerPos'));
607       }
608     }
609     return ret ;
610   },

612   /** @private - click away picker. */
613   modalPaneDidClick: function(evt) {
614     var f = this.get("frame");
615     if(!this.clickInside(f, evt)) this.remove();
616     return YES ; 
617   },
618 
619   mouseDown: function(evt) {
620     return this.modalPaneDidClick(evt);
621   },
622   
623   /** @private
624     internal method to define the range for clicking inside so the picker 
625     won't be clicked away default is the range of contentView frame. 
626     Over-write for adjustments. ex: shadow
627   */
628   clickInside: function(frame, evt) {
629     return SC.pointInRect({ x: evt.pageX, y: evt.pageY }, frame);
630   },

633     Invoked by the root responder. Re-position picker whenever the window resizes. 
634   */
635   windowSizeDidChange: function(oldSize, newSize) {
636     this.positionPane();
637   }
