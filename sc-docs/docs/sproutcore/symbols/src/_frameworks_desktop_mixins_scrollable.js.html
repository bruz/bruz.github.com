  7 /**
  8   @namespace
  9 
 10   Any views you implement that are scrollable should include this mixin to
 11   provide basic support for scrolling actions.  You can also override any 
 12   of these methods as needed for your own specific behaviors.
 13   
 14   Often times instead of adding SC.Scrollable to your view, you should
 15   place your view inside of an SC.ScrollView.  See that class for more 
 16   info.
 17   
 18   Note that isScrollable must always be true.
 19 
 20 */
 21 SC.Scrollable = {
 22   
 23 //@if(debug)
 24   initMixin: function() {
 25     console.warn("SC.Scrollable is deprecated and will be removed in a future version of SproutCore.  Consider pulling the mixin into your own app if you want to keep using it.");
 26   },
 27 //@endif 
 28 
 29   /** Informs the view system that the receiver is scrollable.
 30   
 31     Must always be true.
 32   */
 33   isScrollable: true,
 34   
 35   /** 
 36     Amount to scroll one vertical line.
 37   
 38     Used by the default implementation of scrollDownLine() and scrollUpLine().  Defaults
 39     to 20px.
 40   */
 41   verticalLineScroll: 20,
 42   
 43   /**
 44     Amount to scroll one horizontal line.
 45   
 46     Used by the default implementation of scrollLeftLine() and scrollRightLine(). Defaults
 47     to 20px.
 48   */
 49   horizontalLineScroll: 20,
 50   
 51   /**
 52     Amount to scroll one vertical page.
 53     
 54     Used by the default implementation of scrollUpPage() and scrollDownPage(). Defaults to
 55     current innerFrame height.
 56   */
 57   verticalPageScroll: function() {
 58     return this.get('innerFrame').height ;
 59   }.property('innerFrame'),
 60   
 61   /**
 62     Amount to scroll one horizontal page.
 63     
 64     Used by the default implementation of scrollLeftPage() and scrollRightPage().  Defaults
 65     to current innerFrame width.
 66   */
 67   horizontalPageScroll: function() {
 68     return this.get('innerFrame').width ;  
 69   }.property('innerFrame'),
 70   
 71   /**
 72     Returns true if the receiver has enough vertical content to require 
 73     scrolling.
 74     
 75     If you do not want to allow vertical scrolling, override this to be false 
 76     and set the appropriate CSS.
 77     
 78   */
 79   hasVerticalScroller: function() {
 80     return this.get('scrollFrame').height > this.get('innerFrame').height ;
 81   }.property('scrollFrame'),
 82   
 83   /**
 84     Returns true if the receiver has enough horizontal content to require 
 85     scrolling.
 86     
 87     If you do not want to allow horizontal scrolling, override this to be 
 88     false and set the appropriate CSS.
 89     
 90   */
 91   hasHorizontalScroller: function() {
 92     return this.get('scrollFrame').width > this.get('innerFrame').width ;
 93   }.property('scrollFrame'),
 94 
 95   /**
 96     Scrolls the receiver in the horizontal and vertical directions by the 
 97     amount specified, if allowed.
 98     
 99     @param {Point} amount the amount to scroll.  Must include x, y or both
100     @returns {Point} the actual amount scrolled.
101   */
102   scrollBy: function(amount) {
103     var sf = this.get('scrollFrame') ;
104     var f = this.get('innerFrame') ;
105 
106     if (!this.get('hasVerticalScroller')) amount.y = 0 ;
107     if (sf.height <= f.height) amount.y = 0 ;
108     
109     if (!this.get('hasHorizontalScroller')) amount.x = 0 ; 
110     if (sf.width <= f.width) amount.x = 0 ;
111 
112     // compute new sf
113     var newSf = { x: sf.x - (amount.x || 0), y: sf.y - (amount.y || 0) } ;
114     this.set('scrollFrame', newSf) ;
115     newSf = this.get('scrollFrame') ;
116     
117     return { x: newSf.x - sf.x, y: newSf.y - sf.y }; 
118   },
119 
120   /**
121     Scrolls the receiver to the specified x,y coordinate
122   */
123   scrollTo: function(x,y) {
124     this.set('scrollFrame', { x: 0-x, y: 0-y }) ;  
125   },
126   
127   /**
128     Scroll the view to make the passed frame visible.
129     
130     Frame must be relative to the receiver's offsetParent.
131     
132     @param {SC.ClassicView} view the view you want to make visible
133   */
134   scrollToVisible: function(view) {
135 
136     // get frames and convert them to proper offsets
137     var f = this.get('innerFrame') ;
138     var sf = this.get('scrollFrame') ;
139     
140     // frame of the view, relative to the top of the scroll frame
141     var vf = this.convertFrameFromView(view.get('frame'), view) ;
142     vf.x -= (f.x + sf.x); vf.y -= (f.y + sf.y);
143     
144     // first visible origin
145     var vo = { 
146       x: 0-sf.x, 
147       y: 0-sf.y, 
148       width: f.width, 
149       height: f.height 
150     };
151 
152     // if top edge is not visible, shift origin
153     vo.y -= Math.max(0, SC.minY(vo) - SC.minY(vf)) ;
154     vo.x -= Math.max(0, SC.minX(vo) - SC.minX(vf)) ;
155 
156     // if bottom edge is not visible, shift origin
157     vo.y += Math.max(0, SC.maxY(vf) - SC.maxY(vo)) ;
158     vo.x += Math.max(0, SC.maxX(vf) - SC.maxX(vo)) ;
159 
160     // scroll to that origin.
161     this.scrollTo(vo.x, vo.y) ;
162   },
163   
164   /**
165     Scrolls the receiver down one line if allowed.
166     
167     @param {Number} lines number of lines to scroll
168     @returns {Number} the amount actually scrolled.
169   */
170   scrollDownLine: function(lines) {
171     if (lines === undefined) lines = 1 ;
172     return this.scrollBy({ y: this.get('verticalLineScroll')*lines }).y ;
173   },
174 
175   /**
176     Scrolls the receiver down up line if allowed.
177     
178     @param {Number} lines number of lines to scroll
179     @returns {Number} the amount actually scrolled.
180   */
181   scrollUpLine: function(lines) {
182     if (lines === undefined) lines = 1 ;
183     return 0-this.scrollBy({ y: 0-this.get('verticalLineScroll')*lines }).y ;
184   },
185 
186   /**
187     Scrolls the receiver right one line if allowed.
188     
189     @param {Number} lines number of lines to scroll
190     @returns {Number} the amount actually scrolled.
191   */
192   scrollRightLine: function(lines) {
193     if (lines === undefined) lines = 1 ;
194     return this.scrollTo({ y: this.get('horizontalLineScroll')*lines }).x ;
195   },
196 
197   /**
198     Scrolls the receiver left one line if allowed.
199     
200     @param {Number} lines number of lines to scroll
201     @returns {Number} the amount actually scrolled.
202   */
203   scrollLeftLine: function(lines) {
204     if (lines === undefined) lines = 1 ;
205     return 0-this.scrollTo({ y: 0-this.get('horizontalLineScroll')*lines }).x ;
206   },
207 
208   /**
209     Scrolls the receiver down one page if allowed.
210     
211     @param {Number} pages number of pages to scroll
212     @returns {Number} the amount actually scrolled.
213   */
214   scrollDownPage: function(pages) {
215     if (pages === undefined) pages = 1 ;
216     return this.scrollBy({ y: this.get('verticalPageScroll')*pages }).y ;
217   },
218 
219   /**
220     Scrolls the receiver down up page if allowed.
221     
222     @param {Number} pages number of pages to scroll
223     @returns {Number} the amount actually scrolled.
224   */
225   scrollUpPage: function(pages) {
226     if (pages === undefined) pages = 1 ;
227     return 0-this.scrollBy({ y: 0-this.get('verticalPageScroll')*pages }).y ;
228   },
229 
230   /**
231     Scrolls the receiver right one page if allowed.
232     
233     @param {Number} pages number of pages to scroll
234     @returns {Number} the amount actually scrolled.
235   */
236   scrollRightPage: function(pages) {
237     if (pages === undefined) pages = 1 ;
238     return this.scrollTo({ y: this.get('horizontalPageScroll')*pages }).x ;
239   },
240 
241   /**
242     Scrolls the receiver left one page if allowed.
243     
244     @param {Number} pages number of pages to scroll
245     @returns {Number} the amount actually scrolled.
246   */
247   scrollLeftPage: function(pages) {
248     if (pages === undefined) pages = 1 ;
249     return 0-this.scrollTo({ y: 0-this.get('horizontalPageScroll')*pages }).x ;
250   }
251   
252 } ;
