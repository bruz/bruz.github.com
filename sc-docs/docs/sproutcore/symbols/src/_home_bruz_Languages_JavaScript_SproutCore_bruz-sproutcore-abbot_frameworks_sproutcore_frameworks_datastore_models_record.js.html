  8 /**
  9   @class
 10 
 11   A Record is the core model class in SproutCore. It is analogous to 
 12   NSManagedObject in Core Data and EOEnterpriseObject in the Enterprise
 13   Objects Framework (aka WebObjects), or ActiveRecord::Base in Rails.
 14   
 15   To create a new model class, in your SproutCore workspace, do:
 16   
 17   {{{
 18     $ sc-gen model MyApp.MyModel
 19   }}}
 20   
 21   This will create MyApp.MyModel in clients/my_app/models/my_model.js.
 22   
 23   The core attributes hash is used to store the values of a record in a 
 24   format that can be easily passed to/from the server.  The values should 
 25   generally be stored in their raw string form.  References to external 
 26   records should be stored as primary keys.
 27   
 28   Normally you do not need to work with the attributes hash directly.  
 29   Instead you should use get/set on normal record properties.  If the 
 30   property is not defined on the object, then the record will check the 
 31   attributes hash instead.
 32   
 33   You can bulk update attributes from the server using the 
 34   updateAttributes() method.
 35 
 36   @extends SC.Object
 37   @since SproutCore 1.0
 38 */

316   /**
317     Reads the raw attribute from the underlying data hash.  This method does
318     not transform the underlying attribute at all.
319   
320     @param {String} key the attribute you want to read
321     @returns {Object} the value of the key, or null if it doesn't exist

323   readAttribute: function(key) {
324     var store = this.get('store'), storeKey = this.storeKey;
325     var attrs = store.readDataHash(storeKey);
326     return attrs ? attrs[key] : undefined ;

329   /**
330     Updates the passed attribute with the new value.  This method does not 
331     transform the value at all.  If instead you want to modify an array or 
332     hash already defined on the underlying json, you should instead get 
333     an editable version of the attribute using editableAttribute()
334   
335     @param {String} key the attribute you want to read
336     @param {Object} value the value you want to write
337     @param {Boolean} ignoreDidChange only set if you do NOT want to flag 
338       record as dirty
339     @returns {SC.Record} receiver
340   */
341   writeAttribute: function(key, value, ignoreDidChange) {
342     var store    = this.get('store'), 
343         storeKey = this.storeKey,
344         status   = store.peekStatus(storeKey),
345         attrs;
346     
347     attrs = store.readEditableDataHash(storeKey);
348     if (!attrs) throw SC.Record.BAD_STATE_ERROR;
349 
350     // if value is the same, do not flag record as dirty
351     if (value !== attrs[key]) {
352       if(!ignoreDidChange) this.beginEditing();
353       attrs[key] = value;
354       
355       // If the key is the primaryKey of the record, we need to tell the store
356       // about the change.
357       if (key===this.get('primaryKey')) {
358         SC.Store.replaceIdFor(storeKey, value) ;
359         this.propertyDidChange('id'); // Reset computed value
360       }
361       
362       if(!ignoreDidChange) this.endEditing(key);
363     }
364     return this ;  
365   },

367   /**
368     This will also ensure that any aggregate records are also marked dirty
369     if this record changes.
370     
371     Should not have to be called manually.
372   */
373   propagateToAggregates: function() {
374     var storeKey = this.get('storeKey'),
375         recordType = SC.Store.recordTypeFor(storeKey), 
376         idx, len, key, val, recs;
377     
378     var aggregates = recordType.aggregates;
379     
380     // if recordType aggregates are not set up yet, make sure to 
381     // create the cache first
382     if (!aggregates) {
383       var dataHash = this.get('store').readDataHash(storeKey);
384       aggregates = [];
385       for(var k in dataHash) {
386         if(this[k] && this[k].get && this[k].get('aggregate')===YES) {
387           aggregates.push(k);
388         }
389       }
390       recordType.aggregates = aggregates;
391     }
392     
393     // now loop through all aggregate properties and mark their related
394     // record objects as dirty
395     var K          = SC.Record,
396         dirty      = K.DIRTY,
397         readyNew   = K.READY_NEW,
398         destroyed  = K.DESTROYED,
399         readyClean = K.READY_CLEAN,
400         iter;
401         
402     // If the child is dirty, then make sure the parent gets a dirty
403     // status.  (If the child is created or destroyed, there's no need,
404     // because the parent will dirty itself when it modifies that
405     // relationship.)
406     iter =  function(rec) {
407       var childStatus, parentStatus;
408       
409       if (rec) { 
410         childStatus = this.get('status');
411         if ((childStatus & dirty)  ||  
412             (childStatus & readyNew)  ||  (childStatus & destroyed)) {
413           parentStatus = rec.get('status');
414           if (parentStatus === readyClean) {
415             // Note:  storeDidChangeProperties() won't put it in the
416             //        changelog!
417             rec.get('store').recordDidChange(rec.constructor, null, rec.get('storeKey'), null, YES);
418           }
419         }
420       }
421     };
422         
423     for(idx=0,len=aggregates.length;idx<len;++idx) {
424       key = aggregates[idx];
425       val = this.get(key);
426       recs = SC.kindOf(val, SC.ManyArray) ? val : [val];
427       recs.forEach(iter, this);
428     }
429   },

432     Called by the store whenever the underlying data hash has changed.  This
433     will notify any observers interested in data hash properties that they
434     have changed.
435     
436     @param {Boolean} statusOnly changed
437     @param {String} key that changed (optional)

441     // TODO:  Should this function call propagateToAggregates() at the
442     //        appropriate times?
443     if (statusOnly) this.notifyPropertyChange('status');
444     else {      
445       if (keys) {
446         this.beginPropertyChanges();
447         keys.forEach(function(k) { this.notifyPropertyChange(k); }, this);
448         this.notifyPropertyChange('status'); 
449         this.endPropertyChanges();
450 
451       } else this.allPropertiesDidChange(); 
452     
453       // also notify manyArrays
454       var manyArrays = this.relationships,
455           loc        = manyArrays ? manyArrays.length : 0 ;
456       while(--loc>=0) manyArrays[loc].recordPropertyDidChange(keys);
457     }

460   /**
461     Normalizing a record will ensure that the underlying hash conforms
462     to the record attributes such as their types (transforms) and default 
463     values. 
464     
465     This method will write the conforming hash to the store and return
466     the materialized record.
467     
468     By normalizing the record, you can use .attributes() and be
469     assured that it will conform to the defined model. For example, this
470     can be useful in the case where you need to send a JSON representation
471     to some server after you have used .createRecord(), since this method
472     will enforce the 'rules' in the model such as their types and default
473     values. You can also include null values in the hash with the 
474     includeNull argument.
475     
476     @param {Boolean} includeNull will write empty (null) attributes
477     @returns {SC.Record} the normalized record
478   */
479   
480   normalize: function(includeNull) {    
481     var primaryKey = this.primaryKey, 
482         recordId   = this.get('id'), 
483         store      = this.get('store'), 
484         storeKey   = this.get('storeKey'), 
485         key, valueForKey, typeClass, recHash, attrValue, normChild,  isRecord,
486         isChild, defaultVal, keyForDataHash;
487       
488     var dataHash = store.readEditableDataHash(storeKey) || {};
489     dataHash[primaryKey] = recordId;
490     recHash = store.readDataHash(storeKey);
491     
492     for (key in this) {
493       // make sure property is a record attribute.
494       valueForKey = this[key];
495       if (valueForKey) {
496         typeClass = valueForKey.typeClass;
497         if (typeClass) {
498           keyForDataHash = valueForKey.get('key') || key; // handle alt keys
499           isRecord = SC.typeOf(typeClass.call(valueForKey))===SC.T_CLASS;
500           isChild  = valueForKey.isChildRecordTransform;
501           if (!isRecord && !isChild) {
502             attrValue = this.get(key);
503 
504             if(attrValue!==undefined || (attrValue===null && includeNull)) {
505               dataHash[keyForDataHash] = attrValue;
506             }
507           
508           } else if (isChild) {
509             attrValue = this.get(key);
510 
511             // Sometimes a child attribute property does not refer to a child record.
512             // Catch this and don't try to normalize.
513             if (attrValue && attrValue.normalize) {
514               attrValue.normalize();
515             }
516           } else if (isRecord) {
517             attrValue = recHash[key];
518             if (attrValue !== undefined) {
519               // write value already there
520               dataHash[keyForDataHash] = attrValue;
521             } else {
522               // or write default
523               defaultVal = valueForKey.get('defaultValue');
524 
525               // computed default value
526               if (SC.typeOf(defaultVal)===SC.T_FUNCTION) {
527                 dataHash[keyForDataHash] = defaultVal(this, key, defaultVal);
528               } else {
529                 // plain value                
530                 dataHash[keyForDataHash] = defaultVal;
531               }
532             }
533           }
534         }
535       }
536     }
537 
538     return this;
539   },

543   /**
544     If you try to get/set a property not defined by the record, then this 
545     method will be called. It will try to get the value from the set of 
546     attributes.
547     
548     This will also check is ignoreUnknownProperties is set on the recordType
549     so that they will not be written to dataHash unless explicitly defined
550     in the model schema.
551   
552     @param {String} key the attribute being get/set
553     @param {Object} value the value to set the key to, if present
554     @returns {Object} the value
555   */
556   unknownProperty: function(key, value) {
557     
558     if (value !== undefined) {
559       
560       // first check if we should ignore unknown properties for this 
561       // recordType
562       var storeKey = this.get('storeKey'),
563         recordType = SC.Store.recordTypeFor(storeKey);
564       
565       if(recordType.ignoreUnknownProperties===YES) {
566         this[key] = value;
567         return value;
568       }
569       
570       // if we're modifying the PKEY, then SC.Store needs to relocate where 
571       // this record is cached. store the old key, update the value, then let 
572       // the store do the housekeeping...
573       var primaryKey = this.get('primaryKey');
574       this.writeAttribute(key,value);
575 
576       // update ID if needed
577       if (key === primaryKey) {
578         SC.Store.replaceIdFor(storeKey, value);
579       }
580       
581     }
582     return this.readAttribute(key);
583   },

640   /** @private
641     Creates string representation of record, with status.
642     
643     @returns {String}
644   */
645   
646   toString: function() {
647     // We won't use 'readOnlyAttributes' here because accessing them directly
648     // avoids a SC.clone() -- we'll be careful not to edit anything.
649     var attrs = this.get('store').readDataHash(this.get('storeKey'));
650     return "%@(%@) %@".fmt(this.constructor.toString(), SC.inspect(attrs), this.statusString());
651   },
652   
653   /** @private
654     Creates string representation of record, with status.
655     
656     @returns {String}
657   */
658   
659   statusString: function() {
660     var ret = [], status = this.get('status');
661     
662     for(var prop in SC.Record) {
663       if(prop.match(/[A-Z_]$/) && SC.Record[prop]===status) {
664         ret.push(prop);
665       }
666     }
667     
668     return ret.join(" ");
669   },

671   /**
672    * Registers a child record with this parent record.
673    *
674    * If the parent already knows about the child record, return the cached instance.  If not,
675    * create the child record instance and add it to the child record cache.
676    *
677    * @param {CoreOrion.ChildRecord} recordType The type of the child record to register.
678    * @param {Hash} hash The hash of attributes to apply to the child record.
679    */
680   registerChildRecord: function(recordType, hash) {
681     var pm = recordType.primaryKey || 'childRecordKey';
682     var childKey = hash[pm];
683     var childRecord = null;
684     var crManager = this.get('childRecords');
685     if (childKey && crManager) {
686       childRecord = crManager[childKey];
687     }
688 
689     if (SC.none(childRecord)) childRecord = this.createChildRecord(recordType, hash);
690  
691     return childRecord;
692   },
693   
694   /**
695    * Creates a new child record instance.
696    *
697    * @param {CoreOrion.ChildRecord} recordType The type of the child record to create.
698    * @param {Hash} hash The hash of attributes to apply to the child record. (may be null)
699    */
700   createChildRecord: function(childRecordType, hash) {
701     SC.RunLoop.begin();
702     // Generate the key used by the parent's child record manager.
703     var key = SC.Record._generateChildKey();
704     hash = hash || {}; // init if needed
705     var pm = childRecordType.primaryKey || 'childRecordKey';
706     var childKey = hash[pm];
707     hash[pm] = key;
708     
709     var store = this.get('store');
710     if (SC.none(store)) throw 'Error: during the creation of a child record: NO STORE ON PARENT!';
711     
712     var cr = store.createRecord(childRecordType, hash);
713     cr._parentRecord = this;
714     
715     // ID processing if necessary
716     if(this.generateIdForChild) this.generateIdForChild(cr);
717     
718     // Add the child record to the hash.
719     var crManager = this.get('childRecords');
720     if (SC.none(crManager)) {
721       //console.log('Creating Child Record Manager for (%@)'.fmt(SC.guidFor(this)));
722       crManager = SC.Object.create();
723       this.set('childRecords', crManager);
724     }
725     
726     crManager[key] = cr;
727     SC.RunLoop.end();
728     
729     return cr;
730   },

991   /**
992     Helper method returns a new SC.RecordAttribute instance to map a simple
993     value or to-one relationship.  At the very least, you should pass the 
994     type class you expect the attribute to have.  You may pass any additional

996     
997     Use this helper when you define SC.Record subclasses. 
998     
999     h4. Example
1000     
1001     {{{
1002       MyApp.Contact = SC.Record.extend({
1003         firstName: SC.Record.attr(String, { isRequired: YES })
1004       });
1005     }}}

1007     @param {Class} type the attribute type
1008     @param {Hash} opts the options for the attribute
1009     @returns {SC.RecordAttribute} created instance

1015   /**
1016     Returns an SC.RecordAttribute that describes a fetched attribute.  When

1020     Use this helper when you define SC.Record subclasses.

1022     h4. Example

1026         contacts: SC.Record.fetch('MyApp.Contact')

1038   /**
1039     Returns:
1040     
1041     1: SC.ManyAttribute that describes a record array backed by an 
1042     array of guids stored in the underlying JSON.  
1043     2: SC.ChildrenAttribute that describes a record array backed by a
1044     array of hashes.
1045     
1046     You can edit the contents of this relationship.
1047     
1048     For SC.ManyAttribute, If you set the inverse and isMaster: NO key, 
1049     then editing this array will modify the underlying data, but the 
1050     inverse key on the matching record will also be edited and that 
1051     record will be marked as needing a change.
1052     
1053     @param {SC.Record|String} recordType The type of record to create
1054     @param {Hash} opts the options for the attribute
1055     @returns {SC.ManyAttribute|SC.ChildrenAttribute} created instance
1056   */
1057   toMany: function(recordType, opts) {
1058     opts = opts || {};
1059     var isNested = opts.nested;
1060     var attr;
1061     if(isNested){
1062       attr = SC.ChildrenAttribute.attr(recordType, opts);
1063     }
1064     else {
1065       attr = SC.ManyAttribute.attr(recordType, opts);
1066     }
1067     return attr;
1068   },

1071     Returns:
1072     1. SC.SingleAttribute that converts the underlying ID to a single
1073     record.  If you modify this property, it will rewrite the underyling ID. 
1074     It will also modify the inverse of the relationship, if you set it.
1075     
1076     2. SC.ChildAttribute that you can edit the contents
1077     of this relationship.
1078     
1079     @param {SC.Record|String} recordType the type of the record to create
1080     @param {Hash} opts additional options
1081     @returns {SC.SingleAttribute|SC.ChildAttribute} created instance
1082   */
1083   toOne: function(recordType, opts) {

1090     else {
1091       attr = SC.SingleAttribute.attr(recordType, opts);
1092     }
1093     return attr;

1096   /**
1097     Returns all storeKeys mapped by Id for this record type.  This method is
1098     used mostly by the SC.Store and the Record to coordinate.  You will rarely
1099     need to call this method yourself.
1100     
1101     @returns {Hash}
1102   */
1103   storeKeysById: function() {
1104     var key = SC.keyFor('storeKey', SC.guidFor(this)),
1105         ret = this[key];
1106     if (!ret) ret = this[key] = {};
1107     return ret;

1110   /**
1111     Given a primaryKey value for the record, returns the associated
1112     storeKey.  If the primaryKey has not been assigned a storeKey yet, it 
1113     will be added.
1114     
1115     For the inverse of this method see SC.Store.idFor() and 
1116     SC.Store.recordTypeFor().
1117     
1118     @param {String} id a record id
1119     @returns {Number} a storeKey.
1120   */
1121   storeKeyFor: function(id) {
1122     var storeKeys = this.storeKeysById(),
1123         ret       = storeKeys[id];
1124     
1125     if (!ret) {
1126       ret = SC.Store.generateStoreKey();
1127       SC.Store.idsByStoreKey[ret] = id ;
1128       SC.Store.recordTypesByStoreKey[ret] = this ;
1129       storeKeys[id] = ret ;
1130     }
1131     
1132     return ret ;
1133   },

1141     @returns {Number} a storeKey.
